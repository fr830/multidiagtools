00048 int mdtUsbDeviceDescriptor::fetchAttributes(libusb_device *device, bool fetchActiveConfigOnly)
00049 {
00050   Q_ASSERT(device != 0);
00051 
00052   struct libusb_device_descriptor descriptor;
00053   struct libusb_config_descriptor *configDescriptor;
00054   int err;
00055   quint8 i;

00057 
00058   // Open descriptor
00059   err = libusb_get_device_descriptor(device, &descriptor);
00060   if(err != 0){
00061     return err;
00062   }
00063   // Store attributes
00064   pvbDescriptorType = descriptor.bDescriptorType;
00065   pvbcdUSB = descriptor.bcdUSB;
00066   pvbDeviceClass = descriptor.bDeviceClass;

00069   pvbMaxPacketSize0 = descriptor.bMaxPacketSize0;
00070   pvidVendor = descriptor.idVendor;
00071   pvidProduct = descriptor.idProduct;
00072   pvbcdDevice = descriptor.bcdDevice;
00073   // Get configuration(s)
00074   qDeleteAll(pvConfigs);
00075   pvConfigs.clear();
00076   if(fetchActiveConfigOnly){
00077     err = libusb_get_active_config_descriptor(device, &configDescriptor);
00078     switch(err){
00079       case 0:
00080         config = new mdtUsbConfigDescriptor;
00081         Q_ASSERT(config != 0);
00082         config->fetchAttributes(configDescriptor);
00083         libusb_free_config_descriptor(configDescriptor);
00084         pvConfigs.append(config);
00085         break;
00086       case LIBUSB_ERROR_NOT_FOUND:
00087       {
00088         mdtError e(MDT_USB_IO_ERROR, "Unable to find get current configuration descriptor", mdtError::Warning);
00089         MDT_ERROR_SET_SRC(e, "mdtUsbDeviceDescriptor");
00090         e.commit();
00091         break;
00092       }
00093       default:
00094         return err;
00095     }
00096   }else{
00097     for(i=0; i<descriptor.bNumConfigurations; i++){
00098       err = libusb_get_config_descriptor(device, i, &configDescriptor);
00099       switch(err){
00100         case 0:
00101           config = new mdtUsbConfigDescriptor;
00102           Q_ASSERT(config != 0);
00103           config->fetchAttributes(configDescriptor);
00104           libusb_free_config_descriptor(configDescriptor);
00105           pvConfigs.append(config);
00106           break;
00107         case LIBUSB_ERROR_NOT_FOUND:
00108         {
00109           mdtError e(MDT_USB_IO_ERROR, "Unable to find get configuration descriptor at index " + QString::number(i), mdtError::Warning);
00110           MDT_ERROR_SET_SRC(e, "mdtUsbDeviceDescriptor");
00111           e.commit();
00112           break;
00113         }
00114         default:
00115           qDeleteAll(pvConfigs);
00116           pvConfigs.clear();
00117           return err;
00118       }
00119     }
00120   }
00121 
00122   return 0;
