00103   mdtDeviceIosSegment *segment;

00107   // Check that I/O was not allready set and add it
00108   Q_ASSERT(!pvAnalogInputs.contains(ai));
00109   Q_ASSERT(!pvAnalogInputsByAddressRead.contains(ai->addressRead()));
00110   pvAnalogInputs.append(ai);
00111   pvAnalogInputsByAddressRead.insert(ai->addressRead(), ai);
00112   Q_ASSERT(pvAnalogInputsByAddressRead.values().size() > 0);
00113   Q_ASSERT(pvAnalogInputsByAddressRead.values().at(0) != 0);
00114   // We must reorganize segments - We use the QMap container, because it is allready sorted by keys, ascending
00115   qDeleteAll(pvAnalogInputsSegments);
00116   pvAnalogInputsSegments.clear();
00117   aiList = pvAnalogInputsByAddressRead.values();
00118   Q_ASSERT(aiList.size() > 0);
00119   // We know that we have at least 1 I/O - Just create a segment and add first item
00120   segment = new mdtDeviceIosSegment;
00121   segmentAiList.append(aiList.at(0));
00122   for(i = 1; i < aiList.size(); ++i){
00123     Q_ASSERT(aiList.at(i) != 0);
00124     // If address of current item is not directly successor of previous, we have a hole -> create a new segment
00125     if((aiList.at(i)->addressRead() - aiList.at(i-1)->addressRead() ) != 1){
00126       // Add current segment to list and create a new one
00127       segment->setIos(segmentAiList);
00128       segmentAiList.clear();
00129       pvAnalogInputsSegments.append(segment);
00130       segment = new mdtDeviceIosSegment;
00131     }
00132     segmentAiList.append(aiList.at(i));
00133   }
00134   // Add current segment to list
00135   segment->setIos(segmentAiList);
00136   pvAnalogInputsSegments.append(segment);
