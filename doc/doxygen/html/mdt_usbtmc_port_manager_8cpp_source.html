00258   quint8 status;

00260 
00261   // Flush all endpoints + internal buffers

00263   // Send the INITIATE_ABORT_BULK_IN request and wait on response

00265   maxTry = 10;
00266   while((status == 0x81)&&(maxTry > 0)){
00268     retVal = sendInitiateAbortBulkInRequest(bTag);
00269     if(retVal < 0){
00270       // Error logged by USB port manager

00272     }
00273     if(!waitReadenControlResponse()){
00274       mdtError e(MDT_USB_IO_ERROR, "Timeout by INITIATE_ABORT_BULK_IN request", mdtError::Error);
00275       MDT_ERROR_SET_SRC(e, "mdtUsbtmcPortManager");
00276       e.commit();

00278     }
00279     frames = readenControlResponses();
00280     if(frames.size() != 1){
00281       mdtError e(MDT_USB_IO_ERROR, "INITIATE_ABORT_BULK_IN returned unexpected amount of responses", mdtError::Error);
00282       MDT_ERROR_SET_SRC(e, "mdtUsbtmcPortManager");
00283       e.commit();
00284       return mdtAbstractPort::UnhandledError;
00285     }
00286     frame = frames.at(0);
00287     if(frame.size() != 2){
00288       mdtError e(MDT_USB_IO_ERROR, "INITIATE_ABORT_BULK_IN returned unexpected response size (expected 2, received " + QString::number(frame.size()) + ")", mdtError::Error);
00289       MDT_ERROR_SET_SRC(e, "mdtUsbtmcPortManager");
00290       e.commit();
00291       return mdtAbstractPort::UnhandledError;
00292     }
00293     // Check status
00294     status = frame.at(0);
00295     if(status == 0x81){ // STATUS_TRANSFER_NOT_IN_PROGRESS
00296       wait(1000);
00297     }
00298     maxTry--;
00299   }
00300   if(maxTry < 1){
00301     mdtError e(MDT_USB_IO_ERROR, "INITIATE_ABORT_BULK_IN still pending after 10 try", mdtError::Warning);
00302     MDT_ERROR_SET_SRC(e, "mdtUsbtmcPortManager");
00303     e.commit();
00304     // Continue working (possibly, nothing was to abort)
00305     return 0;
00306   }
00307   // Check returned status
00308   if(status != 0x01){ // 0x01: STATUS_SUCCESS
00309     // No transfer in progress (device FIFO empty)
00310     return 0;
00311   }
00312   // Flush device
00313   readUntilShortPacketReceived();
00314   // Send the CHECK_ABORT_BULK_IN_STATUS request and wait on response
00315   status = 0x02;  // STATUS_PENDING
00316   while(status == 0x02){
00318     retVal = sendCheckAbortBulkInStatusRequest(bTag);
00319     if(retVal < 0){
00320       // Error logged by USB port manager
00321       return retVal;
00322     }
00323     if(!waitReadenControlResponse()){
00324       mdtError e(MDT_USB_IO_ERROR, "Timeout by CHECK_ABORT_BULK_IN_STATUS request", mdtError::Error);
00325       MDT_ERROR_SET_SRC(e, "mdtUsbtmcPortManager");
00326       e.commit();
00327       return mdtAbstractPort::ControlTimeout;
00328     }
00329     frames = readenControlResponses();
00330     if(frames.size() != 1){
00331       mdtError e(MDT_USB_IO_ERROR, "CHECK_ABORT_BULK_IN_STATUS returned unexpected amount of responses", mdtError::Error);
00332       MDT_ERROR_SET_SRC(e, "mdtUsbtmcPortManager");
00333       e.commit();
00334       return mdtAbstractPort::UnhandledError;
00335     }
00336     frame = frames.at(0);
00337     if(frame.size() != 8){
00338       mdtError e(MDT_USB_IO_ERROR, "CHECK_ABORT_BULK_IN_STATUS returned unexpected response size (expected 8, received " + QString::number(frame.size()) + ")", mdtError::Error);
00339       MDT_ERROR_SET_SRC(e, "mdtUsbtmcPortManager");
00340       e.commit();
00341       return mdtAbstractPort::UnhandledError;
00342     }
00343     // Check bmAbortBulkIn.D0
00344     if((quint8)frame.at(1) & 0x01){
00345       qDebug() << "*-* Flushing device ...";
00346       readUntilShortPacketReceived();
00347     }
00348     // Check status
00349     status = frame.at(0);
00350     if(status == 0x02){ // STATUS_PENDING
00351       wait(1000);
