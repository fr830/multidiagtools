00261 {
00262   int columnIndex;

00264   int rowCount;
00265   QModelIndex index;

00275   // Search ...
00276   rowCount = model()->rowCount();
00277   for(row = 0; row < rowCount; ++row){
00278     index = model()->index(row, columnIndex);

00280       if(model()->data(index) == value){
00281         setCurrentIndex(row);

00399 bool mdtAbstractSqlWidget::restorePrimaryKeyDataToModel(const QSqlRecord &previousRecord)
00400 {
00401   Q_ASSERT(pvModel != 0);
00402 
00403   QSqlIndex primaryKey;
00404   int pkIndex;
00405   int indexInRecord;
00406   QSqlRecord currentRecord;
00407   bool pkUpdated = false;
00408 
00409   currentRecord = pvModel->record(currentRow());
00410   primaryKey = pvModel->primaryKey();
00411   for(pkIndex = 0; pkIndex < primaryKey.count(); ++pkIndex){
00412     indexInRecord = previousRecord.indexOf(primaryKey.fieldName(pkIndex));
00413     if(currentRecord.value(indexInRecord) != previousRecord.value(indexInRecord)){
00414       pkUpdated = true;
00415       currentRecord.setValue(indexInRecord, previousRecord.value(indexInRecord));
00416     }
00417   }
00418   if(pkUpdated){
00419     if(!pvModel->setRecord(currentRow(), currentRecord)){
00420       return false;
00421     }
00422   }
00423 
00424   return true;
00425 }
