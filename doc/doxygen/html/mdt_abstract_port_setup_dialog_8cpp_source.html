00169 void mdtAbstractPortSetupDialog::on_buttonBox_clicked(QAbstractButton *button)
00170 {
00171   Q_ASSERT(button != 0);
00172   Q_ASSERT(pvPortManager != 0);
00173 
00174   QDialogButtonBox::StandardButton type;
00175 
00176   type = buttonBox->standardButton(button);
00177   switch(type){
00178     case QDialogButtonBox::Ok:
00179       // If port manager is ready, setup was allready applied, or nothing was changed
00180       if(portManager()->isReady()){
00181         close();
00182       }else{
00183         if(applySetup()){
00184           close();
00185         }
00186       }
00187       break;
00188     case QDialogButtonBox::Apply:
00189       applySetup();
00190       break;
00191     case QDialogButtonBox::Cancel:
00192       close();

00195       mdtError e(MDT_GUI_ERROR, "Recived signal from buttonBox with unknown button type", mdtError::Warning);
