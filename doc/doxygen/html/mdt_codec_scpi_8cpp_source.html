162 mdtValueDouble mdtCodecScpi::decodeValueDouble(const QByteArray & data)
163 {
164  mdtValueDouble x, y;
165  QByteArray str = data.trimmed();
166  bool ok;
167 
168  // Remove 0+ prefix, if present
169  if(str.size() > 2){
170  if(str.left(2) == "0+"){
171  str.remove(0, 2);
172  }
173  }
174  // Convert
175  x = str.toDouble(&ok);
176  if(!ok){
177  pvLastError.setError(QObject::tr("data") + " '" + data + "' " + QObject::tr("contains not a decimal value."), mdtError::Error);
178  MDT_ERROR_SET_SRC(pvLastError, "mdtCodecScpi");
179  pvLastError.commit();
180  return mdtValueDouble();
181  }
182  // Check about NaN
183  y.setValue(9.91e37);
184  if(x == y){
185  x.setValue(std::numeric_limits<double>::quiet_NaN());
186  return x;
187  }
188  // Check about OL flag
189  y.setValue(9.9e37);
190  if(x >= y){
191  x.setValue(9.9e37, true);
192  return x;
193  }
194  y.setValue(-9.9e37);
195  if(x <= y){
196  x.setValue(-9.9e37, true);
197  return x;
198  }
199 
200  return x;

253  fp.parametersRaw = str.mid(idx).trimmed();
