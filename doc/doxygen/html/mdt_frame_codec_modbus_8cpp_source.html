00091     e.commit();

00093   }
00094   // Get the function code
00095   functionCode = pvPdu.at(0);
00096   // Case of error code
00097   if(functionCode > 0x80){
00098     decodeModbusError(pvPdu.at(1));
00099     return functionCode;
00100   }
00101   // Decode PDU according functionCode
00102   switch(functionCode){
00103     // Read coils
00104     case 0x01:
00105       if(!decodeReadCoils()){
00106         return -1;
00107       }
00108       return 0x01;
00109       break;
00110     // Write single coil
00111     case 0x05:
00112       if(!decodeWriteSingleCoil()){
00113         return -1;
00114       }
00115       return 0x05;
00116       break;
00117     // Unknow code
00118     default:
00119       QByteArray num;
00121       num.setNum(functionCode);
