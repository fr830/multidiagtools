00290 mdtAbstractPort::error_t mdtUsbtmcPortThread::abortBulkOut(quint8 bTag, mdtPortThreadHelperPort &threadHelper)
00291 {
00292   Q_ASSERT(usbPort() != 0);
00293 
00294   mdtAbstractPort::error_t portError;
00295   mdtFrameUsbControl *ctlFrame;
00296   quint8 status = 0;
00297   int maxTry;
00298 
00299   qDebug() << "*-* Abort bulk OUT beginns for bTag " << bTag << " ...";
00300   notifyError(mdtAbstractPort::ReadWriteBusy);
00301 
00302   // Check if a query is pending
00303   if(threadHelper.currentWriteFrame() != 0){
00304     if(threadHelper.currentWriteFrame()->waitAnAnswer()){
00305       portError = usbPort()->cancelWriteTransfer();
00306       if(portError != mdtAbstractPort::NoError){
00307         return portError;
00308       }
00309       threadHelper.restoreCurrentWriteFrameToPool();
00310     }
00311   }
00312   // Wait until we can do a control transfer
00313   portError = waitOnControlTransferPossible(true);
00314   if(portError != mdtAbstractPort::NoError){
00315     return portError;
00316   }
00317   // Get a control frame
00318   Q_ASSERT(usbPort()->controlFramesPool().size() > 0);
00319   ctlFrame = usbPort()->controlFramesPool().dequeue();
00320   Q_ASSERT(ctlFrame != 0);
00321   // INITIATE_ABORT_BULK_OUT
00322   maxTry = 5;
00323   while(1){
00324     // Check maxTry
00325     if(maxTry < 1){
00326       mdtError e(MDT_USB_IO_ERROR, "INITIATE_ABORT_BULK_OUT still pending after 5 try", mdtError::Warning);
00327       MDT_ERROR_SET_SRC(e, "mdtUsbtmcPortThread");
00328       e.commit();
00329       usbPort()->controlFramesPool().enqueue(ctlFrame);
00330       return mdtAbstractPort::UnhandledError;
00331     }
00332     --maxTry;
00333     // Do control transfer
00334     portError = initiateAbortBulkOut(bTag, ctlFrame, status);
00335     if(portError != mdtAbstractPort::NoError){
00336       // Control frame was restored by initiateAbortBulkOut()
00337       return portError;
00338     }
00339     // Check status
00340     if(status == 0x01){       // 0x01 - STATUS_SUCCESS - Continue abort process
00341       break;
00342     }else if(status == 0x81){ // 0x81 - STATUS_TRANSFER_NOT_IN_PROGRESS - Retry later
00343       msleep(100);
00344     }else{                    // 0x80 - STATUS_FAILED - Finish abort process (Possibly nothing was to abort)
00345       usbPort()->controlFramesPool().enqueue(ctlFrame);
00346       notifyError(mdtAbstractPort::NoError);
00347       return mdtAbstractPort::NoError;
00348     }
00349   }
00350   // CHECK_ABORT_BULK_OUT_STATUS
00351   maxTry = 5;
00352   while(1){
00353     // Check maxTry
00354     if(maxTry < 1){
00355       mdtError e(MDT_USB_IO_ERROR, "CHECK_ABORT_BULK_OUT_STATUS still pending after 5 try", mdtError::Warning);
00356       MDT_ERROR_SET_SRC(e, "mdtUsbtmcPortThread");
00357       e.commit();
00358       usbPort()->controlFramesPool().enqueue(ctlFrame);
00359       return mdtAbstractPort::UnhandledError;
00360     }
00361     --maxTry;
00362     // Do control transfer
00363     portError = checkAbortBulkOutStatus(bTag, ctlFrame, status);
00364     if(portError != mdtAbstractPort::NoError){
00365       // Control frame was restored by initiateAbortBulkOut()
00366       return portError;
00367     }
00368     // Check status
00369     if(status == 0x02){ // 0x02 - STATUS_PENDING - Retry later
00370       msleep(100);
00371     }else{              // 0x01 - STATUS_SUCCESS or other - Clear halt
00372       break;
00373     }
00374   }
00375   usbPort()->controlFramesPool().enqueue(ctlFrame);
00376   portError = usbPort()->clearWriteEndpointHalt();
00377   if(portError == mdtAbstractPort::NoError){
00378     notifyError(mdtAbstractPort::NoError);
00379   }
00380 
00381   return portError;
