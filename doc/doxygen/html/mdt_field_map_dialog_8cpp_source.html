00128 void mdtFieldMapDialog::updateMappingTableView()
00129 {
00130   QList<mdtFieldMapItem*> items;
00131   QMultiMap<int, mdtFieldMapItem*> sortMap;
00132   mdtFieldMapItem *fmItem;
00133   QTableWidgetItem *twSourceItem, *twDestinationItem;
00134   int row, i;
00135   QString str;
00136   QStringList mappingViewHeader;
00137 
00138   // Get map items and sort them by source field indexes
00139   items = pvFieldMap.items();
00140   i = items.size() - 1;
00141   while(i >= 0){
00142     fmItem = items.at(i);
00143     Q_ASSERT(fmItem != 0);
00144     sortMap.insert(fmItem->sourceFieldIndex(), fmItem);
00145     --i;
00146   }
00147   items = sortMap.values();
00148   // Update view
00149   twMapping->setRowCount(items.size());
00150   // Set header if needed
00151   if(twMapping->horizontalHeaderItem(0) == 0){
00152     mappingViewHeader << tr("Source field") << " -> " << tr("Destination field");
00153     twMapping->setColumnCount(mappingViewHeader.size());
00154     twMapping->setHorizontalHeaderLabels(mappingViewHeader);
00155   }
00156   for(row = 0; row < items.size(); ++row){
00157     fmItem = items.at(row);
00158     Q_ASSERT(fmItem != 0);
00159     // Update source field
00160     twSourceItem = twMapping->item(row, 0);
00161     if(twSourceItem == 0){
00162       twSourceItem = new QTableWidgetItem;
00163       twMapping->setItem(row, 0, twSourceItem);
00164     }
00165     if(fmItem->sourceFieldDisplayText().isEmpty()){
00166       str = fmItem->sourceFieldName();
00167     }else{
00168       str = fmItem->sourceFieldDisplayText();
00169     }
00170     if(fmItem->sourceFieldDataStartOffset() > -1){
00171       str += " [" + QString::number(fmItem->sourceFieldDataStartOffset()) + "-";
00172       if(fmItem->sourceFieldDataEndOffset() > -1){
00173         str += QString::number(fmItem->sourceFieldDataEndOffset());
00174       }else{
00175         str += "*";
00176       }
00177       str += "]";
00178     }
00179     twSourceItem->setData(Qt::DisplayRole, str);
00180     // Update destination field
00181     twDestinationItem = twMapping->item(row, 2);
00182     if(twDestinationItem == 0){
00183       twDestinationItem = new QTableWidgetItem;
00184       twMapping->setItem(row, 2, twDestinationItem);
00185     }
00186     if(fmItem->destinationFieldDisplayText().isEmpty()){
00187       twDestinationItem->setData(Qt::DisplayRole, fmItem->destinationFieldName());
00188     }else{
00189       twDestinationItem->setData(Qt::DisplayRole, fmItem->destinationFieldDisplayText());
00190     }
00191     // Store field map item pointer
00192     twSourceItem->setData(Qt::UserRole, QVariant::fromValue(fmItem));
00193     twDestinationItem->setData(Qt::UserRole, QVariant::fromValue(fmItem));
00194   }
00195   twMapping->resizeColumnsToContents();
00196   twMapping->resizeRowsToContents();
