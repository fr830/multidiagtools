315 void mdtSqlTableWidget::copyTableToClipBoard()
316 {
317  QClipboard *clipboard = QApplication::clipboard();
318  QString data;
319  int row, col;
320 
321  // We call sort, so all data will be fetched (and sorted)
322  sort();
323  // Copy header
324  for(col = 0; col < pvController->columnCount(); ++col){
325  if(!pvTableView->isColumnHidden(col)){
326  data += "\"" + pvController->headerData(col).toString() + "\"";
327  if(col < (pvController->columnCount()-1)){
328  data += "\t";
329  }
330  }
331  }
332  data += "\n";
333  // copy data
334  for(row = 0; row < pvController->rowCount(false); ++row){
335  for(col = 0; col < pvController->columnCount(); ++col){
336  if(!pvTableView->isColumnHidden(col)){
337  data += "\"" + pvController->data(row, col).toString() + "\"";
338  if(col < (pvController->columnCount()-1)){
339  data += "\t";
340  }
341  }
342  }
343  data += "\n";
344  }
345  // Send to clipboard
346  clipboard = QApplication::clipboard();
347  Q_ASSERT(clipboard != 0);
348  QMimeData *mimeData = new QMimeData;
349  mimeData->setText(data);
350  clipboard->setMimeData(mimeData);
351 }

404 void mdtSqlTableWidget::copySelectionToClipBoard()
405 {
406  Q_ASSERT(selectionModel() != 0);
407 
408  QModelIndexList selectionIndexList;
409  QModelIndex previousModelIndex, currentModelIndex;
410  QClipboard *clipboard = QApplication::clipboard();
411  QString data;
412  int i;
413 
414  /*
415  * Get selected indexes. We want that data are token from top-left to bottom right,
416  * not the order of selection.
417  * f.ex. if user selected items from bottom to top,
418  * we want to return items from top to bottom.
419  */
420  selectionIndexList = selectionModel()->selectedIndexes();
421  qSort(selectionIndexList.begin(), selectionIndexList.end());
422  for(i = 0; i < selectionIndexList.size(); ++i){
423  if(!pvTableView->isColumnHidden(selectionIndexList.at(i).column())){
424  if(previousModelIndex.isValid()){
425  if(currentModelIndex.row() == previousModelIndex.row()){
426  data += "\t";
427  }else{
428  data += "\n";
429  }
430  }
431  data += "\"" + pvController->data(currentModelIndex).toString() + "\"";
432  }
433  previousModelIndex = currentModelIndex;
434  }
435  // Send to clipboard
436  clipboard = QApplication::clipboard();
437  Q_ASSERT(clipboard != 0);
438  QMimeData *mimeData = new QMimeData;
439  mimeData->setText(data);
440  clipboard->setMimeData(mimeData);
441 }
