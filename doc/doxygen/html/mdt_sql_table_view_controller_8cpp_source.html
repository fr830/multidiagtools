130 mdtSqlTableSelection mdtSqlTableViewController::currentSelection(const QStringList& fieldList)
131 {
132  Q_ASSERT(model());
133  Q_ASSERT(proxyModel());
134  Q_ASSERT(pvTableView != 0);
135  Q_ASSERT(pvTableView->selectionModel() != 0);
136 
137  mdtSqlTableSelection s;
138  QModelIndexList selectionIndexList;
139  QModelIndexList modelIndexList;
140  int i;
141 
142  /*
143  * Get selected indexes. We want that data are token from top-left to bottom right,
144  * not the order of selection.
145  * f.ex. if user selected items from bottom to top,
146  * we want to return items from top to bottom.
147  */
148  selectionIndexList = pvTableView->selectionModel()->selectedIndexes();
149  qSort(selectionIndexList.begin(), selectionIndexList.end());
150  for(i = 0; i < selectionIndexList.size(); ++i){
151  modelIndexList.append(proxyModel()->mapToSource(selectionIndexList.at(i)));
152  }
153  s.setIndexList(modelIndexList, fieldList, model().get());
154 
155  return s;
156 }

177 void mdtSqlTableViewController::onTableViewcurrentRowChanged(const QModelIndex& current, const QModelIndex& previous)
178 {
179  /*
180  * Table view will send currentRowChanged() when user changed selection,
181  * and then model was updated (rows removed or added).
182  * We only accept this change in Visualizing state.
183  */
191  if(!pvRowChangingByInternalEvent){
192  /*
193  * If model was reset (after select, setFilter,
194  * Table view will not keep a current index.
195  * So, if we receive a invalid index,
196  * we check if we have som rows. If yes, we set the first, 0, as current.
197  */
198  if(current.isValid()){
199  setCurrentRow(current.row());
200  }else{
201  if(rowCount(false) > 0){
202  setCurrentRow(0);
203  }else{
204  setCurrentRow(-1);
205  }
206  }
207  }
208  pvRowChangingByInternalEvent = false;
209 }

314 bool mdtSqlTableViewController::doRemove()
315 {
316  Q_ASSERT(model());
317  Q_ASSERT(pvTableView != 0);
318  Q_ASSERT(pvTableView->selectionModel() != 0);
319 
320  QModelIndexList indexes;
321  int i;
322  int row;
323  QVector<int> rows;
324  QSqlError sqlError;
325 
326  // Build rows list
327  indexes = pvTableView->selectionModel()->selectedIndexes();
328  for(i = 0; i < indexes.size(); ++i){
329  row = indexes.at(i).row();
330  if(!rows.contains(row)){
331  rows.append(row);
332  }
333  }
334  // Delete selected rows
335  for(i = 0; i < rows.size(); ++i){
336  if(!proxyModel()->removeRow(rows.at(i))){
337  QSqlError sqlError;
338  sqlError = model()->lastError();
339  if(sqlError.isValid()){
340  pvLastError.setError(tr("Removing data failed."), mdtError::Error);
341  pvLastError.setSystemError(sqlError.number(), sqlError.text());
342  MDT_ERROR_SET_SRC(pvLastError, "mdtSqlTableViewController");
343  pvLastError.commit();
344  if(messageHandler()){
345  messageHandler()->setError(pvLastError);
346  messageHandler()->displayToUser();
347  }
348  }
349  model()->revertAll();
350  return false;
351  }
352  }
353  if(!beginTransaction()){
354  if(messageHandler()){
355  messageHandler()->setError(pvLastError);
356  messageHandler()->displayToUser();
357  }
358  return false;
359  }
360  if(!model()->submitAll()){
361  model()->revert();
362  sqlError = model()->lastError();
363  pvLastError.setError(tr("Reverting new data failed."), mdtError::Error);
364  pvLastError.setSystemError(sqlError.number(), sqlError.text());
365  MDT_ERROR_SET_SRC(pvLastError, "mdtSqlTableViewController");
366  pvLastError.commit();
367  if(messageHandler()){
368  messageHandler()->setError(pvLastError);
369  messageHandler()->displayToUser();
370  }
371  rollbackTransaction();
372  return false;
373  }
374  if(!commitTransaction()){
375  if(messageHandler()){
376  messageHandler()->setError(pvLastError);
377  messageHandler()->displayToUser();
378  }
379  return false;
380  }
381 
382  return true;
