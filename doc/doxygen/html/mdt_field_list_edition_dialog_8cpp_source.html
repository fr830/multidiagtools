00230     }
00231     // Check about illegal strings
00232     fieldName = correctFieldName(fieldName);
00233     if(fieldName.isEmpty()){
00234       pvResultFields.clear();

00236     }
00237     // Store new name
00238     twItem->setData(Qt::DisplayRole, fieldName);
00239     // Store to result
00240     field = mdtFieldMapField();
00241     twItem = twFields->item(i, 0);
00242     Q_ASSERT(twItem != 0);
00243     fieldIndex = twItem->data(Qt::DisplayRole).toInt();

00246     twItem = twFields->item(i, 2);

00248     fieldDisplayText = twItem->data(Qt::DisplayRole).toString();
00249     field.setDisplayText(fieldDisplayText);
00250     pvResultFields.append(field);

00252   // Check about dupplicate field names a after possible string substitution
