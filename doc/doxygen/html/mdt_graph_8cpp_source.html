00105 bool mdtGraph::removeVertex(const QVariant &key)
00106 {
00107   int i;
00108   int vToRemoveIndex = -1;
00109   mdtGraphVertex *v;
00110 
00111   // Search the vertex that contains data with given key
00112   for(i = 0; i < pvVertexList.size(); ++i){
00113     v = pvVertexList.at(i);
00114     Q_ASSERT(v != 0);
00115     if(v->key() == key){
00116       // If vertex contains other vertices in his adjacency list, we fail
00117       if(v->hasAdjacent()){
00118         return false;
00119       }
00120       // Remember index of vertex to remove
00121       vToRemoveIndex = i;
00122     }else{
00123       // If searched vertex is contained in the adjacency list of another vertex, we fail
00124       if(v->containsAdjacent(key)){
00125         return false;
00126       }
00127     }
00128   }
00129   // If vertex was not found, tell this
00130   if(vToRemoveIndex < 0){
00131     return false;
00132   }
00133   // Remove vertex
00134   v = pvVertexList.at(vToRemoveIndex);
00135   pvVertexList.removeAt(vToRemoveIndex);
00136   if(pvAutoDeleteData){
00137     delete v->data();
00138   }
00139   delete v;
00140 
00141   return true;

00305       // If adjacent vertex is white, we enqueue it and set it gray
00306       if(av->currentColor() == mdtGraphVertex::White){
00307         av->setColor(mdtGraphVertex::Gray);
00308         fifo.enqueue(av);
00309       }
00310     }
00311     // We explored all adjacent vertices of current vertex, set it black
00312     v->setColor(mdtGraphVertex::Black);
00313     Q_ASSERT(v->data() != 0);
