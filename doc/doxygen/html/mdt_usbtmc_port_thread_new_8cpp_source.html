32 void mdtUsbtmcPortThreadNew::aboutToClose()
33 {
34  /*
35  * Don't know how to handle this properly.
36  * One problem is to wake-up USB thread, i.e. sending a event so that libusb_handle_events() returns.
37  * One function that does it is libusb_close() (wich is called by mdtUsbPortThread once we return from here.
38  * Other problem is to cancel transfers in a not to ugly way.
39  * Try to submit transfers cancellation, and give thread a chance to do it in a short time.
40  */
41  pvTransferHandler.submitStopRequest();
42  waitOnState(State_t::Stopped, 1000);
43  setCurrentState(State_t::Stopped); // Will force thread to exit if not allready done

46 void mdtUsbtmcPortThreadNew::run()
47 {
48  int err;
49  libusb_device_handle *handle;
50 
51  // Open USB device
52  if(!openDevice()){
53  setCurrentState(State_t::Error);
54  emit errorOccured(pvLastError);
55  return;
56  }
57  handle = deviceHandle();
58  Q_ASSERT(handle != 0);
59  // Setup transfer handler
60  if(!pvTransferHandler.setup(handle, deviceDescriptor(), bInterfaceNumber())){
61  std::lock_guard<std::mutex> lg(pvLastErrorMutex);
62  pvLastError = pvTransferHandler.lastError();
63  setCurrentState(State_t::Error);
64  emit errorOccured(pvLastError);
65  return;
66  }
67  // Tell main thread that we are now running and run..
68  setCurrentState(State_t::Running);
69  pvTransferHandler.start();
70  while(1){
71  if(currentState() == State_t::Stopped){
72  std::lock_guard<std::mutex> lg(pvLastErrorMutex);
73  pvLastError.setError(QObject::tr("Forced to stop in abnormal way (probably a bug!)"), mdtError::Error);
74  MDT_ERROR_SET_SRC(pvLastError, "mdtUsbtmcPortThreadNew");
75  pvLastError.commit();
76  emit errorOccured(pvLastError);
77  return;
78  }
79  err = libusb_handle_events(usbContext());
80  if(err != 0){
81  std::lock_guard<std::mutex> lg(pvLastErrorMutex);
82  pvLastError.setError(QObject::tr("USB event handling failed (libusb_handle_events())"), mdtError::Error);
83  pvLastError.setSystemError(err, libusb_strerror((libusb_error)err));
84  MDT_ERROR_SET_SRC(pvLastError, "mdtUsbtmcPortThreadNew");
85  pvLastError.commit();
86  setCurrentState(State_t::Error);
87  emit errorOccured(pvLastError);
88  return;
89  }
90  /*
91  * Check if transfer handler must be stopped.
92  * Can be due to 2 main reasons:
93  * - Stop request was submitted by aboutToClose() and all pending transfers where cancelled
94  * - A unhandler error occured (transfer handler will also signal error itself)
95  */
96  if(pvTransferHandler.mustBeStopped()){
97  break;
98  }
99  }
100  pvTransferHandler.stop();
101  // Tell main thread that we are now stopped
102  setCurrentState(State_t::Stopped);
103 }
