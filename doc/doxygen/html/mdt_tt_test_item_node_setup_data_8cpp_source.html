32 void mdtTtTestItemNodeSetupData::addNodeSetupData(const mdtTtTestNodeSetupData & data)
33 {
34  QList<mdtTtTestNodeUnitSetupData> unitSetupDataList;
35  int step;
36  QVector<int> steps;
37  int i, k;
38 
39  // Build list of setps contained in node unit setup data list
40  unitSetupDataList = data.unitSetupList();
41  for(i = 0; i < unitSetupDataList.size(); ++i){
42  step = unitSetupDataList.at(i).value("StepNumber").toInt();
43  if(!steps.contains(step)){
44  steps.append(step);
45  }
46  }
47  // For each step, build a test node setup and add unit setup that are part of the step
48  for(i = 0; i < steps.size(); ++i){
49  // Setup a new node setup
50  mdtTtTestNodeSetupData nodeSetupData;
51  nodeSetupData.setNodeIdentification(data.nodeIdentification());
52  nodeSetupData.setDeviceIdentification(data.deviceIdentification());
53  // Add unit setup for current step
54  for(k = 0; k < unitSetupDataList.size(); ++k){
55  if(unitSetupDataList.at(k).value("StepNumber").toInt() == steps.at(i)){
56  nodeSetupData.addUnitSetup(unitSetupDataList.at(k));
57  }
58  }
59  // Add node setup to map
60  pvNodeSetupDataMap.insert(steps.at(i), nodeSetupData);
61  }
62  pvStepIterator = pvNodeSetupDataMap.begin();
63 }
