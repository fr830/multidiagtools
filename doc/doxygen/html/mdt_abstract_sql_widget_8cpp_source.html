00246 bool mdtAbstractSqlWidget::restorePrimaryKeyDataToModel(const QSqlRecord &previousRecord)
00247 {
00248   Q_ASSERT(pvModel != 0);
00249 
00250   QSqlIndex primaryKey;
00251   int pkIndex;
00252   int indexInRecord;
00253   QSqlRecord currentRecord;
00254   bool pkUpdated = false;
00255 
00256   currentRecord = pvModel->record(currentRow());
00257   primaryKey = pvModel->primaryKey();
00258   for(pkIndex = 0; pkIndex < primaryKey.count(); ++pkIndex){
00259     indexInRecord = previousRecord.indexOf(primaryKey.fieldName(pkIndex));
00260     if(currentRecord.value(indexInRecord) != previousRecord.value(indexInRecord)){
00261       pkUpdated = true;
00262       currentRecord.setValue(indexInRecord, previousRecord.value(indexInRecord));
00263     }
00264   }
00265   if(pkUpdated){
00266     if(!pvModel->setRecord(currentRow(), currentRecord)){
00267       return false;
00268     }
00269   }
00270 
00271   return true;
00272 }
