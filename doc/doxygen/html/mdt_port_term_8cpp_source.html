00109 void mdtPortTerm::sendCmd()
00110 {
00111   QString cmd;
00112   bool cmdIsQuery = false;
00113 
00114   if(!pvRunning){
00115     return;
00116   }
00117   if(pvCurrentPortManager == 0){
00118     qDebug() << "TERM: err, pvCurrentPortManager == 0";
00119     return;
00120   }
00121   if(leCmd->text().size() < 1){
00122     return;
00123   }
00124   cmd = leCmd->text();
00125   if(cmd.trimmed().right(1) == "?"){
00126     cmdIsQuery = true;
00127   }
00128   cmd.append('\n');
00129   //cmd.append((char)0x04);
00131   // Wait until write is possible and write
00132   if(!pvCurrentPortManager->waitOnWriteReady()){
00133     qDebug() << "TERM: cannot write for the moment";
00134     return;
00135   }
00136   if(pvCurrentPortManager->writeData(cmd.toAscii()) < 0){
00137     qDebug() << "TERM: write error";
00138     return;
00139   }
00140   // If we have a query for USBTMC port, send the read request
00141   if((pvCurrentPortManager == pvUsbtmcPortManager)&&(cmdIsQuery)){
00142     if(!pvCurrentPortManager->waitOnWriteReady()){
00143       qDebug() << "TERM: cannot write for the moment";
00144       return;
00145     }
00146     if(pvUsbtmcPortManager->sendReadRequest(false) < 0){
00147       qDebug() << "TERM: cannot send read request";
00148       return;
00149     }
00150   }
00151   leCmd->clear();
00152 }

00196   if(!pvSerialPortManager->openPort()){

00201   if(pvSerialPortManager->start()){
