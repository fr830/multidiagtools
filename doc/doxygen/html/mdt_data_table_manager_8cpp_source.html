189 bool mdtDataTableManager::importFromCsvFile(const QString &csvFilePath, mdtSqlDatabaseManager::createMode_t mode, const QString &dir, const QStringList &pkFields)
190 {
191  QFileInfo fileInfo(csvFilePath);
192  QDir dbDir;
193  mdtCsvFile csvFile(0, pvEncoding);
194  QStringList header;
195  QStringList line;
196  QList<QVariantList> rows;
197  int lineCount;
198  mdtFieldMapItem *mapItem;
199  QList<mdtFieldMapItem*> mapItems;
200  int csvFieldIndex;
201  int modelFieldIndex;
202  mdtSqlSchemaTable table;
203  QProgressDialog *progress = 0;
204  int totalLineCount;
205  int dataLossCount;
206  int i;
207  QSqlField field;
208  QList<mdtFieldMapField> fields;
209  QString autoValuePkFieldName;
210 
211  // Set DB directory
212  if(dir.isEmpty()){
213  dbDir = fileInfo.absoluteDir();
214  }else{
215  dbDir.setPath(dir);
216  }
217  if(!dbDir.exists()){
218  pvLastError.setError(tr("Directory not found: ") + dbDir.path(), mdtError::Error);
219  MDT_ERROR_SET_SRC(pvLastError, "mdtDataTableManager");
220  pvLastError.commit();
221  return false;
222  }
223  // Open CSV file
224  csvFile.setFileName(fileInfo.absoluteFilePath());
225  if(!csvFile.open(QIODevice::ReadOnly)){
226  pvLastError.setError(tr("Cannot open CSV file: ") + fileInfo.absoluteFilePath(), mdtError::Error);
227  MDT_ERROR_SET_SRC(pvLastError, "mdtDataTableManager");
228  pvLastError.commit();
229  return false;
230  }
231  // Read header and create fields
232  header = csvFile.readHeader(pvCsvSeparator, pvCsvDataProtection, pvCsvComment, pvCsvEscapeChar, pvCsvEol);
233  if(header.isEmpty()){
234  pvLastError.setError(tr("No header found in CSV file: ") + fileInfo.absoluteFilePath(), mdtError::Error);
235  MDT_ERROR_SET_SRC(pvLastError, "mdtDataTableManager");
236  pvLastError.commit();
237  csvFile.close();
238  return false;
239  }
240  pvDatabaseHeader.clear();
241  // If no CSV field is part of primary key, we will create one with auto generated value
242  if(pkFields.isEmpty()){
243  autoValuePkFieldName = "id_PK";
244  pvDatabaseHeader << autoValuePkFieldName;
245  }
246  // Create or update field mapping regarding allready set-up map
247  modelFieldIndex = 0;
248  for(csvFieldIndex = 0; csvFieldIndex < header.size(); ++csvFieldIndex){
249  if(!header.at(csvFieldIndex).isEmpty()){
250  mapItems = pvFieldMap.itemsAtSourceFieldName(header.at(csvFieldIndex));
251  if(mapItems.isEmpty()){
252  // No mapping for this header item, generate one
253  mapItem = new mdtFieldMapItem;
254  mapItem->setDestinationFieldName(mdtFieldMapField::getFieldName(header.at(csvFieldIndex)));
255  mapItem->setDestinationFieldDisplayText(header.at(csvFieldIndex));
256  mapItem->setSourceFieldName(header.at(csvFieldIndex));
257  mapItem->setDataType(QVariant::String);
258  pvFieldMap.addItem(mapItem);
259  mapItems.append(mapItem);
260  }
261  // Set indexes
262  for(i = 0; i < mapItems.size(); i++){
263  mapItem = mapItems.at(i);
264  Q_ASSERT(mapItem != 0);
265  mapItem->setSourceFieldIndex(csvFieldIndex);
266  mapItem->setDestinationFieldIndex(modelFieldIndex);
267  modelFieldIndex++;
268  }
269  }
270  }
271  pvDatabaseHeader << pvFieldMap.destinationHeader();
272  // Create a primary key if needed
273  if(pkFields.isEmpty()){
274  // Create primary key field
275  field = QSqlField();
276  field.setName("id_PK");
277  field.setType(QVariant::Int);
278  field.setAutoValue(true);
279  table.addField(field, true);
280  }else{
281  // Primary key is one of the CSV header
282  for(i = 0; i < pkFields.size(); ++i){
283  Q_ASSERT(header.contains(pkFields.at(i)));
284  }
285  }
286  // Create database fields
287  fields = pvFieldMap.mappedDestinationFields();
288  for(i = 0; i < fields.size(); ++i){
289  // Check if field is part of primary key and add it to database table
290  if(pkFields.contains(fields.at(i).displayText())){
291  table.addField(fields.at(i).sqlField(), true);
292  }else{
293  table.addField(fields.at(i).sqlField(), false);
294  }
295  }
296  // Create data set
297  close();
298  pvDataSetName = fileInfo.baseName();
299  if(!createDataSet(dbDir, pvDataSetName, table, mode)){
300  csvFile.close();
301  return false;
302  }
303  // Store data
304  if(pvProgressDialogEnabled){
305  totalLineCount = 0;
306  progress = new QProgressDialog;
307  progress->setWindowTitle(tr("CSV import progress"));
308  progress->setMinimum(0);
309  progress->setMaximum(0);
310  connect(progress, SIGNAL(canceled()), this, SLOT(setAbortFlag()));
311  progress->show();
312  }
313  lineCount=0;
314  dataLossCount = 0;
315  pvAbort = false;
316  while(csvFile.hasMoreLines()){
317  // Check if we have to abort
318  if(pvAbort){
319  csvFile.close();
320  close();
321  return false;
322  }
323  // Read a line in CSV file
324  line = csvFile.readLine(pvCsvSeparator, pvCsvDataProtection, pvCsvComment, pvCsvEscapeChar, pvCsvEol);
325  // On some parse error, it can happen that we have to much columns
326  // Common known issue is the ending ; with no more data, but we don't loose data in this case.
327  while(line.size() > table.fieldCount()){
328  if(!line.last().isEmpty()){
329  dataLossCount++;
330  }
331  line.removeLast();
332  }
333  rows.append(pvFieldMap.destinationDataRow(line));
334  if(lineCount > 100){
335  if(!commitRowsToDatabase(rows, fields, autoValuePkFieldName)){
336  csvFile.close();
337  close();
338  pvLastError.commit();
339  return false;
340  }
341  // Update progress dialog
342  if(progress != 0){
343  totalLineCount += rows.size();
344  progress->setLabelText(tr("Importing line ") + QString::number(totalLineCount));
345  qApp->processEvents();
346  }
347  rows.clear();
348  lineCount=0;
349  }
350  lineCount++;
351  }
352  // Commit last lines
353  if(!commitRowsToDatabase(rows, fields, autoValuePkFieldName)){
354  csvFile.close();
355  close();
356  pvLastError.commit();
357  return false;
358  }
359  csvFile.close();
360  if(progress != 0){
361  delete progress;
362  }
363  // Log a warning if some data were lost
364  if(dataLossCount > 0){
365  pvLastError.setError(tr("Some data has been lost during import of CSV file: ") + fileInfo.absoluteFilePath(), mdtError::Warning);
366  MDT_ERROR_SET_SRC(pvLastError, "mdtDataTableManager");
367  pvLastError.commit();
368  }
369 
370  return true;
371 }

373 bool mdtDataTableManager::copyTable(const QString & sourceTableName, const QString & destinationTableName, mdtSqlDatabaseManager::createMode_t mode, QSqlDatabase sourceDatabase, QSqlDatabase destinationDatabase)
374 {
375  Q_ASSERT(sourceDatabase.isOpen());
376  Q_ASSERT(destinationDatabase.isOpen());
377 
378  QSqlRecord record;
379  QList<mdtFieldMapField> sourceFields, destinationFields;
380  QList<mdtFieldMapItem*> mapItems;
381  mdtFieldMapItem *mapItem;
382  int i, sourceFieldIndex, destinationFieldIndex;
383  mdtSqlSchemaTable destinationTable;
384  QSqlIndex pkFields;
385  QProgressDialog *progress = 0;
386  int lineCount, totalLineCount;
387  QList<QVariant> rowData;
388  QList<QVariantList> rows;
389  QString sql;
390  QSqlError sqlError;
391  QString errorText;
392  QSqlDriver *sourceSqlDriver;
393 
394  // Build source fields list
395  record = sourceDatabase.record(sourceTableName);
396  for(i = 0; i < record.count(); ++i){
397  mdtFieldMapField field;
398  field.setSqlField(record.field(i));
399  sourceFields.append(field);
400  }
401  pvFieldMap.setSourceFields(sourceFields);
402  // Create or update field mapping regarding allready set-up map
403  destinationFieldIndex = 0;
404  for(sourceFieldIndex = 0; sourceFieldIndex < sourceFields.size(); ++sourceFieldIndex){
405  mapItems = pvFieldMap.itemsAtSourceFieldName(sourceFields.at(sourceFieldIndex).name());
406  if(mapItems.isEmpty()){
407  // No mapping for this source item, generate one
408  mapItem = new mdtFieldMapItem;
409  mdtFieldMapField field;
410  field.setSqlField(sourceFields.at(sourceFieldIndex).sqlField());
411  mapItem->setSourceField(field);
412  mapItem->setDestinationField(field);
413  pvFieldMap.addItem(mapItem);
414  mapItems.append(mapItem);
415  }
416  // Set indexes
417  for(i = 0; i < mapItems.size(); ++i){
418  mapItem = mapItems.at(i);
419  Q_ASSERT(mapItem != 0);
420  mapItem->setSourceFieldIndex(sourceFieldIndex);
421  mapItem->setDestinationFieldIndex(destinationFieldIndex);
422  destinationFieldIndex++;
423  }
424  }
425  sourceFields = pvFieldMap.mappedSourceFields();
426  // Create destination table
427  pkFields = sourceDatabase.primaryIndex(sourceTableName);
428  destinationFields = pvFieldMap.mappedDestinationFields();
429  destinationTable.setTableName(destinationTableName);
430  pvDataSetTableName = destinationTableName;
431  for(i = 0; i < destinationFields.size(); ++i){
432  // Check if field is part of primary key and add it to database table
433  if(pkFields.contains(destinationFields.at(i).name())){
434  destinationTable.addField(destinationFields.at(i).sqlField(), true);
435  }else{
436  destinationTable.addField(destinationFields.at(i).sqlField(), false);
437  }
438  }
439  pvDatabaseManager->setDatabase(destinationDatabase);
440  if(!pvDatabaseManager->createTable(destinationTable, mode)){
441  pvLastError = pvDatabaseManager->lastError();
442  return false;
443  }
444  // Prepare copy
445  if(pvProgressDialogEnabled){
446  totalLineCount = 0;
447  progress = new QProgressDialog;
448  progress->setWindowTitle(tr("Database table copy progress"));
449  progress->setMinimum(0);
450  progress->setMaximum(0);
451  connect(progress, SIGNAL(canceled()), this, SLOT(setAbortFlag()));
452  progress->show();
453  }
454  sourceSqlDriver = sourceDatabase.driver();
455  Q_ASSERT(sourceSqlDriver != 0);
456  sql = "SELECT ";
457  sourceFields = pvFieldMap.mappedSourceFields();
458  for(i = 0; i < sourceFields.size(); ++i){
459  sql += sourceSqlDriver->escapeIdentifier(sourceFields.at(i).name(), QSqlDriver::FieldName);
460  if(i < (sourceFields.size() - 1)){
461  sql += ",";
462  }
463  }
464  sql += "\n FROM '" + sourceTableName +"'";
465  QSqlQuery sourceDataQuery(sourceDatabase);
466  if(!sourceDataQuery.exec(sql)){
467  QString msg;
468  sqlError = sourceDataQuery.lastError();
469  msg = tr("Unable to get source data from table '") + sourceTableName + tr("' for copy.");
470  msg += tr(" (Generated SQL statement: ") + sql + tr(")");
471  pvLastError.setError(msg, mdtError::Error);
472  pvLastError.setSystemError(sqlError.number(), sqlError.text());
473  MDT_ERROR_SET_SRC(pvLastError, "mdtDataTableManager");
474  pvLastError.commit();
475  return false;
476  }
477  // Copy data
478  lineCount = 0;
479  pvAbort = false;
480  while(sourceDataQuery.next()){
481  // Check if we have to abort
482  if(pvAbort){
483  close();
484  return false;
485  }
486  // Get a row od data from source
487  record = sourceDataQuery.record();
488  rowData.clear();
489  for(i = 0; i < record.count(); ++i){
490  rowData.append(sourceDataQuery.value(i));
491  }
492  // Copy row
493  rows.append(pvFieldMap.destinationDataRow(rowData));
494  if(lineCount > 100){
495  if(!commitRowsToDatabase(rows, destinationFields, "")){
496  close();
497  errorText = pvLastError.text();
498  errorText += " (Source table: " + sourceTableName + " , destination table: " + destinationTableName + ")";
499  pvLastError.updateText(errorText);
500  pvLastError.commit();
501  return false;
502  }
503  // Update progress dialog
504  if(progress != 0){
505  totalLineCount += rows.size();
506  progress->setLabelText(tr("Importing row ") + QString::number(totalLineCount));
507  qApp->processEvents();
508  }
509  rows.clear();
510  lineCount = 0;
511  }
512  lineCount++;
513  }
514  // Commit last lines
515  if(!commitRowsToDatabase(rows, destinationFields, "")){
516  errorText = pvLastError.text();
517  errorText += " (Source table: " + sourceTableName + " , destination table: " + destinationTableName + ")";
518  pvLastError.updateText(errorText);
519  pvLastError.commit();
520  close();
521  return false;
522  }
523  if(progress != 0){
524  delete progress;
525  }
526 
527  return true;
528 }

611 bool mdtDataTableManager::commitRowsToDatabase(const QList<QVariantList> &rows, const QList<mdtFieldMapField> &fields, const QString & autoValuePkFieldName)
612 {
613  QSqlDatabase db = pvDatabaseManager->database();
614  QSqlQuery query(db);
615  QString sql;
616  QSqlError sqlError;
617  int row, col, autoValuePkFieldCount;
618 
619  // Begin transaction (without, insertions are very slow)
620  if(!db.transaction()){
621  pvLastError.setError(tr("Unable to begin transaction, data set: ") + pvDataSetName, mdtError::Error);
622  pvLastError.setSystemError(db.lastError().number(), db.lastError().text());
623  MDT_ERROR_SET_SRC(pvLastError, "mdtDataTableManager");
624  pvLastError.commit();
625  return false;
626  }
627  // Prepare query for insertion
628  sql = insertBindValuesPrepareStatement(fields, autoValuePkFieldName);
629  if(!query.prepare(sql)){
630  sqlError = query.lastError();
631  pvLastError.setError("Cannot prepare query for insertion in database", mdtError::Error);
632  pvLastError.setSystemError(sqlError.number(), sqlError.text());
633  MDT_ERROR_SET_SRC(pvLastError, "mdtDataTableManager");
634  pvLastError.commit();
635  db.rollback();
636  return false;
637  }
638  // Set data
639  for(row = 0; row < rows.size(); ++row){
640  // Add auto generated PK
641  if(autoValuePkFieldName.isEmpty()){
642  autoValuePkFieldCount = 0;
643  }else{
644  autoValuePkFieldCount = 1;
645  query.bindValue(0, QVariant());
646  }
647  for(col = 0; col < rows.at(row).size(); ++col){
648  query.bindValue(col + autoValuePkFieldCount, rows.at(row).at(col));
649  }
650  // Execute query
651  if(!query.exec()){
652  sqlError = query.lastError();
653  pvLastError.setError("Cannot execute query for insertion in database", mdtError::Error);
