00040 bool mdtUicNumber::setNumber(const QString &uicNumber)
00041 {
00042   QString number;
00043   int i, digit;
00044 
00045   // Remove unusable chars
00046   number = uicNumber.simplified();
00047   number.remove(" ");
00048   number.remove("-");
00049   // Check length:
00050   //  - We only know about 6 and 11 digits UIC numbers
00051   //  - If we have 7 or 12 digits, the control key is given
00052   if((number.size() == 7)||(number.size() == 12)){
00053     number.remove(number.size()-1, 1);
00054   }
00055   if((number.size() != 6)&&(number.size() != 11)){
00056     mdtError e(MDT_PARSE_ERROR, "UIC number has invalid digits count (allowed: 6, 7, 11 or 12 digits)", mdtError::Error);
00057     MDT_ERROR_SET_SRC(e, "mdtUicNumber");
00058     e.commit();
00059     return false;
00060   }
00061   // Store digits
00062   pvUicNumberDigits.clear();
00063   for(i=0; i<number.size(); ++i){
00064     digit = number.at(i).digitValue();
00065     if(digit<0){
00066       mdtError e(MDT_PARSE_ERROR, "Invalid UIC number (contains unallowed chars)", mdtError::Error);
00067       MDT_ERROR_SET_SRC(e, "mdtUicNumber");
00068       e.commit();
00069       pvUicNumberDigits.clear();
00070       return false;
00071     }
00072     pvUicNumberDigits.append(digit);
00073   }
00074 
00075   return true;
00076 }
