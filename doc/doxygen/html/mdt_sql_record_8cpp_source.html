46 bool mdtSqlRecord::addField(const QString & fieldName, const QString & tableName, const QSqlDatabase & db)
47 {
48  QSqlRecord dbRecord;
49  QSqlField fld;
50 
51  // Get record object to fetch attributes
52  dbRecord = db.record(tableName);
53  if(dbRecord.isEmpty()){
54  QSqlError sqlError = db.lastError();
55  pvLastError.setError("Cannot get field information. Field: " + fieldName + ", table: " + tableName, mdtError::Error);
56  if(sqlError.isValid()){
57  pvLastError.setSystemError(sqlError.number(), sqlError.text());
58  }
59  MDT_ERROR_SET_SRC(pvLastError, "mdtSqlRecord");
60  pvLastError.commit();
61  return false;
62  }
63  // Get field and add it
64  fld = dbRecord.field(fieldName);
65  if(!fld.isValid()){
66  pvLastError.setError("Field '" + fieldName + "' was not found in table '" + tableName + "'", mdtError::Error);
67  MDT_ERROR_SET_SRC(pvLastError, "mdtSqlRecord");
68  pvLastError.commit();
69  return false;
70  }
71  fld.setGenerated(false);
72  pvRecord.append(fld);
73 
74  return true;
75 }

211  sql += ",";

213  }
214  // Add row condition
215  if(matchData.count() > 0){
216  field = matchData.field(0);
217  delimiter = sqlDataDelimiter(field.type());
218  sql += " WHERE " + field.name() + "=" + delimiter + field.value().toString() + delimiter;
219  for(i = 1; i < matchData.count(); ++i){
220  field = matchData.field(i);
221  delimiter = sqlDataDelimiter(field.type());
222  sql += " AND " + field.name() + "=" + delimiter + field.value().toString() + delimiter;
223  }
224  }
225 
226  return sql;
