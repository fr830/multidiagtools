51 bool mdtSqlDataWidgetController::mapFormWidgets(QWidget* widget, const QString& firstWidgetInTabOrder)
52 {
53  Q_ASSERT(widget != 0);
54  Q_ASSERT(widget->layout() != 0);
55  Q_ASSERT(model());
56  Q_ASSERT(currentState() == Stopped);
57 
58  int i;
59  QString fieldName;
60  QWidget *w;
61  mdtSqlSchemaTable st;
62  bool isFirstWidgetInTabOrder;
63 
64  // Clear previous mapping
65  qDeleteAll(pvFieldHandlers);
66  pvFieldHandlers.clear();
67  pvWidgetMapper.clearMapping();
68  pvFirstDataWidget = 0;
69  // Search widgets with fld_ as prefix in they objectName
70  buildWidgetsList(widget, "fld_");
71  // Fetch table information - record returned by QSqlDatabase does not return Date and DateTime field infomration
72  if(!st.setupFromTable(model()->tableName(), model()->database())){
73  pvLastError = st.lastError();
74  return false;
75  }
76  // Map found widgets
77  for(i = 0; i < pvFoundWidgets.size(); ++i){
78  w = pvFoundWidgets.at(i);
79  Q_ASSERT(w != 0);
80  fieldName = w->objectName();
81  fieldName = fieldName.right(fieldName.length()-4);
82  isFirstWidgetInTabOrder = (w->objectName() == firstWidgetInTabOrder);
83  if(!addMapping(w, fieldName, st, isFirstWidgetInTabOrder)){
84  qDeleteAll(pvFieldHandlers);
85  pvFieldHandlers.clear();
86  pvWidgetMapper.clearMapping();
87  pvFirstDataWidget = 0;
88  return false;
89  }
90  }
91  // Update UI
92  updateMappedWidgets();
93  updateNavigationControls();
94  // Add data validator
95  addDataValidator(std::shared_ptr<mdtSqlFormWidgetDataValidator>(new mdtSqlFormWidgetDataValidator(0, pvFieldHandlers)));
96  // Found widgets are not needed anymore
97  pvFoundWidgets.clear();
98 
99  return true;
100 }

118 bool mdtSqlDataWidgetController::addMapping(QWidget* widget, const QString& fieldName, const mdtSqlSchemaTable& st, bool isFirstWidgetInTabOrder)
119 {
120  Q_ASSERT(widget != 0);
121  Q_ASSERT(model());
122  Q_ASSERT(currentState() == Stopped);
123 
124  int fieldIndex;
125  mdtSqlFieldHandler *fieldHandler;
126 
127  // Search matching field in model
128  fieldIndex = model()->record().indexOf(fieldName);
129  // If field was found, map it
130  if(fieldIndex >= 0){
131  fieldHandler = new mdtSqlFieldHandler;
132  fieldHandler->setField(st.field(fieldName));
133  fieldHandler->setDataWidget(widget);
134  connect(fieldHandler, SIGNAL(dataEdited()), this, SIGNAL(dataEdited()));
135  pvFieldHandlers.append(fieldHandler);
136  // If this widget is the first in focus chain, ref it
137  if(isFirstWidgetInTabOrder){
138  pvFirstDataWidget = widget;
139  }
140  // Add to widget mapper
141  pvWidgetMapper.addMapping(fieldHandler, fieldIndex, "data");
142  }else{
143  widget->setEnabled(false);
144  mdtError e(tr("For table '") + tableName() + tr("': cannot find field for widget '") + widget->objectName() + tr("'"), mdtError::Warning);
145  MDT_ERROR_SET_SRC(e, "mdtSqlFormWidget");
146  e.commit();
147  }
148 
149  return true;
150 }

300 bool mdtSqlDataWidgetController::doRemove()
301 {
302  Q_ASSERT(model());
303 
304  QSqlError sqlError;
305 
306  // Remove current row
307  if(!model()->removeRow(currentRow())){
308  sqlError = model()->lastError();
309  pvLastError.setError(tr("Removing data failed."), mdtError::Error);
310  pvLastError.setSystemError(sqlError.number(), sqlError.text());
311  MDT_ERROR_SET_SRC(pvLastError, "mdtSqlDataWidgetController");
312  pvLastError.commit();
313  if(messageHandler()){
314  messageHandler()->setError(pvLastError);
315  messageHandler()->displayToUser();
316  }
317  return false;
318  }
319  /*
320  * We use QDataWidgetMapper::ManualSubmit submit policy and QSqlTableModel::OnManualSubmit edit strategy.
321  * Widget mapper calls submit() on model, but this has no effect with OnManualSubmit edit strategy,
322  * so we have to call submitAll() on model.
323  */
324  if(!model()->submitAll()){
325  model()->revertRow(currentRow());
326  sqlError = model()->lastError();
327  pvLastError.setError(tr("Removing data failed."), mdtError::Error);
328  pvLastError.setSystemError(sqlError.number(), sqlError.text());
329  MDT_ERROR_SET_SRC(pvLastError, "mdtSqlDataWidgetController");
330  pvLastError.commit();
331  if(messageHandler()){
332  messageHandler()->setError(pvLastError);
333  messageHandler()->displayToUser();
334  }
335  return false;
336  }
337 
338  return true;
339 }
