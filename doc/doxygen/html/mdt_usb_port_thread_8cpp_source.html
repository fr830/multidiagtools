00061   // Write the frame to port

00097         // Restore frame to pool and return

00099         return mdtAbstractPort::NoError;
00100       }else{
00101         return portError;
00102       }
00103     }

00122     if(port->writeTimeoutOccured()){
00123       // Cannot write now, sleep some time and try later
00124       port->unlockMutex();

00133       }
00134       frame->take(written);
00135       // Update cursor and toWrite
00136       if(frame->isEmpty()){
00137         toWrite = 0;
00138       }else{
00139         bufferCursor = frame->data();
00140         Q_ASSERT(bufferCursor < (frame->data() + frame->size()));
00141         toWrite -= written;
00142         Q_ASSERT(toWrite >= 0);
00143       }
00144     }
00145   }
00146   // Here, frame is completly sent
00147   Q_ASSERT(frame->isEmpty());
