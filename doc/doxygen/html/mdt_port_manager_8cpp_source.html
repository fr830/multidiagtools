00319 int mdtPortManager::sendData(mdtPortTransaction *transaction)
00320 {
00321   Q_ASSERT(pvPort != 0);
00322   Q_ASSERT(transaction != 0);
00323   Q_ASSERT(!pvTransactionsPending.contains(transaction->id()));
00324   Q_ASSERT(!transactionsDoneContains(transaction->id()));
00325 
00326   mdtFrame *frame = 0;
00327 
00328   // Wait until we can write
00329   while(1){
00330     // If port manager was stopped, we return
00331     if(isClosed()){
00332       restoreTransaction(transaction);
00333       return mdtAbstractPort::WriteCanceled;
00334     }
00335     if(isReady()){
00336       lockPortMutex();
00337       if(pvPort->writeFramesPool().size() > 0){
00338         frame = pvPort->writeFramesPool().dequeue();
00339         Q_ASSERT(frame != 0);
00340         // Here we are ready to write - we keep port mutext locked
00341         break;
00342       }
00343       unlockPortMutex();
00344     }
00345     QCoreApplication::processEvents(QEventLoop::AllEvents | QEventLoop::WaitForMoreEvents);
00346   }
00347   if(!waitTransactionPossible()){
00348     restoreTransaction(transaction);
00349     unlockPortMutex();
00350     return mdtAbstractPort::WriteCanceled;
00351   }
00352   // We are ready to write
00353   Q_ASSERT(frame != 0);
00354   frame->clear();
00355   frame->clearSub();
00356   // Store data and add frame to write queue
00357   frame->append(transaction->data());
00358   pvPort->addFrameToWrite(frame);
00359   addTransactionPending(transaction);
00360   unlockPortMutex();
00361 
00362   return transaction->id();
00363 }

00381 bool mdtPortManager::waitTransactionDone(int id)
00382 {
00383   Q_ASSERT(pvPort != 0);
00384 
00385   // Check if transaction is allready done
00386   if(transactionsDoneContains(id)){
00387     return true;
00388   }
00389   // Check if transaction exists in pending queue
00390   if(!pvTransactionsPending.contains(id)){
00391     mdtError e(MDT_PORT_IO_ERROR, "Wait on a frame that was never added to pending queue, id: " + QString::number(id), mdtError::Warning);
00392     MDT_ERROR_SET_SRC(e, "mdtPortManager");
00393     e.commit();
00394     emit pmUnhandledErrorEvent();
00395     return false;
00396   }
00397   /*
00398    * If thread handles timeout/disconnection:
00399    *  In normal way, the thread notifies the error, the state becomes != Ready and this method returns false
00400    *  But, if many errors are notified, state changes often,
00401    *   and it happens that we are waiting, in Ready state, on a ID that was flushed, and this for a eternity.
00402    *   (Keep in mind that we let event loop running, and this way new transactions can be initalized, etc...)
00403    *  Another problem that is not signaled from thread is a frame the never completes.
00404    *  To avoid this problem, and keep coherence with port thread,
00405    *   we add a offset to wait timeout, and return if this time was elapsed.
00406    * If thread does not handle timeout:
00407    *  We simply return after timeout
00408    */
00409   if(pvThreadHandlesReadTimeout){
00410     startTransactionTimer(config().readTimeout() + 200);
00411   }else{
00412     startTransactionTimer(config().readTimeout());
00413   }
00414   // Try until success or timeout/error
00415   while(!transactionsDoneContains(id)){
00416     // Check about timeout or other error
00417     //  Note: on busy state, we continue waiting (f.ex. ReadPoolEmpty error)
00418     if((!isReady())&&(currentState() != Busy)){
00419       // Transactions are restored by onThreadsErrorOccured()
00420       stopTransactionTimer();
00421       return false;
00422     }
00423     // Check about timeout
00424     if(transactionTimeoutOccured()){
00425       // Transactions are restored by onTransactionTimeoutOccured()
00426       return false;
00427     }
00428     // Wait
00429     QCoreApplication::processEvents(QEventLoop::AllEvents | QEventLoop::WaitForMoreEvents);
00430   }
00431   stopTransactionTimer();
00432 
00433   return true;
00434 }

00709   mdtPortTransaction *transaction;
