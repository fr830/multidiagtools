57 int mdtFrameUsbTmc::putData(const char *data, int maxLen)
58 {
59  Q_ASSERT(capacity() > 12);
60 
61  int frameSize = 0;
62  int stored = 0;
63 
64  // If frame is allready full, return
65  if(bytesToStore() < 1){
66  return 0;
67  }
68 
69  // Check wat's possible to store
70  if(maxLen > remainCapacity()){
71  maxLen = remainCapacity();
72  }
73  // store
74  append(data, maxLen);
75  stored = maxLen;
76 
77  // Check if we have enougth data to decode header
78  if(size() >= 12){
79  // Check if header was not allready decoded
80  if(pvTransferSize == 0){
81  // We have a header here, decode it
82  pvMsgID = at(0);
83  pvbTag = at(1);
84  pvbTagInverse = at(2);
85  // Check bTag
86  if(pvbTagInverse != (quint8)(~pvbTag)){
87  // Sync lost with device (or other error).
88  // We simply abort here, because transfer size will be wrong.
89  // Port manager will check bTag coherence and handle the error.
90  pvEOFcondition = true;
91  qDebug() << "mdtFrameUsbTmc::putData(): bTag/bTagInverse incoherent, bTag: " << pvbTag << " , expected inverse: " << (quint8)~pvbTag << " , rx inverse: " << pvbTagInverse;
92  return stored;
93  }else{
94  pvbTagOk = true;
95  }
96  pvTransferSize = at(4);
97  pvTransferSize += (at(5) << 8);
98  pvTransferSize += (at(6) << 16);
99  pvTransferSize += (at(7) << 24);
100  // MsgID specific part
101  if(pvMsgID == DEV_DEP_MSG_IN){
102  pvMsgIDsupported = true;
103  // EOM
104  if(at(8) & 0x01){
105  pvEOM = true;
106  }else{
107  pvEOM = false;
108  }
110  // Port manager will check received MsgID, and warn if unhadled one was received
111  }
117  }
118  // Calc total frame size, without alignment bytes
119  frameSize = pvTransferSize + 12;
120  // Check if we have a complete frame
121  if(size() >= frameSize){
122  pvEOFcondition = true;
123  pvMessageData += right(size()-12).left(pvTransferSize);
124  }
125  }
126 
127  return stored;
128 }

187 void mdtFrameUsbTmc::encode()
188 {
189  // Check minimal needed capacity
190  if(capacity() < 16){
191  reserve(16);
192  }
193  mdtFrame::clear();
194  // Get message data size
195  // Ajust command specific flags
196  if(pvMsgID == DEV_DEP_MSG_OUT){
197  pvTransferSize = pvMessageData.size();
198  }else if(pvMsgID == DEV_DEP_MSG_IN){
199  pvEOM = false;
200  pvMessageData.clear();
201  }else{
202  pvTransferSize = 0;
203  }
204  // Build USBTMC header
205  append((char)pvMsgID);
206  append((char)pvbTag);
207  append((char)pvbTagInverse);
208  append((char)0x00);
209  append((char)pvTransferSize & 0xFF); // TransferSize, LLSB
210  append((char)((pvTransferSize >> 8) & 0xFF)); // TransferSize, LSB
211  append((char)((pvTransferSize >> 16) & 0xFF)); // TransferSize, LSB
212  append((char)((pvTransferSize >> 24) & 0xFF)); // TransferSize, LSB
213  if(pvEOM){
214  append((char)0x01); // bmTransferAttributes
215  }else{
216  append((char)0x00); // bmTransferAttributes
217  }
218  append((char)0x00); // Reserved
219  append((char)0x00); // Reserved
220  append((char)0x00); // Reserved
221  // Add data
222  append(pvMessageData);
223  // Add alignment bytes if requierd
224  while((size() % 4) != 0){
225  append((char)0x00);
226  }
227 }
