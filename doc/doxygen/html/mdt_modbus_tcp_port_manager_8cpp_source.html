00325   pdu = codec.encodeReadDiscreteInputs(startFrom, bitsCount);
00326   if(pdu.isEmpty()){
00327     return (int)mdtAbstractPort::UnhandledError;
00328   }
00329   // Get a new transaction
00330   transaction = getNewTransaction();
00331   // Send query
00332   transaction->setQueryReplyMode(true);
00333   transaction->setData(pdu);
00334   transactionId = writeData(transaction);
00335   if(transactionId < 0){
00336     return transactionId;
00337   }
00338   // Wait on result (use device's defined timeout)
00339   if(!waitTransactionDone(transactionId)){
00340     return (int)mdtAbstractPort::ReadTimeout;
00341   }
00342   // Get reply and decode
00343   if(codec.decode(readenFrame(transactionId)) != 0x02){
00344     return (int)mdtAbstractPort::UnhandledError;
00345   }
00346   // Check that we have enough states (can be more, because digital inputs are returned as multiple of 8)
00347   if(codec.values().size() < bitsCount){
00348     mdtError e(MDT_TCP_IO_ERROR, "Have not received enough input states", mdtError::Error);
00349     MDT_ERROR_SET_SRC(e, "mdtModbusTcpPortManager");
00350     e.commit();
00351     return (int)mdtAbstractPort::UnhandledError;
00352   }
00353   // Ok, have the states, let's decode ID
00354   id = 0;
00355   for(i=0; i<bitsCount; i++){
00356     if((codec.values().at(i).isValid())&&(codec.values().at(i).toBool() == true)){
00357       id |= (1<<i);
