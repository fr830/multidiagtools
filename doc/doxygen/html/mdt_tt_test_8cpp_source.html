578 QVariant mdtTtTest::getConnectionId(const QVariant & testNodeId, const QString & schemaPosition, const QString & connector, const QString & contact, bool & ok)
579 {
580  QString sql;
581  QList<QVariant> dataList;
582  QVariant connectionId;

584  sql = "SELECT UCNX.Id_PK FROM UnitConnection_tbl UCNX";

586  sql += " JOIN Unit_tbl U ON U.Id_PK = UCNX.Unit_Id_FK";

613 
614  return dataList.at(0);
