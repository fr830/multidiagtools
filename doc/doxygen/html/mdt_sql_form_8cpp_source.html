178 bool mdtSqlForm::select(const QString &tableName)
179 {
180  if(pvController->tableName() == tableName){
181  if(!pvController->select()){
182  pvLastError = pvController->lastError();
183  return false;
184  }
185  return true;
186  }
187  std::shared_ptr<mdtSqlTableViewController> controller;
188  controller = pvController->childController<mdtSqlTableViewController>(tableName);
189  if(!controller){
190  pvLastError.setError(tr("Cannot find controller that acts on table '") + tableName + tr("'"), mdtError::Error);
191  MDT_ERROR_SET_SRC(pvLastError, "mdtSqlForm");
192  pvLastError.commit();
193  return false;
194  }
195  if(!controller->select()){
196  pvLastError = controller->lastError();
197  return false;
198  }
199  return true;
200 }
