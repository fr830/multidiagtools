00246   if(!pvControlTransferPending){
00247     return NoError;
00248   }
00249   // If transfer is null, port was closed
00250   if(pvControlTransfer == 0){
00251     // Stop thread
00252     return UnhandledError;
00253   }
00254   // Get and update complete flag
00255   pComplete = (int*)pvControlTransfer->user_data;
00256   pvControlTransferComplete = *pComplete;
00257   // If transfer is not complete, we simply return
00258   if(pvControlTransferComplete == 0){
00259     return NoError;
00260   }
00261   // Transfer complete, take the frame and check flags
00262   pvControlTransferPending = false;
00263   frame = pvControlQueryFrames.dequeue();
00264   Q_ASSERT(frame != 0);
00265   // Check if transfer has timed out
00266   if(pvControlTransfer->status & LIBUSB_TRANSFER_TIMED_OUT){
00267     updateWriteTimeoutState(true);
00268   }
00269   // Check if transfer was cancelled
00270   if(pvControlTransfer->status & LIBUSB_TRANSFER_CANCELLED){
00271     qDebug() << "mdtUsbPort::handleControlResponses() return ReadCanceled";
00272     pvControlFramesPool.enqueue(frame);
00273     return WriteCanceled;
00274   }
00275   // Check if device is disconnected (not valid if cancelled)
00276   if(pvControlTransfer->status & LIBUSB_TRANSFER_NO_DEVICE){
00277     qDebug() << "mdtUsbPort::handleControlResponses(): device disconnected";
00278     pvControlFramesPool.enqueue(frame);
00279     return Disconnected;
00280   }
00281   // Check if transfer stall
00282   if(pvControlTransfer->status & LIBUSB_TRANSFER_STALL){
00283     mdtError e(MDT_USB_IO_ERROR, "Transfer status LIBUSB_TRANSFER_STALL during a control transfer", mdtError::Error);
00284     MDT_ERROR_SET_SRC(e, "mdtUsbPort");
00285     e.commit();
00286     pvControlFramesPool.enqueue(frame);
00287     return UnhandledError;
00288   }
00289   // Check if transfer failed (not valid if cancelled)
00290   if(pvControlTransfer->status & LIBUSB_TRANSFER_ERROR){
00291     mdtError e(MDT_USB_IO_ERROR, "Transfer status LIBUSB_TRANSFER_ERROR during a control transfer", mdtError::Error);
00292     MDT_ERROR_SET_SRC(e, "mdtUsbPort");
00293     e.commit();
00294     pvControlFramesPool.enqueue(frame);
00295     return UnhandledError;
00296   }
00297   // Check transfer overflow
00298   if(pvControlTransfer->status & LIBUSB_TRANSFER_OVERFLOW){
00299     mdtError e(MDT_USB_IO_ERROR, "Transfer status LIBUSB_TRANSFER_OVERFLOW during a control transfer", mdtError::Error);
00300     MDT_ERROR_SET_SRC(e, "mdtUsbPort");
00301     e.commit();
00302     pvControlFramesPool.enqueue(frame);
00303     return UnhandledError;
00304   }
00305   // Check transfer short frames NOK
00306   if(pvControlTransfer->status & LIBUSB_TRANSFER_SHORT_NOT_OK){
00307     mdtError e(MDT_USB_IO_ERROR, "Transfer status LIBUSB_TRANSFER_SHORT_NOT_OK during a control transfer", mdtError::Error);
00308     MDT_ERROR_SET_SRC(e, "mdtUsbPort");
00309     e.commit();
00310     pvControlFramesPool.enqueue(frame);
00311     return UnhandledError;
00312   }
00313   // All is Ok, copy data and enqueue to responses queue
00314   frame->clear();
00315   frame->clearSub();
00316   controlSetup =  libusb_control_transfer_get_setup(pvControlTransfer);
00317   Q_ASSERT(controlSetup != 0);
00318   frame->setbmRequestType(controlSetup->bmRequestType);
00319   frame->setbRequest(controlSetup->bRequest);
00320   frame->setwValue(controlSetup->wValue);
00321   frame->setwIndex(controlSetup->wIndex);
00322   frame->setwLength(controlSetup->wLength);
00323   controlData = (char*)libusb_control_transfer_get_data(pvControlTransfer);
00324   frame->append(controlData, pvControlTransfer->actual_length);
00325   pvControlResponseFrames.enqueue(frame);
00326 
00327   return NoError;
00328 }
