00064 QList<mdtPortInfo*> mdtUsbPortManager::scan(int bDeviceClass, int bDeviceSubClass, int bDeviceProtocol)
00065 {
00066   Q_ASSERT(!isRunning());
00067 
00068   QList<mdtPortInfo*> portInfoList;
00069   mdtDeviceInfo *deviceInfo;
00070   mdtPortInfo *portInfo;
00071   libusb_context *ctx;
00072   libusb_device **devicesList;
00073   mdtUsbDeviceDescriptor deviceDescriptor;
00074   mdtUsbConfigDescriptor *configDescriptor;
00075   mdtUsbInterfaceDescriptor *ifaceDescriptor;
00076   QString portName;
00077   libusb_device *device;
00078   ssize_t devicesCount;
00079   ssize_t i;
00080   int j, k;
00081   int err;
00082   QString str;
00083 
00084   // Init libusb
00085   err = libusb_init(&ctx);
00086   if(err != 0){
00087     mdtError e(MDT_USB_IO_ERROR, "Cannot init libusb", mdtError::Error);
00088     MDT_ERROR_SET_SRC(e, "mdtUsbPortManager");
00089     e.commit();
00090     return portInfoList;
00091   }
00092   // Get devices list
00093   devicesCount = libusb_get_device_list(ctx, &devicesList);
00094   if(devicesCount < 0){
00095     mdtError e(MDT_USB_IO_ERROR, "Cannot list devices", mdtError::Error);
00096     MDT_ERROR_SET_SRC(e, "mdtUsbPortManager");
00097     e.commit();
00098     return portInfoList;
00099   }
00100   // Scan ...
00101   for(i=0; i<devicesCount; i++){
00102     device = devicesList[i];
00103     err = deviceDescriptor.fetchAttributes(device, false);
00104     if(err != 0){
00105       mdtError e(MDT_USB_IO_ERROR, "Device attributes fetch failed", mdtError::Error);
00106       MDT_ERROR_SET_SRC(e, "mdtUsbPortManager");
00107       e.commit();
00108       continue;
00109     }
00110     // Create a port info object
00111     portInfo = new mdtPortInfo;
00112     // Search interfaces in various configurations
00113     for(j=0; j<deviceDescriptor.configurations().size(); j++){
00114       configDescriptor = deviceDescriptor.configurations().at(j);
00115       Q_ASSERT(configDescriptor != 0);
00116       for(k=0; k<configDescriptor->interfaces().size(); k++){
00117         ifaceDescriptor = configDescriptor->interfaces().at(k);
00118         Q_ASSERT(ifaceDescriptor != 0);
00119         // See if a specific bDeviceClass was requested
00120         if(bDeviceClass >= 0){
00121           if(ifaceDescriptor->bInterfaceClass() != (quint8)bDeviceClass){
00122             continue;
00123           }
00124         }
00125         // See if a specific bDeviceSubClass was requested
00126         if(bDeviceSubClass >= 0){
00127           if(ifaceDescriptor->bInterfaceSubClass() != (quint8)bDeviceSubClass){
00128             continue;
00129           }
00130         }
00131         // See if a specific bDeviceProtocol was requested
00132         if(bDeviceProtocol >= 0){
00133           if(ifaceDescriptor->bInterfaceProtocol() != (quint8)bDeviceProtocol){
00134             continue;
00135           }
00136         }
00137         // Interresting interface
00138         deviceInfo = new mdtDeviceInfo;
00139         // Build the display text
00140         str = deviceDescriptor.vendorName();
00141         str += ", ";
00142         str += deviceDescriptor.productName();
00143         if(!deviceDescriptor.serialNumber().isEmpty()){
00144           str += " (SN: " + deviceDescriptor.serialNumber() + ")";
00145         }
00146         portInfo->setDisplayText(str);
00147         // Set VID, PID, SID and protocol ID
00148         deviceInfo->setVendorId(deviceDescriptor.idVendor());
00149         deviceInfo->setProductId(deviceDescriptor.idProduct());
00150         deviceInfo->setSerialId(deviceDescriptor.serialNumber());
00151         deviceInfo->setProtocolId(ifaceDescriptor->bInterfaceProtocol());
00152         // Add interface to list
00153         portInfo->addDevice(deviceInfo);
00154       }
00155     }
00156     // If port info has no device, delete it, else add it to list
00157     if(portInfo->deviceInfoList().size() < 1){
00158       delete portInfo;
00159     }else{
00160       // Build port name
00161       deviceInfo = portInfo->deviceInfoList().at(0);
00162       Q_ASSERT(deviceInfo != 0);
00163       portName = "VID=0x";
00164       portName += QString::number((unsigned int)deviceInfo->vendorId(), 16);
00165       portName += ":PID=0x";
00166       portName += QString::number((unsigned int)deviceInfo->productId(), 16);
00167       if(!deviceDescriptor.serialNumber().isEmpty()){
00168         portName += ":SID=" + deviceDescriptor.serialNumber();
00169       }
00170       portInfo->setPortName(portName);
00171       // Add port info to list
00172       portInfoList.append(portInfo);
00173     }
00174   }
00175   // Release ressources
00176   libusb_free_device_list(devicesList, 1);
00177   libusb_exit(ctx);
00178 
00179   return portInfoList;
00180 }

00182 int mdtUsbPortManager::sendControlRequest(const mdtFrameUsbControl &request, bool setwIndexAsbInterfaceNumber)
00183 {
00184   Q_ASSERT(pvPort != 0);
00185 
00186   mdtUsbPort *port;
00187   mdtFrameUsbControl *frame;
00188 
00189   // We need a mdtUsbPort object
00190   port = dynamic_cast<mdtUsbPort*>(pvPort);
00191   Q_ASSERT(port != 0);
00192   // Get a frame in pool
00193   port->lockMutex();
00194   if(port->controlFramesPool().size() < 1){
00195     port->unlockMutex();
00196     mdtError e(MDT_PORT_IO_ERROR, "No frame available in control frames pool", mdtError::Error);
00197     MDT_ERROR_SET_SRC(e, "mdtUsbPortManager");
00198     e.commit();
00199     return mdtAbstractPort::WritePoolEmpty;
00200   }
00201   frame = dynamic_cast<mdtFrameUsbControl*> (port->controlFramesPool().dequeue());
00202   Q_ASSERT(frame != 0);
00203   frame->clear();
00204   frame->clearSub();
00205   // Store data, encode and add frame to queue
00206   frame->setDirectionDeviceToHost(request.directionIsDeviceToHost());
00207   frame->setbmRequestType(request.bmRequestType());
00208   frame->setbRequest(request.bRequest());
00209   frame->setwValue(request.wValue());
00210   frame->setwIndex(request.wIndex());
00211   frame->setwLength(request.wLength());
00212   frame->encode();
00213   port->addControlRequest(frame, setwIndexAsbInterfaceNumber);
00214   port->unlockMutex();
00215 
00216   return 0;
