44 int mdtPortLock::openLocked(const QString &portName, int flags)
45 {
46  int i;
47  int err;
48 
49  // If this method is called second time without unlocking port before, it's a error
50  if(pvIsLocked){
51  mdtError e(MDT_PORT_IO_ERROR , "Port " + portName + " is allready locked by this instance, please unlock it first", mdtError::Info);
52  MDT_ERROR_SET_SRC(e, "mdtPortLock");
53  e.commit();
54  return -1;
55  }
56 
57  pvPort.setFile(portName);
58  pvIsLockedByAnother = false;
59 
60  // Search about lock files
61  buildLockFilesList();
62  for(i=0; i<pvLockfiles.size(); i++){
63  if(pvLockfiles.at(i).exists()){
64  mdtError e(MDT_PORT_IO_ERROR , "Port " + pvPort.absoluteFilePath() + " is allready locked (Lockfile" + pvLockfiles.at(i).filePath() + " exists)", mdtError::Info);
65  MDT_ERROR_SET_SRC(e, "mdtPortLock");
66  e.commit();
67  pvIsLockedByAnother = true;
68  return -1;
69  }
70  }
71  // No lock files found, create all
72  for(i=0; i<pvLockfiles.size(); i++){
73  if(!createLockFile(pvLockfiles.at(i))){
74  return -1;
75  }
76  }
77  // Try to open the port
78  pvFd = ::open(pvPort.absoluteFilePath().toStdString().c_str(), flags);
79  if(pvFd < 0){
80  err = errno;
81  mdtError e(MDT_PORT_IO_ERROR, "Unable to open " + pvPort.absoluteFilePath(), mdtError::Error);
82  e.setSystemError(err, strerror(err));
83  MDT_ERROR_SET_SRC(e, "mdtPortLock");
84  e.commit();
85  removeLockFiles();
86  errno = err;
87  return -1;
88  }
89  // Lock the port
90  pvLock.l_type = F_WRLCK;
91  pvLock.l_whence = SEEK_SET;
92  pvLock.l_start = 0;
93  pvLock.l_len = 0;
94  if(fcntl(pvFd, F_SETLK, &pvLock) < 0){
95  err =errno;
96  mdtError e(MDT_PORT_IO_ERROR, "Unable to lock " + pvPort.absoluteFilePath(), mdtError::Error);
97  e.setSystemError(err, strerror(err));
98  MDT_ERROR_SET_SRC(e, "mdtPortLock");
99  e.commit();
100  removeLockFiles();
101  ::close(pvFd);
102  pvFd = -1;
103  errno = err;
104  return -1;
105  }
106  // Ok, port is open and locked here
107  pvIsLocked = true;
108  pvIsLockedByAnother = false;
109 
110  return pvFd;
111 }

137 void mdtPortLock::scanForLockDirectories()
138 {
139  QDir dir;
140  QFileInfo info;
141 
142  // Try /var/lock
143  dir.setPath("/var/lock");
144  if(dir.exists()){
145  info.setFile(dir.path());
146  if(info.isWritable()){
147  pvWritableLockDirectories << dir.path();
148  }
149  }
150  // Try /tmp
151  dir.setPath("/tmp");
152  if(dir.exists()){
153  info.setFile(dir.path());
154  if(info.isWritable()){
155  pvWritableLockDirectories << dir.path();
156  }
157  }
158  // Warn if no directory was found
159  if(pvWritableLockDirectories.size() < 1){
160  mdtError e(MDT_PORT_IO_ERROR, "No lock directory found", mdtError::Warning);
161  MDT_ERROR_SET_SRC(e, "mdtPortLock");
162  e.commit();
163  }
164 }

178 bool mdtPortLock::createLockFile(const QFileInfo &file)
179 {
180  QFile f(file.absoluteFilePath());
181 
182  // Check if file exists
183  if(f.exists()){
184  mdtError e(MDT_PORT_IO_ERROR, "File " + f.fileName() + " exists", mdtError::Error);
185  MDT_ERROR_SET_SRC(e, "mdtPortLock");
186  e.commit();
187  pvIsLocked = true;
188  return false;
189  }
190  // Try to create
191  if(!f.open(QIODevice::ReadWrite)){
192  mdtError e(MDT_PORT_IO_ERROR, "File " + f.fileName() + " cannot be created", mdtError::Error);
193  MDT_ERROR_SET_SRC(e, "mdtPortLock");
194  e.commit();
195  return false;
196  }
197  pvCreatedLockFiles << file;
198  f.close();
199 
200  return true;
201 }
