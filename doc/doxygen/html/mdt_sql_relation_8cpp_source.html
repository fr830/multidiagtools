00056 bool mdtSqlRelation::addRelation(const QString &parentFieldName, const QString &childFieldName, bool copyParentToChildOnInsertion, const QString &operatorWithPreviousItem)
00057 {
00058   Q_ASSERT(pvParentModel != 0);
00059   Q_ASSERT(pvChildModel != 0);
00060 
00061   QSqlRecord record;
00062   int parentFieldIndex;
00063   QSqlField parentField;
00064   int childFieldIndex;
00065   QSqlField childField;
00066 
00067   // Get parent  field index, check that it exist
00068   record = pvParentModel->record();
00069   parentFieldIndex = record.indexOf(parentFieldName);
00070   if(parentFieldIndex < 0){
00071     mdtError e(MDT_DATABASE_ERROR, "Field '" + parentFieldName + "' not found in parent table", mdtError::Error);
00072     MDT_ERROR_SET_SRC(e, "mdtSqlRelation");
00073     e.commit();
00074     return false;
00075   }
00076   parentField = record.field(parentFieldIndex);
00077   // Get child field index, check that it exist
00078   record = pvChildModel->record();
00079   childFieldIndex = record.indexOf(childFieldName);
00080   if(childFieldIndex < 0){
00081     mdtError e(MDT_DATABASE_ERROR, "Field '" + childFieldName + "' not found in child table", mdtError::Error);
00082     MDT_ERROR_SET_SRC(e, "mdtSqlRelation");
00083     e.commit();
00084     return false;
00085   }
00086   childField = record.field(childFieldIndex);
00087   // Add relation informations
00088   mdtSqlRelationItem *item = new mdtSqlRelationItem;
00089   item->setParentField(parentField);
00090   item->setParentFieldIndex(parentFieldIndex);
00091   item->setChildField(childField);
00092   item->setChildFieldIndex(childFieldIndex);
00093   item->setCopyParentToChildOnInsertion(copyParentToChildOnInsertion);
00094   item->setRelationOperatorWithPreviousItem(" " + operatorWithPreviousItem.trimmed() + " ");
00095   pvRelations.append(item);
00096 
00097   return true;
00098 }

00181 void mdtSqlRelation::generateChildModelRelationFilter(int row)
00182 {
00183   Q_ASSERT(pvParentModel != 0);
00184   Q_ASSERT(pvChildModel != 0);
00186 
00187   int i;
00188   mdtSqlRelationItem *item;
00189   QSqlRecord record;
00190   QVariant data;
00191 
00192   // Get data record of the parent model
00193   record = pvParentModel->record(row);
00194   // Build filter
00195   pvChildModelRelationFilter.clear();
00196   for(i=0; i<pvRelations.size(); ++i){
00197     item = pvRelations.at(i);
00198     Q_ASSERT(item != 0);
00200     // Get parent model's data
00201     if(!record.isNull(item->parentFieldIndex())){
00202       data = record.value(item->parentFieldIndex());
00203     }
00204     if(i>0){
00206       pvChildModelRelationFilter += item->relationOperatorWithPreviousItem();
00207     }
00209     pvChildModelRelationFilter += item->nameProtection() + pvChildModel->tableName() + item->nameProtection() + "." + item->nameProtection() + item->childFieldName() + item->nameProtection() + "=";
00210     if(data.isValid()){
00211       pvChildModelRelationFilter += item->dataProtection() + data.toString() + item->dataProtection();
00212     }else{
00213       pvChildModelRelationFilter += item->dataProtection() + item->valueForNoDataFilter().toString() + item->dataProtection();
00214     }
00215   }
00216   // Apply filter
00217   generateChildModelFilter();
00218   pvChildModel->setFilter(pvChildModelFilter);
00219   emit childModelFilterApplied();
00220   if(pvChildModel->rowCount() < 1){
00221     emit childModelIsEmpty();
00222   }
