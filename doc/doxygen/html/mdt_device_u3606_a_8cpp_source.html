197 mdtDeviceU3606A::MeasureType_t mdtDeviceU3606A::getMeasureConfiguration()
198 {
199  QByteArray data;
200  mdtCodecScpi codec;
201  mdtScpiFunctionFarameter fp;
202 
203  // Get configuration from device
204  data = sendQuery("CONF?\n");
205  if(data.isEmpty()){
206  return MeasureType_t::Unknown;
207  }
208  // Deduce current device configuration
209  fp = codec.decodeFunctionParameters(data);
210  switch(fp.fundamentalMeasurementCommand()){
211  case mdtScpiFunctionFarameter::FundamentalMeasurementCommand_t::Unknown:
212  return MeasureType_t::Unknown;
213  case mdtScpiFunctionFarameter::FundamentalMeasurementCommand_t::Voltage:
214  switch(fp.simpleMeasurementCommand()){
215  case mdtScpiFunctionFarameter::SimpleMeasurementCommand_t::Dc:
216  return MeasureType_t::VoltageDc;
217  case mdtScpiFunctionFarameter::SimpleMeasurementCommand_t::Ac:
218  return MeasureType_t::VoltageAc;
219  default:
220  return MeasureType_t::Unknown;
221  }
222  break;
223  case mdtScpiFunctionFarameter::FundamentalMeasurementCommand_t::Current:
224  switch(fp.simpleMeasurementCommand()){
225  case mdtScpiFunctionFarameter::SimpleMeasurementCommand_t::Dc:
226  return MeasureType_t::CurrentDc;
227  case mdtScpiFunctionFarameter::SimpleMeasurementCommand_t::Ac:
228  return MeasureType_t::CurrentAc;
229  default:
230  return MeasureType_t::Unknown;
231  }
232  break;
233  case mdtScpiFunctionFarameter::FundamentalMeasurementCommand_t::Resistance:
234  return MeasureType_t::Resistance;
235  case mdtScpiFunctionFarameter::FundamentalMeasurementCommand_t::Fresistance:
236  return MeasureType_t::Lresistance;
237  }
238 }
