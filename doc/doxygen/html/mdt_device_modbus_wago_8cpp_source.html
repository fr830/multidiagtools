00084 bool mdtDeviceModbusWago::isWago750()
00085 {
00086   // If device is not from Wago, server can return a error (Something like: invalid address range)
00087   // We put a information into the log, so that we know that we are scanning.
00088   mdtError e1(MDT_DEVICE_ERROR, "Device " + name() + ": checking if device is from Wago ...", mdtError::Info);
00089   MDT_ERROR_SET_SRC(e1, "mdtDeviceModbusWago");
00090   e1.commit();
00091 
00092   if(!getRegisterValues(0x2011, 1)){
00093     return false;
00094   }
00095   if(registerValues().size() != 1){
00096     return false;
00097   }
00098   mdtError e2(MDT_DEVICE_ERROR, "Device " + name() + ": device is from Wago", mdtError::Info);
00099   MDT_ERROR_SET_SRC(e2, "mdtDeviceModbusWago");
00100   e2.commit();
00101 
00102   return (registerValues().at(0) == 750);
00103 }
