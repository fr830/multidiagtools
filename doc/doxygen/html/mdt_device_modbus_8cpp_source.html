00276 void mdtDeviceModbus::decodeReadenFrame(mdtPortTransaction *transaction)
00277 {
00278   Q_ASSERT(transaction != 0);
00279   if(ios() == 0){
00280     return;
00281   }
00282 
00283   int fc;
00284   QVariant var;
00285 
00287   // Decode readen frame and update I/O's
00288   fc = pvCodec->decode(transaction->data());
00289   switch(fc){
00290     case 0x01:  // Read coils
00291       // Check if we have a answer for a single I/O or multiple I/Os
00292       if((transaction->ioCount() == 1)&&(transaction->digitalIo() != 0)){
00293         // If one DO was requested, 1 byte is returned with just first bit set
00294         if(pvCodec->values().size() == 8){
00295           var = pvCodec->values().at(0);
00296           if((var.isValid())&&(var.type() == QVariant::Bool)){
00297             transaction->digitalIo()->setValue(var.toBool(), false);
00298           }else{
00299             transaction->digitalIo()->setValue(mdtValue(), false);
00300           }
00301         }else{
00302           transaction->digitalIo()->setValue(mdtValue(), false);
00303         }
00304         transaction->digitalIo()->setEnabled(true);
00305       }else{
00306         // Transaction contains first I/O address (for read access) and qty of I/Os
00307         ios()->updateDigitalOutputValues(pvCodec->values(), transaction->address(), mdtDeviceIos::Read , transaction->ioCount(), true);
00308         ios()->setDigitalOutputsEnabled(true);
00309       }
00310       break;
00311     case 0x02:  // Read discrete inputs
00312       // Check if we have a answer for a single I/O or multiple I/Os
00313       if((transaction->ioCount() == 1)&&(transaction->digitalIo() != 0)){
00314         // If one DI was requested, 1 byte is returned with just first bit set
00315         if(pvCodec->values().size() == 8){
00316           var = pvCodec->values().at(0);
00317           if((var.isValid())&&(var.type() == QVariant::Bool)){
00318             transaction->digitalIo()->setValue(var.toBool(), false);
00319           }else{
00320             transaction->digitalIo()->setValue(mdtValue(), false);
00321           }
00322         }else{
00323           transaction->digitalIo()->setValue(mdtValue(), false);
00324         }
00325       }else{
00326         // Transaction contains first I/O address and qty of I/Os
00327         ios()->updateDigitalInputValues(pvCodec->values(), transaction->address(), transaction->ioCount(), true);
00328       }
00329       break;
00330     case 0x03:  // Read holding registers
00331       // Check if we have a answer for a single I/O or multiple I/Os
00332       if((transaction->ioCount() == 1)&&(transaction->analogIo() != 0)){
00333         // Single I/O reply
00334         if(pvCodec->values().size() == 1){
00335           var = pvCodec->values().at(0);
00336           if((var.isValid())&&(var.type() == QVariant::Int)){
00337             // Codec only get raw data, so let I/O do needed conversions
00338             transaction->analogIo()->setValue(var.toInt(), false);
00339           }else{
00340             transaction->analogIo()->setValue(mdtValue(), false);
00341           }
00342         }else{
00343           transaction->analogIo()->setValue(mdtValue(), false);
00344         }
00345         transaction->analogIo()->setEnabled(true);
00346       }else{
00347         // Multiple I/O reply
00348         ios()->updateAnalogOutputValues(pvCodec->values(), transaction->address(), mdtDeviceIos::Read, transaction->ioCount(), true);
00349         ios()->setAnalogOutputsEnabled(true);
00350       }
00351       break;
00352     case 0x04:  // Read input registers
00353       // Check if we have a answer for a single I/O or multiple I/Os
00354       if((transaction->ioCount() == 1)&&(transaction->analogIo() != 0)){
00355         // Single I/O reply
00356         if(pvCodec->values().size() == 1){
00357           var = pvCodec->values().at(0);
00358           if((var.isValid())&&(var.type() == QVariant::Int)){
00359             // Codec only get raw data, so let I/O do needed conversions
00360             transaction->analogIo()->setValue(var.toInt(), false);
00361           }else{
00362             transaction->analogIo()->setValue(mdtValue(), false);
00363           }
00364         }else{
00365           transaction->analogIo()->setValue(mdtValue(), false);
00366         }
00367       }else{
00368         // Multiple I/O reply
00369         ios()->updateAnalogInputValues(pvCodec->values(), transaction->address(), transaction->ioCount(), true);
00370       }
00371       break;
00372     case 0x05:  // Write single coil reply
00373       if(transaction->digitalIo() != 0){
00374         // Check validity and update
00375         if(pvCodec->values().size() == 2){
00376           // Address
00377           var = pvCodec->values().at(0);
00378           if((var.isValid())&&(var.type() == QVariant::Int)&&(var.toInt() == transaction->digitalIo()->addressWrite())){
00379             // Value
00380             var = pvCodec->values().at(1);
00381             if((var.isValid())&&(var.type() == QVariant::Bool)){
00382               transaction->digitalIo()->setValue(var.toBool(), false);
00383             }else{
00384               transaction->digitalIo()->setValue(mdtValue(), false);
00385             }
00386           }else{
00387             transaction->digitalIo()->setValue(mdtValue(), false);
00388           }
00389         }else{
00390           transaction->digitalIo()->setValue(mdtValue(), false);
00391         }
00392         transaction->digitalIo()->setEnabled(true);
00393       }
00394       break;
00395     case 0x06:  // Write single register reply
00396       if(transaction->analogIo() != 0){
00397         // Check validity and update
00398         if(pvCodec->values().size() == 2){
00399           // Address
00400           var = pvCodec->values().at(0);
00401           if((var.isValid())&&(var.type() == QVariant::Int)&&(var.toInt() == transaction->analogIo()->addressWrite())){
00402             // Value
00403             var = pvCodec->values().at(1);
00404             if((var.isValid())&&(var.type() == QVariant::Int)){
00405               // Update (G)UI - Codec only get raw data, so let I/O do needed conversions
00406               transaction->analogIo()->setValue(var.toInt(), false);
00407             }
00408           }else{
00409             transaction->analogIo()->setValue(mdtValue(), false);
00410           }
00411         }else{
00412           transaction->analogIo()->setValue(mdtValue(), false);
00413         }
00414         transaction->analogIo()->setEnabled(true);
00415       }
00416       break;
00417     case 0x0F:  // Write multiple coils
00418       // Check validitiy and update
00419       if(pvCodec->values().size() == 2){
00420         // Start address
00421         var = pvCodec->values().at(0);
00422         if((var.isValid())&&(var.type() == QVariant::Int)&&(var.toInt() == transaction->address())){
00423           // Qty of outputs set
00424           var = pvCodec->values().at(1);
00425           if((!var.isValid())||(var.type() != QVariant::Int)||(var.toInt() != transaction->ioCount())){
00426             ios()->setDigitalOutputsValue(mdtValue());
00427           }
00428         }else{
00429           ios()->setDigitalOutputsValue(mdtValue());
00430         }
00431       }else{
00432         ios()->setDigitalOutputsValue(mdtValue());
00433       }
00434       ios()->setDigitalOutputsEnabled(true);
00435     case 0x10:  // Write multiple registers reply
00436       // Check validitiy and update
00437       if(pvCodec->values().size() == 2){
00438         // Start address
00439         var = pvCodec->values().at(0);
00440         if((var.isValid())&&(var.type() == QVariant::Int)&&(var.toInt() == transaction->address())){
00441           // Qty of outputs set
00442           var = pvCodec->values().at(1);
00443           if((!var.isValid())||(var.type() != QVariant::Int)||(var.toInt() != transaction->ioCount())){
00444             ios()->setAnalogOutputsValue(mdtValue());
00445           }
00446         }else{
00447           ios()->setAnalogOutputsValue(mdtValue());
00448         }
00449       }else{
00450         ios()->setAnalogOutputsValue(mdtValue());
00451       }

00453       break;

00457       MDT_ERROR_SET_SRC(e, "mdtDeviceModbus");
00458       e.commit();
00459   }
