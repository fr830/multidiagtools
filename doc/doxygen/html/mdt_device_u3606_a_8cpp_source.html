193 mdtDeviceU3606A::MeasureType_t mdtDeviceU3606A::getMeasureConfiguration()
194 {
195  QByteArray data;
196  mdtCodecScpi codec;
197  mdtScpiFunctionFarameter fp;
198 
199  // Get configuration from device
200  data = sendQuery("CONF?\n");
201  if(data.isEmpty()){
202  return MeasureType_t::Unknown;
203  }
204  // Deduce current device configuration
205  fp = codec.decodeFunctionParameters(data);
206  switch(fp.fundamentalMeasurementCommand()){
207  case mdtScpiFunctionFarameter::FundamentalMeasurementCommand_t::Unknown:
208  return MeasureType_t::Unknown;
209  case mdtScpiFunctionFarameter::FundamentalMeasurementCommand_t::Voltage:
210  switch(fp.simpleMeasurementCommand()){
211  case mdtScpiFunctionFarameter::SimpleMeasurementCommand_t::Dc:
212  return MeasureType_t::VoltageDc;
213  case mdtScpiFunctionFarameter::SimpleMeasurementCommand_t::Ac:
214  return MeasureType_t::VoltageAc;
215  default:
216  return MeasureType_t::Unknown;
217  }
218  break;
219  case mdtScpiFunctionFarameter::FundamentalMeasurementCommand_t::Current:
220  switch(fp.simpleMeasurementCommand()){
221  case mdtScpiFunctionFarameter::SimpleMeasurementCommand_t::Dc:
222  return MeasureType_t::CurrentDc;
223  case mdtScpiFunctionFarameter::SimpleMeasurementCommand_t::Ac:
224  return MeasureType_t::CurrentAc;
225  default:
226  return MeasureType_t::Unknown;
227  }
228  break;
229  case mdtScpiFunctionFarameter::FundamentalMeasurementCommand_t::Resistance:
230  return MeasureType_t::Resistance;
231  case mdtScpiFunctionFarameter::FundamentalMeasurementCommand_t::Fresistance:
232  return MeasureType_t::Lresistance;
233  }
234 }
