00061 bool mdtAnalogIo::setRange(double min, double max, int intValueBitsCount, int intValueLsbIndex, bool intValueSigned, bool scaleFromMinToMax, double conversionFactor)
00062 {
00063   Q_ASSERT(max > min);
00064   Q_ASSERT((qAbs(conversionFactor) - DBL_EPSILON) > 0.0);
00065 
00066   int i;
00067 
00068   // Check that lsb index and bits count are in acceptable range
00069   if(intValueLsbIndex < 0){
00070     mdtError e(MDT_PARSE_ERROR, "Specified intValueLsbIndex is < 0, using current conversion parameters", mdtError::Warning);
00071     MDT_ERROR_SET_SRC(e, "mdtAnalogIo");
00072     e.commit();
00073     return false;
00074   }
00075   if(intValueBitsCount < 0){
00076     mdtError e(MDT_PARSE_ERROR, "Specified intValueBitsCount is < 0, using current conversion parameters", mdtError::Warning);
00077     MDT_ERROR_SET_SRC(e, "mdtAnalogIo");
00078     e.commit();
00079     return false;
00080   }
00081   if((unsigned int)(intValueLsbIndex+intValueBitsCount) > (unsigned int)(8*sizeof(int))){
00082     mdtError e(MDT_PARSE_ERROR, "Specified intValueLsbIndex + intValueBitsCount is to big, using current conversion parameters", mdtError::Warning);
00083     MDT_ERROR_SET_SRC(e, "mdtAnalogIo");
00084     e.commit();
00085     return false;
00086   }
00087   // Store lsbIndex
00088   pvIntValueLsbIndex = intValueLsbIndex;
00089   pvIntValueLsbIndexEnc = intValueLsbIndex;
00090   // Setup mask
00091   pvIntValueMask = 0;
00092   pvIntValueMaskEnc = 0;
00093   for(i = 0; i < intValueBitsCount; ++i){
00094     pvIntValueMask += (1<<i);
00095     pvIntValueMaskEnc += (1<<i);
00096   }
00097   // Set factors + sign mask
00098   pvIntValueSigned = intValueSigned;
00099   pvScaleFromMinToMax = scaleFromMinToMax;
00100   if(pvScaleFromMinToMax){
00101     pvStepQ = conversionFactor * (max-min) / (pow(2.0, intValueBitsCount));
00102   }else{
00103     pvStepQ = conversionFactor;
00104   }
00105   pvIntValueSignMask = (1 << (intValueBitsCount-1) );
00106   pvStepAD = 1.0/pvStepQ;
00107 
00108   // Store min and max and signal
00109   pvMinimum = min;
00110   pvMaximum = max;
00111   emit(rangeChangedForUi(min, max));
00112   // Set value to the minimum
00113   pvNotifyUi = true;
00114   setValueFromDouble(min, false);
00115   pvValue.setDefaultValue(pvValue.valueDouble());
00116   pvValue.setDefaultValue(pvValue.valueInt());
00117   pvValue.clear();
00118 
00119   return true;
00120 }

00154 {
00155   mdtValue v;
00156 
00157   // If setValue() is called, valueChangedForUi() is emited
00158   //  and UI will receive the new value. But, because of this update,

00167 void mdtAnalogIo::setValueFromDouble(const mdtValue &value, bool emitValueChanged)
00168 {
00169   int m;
00170   double x = value.valueDouble();
00171 
00172   // Check if new value has changed
00173   if((fabs(x - pvValue.valueDouble()) <= pvStepQ)&&(value.isValid() == pvValue.isValid())){
00174     return;
00175   }
00176   // Check validity of new value and process if true
00177   if((value.isValid())&&(value.hasValueDouble())){
00178     // We must clip value, else we can have strongly wrong errors (underflow/overflow errors)
00179     if(x <= (pvMinimum + pvStepQ)){
00180       x = pvMinimum;
00181     }else if(x >= (pvMaximum - pvStepQ)){
00182       x = pvMaximum - pvStepQ;
00183     }
00184     // Make conversion
00185     if(pvIntValueSigned){
00186       m = pvStepAD * x;
00187     }else{
00188       if(pvScaleFromMinToMax){
00189         m = pvStepAD * (x - pvMinimum);
00190       }else{
00191         m = pvStepAD * x;
00192       }
00193     }
00194     // Apply C2 and negate if value < 0
00195     if(pvIntValueSigned && (x < 0.0)){
00196       m = -m;
00197       m = (~m) & pvIntValueMaskEnc; // C1
00198       ++m;  // C2
00199     }
00200     m &= pvIntValueMaskEnc;
00201     m = m << pvIntValueLsbIndexEnc;
00202     // Store value
00203     pvValue.setValue(m);
00204     pvValue.setValue(x, value.isMinusOl(), value.isPlusOl());
00205   }else{
00206     pvValue.clear();
00207   }
00208   // Check if new value has changed
00209   // Notify
00210   if(pvNotifyUi){
00211     emit(valueChangedForUi(pvValue));
00212   }
00213   if(emitValueChanged){
00214     emit(valueChanged(this));
00215     emit(valueChanged(pvValue));
00216   }
00217 }
