49 void mdtUsbtmcPortSetupDialog::rescan()
50 {
51  setStateScanning();
52  // Clear previous scan result
53  cbInterfaces->clear();
54  cbDevices->clear();
55  pvDeviceDescriptorsList.clear();
56  // Enumerate USBTMC devices
57  pvDeviceDescriptorsList = pvPort.scan();
58  if(pvDeviceDescriptorsList.isEmpty()){
59  pvLastError = pvPort.lastError();
60  displayLastError();
61  setStateNoDeviceFound();
62  return;
63  }
64  // Build comboboxes - We store pvDeviceDescriptorsList index as user data
65  int i = 0;
66  for(auto & deviceDescriptor : pvDeviceDescriptorsList){
67  cbDevices->addItem(deviceDisplayText(deviceDescriptor), i);
68  ++i;
69  }
70  // Update device info
71  if(pvSelectedDevice.isEmpty()){
72  setStateNoDeviceFound();
73  return;
74  }
75  // We have at least 1 device here
76  displayDeviceInformations();
77  setStateDeviceSelected();
78 }

121 void mdtUsbtmcPortSetupDialog::displayDeviceInformations()
122 {
123  mdtCodecScpi codec;
124  mdtScpiIdnResponse idnResponse;
125  QByteArray data;
126  // Clear previous resulsts
127  lbManufacturer->clear();
128  lbModel->clear();
129  lbSerial->clear();
130  lbFirmware->clear();
131  setStateScanning();
132  // Open device
133  if(pvSelectedDevice.isEmpty()){
134  setStateNoDeviceFound();
135  return;
136  }
137  if(!pvPort.openDevice(pvSelectedDevice, pvSelectedbInterfaceNumber, false)){
138  pvLastError = pvPort.lastError();
139  displayLastError();
140  setStateNoDeviceFound();
141  return;
142  }
143  // Get device iformation and display them
144  data = pvPort.sendQuery("*IDN?\n");
145  if(data.isEmpty()){
146  pvLastError = pvPort.lastError();
147  pvPort.close();
148  displayLastError();
149  setStateNoDeviceFound();
150  return;
151  }
152  idnResponse = codec.decodeIdn(data);
153  pvPort.close();
154  lbManufacturer->setText(idnResponse.manufacturer);
155  lbModel->setText(idnResponse.model);
156  lbSerial->setText(idnResponse.serial);
157  lbFirmware->setText(idnResponse.firmwareLevel);
158  setStateDeviceSelected();
159 }
