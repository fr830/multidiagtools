00030 bool mdtFrameCodecScpiU3606A::decodeConfigure(const QByteArray &data)
00031 {
00032   QString str;
00033   QVariant value;
00034 
00035   // Clear previous results
00036   pvValues.clear();
00037   pvMeasureType = MT_UNKNOW;
00038   pvRange.clear();
00039   pvResolution.clear();
00040 
00041   // Extract <function> and <parameters>
00042   if(!decodeFunctionParameters(data)){
00043     return false;
00044   }
00045   Q_ASSERT(pvValues.size() > 0);
00046   Q_ASSERT(pvValues.at(0).type() == QVariant::String);
00047   // Decode <function>
00048   pvNodes = pvValues.at(0).toString().split(':', QString::SkipEmptyParts);
00049   if(pvNodes.size() < 1){
00050     mdtError e(MDT_FRAME_DECODE_ERROR, "Function part contains no data", mdtError::Error);
00051     MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpiU3606A");
00052     e.commit();
00053     return false;
00054   }
00055   // Get "first level" type (voltage, current, resistance, ...) and set default sub type (DC, AC, ...)
00056   str = pvNodes.at(0);
00057   if(str.left(4) == "VOLT"){
00058     pvMeasureType = MT_VOLTAGE_DC;
00059   }else if(str.left(4) == "CURR"){
00060     pvMeasureType = MT_CURRENT_DC;
00061   }else if(str.left(3) == "RES"){
00062     pvMeasureType = MT_RESISTANCE;
00063   }else if(str.left(4) == "CONT"){
00064     pvMeasureType = MT_CONTINUITY;
00065   }else if(str.left(4) == "LRES"){
00066     pvMeasureType = MT_LRESISTANCE;
00067   }else if(str.left(3) == "CAP"){
00068     pvMeasureType = MT_CAPACITANCE;
00069   }else if(str.left(4) == "DIOD"){
00070     pvMeasureType = MT_DIODE;
00071   }else if(str.left(4) == "FREQ"){
00072     pvMeasureType = MT_FREQUENCY;
00073   }else if(str.left(4) == "PWID"){
00074     pvMeasureType = MT_PWIDTH;
00075   }else if(str.left(4) == "DCYC"){
00076     pvMeasureType = MT_DCYCLE;
00077   }else{
00078     pvMeasureType = MT_UNKNOW;
00079   }
00080   // Get optional sub types (AC)
00081   if(pvNodes.size() > 1){
00082     str = pvNodes.at(1);
00083     switch(pvMeasureType){
00084       case MT_VOLTAGE_DC:
00085         if(str.left(2) == "AC"){
00086           pvMeasureType = MT_VOLTAGE_AC;
00087         }
00088         break;
00089       case MT_CURRENT_DC:
00090         if(str.left(2) == "AC"){
00091           pvMeasureType = MT_CURRENT_AC;
00092         }
00093       default:
00094         // No other case to check
00095         break;
00096     }
00097   }
00098   // Get <parameters>
00099   if(pvValues.size() == 3){
00100     value = pvValues.at(1);
00101     if(value.convert(QVariant::Double)){
00102       pvRange = checkFloatingValueValidity(value.toDouble());
00103     }
00104     value = pvValues.at(2);
00105     if(value.convert(QVariant::Double)){
00106       pvResolution = checkFloatingValueValidity(value.toDouble());
00107     }
00108   }
00109 
00110   return true;
00111 }
