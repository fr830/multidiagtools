00093 {
00094   // Remove white spaces at beginning and end
00095   pvAsciiData = pvAsciiData.trimmed();
00096   // Remove EOF
00097   if(!removeEofSeq()){

00100   // Check that data are available

00102     mdtError e(MDT_FRAME_DECODE_ERROR, "Frame contains no data (after clean)", mdtError::Error);
00103     MDT_ERROR_SET_SRC(e, "mdtFrameCodecAscii");

00107

00111 bool mdtFrameCodecAscii::removeEofSeq()
00112 {
00113   // If EofSeq was not set, simply return
00114   if(pvEofSeq.size() < 1){
00115     return true;
00116   }
00117   // Check data length
00118   if(pvAsciiData.size() <= pvEofSeq.size()){
00119     mdtError e(MDT_FRAME_DECODE_ERROR, "Frame is smaller than End Of Frame Sequence" , mdtError::Error);
00120     MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpi");
00121     e.commit();
00122     return false;
00123   }
00124   // Remove the end of frame sequence
00125   pvAsciiData.remove(pvAsciiData.size()-pvEofSeq.size(), pvEofSeq.size());
00126 
00127   return true;
