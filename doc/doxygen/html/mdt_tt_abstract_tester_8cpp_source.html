177  // Create the new test
178  testId = pvTest->createTest(testModelId);

270  displayLastError();
271  return;
272  }
273  // Get part for TestItem_tbl
274  fields.clear();
275  fields << "Id_PK" << "Test_Id_FK" << "TestModelItem_Id_FK" << "Date" << "MeasuredValue" << "InstrumentRangeMin" << "InstrumentRangeMax"\
276  << "ResultValue" << "Result" << "Remark";
277  rowCount = pvTestItemViewController->rowCount(false);
278  for(row = 0; row < rowCount; ++row){
279  testItemRecords << pvTestItemViewController->record(row, fields, ok);
280  if(!ok){
281  pvLastError = pvTestItemViewController->lastError();
282  displayLastError();
283  return;
284  }
285  }
286  // Save test
287  if(!pvTest->saveTest(testRecord, testItemRecords)){
288  pvLastError = pvTest->lastError();
289  displayLastError();
290  return;
291  }
292  // Return to current test - will force controller to requery data from database (without that, we keep disrty indexes)
293  if(!pvTestViewController->setFilter("Id_PK", testRecord.value("Id_PK"))){
294  pvLastError = pvTestViewController->lastError();
295  return;
