141 bool mdtTtAbstractTestNodeCalibrationTool::saveTestNodeUnitCalibrationOffsets()
142 {
143  Q_ASSERT(pvTestNodeUnitTableController);
144 
145  mdtSqlRecord record;
146  int N;
147  int i;
148  bool ok;
149  QVariant tnuId;
150  QVariant var;
151 
152  // Setup record with fields to update
153  if(!record.addField("CalibrationOffset", "TestNodeUnit_tbl", pvDatabase)){
154  pvLastError = record.lastError();
155  return false;
156  }
157  if(!record.addField("CalibrationDate", "TestNodeUnit_tbl", pvDatabase)){
158  pvLastError = record.lastError();
159  return false;
160  }
161  // Beginn a transaction
162  if(!beginTransaction()){
163  return false;
164  }
165  // Save each value
166  N = pvTestNodeUnitTableController->rowCount(true);
167  for(i = 0; i < N; ++i){
168  record.clearValues();
169  // Get test node unit ID
170  tnuId = pvTestNodeUnitTableController->data(i, "Unit_Id_FK_PK", ok);
171  if(!ok){
172  pvLastError = pvTestNodeUnitTableController->lastError();
173  rollbackTransaction();
174  return false;
175  }
176  // Get offset value
177  var = pvTestNodeUnitTableController->data(i, "CalibrationOffset", ok);
178  if(!ok){
179  pvLastError = pvTestNodeUnitTableController->lastError();
180  rollbackTransaction();
181  return false;
182  }
183  record.setValue("CalibrationOffset", var);
184  // Get offset date
185  var = pvTestNodeUnitTableController->data(i, "CalibrationDate", ok);
186  if(!ok){
187  pvLastError = pvTestNodeUnitTableController->lastError();
188  rollbackTransaction();
189  return false;
190  }
191  record.setValue("CalibrationDate", var);
192  // Save to TestNodeUnit_tbl
193  if(!pvTest->updateRecord("TestNodeUnit_tbl", record, "Unit_Id_FK_PK", tnuId)){
194  pvLastError = pvTest->lastError();
195  rollbackTransaction();
196  return false;
197  }
198  }
199  // Commit trasaction
200  if(!commitTransaction()){
201  return false;
202  }
203 
204  return true;
205 }

262 
263  // Get double value
264  if(value.isMinusOl()){

267  x = std::numeric_limits<double>::max();
268  }else{

272  if(!mdtTtTest::isInRange(x, min, max)){
273  QString msg;

275  msg = tr("Value") + " " + QString::number(x) + " " + tr("is out of range.") + " ";
276  msg += tr("Allowed range is: ") + "[" + QString::number(min) + ";" + QString::number(max) + "]";
277  pvLastError.setInformativeText(msg);
