00281   QSqlTableModel *m;

00300     pvLastError.commit();

00304   // Choose delimiter regarding data type

00309       ;
00310   }
00311   // Setup and apply filter
00312   filter = fieldName + " = " + delimiter + matchData.toString() + delimiter;
