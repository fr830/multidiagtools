00069 void mdtDevice::setIos(mdtDeviceIos *ios, bool autoOutputUpdate)
00070 {
00071   Q_ASSERT(ios != 0);
00072 
00073   int i;
00074   QList<mdtAnalogIo*> analogOutputs;
00075   QList<mdtDigitalIo*> digitalOutputs;
00076 
00077   // Clear prevous I/Os
00078   if(pvIos != 0){
00079     analogOutputs = pvIos->analogOutputs();
00080     for(i=0; i<analogOutputs.size(); i++){
00081       Q_ASSERT(analogOutputs.at(i) != 0);
00082       disconnect(analogOutputs.at(i), SIGNAL(valueChanged(mdtAnalogIo*)), this, SLOT(setAnalogOutputValue(mdtAnalogIo*)));
00083     }
00084     digitalOutputs = pvIos->digitalOutputs();
00085     for(i=0; i<digitalOutputs.size(); i++){
00086       Q_ASSERT(digitalOutputs.at(i) != 0);
00087       disconnect(digitalOutputs.at(i), SIGNAL(valueChanged(mdtDigitalIo*)), this, SLOT(setDigitalOutputValue(mdtDigitalIo*)));
00088     }
00089     if(pvIos != ios){
00090       pvIos->deleteIos();
00091       delete pvIos;
00092     }
00093   }
00094   // Set new I/Os
00095   pvIos = ios;
00096   if(autoOutputUpdate){
00097     analogOutputs = pvIos->analogOutputs();
00098     for(i=0; i<analogOutputs.size(); i++){
00099       Q_ASSERT(analogOutputs.at(i) != 0);
00100       connect(analogOutputs.at(i), SIGNAL(valueChanged(mdtAnalogIo*)), this, SLOT(setAnalogOutputValue(mdtAnalogIo*)));
00101     }
00102     digitalOutputs = pvIos->digitalOutputs();
00103     for(i=0; i<digitalOutputs.size(); i++){
00104       Q_ASSERT(digitalOutputs.at(i) != 0);
00105       connect(digitalOutputs.at(i), SIGNAL(valueChanged(mdtDigitalIo*)), this, SLOT(setDigitalOutputValue(mdtDigitalIo*)));
00106     }
00107   }
