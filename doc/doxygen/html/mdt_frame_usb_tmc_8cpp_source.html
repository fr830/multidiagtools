00053 int mdtFrameUsbTmc::putData(const char *data, int maxLen)
00054 {
00055   Q_ASSERT(capacity() > 12);
00056 
00057   int frameSize = 0;
00058   int stored = 0;
00059 
00060   // If frame is allready full, return
00061   if(bytesToStore() < 1){
00062     return 0;
00063   }
00064 
00065   // Check wat's possible to store
00066   if(maxLen > remainCapacity()){
00067     maxLen = remainCapacity();
00068   }
00069   // store
00070   append(data, maxLen);
00071   stored = maxLen;
00072 
00073   // Check if we have enougth data to decode header
00074   if(size() >= 12){
00075     // Check if header was not allready decoded
00076     if(pvTransferSize == 0){
00077       // We have a header here, decode it
00078       pvMsgID = at(0);
00079       pvbTag = at(1);
00080       pvbTagInverse = at(2);
00081       // Check bTag
00082       if(pvbTagInverse != (quint8)(~pvbTag)){
00083         // Sync lost with device (or other error).
00084         // We simply abort here, because transfer size will be wrong.
00085         // Port manager will check bTag coherence and handle the error.
00086         pvEOFcondition = true;
00087         qDebug() << "mdtFrameUsbTmc::putData(): bTag/bTagInverse incoherent, bTag: " << pvbTag << " , expected inverse: " << (quint8)~pvbTag << " , rx inverse: " << pvbTagInverse;
00088         return stored;
00089       }
00090       pvTransferSize = at(4);
00091       pvTransferSize += (at(5) << 8);
00092       pvTransferSize += (at(6) << 16);
00093       pvTransferSize += (at(7) << 24);
00094       // MsgID specific part
00095       if(pvMsgID == DEV_DEP_MSG_IN){
00096         // EOM
00097         if(at(8) & 0x01){
00098           pvEOM = true;
00099         }else{
00100           pvEOM = false;
00101         }
00103       // Port manager will check received MsgID, and warn if unhadled one was received
00104       }
00110     }
00111     // Calc total frame size, without alignment bytes
00112     frameSize = pvTransferSize + 12;
00113     // Check if we have a complete frame
00114     if(size() >= frameSize){
00115       pvEOFcondition = true;
00116       pvMessageData += right(size()-12).left(pvTransferSize);
00117     }
00118   }
00119 
00120   return stored;
00121 }

00170 void mdtFrameUsbTmc::encode()
00171 {
00172   // Check minimal needed capacity
00173   if(capacity() < 16){
00174     reserve(16);
00175   }
00176   mdtFrame::clear();
00177   // Get message data size
00178   // Ajust command specific flags
00179   if(pvMsgID == DEV_DEP_MSG_OUT){
00180     pvTransferSize = pvMessageData.size();
00181   }else if(pvMsgID == DEV_DEP_MSG_IN){
00182     pvEOM = false;
00183     pvMessageData.clear();
00184   }else{
00185     pvTransferSize = 0;
00186   }
00187   // Build USBTMC header
00188   append((char)pvMsgID);
00189   append((char)pvbTag);
00190   append((char)pvbTagInverse);
00191   append((char)0x00);
00192   append((char)pvTransferSize & 0xFF);            // TransferSize, LLSB
00193   append((char)((pvTransferSize >> 8) & 0xFF));   // TransferSize, LSB
00194   append((char)((pvTransferSize >> 16) & 0xFF));  // TransferSize, LSB
00195   append((char)((pvTransferSize >> 24) & 0xFF));  // TransferSize, LSB
00196   if(pvEOM){
00197     append((char)0x01);                           // bmTransferAttributes
00198   }else{
00199     append((char)0x00);                           // bmTransferAttributes
00200   }
00201   append((char)0x00);                             // Reserved
00202   append((char)0x00);                             // Reserved
00203   append((char)0x00);                             // Reserved
00204   // Add data
00205   append(pvMessageData);
00206   // Add alignment bytes if requierd
00207   while((size() % 4) != 0){
00208     append((char)0x00);
00209   }
00210 }
