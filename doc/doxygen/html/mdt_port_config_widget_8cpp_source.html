00088 void mdtPortConfigWidget::updateConfig(mdtPortConfig &config)
00089 {
00090   int index;
00091 
00092   // Frame type
00093   index = cbFrameType->currentIndex();
00094   if(index >= 0){
00095     switch(cbFrameType->itemData(index).toInt()){
00096       case mdtFrame::FT_RAW:
00097         config.setFrameType(mdtFrame::FT_RAW);
00098         break;
00099       case mdtFrame::FT_RAW_TOP:
00100         config.setFrameType(mdtFrame::FT_RAW_TOP);
00101         break;
00102       case mdtFrame::FT_ASCII:
00103         config.setFrameType(mdtFrame::FT_ASCII);
00104         break;
00105       case mdtFrame::FT_MODBUS_TCP:
00106         config.setFrameType(mdtFrame::FT_MODBUS_TCP);
00107         break;
00108     }
00109   }
00110   // End of frame sequence - Only for ASCII frame type
00111   if(config.frameType() == mdtFrame::FT_ASCII){
00112     index = cbEndOfFrameSeq->currentIndex();
00113     if(index > -1){
00114       config.setEndOfFrameSeq(cbEndOfFrameSeq->itemData(index).toByteArray());
00115     }
00116   }else{
00117     config.setEndOfFrameSeq("");
00118   }
00119   // Read options
00120   config.setReadFrameSize(sbReadFrameSize->value());
00121   config.setReadQueueSize(sbReadQueueSize->value());
00122   config.setReadTimeout(sbReadTimeout->value());
00123   config.setUseReadTimeoutProtocol(cbUseReadTimeoutProtocol->isChecked());
00124   // Write options
00125   config.setWriteFrameSize(sbWriteFrameSize->value());
00126   config.setWriteQueueSize(sbWriteQueueSize->value());
00127   config.setWriteTimeout(sbWriteTimeout->value());
00128   config.setBytePerByteWrite(cbBytePerByteWrite->isChecked(), sbWriteInterbyteWaitTime->value());
00129   config.setWriteInterframeTime(sbWriteInterframeTime->value());
00130 
00131   // Display real applied setup
00132   displayConfig(config);
