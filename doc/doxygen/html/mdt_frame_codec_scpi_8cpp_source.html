00069 bool mdtFrameCodecScpi::decodeValues(const QByteArray &data, QString sep)
00070 {
00071   int i;
00072   QVariant value;
00073   double fltValue;
00074 
00075   // Clear previous results
00076   pvValues.clear();
00077   pvNodes.clear();
00078 
00079   // Store raw data in local QString constainer
00080   pvAsciiData = data;
00081 
00082   // Remove begin/end spaces and EOF
00083   if(!clean()){
00084     return false;
00085   }
00086 
00087   // Extract nodes
00088   pvNodes = pvAsciiData.split(sep);
00089   for(i=0; i<pvNodes.size(); i++){
00090     // We ignore empty nodes
00091     if(pvNodes.at(i).isEmpty()){
00092       continue;
00093     }
00094     // Convert node
00095     value = convertData(pvNodes.at(i));
00096     // Check limits if type is double
00097     if(value.type() == QVariant::Double){
00098       fltValue = value.toDouble();
00099       // Check about Nan
00100       if(qAbs(fltValue - pvNan) <= FLT_EPSILON){
00101         value.clear();
00102       }else{
00103         // Check about -OL and OL
00104         if((fltValue <= (pvNinfinity+FLT_EPSILON))||(fltValue >= (pvInfinity-FLT_EPSILON))){
00105           value.clear();
00106         }
00107       }
00108     }
00109     // Add converted value
00110     pvValues.append(value);
00111   }
00112 
00113   return true;
00114 }

00182 bool mdtFrameCodecScpi::decodeIEEEblock(QString data)
00183 {
00184   int i;
00185   bool ok = false;
00186   QString dataLen;
00187 
00188   // If string length is < 2: shure: not an IEEE block
00189   if(data.size() < 2){
00190     mdtError e(MDT_FRAME_DECODE_ERROR, "Frame is not a IEEE block" , mdtError::Warning);
00191     MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpi");
00192     e.commit();
00193     return false;
00194   }
00195   // If first char is '#' , simply remove it
00196   if(data.at(0) == QChar('#')){
00197     data.remove(0, 1);
00198   }
00199   // We must have a number that tells how many info digits that follows
00200   i = QString(data.at(0)).toInt(&ok); // Note: we need a QString, not a QChar.
00201   if(!ok){
00202     mdtError e(MDT_FRAME_DECODE_ERROR, "Frame contains no block header length" , mdtError::Warning);
00203     MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpi");
00204     e.commit();
00205     return false;
00206   }
00207   data.remove(0, 1);
00208   // Check that we have enouth chars to decode data length
00209   if(data.size() < i){
00210     mdtError e(MDT_FRAME_DECODE_ERROR, "Frame contains no block header with data length" , mdtError::Warning);
00211     MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpi");
00212     e.commit();
00213     return false;
00214   }
00215   // Get the data length
00216   dataLen = data.left(i);
00217   data.remove(0, i);
00218   i = dataLen.toInt(&ok);
00219   if(!ok){
00220     mdtError e(MDT_FRAME_DECODE_ERROR, "Unable to decode data length in block header" , mdtError::Warning);
00221     MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpi");
00222     e.commit();
00223     return false;
00224   }
00225   // Ok, let's decode the data
00226   return decodeIEEEdata(data.toAscii());
00227 }
