00060 QList<mdtPortInfo*> mdtUsbPortManager::scan(int bDeviceClass, int bDeviceSubClass, int bDeviceProtocol)
00061 {
00062   Q_ASSERT(!isRunning());
00063 
00064   QList<mdtPortInfo*> portInfoList;
00065   mdtDeviceInfo *deviceInfo;
00066   mdtPortInfo *portInfo;
00067   libusb_context *ctx;
00068   libusb_device **devicesList;
00069   mdtUsbDeviceDescriptor deviceDescriptor;
00070   mdtUsbConfigDescriptor *configDescriptor;
00071   mdtUsbInterfaceDescriptor *ifaceDescriptor;
00072   QString portName;
00073   libusb_device *device;
00074   ssize_t devicesCount;
00075   ssize_t i;
00076   int j, k;
00077   int err;
00078   QString str;
00079 
00080   // Init libusb
00081   err = libusb_init(&ctx);
00082   if(err != 0){
00083     mdtError e(MDT_USB_IO_ERROR, "Cannot init libusb", mdtError::Error);
00084     MDT_ERROR_SET_SRC(e, "mdtUsbPortManager");
00085     e.commit();
00086     return portInfoList;
00087   }
00088   // Get devices list
00089   devicesCount = libusb_get_device_list(ctx, &devicesList);
00090   if(devicesCount < 0){
00091     mdtError e(MDT_USB_IO_ERROR, "Cannot list devices", mdtError::Error);
00092     MDT_ERROR_SET_SRC(e, "mdtUsbPortManager");
00093     e.commit();
00094     return portInfoList;
00095   }
00096   // Scan ...
00097   for(i=0; i<devicesCount; i++){
00098     device = devicesList[i];
00099     err = deviceDescriptor.fetchAttributes(device, false);
00100     if(err != 0){
00101       mdtError e(MDT_USB_IO_ERROR, "Device attributes fetch failed", mdtError::Error);
00102       MDT_ERROR_SET_SRC(e, "mdtUsbPortManager");
00103       e.commit();
00104       continue;
00105     }
00106     // Create a port info object
00107     portInfo = new mdtPortInfo;
00108     // Search interfaces in various configurations
00109     for(j=0; j<deviceDescriptor.configurations().size(); j++){
00110       configDescriptor = deviceDescriptor.configurations().at(j);
00111       Q_ASSERT(configDescriptor != 0);
00112       for(k=0; k<configDescriptor->interfaces().size(); k++){
00113         ifaceDescriptor = configDescriptor->interfaces().at(k);
00114         Q_ASSERT(ifaceDescriptor != 0);
00115         // See if a specific bDeviceClass was requested
00116         if(bDeviceClass >= 0){
00117           if(ifaceDescriptor->bInterfaceClass() != (quint8)bDeviceClass){
00118             continue;
00119           }
00120         }
00121         // See if a specific bDeviceSubClass was requested
00122         if(bDeviceSubClass >= 0){
00123           if(ifaceDescriptor->bInterfaceSubClass() != (quint8)bDeviceSubClass){
00124             continue;
00125           }
00126         }
00127         // See if a specific bDeviceProtocol was requested
00128         if(bDeviceProtocol >= 0){
00129           if(ifaceDescriptor->bInterfaceProtocol() != (quint8)bDeviceProtocol){
00130             continue;
00131           }
00132         }
00133         // Interresting interface
00134         deviceInfo = new mdtDeviceInfo;
00135         // Build the display text
00136         str = deviceDescriptor.vendorName();
00137         str += ", ";
00138         str += deviceDescriptor.productName();
00139         if(!deviceDescriptor.serialNumber().isEmpty()){
00140           str += " (SN: " + deviceDescriptor.serialNumber() + ")";
00141         }
00142         portInfo->setDisplayText(str);
00143         // Set VID, PID, SID and protocol ID
00144         deviceInfo->setVendorId(deviceDescriptor.idVendor());
00145         deviceInfo->setProductId(deviceDescriptor.idProduct());
00146         deviceInfo->setSerialId(deviceDescriptor.serialNumber());
00147         deviceInfo->setProtocolId(ifaceDescriptor->bInterfaceProtocol());
00148         // Add interface to list
00149         portInfo->addDevice(deviceInfo);
00150       }
00151     }
00152     // If port info has no device, delete it, else add it to list
00153     if(portInfo->deviceInfoList().size() < 1){
00154       delete portInfo;
00155     }else{
00156       // Build port name
00157       deviceInfo = portInfo->deviceInfoList().at(0);
00158       Q_ASSERT(deviceInfo != 0);
00159       portName = "0x";
00160       portName += QString::number((unsigned int)deviceInfo->vendorId(), 16);
00161       portName += ":0x";
00162       portName += QString::number((unsigned int)deviceInfo->productId(), 16);
00163       if(!deviceDescriptor.serialNumber().isEmpty()){
00164         portName += ":" + deviceDescriptor.serialNumber();
00165       }
00166       portInfo->setPortName(portName);
00167       // Add port info to list
00168       portInfoList.append(portInfo);
00169     }
00170   }
00171   // Release ressources
00172   libusb_free_device_list(devicesList, 1);
00173   libusb_exit(ctx);
00174 
00175   return portInfoList;
00176 }
