00045 int mdtFrameModbusTcp::putData(const char *data, int maxLen)
00046 {
00047   Q_ASSERT(capacity() >= 7);
00048 
00049   int frameSize = 0;
00050   int stored = 0;
00051 
00052   // If frame is allready full, return
00053   if(bytesToStore() < 1){
00054     return 0;
00055   }
00056 
00057   // Check if it's possible to get size of incomming frame
00058   if((size()+maxLen) >= 6){
00059     // Check if some data must be stored before we can get the frame size
00060     if(size() < 6){
00061       stored = 6-size();
00062       append(data, stored);
00063       Q_ASSERT((data + stored) <= (data + maxLen));
00064       data += stored;
00065       maxLen -= stored;
00066     }
00067     frameSize = (at(4) << 8) + at(5);
00068     // We have MODBUS frame size , = Unit ID + PDU. Add missing size
00069     frameSize += 6;
00070     // Check if it is possible to store this frame
00071     if(frameSize > capacity()){
00072       // Problem here: fill the frame with null values, so bytesToStore() will return 0 next time
00073       fill(0, capacity());
00074       stored = capacity();
00075       return stored;
00076     }
00077   }
00078   // Check wat's possible to store
00080   if(maxLen > remainCapacity()){
00081     maxLen = remainCapacity();
00082   }
00084   // Store
00085   append(data, maxLen);
00086   stored += maxLen;
00088   // Check if frame is complete
00089   if(size() == frameSize){
00090     pvEOFcondition = true;
00091     // Store members
00092     pvTransactionId = (at(0) << 8) + at(1);
00093     pvProtocolId = (at(2) << 8) + at(3);
00094     pvUnitId = at(6);
00095     // Store PDU if exists
00096     if(size() > 7){
00097       pvPdu = right(size()-7);
00098     }
00099   }
00100 
00101   return stored;
00102 }

00142 void mdtFrameModbusTcp::encode()
00143 {
00144   Q_ASSERT(capacity() >= 7);
00145 
00146   quint16 length = pvPdu.size() + 1;
00147 
00149   // Transaction ID
00150   append(pvTransactionId >> 8);
00151   append(pvTransactionId & 0x00FF);
00152   // Protocol ID
00153   append(pvProtocolId >> 8);
00154   append(pvProtocolId & 0x00FF);
00155   // Length
00156   append(length >> 8);
00157   append(length & 0x00FF);
00158   // Unit ID
00159   append(pvUnitId);
00160   // PDU
00161   append(pvPdu);
