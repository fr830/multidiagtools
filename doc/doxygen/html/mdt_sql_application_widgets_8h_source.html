226 template <typename T>
227 bool mdtSqlApplicationWidgets<T>::closeOpenSqlWindows()
228 {
229  mdtSqlForm *form;
230 
231  // Close all editors handled by a generic SQL window
232  for(auto & window : pvOpenEditorWidows){
233  Q_ASSERT(window);
234  form = window->sqlForm();
235  if(form != 0){
236  if(!form->allDataAreSaved()){
237  return false;
238  }
239  }
240  window->close();
241  }
242 
243  return true;
244 }
