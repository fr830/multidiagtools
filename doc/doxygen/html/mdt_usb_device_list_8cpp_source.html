76 mdtUsbDeviceDescriptor mdtUsbDeviceList::findDevice ( uint16_t idVendor, uint16_t idProduct, const QString& serialNumber )
77 {
78  // Be shure we have at least one device in our list
79  if(pvDeviceDescriptors.isEmpty()){
80  pvLastError.setError(QObject::tr("Failed to find any device because descriptors list is empty (did you scan ?)"), mdtError::Error);
81  MDT_ERROR_SET_SRC(pvLastError, "mdtUsbDeviceList");
82  pvLastError.commit();
83  return mdtUsbDeviceDescriptor();
84  }
85  // Search requested device
86  auto foundIt = pvDeviceDescriptors.end();
87  if(serialNumber.isEmpty()){
88  // Search by idVendor and idProduct
89  deviceMatchDataVidPid md;
90  md.idProduct = idProduct;
91  md.idVendor = idVendor;
92  foundIt = std::find_if(pvDeviceDescriptors.begin(), pvDeviceDescriptors.end(), md);
93  }else{
94  // Search by idVendor, idProduct and SN
95  deviceMatchDataVidPidSn md;
96  md.idProduct = idProduct;
97  md.idVendor = idVendor;
98  md.serialNumber = serialNumber;
99  foundIt = std::find_if(pvDeviceDescriptors.begin(), pvDeviceDescriptors.end(), md);
100  }
101  if(foundIt == pvDeviceDescriptors.end()){
102  // Device not found
103  clear();
104  QString str;
105  str = "ID ";
106  str += QString("0x%1:0x%2 ").arg(idVendor, 4, 16, QChar('0')).arg(idProduct, 4, 16, QChar('0'));
107  str += " (SN:" + serialNumber + ")";
108  pvLastError.setError(QObject::tr("Could not find device ") + str, mdtError::Error);
109  MDT_ERROR_SET_SRC(pvLastError, "mdtUsbDeviceList");
110  pvLastError.commit();
111  return mdtUsbDeviceDescriptor();
112  }
113 
114  return *foundIt;
115 }

192 bool mdtUsbDeviceList::buildDevicesList(bool fetchDeviceActiveConfigOnly)
193 {
194  ssize_t devCount;
195  ssize_t i;
196 
197  // Let libusb build his USB devices list
198  devCount = libusb_get_device_list(pvUsbContext, &pvLibusDeviceList);
199  if(devCount < 0){
200  pvLastError.setError(QObject::tr("Failed to get devices list."), mdtError::Error);
201  pvLastError.setSystemError(devCount, libusb_error_name(devCount));
202  MDT_ERROR_SET_SRC(pvLastError, "mdtUsbDeviceList");
203  pvLastError.commit();
204  return false;
205  }
206  // Build descriptors for each device
207  for(i = 0; i < devCount; ++i){
208  Q_ASSERT(pvLibusDeviceList[i] != 0);
209  mdtUsbDeviceDescriptor device;
210  if(!device.fetchAttributes(pvLibusDeviceList[i], fetchDeviceActiveConfigOnly)){
211  pvLastError = device.lastError();
212  return false;
213  }
214  pvDeviceDescriptors.append(device);
215  }
216 
217  return true;
218 }
