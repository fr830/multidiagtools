00129 bool mdtFrameCodecScpiU3606A::decodeNodeValues(QString data)
00130 {
00131   QStringList nodeValues;

00133   // Case of VOLT
00134   if(data.left(4) == "VOLT"){

00143     data = data.remove(0, 2);
00144   // Case of AC
00145   }else if(data.left(2) == "AC"){
00146     // Select the correct measure type
00147     switch(pvMeasureType){
00148       case MT_VOLTAGE_DC:
00149         pvMeasureType = MT_VOLTAGE_AC;
00150         break;
00151       case MT_CURRENT_DC:
00152         pvMeasureType = MT_CURRENT_AC;
00153         break;
00154       default:
00155         // Something goes wrong
00156         pvMeasureType = MT_UNKNOW;
00157         mdtError e(MDT_FRAME_DECODE_ERROR, "Unknow measure type" , mdtError::Warning);
00158         MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpiU3606A");
00159         e.commit();
00160         return false;
00161     }
00162   // Unknow measure type - Simply add to values
00163   }else{
00164     pvValues << data;
00165     return true;
00166   }
00167 
00168   // Continue parsing, it's possible to find range and resolution
00169   data = data.remove(0, 2);
00170   data = data.trimmed();
00171   // case of no range
00172   if(data.size() < 1){
00173     return true;
00174   }
00175   nodeValues = data.split(",");
00176   // We must have a couple range,resolution
00177   if(nodeValues.size() != 2){
00178     pvMeasureType = MT_UNKNOW;
00179     mdtError e(MDT_FRAME_DECODE_ERROR, "Range,resolution values not found" , mdtError::Warning);
00180     MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpiU3606A");
00181     e.commit();
00182     return false;
00183   }
00184   pvRange = convertToDouble(nodeValues.at(0));
00185   pvResolution = convertToDouble(nodeValues.at(1));
00186
