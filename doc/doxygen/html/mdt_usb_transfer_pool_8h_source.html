289  template<typename TransferHandlerType, typename A>
290  bool allocTransfers(unsigned int n, const A & allocator)
291  {
292  Q_ASSERT(n <= pvMaxTransfers);
293 
294  int i;
295  T *transfer;
296 
297  clear();
298  pvDeviceHandle = allocator.deviceHandle;
299  pvTransfers.reserve(n);
300  for(i = 0; i < n; ++i){
301  transfer = allocator.allocateNewTransfer();
302  if(transfer == 0){
303  clear();
304  return false;
305  }
306  pvTransfers.push_back(transfer);
307  }
308  pvAllocatedTransfers = pvTransfers.size();
309 
310  return true;
311  }

322  // Allocate a new transfer if required and permitted
323  if(pvTransfers.empty()){
324  Q_ASSERT(pvAllocatedTransfers < pvMaxTransfers);
325  if(pvAllocatedTransfers < pvMaxTransfers){
326  ++pvAllocatedTransfers;
327  transfer = allocator.allocateNewTransfer();
328  pvPendingTransfers.push_back(transfer);
329  return transfer;

336  Q_ASSERT(transfer != 0);
337  pvTransfers.pop_back();
338  pvPendingTransfers.push_back(transfer);
339 
340  return transfer;
341  }
