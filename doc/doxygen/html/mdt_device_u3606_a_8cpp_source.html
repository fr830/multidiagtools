260 mdtFrameCodecScpiU3606A::measure_type_t mdtDeviceU3606A::getMeasureConfiguration()
261 {
262  int bTag;
263  mdtPortTransaction *transaction;
264  mdtUsbtmcPortManager *upm;
265 
266  // We need a specific port manager
267  upm = dynamic_cast<mdtUsbtmcPortManager*>(portManager());
268  Q_ASSERT(upm != 0);
269  // Send query
270  bTag = upm->sendData("CONF?");
271  if(bTag < 0){
272  return mdtFrameCodecScpiU3606A::MT_UNKNOW;
273  }
274  // Setup transaction
275  transaction = getNewTransaction();
276  transaction->setType(mdtFrameCodecScpi::QT_CONF);
277  transaction->setQueryReplyMode(true);
278  // Send read request - transaction is restored by sendReadRequest() on failure
279  bTag = upm->sendReadRequest(transaction);
280  if(bTag < 0){
281  return mdtFrameCodecScpiU3606A::MT_UNKNOW;
282  }
283  // Wait on response - transaction are restored automatically during waitTransactionDone() on failure
284  if(!waitTransactionDone(bTag)){
285  return mdtFrameCodecScpiU3606A::MT_UNKNOW;
286  }
287 
288  return pvCodec->measureType();

360 void mdtDeviceU3606A::decodeReadenFrame(mdtPortTransaction *transaction)
361 {
362  Q_ASSERT(transaction != 0);
363 
364  bool ok;
365 
367 
368  switch(transaction->type()){
369  case mdtFrameCodecScpi::QT_VALUE:
370  if(transaction->analogIo() == 0){
371  lastErrorW().setError(tr("Device ") + name() + tr(": received value response from device but no I/O was affected to query"), mdtError::Warning);
372  MDT_ERROR_SET_SRC(lastErrorW(), "mdtDeviceU3606A");
373  lastErrorW().commit();
374  }else{
375  ok = pvCodec->decodeValues(transaction->data());
376  if(ok && (pvCodec->values().size() == 1)){
377  transaction->analogIo()->setValue(pvCodec->values().at(0).value<mdtValue>(), false);
378  }else{
379  transaction->analogIo()->setValue(mdtValue(), false);
380  }
381  }
382  break;
383  case mdtFrameCodecScpi::QT_SINGLE_VALUE_FLT:
384  if(transaction->analogIo() == 0){
385  lastErrorW().setError(tr("Device ") + name() + tr(": received value response from device but no I/O was affected to query"), mdtError::Warning);
386  MDT_ERROR_SET_SRC(lastErrorW(), "mdtDeviceU3606A");
387  lastErrorW().commit();
388  }else{
389  qDebug() << "decodeReadenFrame() - data: " << transaction->data();
390  transaction->analogIo()->setValue(pvCodec->decodeSingleValueDouble(transaction->data()), false);
391  qDebug() << "decodeReadenFrame() - decoded data: " << pvCodec->decodeSingleValueDouble(transaction->data());
392  qDebug() << "decodeReadenFrame() - decoded data: " << transaction->analogIo()->value().valueDouble();
393  }
394  break;
395  case mdtFrameCodecScpi::QT_ERR:
396  ok = pvCodec->decodeError(transaction->data());
397  if(!ok){
398  lastErrorW().setError(tr("Device ") + name() + tr(": SYST:ERR? reply could not be decoded"), mdtError::Warning);
399  MDT_ERROR_SET_SRC(lastErrorW(), "mdtDeviceU3606A");
400  lastErrorW().commit();
401  break;
402  }
403  handleDeviceError(pvCodec->values());
404  break;
405  case mdtFrameCodecScpi::QT_CONF:
406  if(!pvCodec->decodeConfigure(transaction->data())){
407  lastErrorW().setError(tr("Device ") + name() + tr(": CONF? reply could not be decoded"), mdtError::Warning);
408  MDT_ERROR_SET_SRC(lastErrorW(), "mdtDeviceU3606A");
409  lastErrorW().commit();
410  break;
411  }
412  updateMeasureSetup();
413  break;
414  default:
415  lastErrorW().setError(tr("Device ") + name() + tr(": received unsupported response from device (type: ") + QString::number(transaction->type()) + tr(")"), mdtError::Warning);
416  MDT_ERROR_SET_SRC(lastErrorW(), "mdtDeviceU3606A");
417  lastErrorW().commit();
418  }
419 }

421 int mdtDeviceU3606A::readAnalogInput(mdtPortTransaction *transaction)
422 {
423  Q_ASSERT(ios() != 0);
424  Q_ASSERT(transaction != 0);
425 
426  int bTag;
427  mdtAnalogIo *aio;
428 
429  aio = transaction->analogIo();
430  Q_ASSERT(aio != 0);
431  switch(aio->addressRead()){
432  case 0: // Measure
433  transaction->setData("READ?");
434  break;
435  case 1: // Sense U (voltage)
436  transaction->setData("SENS:VOLT?");
437  break;
438  case 2: // Sense I (current)
439  transaction->setData("SENS:CURR?");
440  break;
441  default:
442  lastErrorW().setError(tr("Requested to read a unknown analog input. Address: ") + QString::number(aio->addressRead()), mdtError::Error);
443  MDT_ERROR_SET_SRC(lastErrorW(), "mdtDeviceU3606A");
444  lastErrorW().commit();
445  return mdtAbstractPort::UnhandledError;
446  }
447  // Send query
448  bTag = pvUsbtmcPortManager->sendData(transaction);
449  if(bTag < 0){
450  return bTag;
451  }
452  // Remember query type.
453  transaction->setType(mdtFrameCodecScpi::QT_SINGLE_VALUE_FLT);
454  // Send read request
455  bTag = pvUsbtmcPortManager->sendReadRequest(transaction);
456 
457  return bTag;
458 }
