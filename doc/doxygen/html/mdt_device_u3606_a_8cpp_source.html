00069   qDebug() << "mdtDeviceU3606A::decodeReadenFrame() , ID: " << transaction.id();
00070   
00071   switch(transaction.type()){
00072     case MDT_FC_SCPI_VALUE:
00073       if(transaction.analogIo() == 0){
00074         mdtError e(MDT_DEVICE_ERROR, "Device " + name() + ": received value response from device but no I/O was affected to query", mdtError::Warning);
00075         MDT_ERROR_SET_SRC(e, "mdtDeviceU3606A");
00076         e.commit();

00078         ok = pvCodec->decodeValues(transaction.data());
00079         if(ok && (pvCodec->values().size() == 1)){
00080           transaction.analogIo()->setValue(pvCodec->values().at(0), false);
00081         }else{
00082           transaction.analogIo()->setValue(QVariant(), false);
00083         }
00084       }
00085       break;
00086     default:
00087       mdtError e(MDT_DEVICE_ERROR, "Device " + name() + ": received unsupported response from device (type: " + QString::number(transaction.type()) + ")", mdtError::Error);
00088       MDT_ERROR_SET_SRC(e, "mdtDeviceU3606A");
