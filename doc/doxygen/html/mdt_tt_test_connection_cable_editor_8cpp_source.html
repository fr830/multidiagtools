179 void mdtTtTestConnectionCableEditor::generateLinks()
180 {
181  mdtTtTestConnectionCable tcc(this, database());
182  mdtTtTestNode tn(this, database());
183  QVariant cableId;
184  QVariant dutVehicleId;
185  QVariant dutUnitId;
186  QVariant testNodeId;
187  QVariant dutStartConnectorId;
188  QList<QVariant> dutStartConnectionIdList;
189  QList<QVariant> dutLinkedConnectorIdList;
190  QList<QVariant> dutEndConnectorIdList;
191  QList<QVariant> dutEndConnectionIdList;
192  QList<QVariant> busAtestConnectionIdList;
193  QList<QVariant> busBtestConnectionIdList;
194  QList<QVariant> idList;
195  int i, k;
196 
197  // Get cable ID
198  cableId = currentData("TestCable_tbl", "Id_PK");
199  if(cableId.isNull()){
200  return;
201  }
202  // Let user choose DUT unit and test node
203  dutVehicleId = selectDutVehicleId();
204  if(dutVehicleId.isNull()){
205  return;
206  }
207  dutUnitId = selectDutUnitId(dutVehicleId);
208  if(dutUnitId.isNull()){
209  return;
210  }
211  testNodeId = selectTestNode();
212  if(testNodeId.isNull()){
213  return;
214  }
215  // Get test node's BUS channel test connections
216  busAtestConnectionIdList = tn.getChannelTestConnectionIdList(testNodeId, "BUSA");
217  if(busAtestConnectionIdList.isEmpty()){
218  qDebug() << "There is no channel test connection available on BUSA";
219  return;
220  }
221  busBtestConnectionIdList = tn.getChannelTestConnectionIdList(testNodeId, "BUSB");
222  if(busBtestConnectionIdList.isEmpty()){
223  qDebug() << "There is no channel test connection available on BUSB";
224  return;
225  }
226  qDebug() << "BUSA cnn: " << busAtestConnectionIdList;
227  qDebug() << "BUSB cnn: " << busBtestConnectionIdList;
228  // Select a start connector
229  dutStartConnectorId = selectStartConnectorId(dutUnitId);
230  if(dutStartConnectorId.isNull()){
231  return;
232  }
233  qDebug() << "Selected DUT: " << dutUnitId << " , Test node: " << testNodeId;
234  // Load link list
235  if(!tcc.loadLinkList()){
236  pvLastError = tcc.lastError();
237  return;
238  }
239  // Get start unit connections for selected connector
240  dutStartConnectionIdList = tcc.getToUnitConnectorRelatedUnitConnectionIdList(dutStartConnectorId);
241  qDebug() << "Start connections: " << dutStartConnectionIdList;
242  for(i = 0; i < dutStartConnectionIdList.size(); ++i){
243  // Get list of unit connectors that is linked to selected start unit connection
244  idList = tcc.getToUnitConnectionLinkedUnitConnectorIdList(dutStartConnectionIdList.at(i));
245  for(k = 0; k < idList.size(); ++k){
246  if(!dutLinkedConnectorIdList.contains(idList.at(k))){
247  dutLinkedConnectorIdList.append(idList.at(k));
248  }
249  }
250  }
251  // Select end connector(s)
252  dutEndConnectorIdList = selectEndConnectorIdList(dutLinkedConnectorIdList);
253  qDebug() << "End connectors: " << dutEndConnectorIdList;
254  // Get end connections that are part of selected end connectors
255  dutEndConnectionIdList = tcc.getToUnitConnectionIdListLinkedUnitConnectionIdListPartOfUnitConnectorList(dutStartConnectionIdList, dutEndConnectorIdList);
256  qDebug() << "Start connections: " << dutStartConnectionIdList;
257  qDebug() << "End connections: " << dutEndConnectionIdList;
258  // Check that test node has needed amount of busses lines
259  if(busAtestConnectionIdList.size() < dutStartConnectionIdList.size()){
260  qDebug() << "Test node has to less BUSA test connections";
261  return;
262  }
263  if(busBtestConnectionIdList.size() < dutEndConnectionIdList.size()){
264  qDebug() << "Test node has to less BUSB test connections";
265  return;
266  }
267  // Add links to link table
273  if(!tcc.addLinks(testNodeId, cableId, busAtestConnectionIdList, dutStartConnectionIdList)){
274  pvLastError = tcc.lastError();
275  return;
276  }
277  if(!tcc.addLinks(testNodeId, cableId, busBtestConnectionIdList, dutEndConnectionIdList)){
278  pvLastError = tcc.lastError();
279  return;
280  }
281  // Update GUI
282  select("TestLink_view");
283 }
