98  {

139  double binaryToDouble(T m)
140  {
141  double x;
142 
143  // Extract bits in correct range
144  m = m >> pvValueLsbIndex;
145  m &= pvValueMask;
146  // Apply C2 and negate if sign bit is present
147  if((pvValueIsSigned)&&(m & pvSignMask)){
148  m = ( (~m) & pvValueMask ); // C1
149  ++m; // C2
150  m = -m;
151  }
152  // Make conversion
153  if(pvValueIsSigned){
154  x = pvQuantizationStep*(double)m;
155  }else{
156  if(pvScaleFromMinToMax){
157  x = pvQuantizationStep*(double)m + pvMinimum;
158  }else{
159  x = pvQuantizationStep*(double)m;
160  }
161  }
162 
163  return x;

168  T doubleToBinary(double x)
169  {
170  T m;
171 
172  // We must clip value, else we can have strongly wrong errors (underflow/overflow errors)
173  if(x < (pvMinimum + pvQuantizationStep)){
174  x = pvMinimum;
175  }else if(x > (pvMaximum - pvQuantizationStep)){
176  x = pvMaximum - pvQuantizationStep;
177  }
178  // Make conversion
179  if(pvValueIsSigned){
180  m = pvInvQuantizationStep * x;
181  }else{
182  if(pvScaleFromMinToMax){
183  m = pvInvQuantizationStep * (x - pvMinimum);
184  }else{
185  m = pvInvQuantizationStep * x;
186  }
187  }
188  // Apply C2 and negate if value < 0
189  if(pvValueIsSigned && (x < 0.0)){
190  m = -m;
191  m = (~m) & pvValueMask; // C1
192  ++m; // C2
193  }
194  m &= pvValueMask;
195  m = m << pvValueLsbIndex;
196
