117 bool mdtCsvFile::readLine(QByteArray &line, const QString &dataProtection, const QString &comment, const QChar &escapeChar, QByteArray eol)
118 {
119  QByteArray buffer;
120  bool eolFound = false;
121  int eolCursor;
122  int dpCursor;
123 
124  buffer = pvReadLineBuffer;
125  // Read until EOL found or EOF
126  while(!eolFound){
127  eolCursor = buffer.indexOf(eol);
128  if(eolCursor < 0){
129  if(atEnd()){
130  line.append(buffer);
131  buffer.clear();
132  // Now check that line is not commented
133  if(indexOfCommentedLineBeginning(line, comment) > -1){
134  line.clear();
135  }
136  break;
137  }
138  buffer.append(read(pvReadLineBufferSize));
139  }else{
140  // A EOL was found, check if it is in a data protection or not
141  line.append(buffer.left(eolCursor));
142  buffer.remove(0, eolCursor);
143  if(dataProtectionSectionBegins(line, dataProtection, escapeChar)){
144  // Search until we find a closing data protrection to confirm that we found a opening one
145  readUntilDataProtection(buffer, dpCursor, dataProtection, escapeChar);
146  if(dpCursor < 0){
147  mdtError e(MDT_FILE_IO_ERROR, "In file " + fileName() + " : format error (missing a closing " + dataProtection + ")", mdtError::Error);
148  MDT_ERROR_SET_SRC(e, "mdtCsvFile");
149  e.commit();
150  return false;
151  }else{
152  line.append(buffer.left(dpCursor+dataProtection.size()));
153  buffer.remove(0, dpCursor+dataProtection.size());
154  }
155  }else{
156  // Here we found a EOL in a unprotected section.
157  buffer.remove(0, eol.size());
158  // Now check that line is not commented
159  if(indexOfCommentedLineBeginning(line, comment) > -1){
160  line.clear();
161  }else{
162  eolFound = true;
163  }
164  }
165  }
166  }
167  pvReadLineBuffer = buffer;
168 
169  return true;
170 }

249 bool mdtCsvFile::dataProtectionSectionBegins(const QByteArray &line, const QString &dataProtection, const QChar &escapeChar)
250 {
251  bool escapeEqualProtection = QString(escapeChar) == dataProtection;
252  QByteArray escapedDataProtection;
253  int dpCount;
254  int esDpCount;
255 
256  // If no data protection is given, we are never in a protected section
257  if(dataProtection.isEmpty()){
258  return false;
259  }
260  // Build the escaped data protection version
261  escapedDataProtection.append(escapeChar.toAscii());
262  escapedDataProtection += dataProtection.toAscii();
263  // Count data protection occurences
264  dpCount = line.count(dataProtection.toAscii());
265  // Count escaped data protection occurences
266  if(escapeEqualProtection){
267  esDpCount = 0;
268  }else{
269  esDpCount = line.count(escapedDataProtection);
270  }
271 
272  return ((dpCount-esDpCount)%2);
273 }

275 void mdtCsvFile::readUntilDataProtection(QByteArray &buffer, int &dpIndex, const QString &dataProtection, const QChar &escapeChar)
276 {
277  bool dpFound = false;
278  bool escapeEqualProtection = QString(escapeChar) == dataProtection;
279  char es = escapeChar.toAscii();
280  int start = 0;
281 
282  while((!dpFound)/*&&(!atEnd())*/){
283  // Search data protection
284  dpIndex = buffer.indexOf(dataProtection, start);
285  if(escapeEqualProtection){
286  if(dpIndex > -1){
287  // We need enougth elements to check about escape
288  if(buffer.size() > (dpIndex+1)){
289  // Check that data protection is not escaped
290  if(buffer.at(dpIndex+1) == es){
291  // Escaped data protection
292  start = dpIndex+2;
293  dpIndex = -1;
294  }
295  }else{
296  // Not enougth data, read more
297  dpIndex = -1;
298  }
299  }
300  }
301  if(dpIndex > -1){
302  dpFound = true;
303  }else{
304  if(atEnd()){
305  buffer.append(pvReadLineBuffer);
306  pvReadLineBuffer.clear();
307  dpIndex = buffer.indexOf(dataProtection);
308  break;
309  }else{
310  buffer.append(read(pvReadLineBufferSize));
311  }
312  }
313  }
