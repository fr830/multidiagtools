565 bool mdtTtTestNodeEditor::setupTestNodeTable()
566 {
567  Ui::mdtTtTestNodeEditor tne;
568  QSqlTableModel *testNodeModel;
569  QDataWidgetMapper *baseVehicleTypeMapper;
570  QSqlTableModel *baseVehicleTypeModel;
571  mdtSqlRelation *baseVehicleTypeRelation;
572 
573  // Setup main form widget
574  tne.setupUi(mainSqlWidget());
575  connect(tne.pbSetVehicleType, SIGNAL(clicked()), this, SLOT(setBaseVehicleType()));
577  // Setup form
578  if(!setMainTable("TestNode_tbl", "Test node", database())){
579  return false;
580  }
587  /*
588  * Setup base unit widget mapping
589  */
590  testNodeModel = model("TestNode_tbl");
591  Q_ASSERT(testNodeModel != 0);
592  // Setup base article model
593  baseVehicleTypeModel = new QSqlTableModel(this, database());
594  baseVehicleTypeModel->setTable("VehicleType_tbl");
595  if(!baseVehicleTypeModel->select()){
596  return false;
597  }
598  // Setup base article widget mapper
599  baseVehicleTypeMapper = new QDataWidgetMapper(this);
600  baseVehicleTypeMapper->setModel(baseVehicleTypeModel);
601  baseVehicleTypeMapper->addMapping(tne.leType, baseVehicleTypeModel->fieldIndex("Type"));
602  baseVehicleTypeMapper->addMapping(tne.leSubType, baseVehicleTypeModel->fieldIndex("SubType"));
603  baseVehicleTypeMapper->addMapping(tne.leSeriesNumber, baseVehicleTypeModel->fieldIndex("SeriesNumber"));
604  // Setup base article relation
605  baseVehicleTypeRelation = new mdtSqlRelation(this);
606  baseVehicleTypeRelation->setParentModel(testNodeModel);
607  baseVehicleTypeRelation->setChildModel(baseVehicleTypeModel);
608  if(!baseVehicleTypeRelation->addRelation("VehicleType_Id_FK_PK", "Id_PK", false)){
609  return false;
610  }
611  connect(mainSqlWidget(), SIGNAL(currentRowChanged(int)), baseVehicleTypeRelation, SLOT(setParentCurrentIndex(int)));
612  connect(baseVehicleTypeRelation, SIGNAL(childModelFilterApplied()), baseVehicleTypeMapper, SLOT(toFirst()));
613  connect(baseVehicleTypeRelation, SIGNAL(childModelIsEmpty()), baseVehicleTypeMapper, SLOT(revert()));
614  // Force a update
615  mainSqlWidget()->setCurrentIndex(mainSqlWidget()->currentRow());
616 
617  return true;
618 }
