00288 {
00289   mdtPortInfo portInfo;
00290 
00291   // Check that host is set
00292   if(leHost->text().trimmed().isEmpty()){
00293     showStatusMessage(tr("Please enter a host name or IP address"), 3000);
00294     return false;
00295   }
00296   setStateConnecting();
00297   // Set port name
00298   portInfo.setPortName(leHost->text().trimmed() + ":" + QString::number(sbPort->value()));
00299   portInfo.setDisplayText(tr("Host: ") + leHost->text().trimmed() + tr("   ,   Port: ") + QString::number(sbPort->value()));
00300   // We must close the port
00301   portManager()->closePort();
00302   // Apply current configuration
00303   updateConfig();
00304   portManager()->setPortInfo(portInfo);
00305   // Try to open port and start port manager
00306   showStatusMessage(tr("Trying to connect ..."));
00307   if(!portManager()->openPort()){
00308     setStateDisconnected();
00309     showStatusMessage(tr("Cannot connect to ") + portManager()->portName());
00310     return false;
00311   }
00312   if(!portManager()->start()){
00313     showStatusMessage(tr("Cannot start thread"));
00314     portManager()->closePort();
00315     setStateDisconnected();
00316     return false;
00317   }
00318   showStatusMessage(tr("Connected"), 3000);
00319 
00320   return true;
