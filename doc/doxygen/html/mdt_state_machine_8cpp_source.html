49 bool mdtStateMachine::waitOnState(int state, int timeout)
50 {
51  pvTimeoutOccured = false;
52  if(currentState() == state){
53  return true;
54  }
55  if(timeout > 0){
56  pvTimeoutTimer->start(timeout);
57  }
58  while(currentState() != state){
59  if(pvTimeoutOccured){
60  return false;
61  }
62  QCoreApplication::processEvents(QEventLoop::AllEvents | QEventLoop::WaitForMoreEvents);
63  }
64  pvTimeoutTimer->stop();
65 
66  return true;
67 }
