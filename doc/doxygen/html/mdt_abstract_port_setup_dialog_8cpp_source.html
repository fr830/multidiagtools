169 void mdtAbstractPortSetupDialog::on_buttonBox_clicked(QAbstractButton *button)
170 {
171  Q_ASSERT(button != 0);
172  Q_ASSERT(pvPortManager != 0);
173 
174  QDialogButtonBox::StandardButton type;
175 
176  type = buttonBox->standardButton(button);
177  switch(type){
178  case QDialogButtonBox::Ok:
179  // If port manager is ready, setup was allready applied, or nothing was changed
180  if(portManager()->isReady()){
181  close();
182  }else{
183  if(applySetup()){
184  close();
185  }
186  }
187  break;
188  case QDialogButtonBox::Apply:
189  applySetup();
190  break;
191  case QDialogButtonBox::Cancel:
192  close();
193  break;
194  default:
195  mdtError e(MDT_GUI_ERROR, "Recived signal from buttonBox with unknown button type", mdtError::Warning);
196  MDT_ERROR_SET_SRC(e, "mdtAbstractPortSetupDialog");
197  e.commit();
198  }
199 }
