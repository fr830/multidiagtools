00259   Q_ASSERT(pvPort != 0);
00260   Q_ASSERT(threadHelper.currentWriteFrame() != 0);
00261 
00262   mdtUsbPort *port;
00263   mdtAbstractPort::error_t portError;
00264   QQueue<mdtFrameUsbControl*> ctlFramesPoolCopy;
00265   mdtFrameUsbControl *ctlFrame;
00266   quint8 status;
00267   int maxTry;
00268 
00269   // We need a mdtUsbPort object
00270   port = dynamic_cast<mdtUsbPort*>(pvPort);
00271   Q_ASSERT(port != 0);
00272 
00273   // Check if a query is pending
00274   if(threadHelper.currentWriteFrame() != 0){
00275     if(threadHelper.currentWriteFrame()->waitAnAnswer()){
00276       portError = port->cancelWriteTransfer();
00277       if(portError != mdtAbstractPort::NoError){
00278         return portError;
00279       }
00280       threadHelper.restoreCurrentWriteFrameToPool();
00281     }
00282   }
00283   // Flush pending control frames
00284   while(port->controlQueryFrames().size() > 0){
00285     // Warn and remove frame
00286     mdtError e(MDT_USB_IO_ERROR, "Aborting a pending control transfer", mdtError::Warning);
00287     MDT_ERROR_SET_SRC(e, "mdtUsbtmcPortThread");
00288     e.commit();
00289     port->controlFramesPool().enqueue(port->controlQueryFrames().dequeue());
00290   }
00291   // Cancel possibly running control transfer
00292   portError = port->cancelControlTransfer();
00293   if(portError != mdtAbstractPort::NoError){
00294     return portError;
00295   }
00296   while(port->controlResponseFrames().size() > 0){
00297     // Warn and remove frame
00298     mdtError e(MDT_USB_IO_ERROR, "Aborting a pending control transfer", mdtError::Warning);
00299     MDT_ERROR_SET_SRC(e, "mdtUsbtmcPortThread");
00300     e.commit();
00301     port->controlFramesPool().enqueue(port->controlResponseFrames().dequeue());
00302   }
00303   // Get a control frame for working
00304   Q_ASSERT(port->controlFramesPool().size() > 0);
00305   ctlFrame = port->controlFramesPool().dequeue();
00306   Q_ASSERT(ctlFrame != 0);
00307   // Take all control frames from pool, so port manager cannot init a control transfer
00308   while(port->controlFramesPool().size() > 0){
00309     ctlFramesPoolCopy.enqueue(port->controlFramesPool().dequeue());
00310   }
00311   // Send the INITIATE_ABORT_BULK_IN request and wait on response
00312   status = 0x81;  // STATUS_TRANSFER_NOT_IN_PROGRESS
00313   maxTry = 5;
00314   while((status == 0x81)&&(maxTry > 0)){
00315     // Check about stoping
00316     if(!pvRunning){
00317       // restore frames and return noerror
00318       port->controlFramesPool().enqueue(ctlFrame);
00319       while(ctlFramesPoolCopy.size() > 0){
00320         port->controlFramesPool().enqueue(ctlFramesPoolCopy.dequeue());
00321       }
00322       return mdtAbstractPort::NoError;
00323     }
00324     // Send INITIATE_ABORT_BULK_IN request
00326     portError = sendInitiateAbortBulkInRequest(bTag, ctlFrame);
00327     if(portError != mdtAbstractPort::NoError){
00328       // restore frames and abort
00329       while(ctlFramesPoolCopy.size() > 0){
00330         port->controlFramesPool().enqueue(ctlFramesPoolCopy.dequeue());
00331       }
00332       return portError;
00333     }
00334     // more than 1 frame available should only happen if device had a control response queued
00335     if(port->controlResponseFrames().size() != 1){
00336       mdtError e(MDT_USB_IO_ERROR, "INITIATE_ABORT_BULK_IN returned unexpected amount of responses", mdtError::Error);
00337       MDT_ERROR_SET_SRC(e, "mdtUsbtmcPortThread");
00338       e.commit();
00339       // restore frames and abort
00340       while(ctlFramesPoolCopy.size() > 0){
00341         port->controlFramesPool().enqueue(ctlFramesPoolCopy.dequeue());
00342       }
00343       return mdtAbstractPort::UnhandledError;
00344     }
00345     ctlFrame = port->controlResponseFrames().dequeue();
00346     if(ctlFrame->size() != 2){
00347       mdtError e(MDT_USB_IO_ERROR, "INITIATE_ABORT_BULK_IN returned unexpected response size (expected 2, received " + QString::number(ctlFrame->size()) + ")", mdtError::Error);
00348       MDT_ERROR_SET_SRC(e, "mdtUsbtmcPortThread");
00349       e.commit();
00350       // Restore frames and abort
00351       port->controlFramesPool().enqueue(ctlFrame);
00352       while(ctlFramesPoolCopy.size() > 0){
00353         port->controlFramesPool().enqueue(ctlFramesPoolCopy.dequeue());
00354       }
00355       return mdtAbstractPort::UnhandledError;
00356     }
00357     // Check status
00358     status = ctlFrame->at(0);
00360     if(status == 0x81){ // STATUS_TRANSFER_NOT_IN_PROGRESS
00361       msleep(500);
00362     }
00363     maxTry--;
00364   }
00365   if(maxTry < 1){
00366     mdtError e(MDT_USB_IO_ERROR, "INITIATE_ABORT_BULK_IN still pending after 10 try", mdtError::Warning);
00367     MDT_ERROR_SET_SRC(e, "mdtUsbtmcPortThread");
00368     e.commit();
00369     // Continue working (possibly, nothing was to abort)
00370     port->controlFramesPool().enqueue(ctlFrame);
00371     while(ctlFramesPoolCopy.size() > 0){
00372       port->controlFramesPool().enqueue(ctlFramesPoolCopy.dequeue());
00373     }
00374     if(pvRunning){
00375       notifyError(mdtAbstractPort::ReadCanceled, true);
00376     }
00377     return mdtAbstractPort::NoError;
00378   }
00379   // Check returned status
00380   if(status != 0x01){ // 0x01: STATUS_SUCCESS
00381     // No transfer in progress (device FIFO empty)
00382     port->controlFramesPool().enqueue(ctlFrame);
00383     while(ctlFramesPoolCopy.size() > 0){
00384       port->controlFramesPool().enqueue(ctlFramesPoolCopy.dequeue());
00385     }
00386     portError = port->cancelReadTransfer();
00387     if(portError != mdtAbstractPort::NoError){
00388       return portError;
00389     }
00390     if(pvRunning){
00391       notifyError(mdtAbstractPort::ReadCanceled, true);
00392     }
00393     return mdtAbstractPort::NoError;
00394   }
00395   // Flush device
00396   portError = readUntilShortPacketReceived(100);
00397   if(portError != mdtAbstractPort::NoError){
00398     // restore frames and abort
00399     port->controlFramesPool().enqueue(ctlFrame);
00400     while(ctlFramesPoolCopy.size() > 0){
00401       port->controlFramesPool().enqueue(ctlFramesPoolCopy.dequeue());
00402     }
00403     return portError;
00404   }
00405   // Send the CHECK_ABORT_BULK_IN_STATUS request and wait on response
00406   status = 0x02;  // STATUS_PENDING
00407   maxTry = 10;
00408   while((status == 0x02)&&(maxTry > 0)){
00409     // Send CHECK_ABORT_BULK_IN_STATUS
00411     portError = sendCheckAbortBulkInStatusRequest(bTag, ctlFrame);
00412     if(portError != mdtAbstractPort::NoError){
00413       // restore frames and abort
00414       while(ctlFramesPoolCopy.size() > 0){
00415         port->controlFramesPool().enqueue(ctlFramesPoolCopy.dequeue());
00416       }
00417       return portError;
00418     }
00419     // more than 1 frame available should only happen if device had a control response queued
00420     if(port->controlResponseFrames().size() != 1){
00421       mdtError e(MDT_USB_IO_ERROR, "CHECK_ABORT_BULK_IN_STATUS returned unexpected amount of responses", mdtError::Error);
00422       MDT_ERROR_SET_SRC(e, "mdtUsbtmcPortThread");
00423       e.commit();
00424       // restore frames and abort
00425       while(ctlFramesPoolCopy.size() > 0){
00426         port->controlFramesPool().enqueue(ctlFramesPoolCopy.dequeue());
00427       }
00428       return mdtAbstractPort::UnhandledError;
00429     }
00430     ctlFrame = port->controlResponseFrames().dequeue();
00431     if(ctlFrame->size() != 8){
00432       mdtError e(MDT_USB_IO_ERROR, "CHECK_ABORT_BULK_IN_STATUS returned unexpected response size (expected 8, received " + QString::number(ctlFrame->size()) + ")", mdtError::Error);
00433       MDT_ERROR_SET_SRC(e, "mdtUsbtmcPortThread");
00434       e.commit();
00435       // restore frames and abort
00436       port->controlFramesPool().enqueue(ctlFrame);
00437       while(ctlFramesPoolCopy.size() > 0){
00438         port->controlFramesPool().enqueue(ctlFramesPoolCopy.dequeue());
00439       }
00440       return mdtAbstractPort::UnhandledError;
00441     }
00442     // Check status
00443     status = ctlFrame->at(0);
00444     if(status == 0x02){ // STATUS_PENDING
00445       // Check bmAbortBulkIn.D0
00446       if((quint8)ctlFrame->at(1) & 0x01){
00447         qDebug() << "*-* Flushing device ...";
00448         // Flush device

00462   }

00470   while(ctlFramesPoolCopy.size() > 0){
00471     port->controlFramesPool().enqueue(ctlFramesPoolCopy.dequeue());
00472   }
00473   if(pvRunning){
