196 void mdtLed::paintEvent(QPaintEvent *)
197 {
198  QPainter painter(this);
199  QPointF center( (qreal)size().width()/2.0 , (qreal)size().height()/2.0 );
200 
201  painter.setPen(pvPen);
202 
203  // Set brush parameters
204  if(!pvTextMode){
205  if(pvGlint){
206  QRadialGradient gradient( center , center.x() , center);
207  gradient.setColorAt(0.0 , pvCurrentGlance);
208  gradient.setColorAt(1.0 , pvCurrentColor);
209  painter.setBrush(gradient);
210  }else{
211  pvFlatBrush.setColor(pvCurrentColor);
212  painter.setBrush(pvFlatBrush);
213  }
214  }
215 
216  if(pvAntialiasing){
217  painter.setRenderHint(QPainter::Antialiasing, true);
218  }else{
219  painter.setRenderHint(QPainter::Antialiasing, false);
220  }
221 
222  // Draw
223  if(pvFixedSize.isValid()){
224  int leftBorder = ( size().width() - pvFixedSize.width() ) / 2 ;
225  int topBorder = ( size().height() - pvFixedSize.height() ) / 2;
226  painter.drawEllipse(leftBorder , topBorder , pvFixedSize.width() , pvFixedSize.height());
227  if(pvTextMode){
228  painter.drawText(leftBorder, topBorder, pvFixedSize.width(), pvFixedSize.height(), Qt::AlignHCenter | Qt::AlignVCenter , pvText);
229  }
230  }else{
231  painter.drawEllipse(pvHorzontalBorder , pvVerticalBorder , size().width()-pv2xHorzontalBorder , size().height()-pv2xVerticalBorder);
232  if(pvTextMode){
233  painter.drawText(pvHorzontalBorder, pvVerticalBorder, size().width()-pv2xHorzontalBorder, size().height()-pv2xVerticalBorder, Qt::AlignHCenter | Qt::AlignVCenter , pvText);
234  }
235  }
236 }
