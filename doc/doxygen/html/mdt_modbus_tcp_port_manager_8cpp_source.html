113 mdtDeviceAddressList mdtModbusTcpPortManager::scan(const QNetworkInterface & iface, const mdtModbusHwNodeIdList& expectedHwNodeAddresses,
114  quint16 port, int timeout )
115 {
116  Q_ASSERT(isClosed());
117 
118  quint32 subNet;
119  quint32 bCast;
120  QHostAddress firstIp;
121  quint32 firstIpV4;
122  QHostAddress lastIp;
123  quint32 lastIpV4;
124  QHostAddress currentIp;
125  QList<QNetworkAddressEntry> entries;
126  QList<mdtNetworkHost> hosts;
127  quint32 j;
128 
129  // Get address entries of interface
130  entries = iface.addressEntries();
131  if(entries.size() < 1){
132  pvLastError.setError(tr("Interface ") + iface.name() + tr(" has no IP address"), mdtError::Error);
133  MDT_ERROR_SET_SRC(pvLastError, "mdtModbusTcpPortManager");
134  pvLastError.commit();
135  return mdtDeviceAddressList();
136  }
137  // Build the list of IPs
138  for(const auto & entry : entries){
139  // Check protocol
140  if(entry.broadcast().protocol() != QAbstractSocket::IPv4Protocol){
141  continue;
142  }
143  bCast = entry.broadcast().toIPv4Address();
144  subNet = bCast & entry.netmask().toIPv4Address();
145  // We need min. place for 2 machines in subnet
146  if((bCast-subNet-4) < 0){
147  pvLastError.setError(tr("Interface '") + iface.name() + tr("' has a subnet for less than 2 machines"), mdtError::Warning);
148  MDT_ERROR_SET_SRC(pvLastError, "mdtModbusTcpPortManager");
149  pvLastError.commit();
150  continue;
151  }
152  // Extract first and last IPs
153  firstIp.setAddress(subNet+1);
154  lastIp.setAddress(bCast-1);
155  firstIpV4 = firstIp.toIPv4Address();
156  lastIpV4 = lastIp.toIPv4Address();
157  // Check that we have not to many hosts to scan
158  if((lastIpV4-firstIpV4) > 1000){
159  pvLastError.setError(tr("Interface '") + iface.name() + ("' has a subnet for more than 1000 machines (do you try to scan the planet ?)"), mdtError::Warning);
160  MDT_ERROR_SET_SRC(pvLastError, "mdtModbusTcpPortManager");
161  pvLastError.commit();
162  return mdtDeviceAddressList();
163  }
164  // Build list of IP addresses and add to hosts
165  for(j = firstIpV4; j <= lastIpV4; ++j){
166  mdtNetworkHost host;
167  currentIp.setAddress(j);
168  host.hostName = currentIp.toString();
169  host.port = port;
170  hosts.append(host);
171  }
172  }
173 
174  return scan(hosts, expectedHwNodeAddresses, timeout);
175 }

258  QTcpSocket socket;

284  ok = true;
285  }

287  socket.abort();
288  while((socket.state() != QAbstractSocket::ClosingState)&&(socket.state() != QAbstractSocket::UnconnectedState)){
289  QCoreApplication::processEvents(QEventLoop::AllEvents | QEventLoop::WaitForMoreEvents);

371 mdtModbusHwNodeId mdtModbusTcpPortManager::getHardwareNodeAddress(int bitsCount, int startFrom)
372 {
373  Q_ASSERT(isReady());
374  Q_ASSERT(bitsCount > 0);
375  Q_ASSERT(startFrom >= 0);
376 
377  mdtModbusHwNodeId nid;
378  int transactionId;
379  mdtPortTransaction *transaction;
380  QByteArray pdu;
381  mdtFrameCodecModbus codec;
383 
384  // Setup MODBUS PDU
385  pdu = codec.encodeReadDiscreteInputs(startFrom, bitsCount);
386  if(pdu.isEmpty()){
387  pvLastError.setError(tr("Could not get MODBUS hardware node ID (PDU encoding failed)"), mdtError::Error);
388  MDT_ERROR_SET_SRC(pvLastError, "mdtModbusTcpPortManager");
389  pvLastError.commit();
390  return nid;
392  }
393  // Get a new transaction
394  transaction = getNewTransaction();
395  // Send query
396  transaction->setQueryReplyMode(true);
397  transaction->setData(pdu);
398  transactionId = sendData(transaction);
399  if(transactionId < 0){
401  return nid;
403  }
404  // Wait on result (use device's defined timeout)
405  if(!waitTransactionDone(transactionId)){
407  return nid;
409  }
410  // Get reply and decode
411  if(codec.decode(readenFrame(transactionId)) != 0x02){
412  pvLastError.setError(tr("Could not get MODBUS hardware node ID (PDU decoding failed)"), mdtError::Error);
413  MDT_ERROR_SET_SRC(pvLastError, "mdtModbusTcpPortManager");
414  pvLastError.commit();
415  return nid;
417  }
418  // Check that we have enough states (can be more, because digital inputs are returned as multiple of 8)
419  if(codec.values().size() < bitsCount){
420  pvLastError.setError(tr("Have not received enough input states"), mdtError::Error);
421  MDT_ERROR_SET_SRC(pvLastError, "mdtModbusTcpPortManager");
422  pvLastError.commit();
423  return nid;
425  }
426  // Ok, have the states, let's decode ID
428  std::vector<bool> bits;
429  bits.reserve(codec.values().size());
430  for(auto & var : codec.values()){
431  Q_ASSERT(!var.isNull());
432  bits.push_back(var.toBool());
433  }
434  nid.setId(bits, bitsCount);
444  return nid;
446 }

565 int mdtModbusTcpPortManager::sendData(mdtPortTransaction *transaction)
566 {
567  Q_ASSERT(pvPort != 0);
568  Q_ASSERT(transaction != 0);
569  Q_ASSERT(!transactionsDoneContains(transaction->id()));
570 
571  mdtFrameModbusTcp *frame = 0;
572 
573  // Wait until we can write
574  while(1){
575  // If port manager was stopped, we return
576  if(isClosed()){
577  restoreTransaction(transaction);
578  return mdtAbstractPort::WriteCanceled;
579  }
580  if(isReady()){
581  lockPortMutex();
582  if(pvPort->writeFramesPool().size() > 0){
583  frame = dynamic_cast<mdtFrameModbusTcp*> (pvPort->writeFramesPool().dequeue());
584  Q_ASSERT(frame != 0);
585  // Here we are ready to write - we keep port mutext locked
586  break;
587  }
588  unlockPortMutex();
589  }
590  QCoreApplication::processEvents(QEventLoop::AllEvents | QEventLoop::WaitForMoreEvents);
591  }
592  if(!waitTransactionPossible()){
593  restoreTransaction(transaction);
594  unlockPortMutex();
595  return mdtAbstractPort::WriteCanceled;
596  }
597  // We are ready to write
598  Q_ASSERT(frame != 0);
599  frame->clear();
600  frame->clearSub();
601  // Store data and add frame to write queue
602  incrementCurrentTransactionId(0, 65535);
603  frame->setTransactionId(currentTransactionId());
604  frame->setUnitId(0xFF);
605  frame->setPdu(transaction->data());
606  frame->encode();
607  // We enable waitAnAnswer , used by mdtTcpSocketThread for timeout detection
608  frame->setWaitAnAnswer(true);
609  pvPort->addFrameToWrite(frame);
610  transaction->setId(currentTransactionId());
611  addTransactionPending(transaction);
612  unlockPortMutex();
613 
614  return transaction->id();
615 }
