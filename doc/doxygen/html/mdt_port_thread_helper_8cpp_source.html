70 mdtAbstractPort::error_t mdtPortThreadHelper::handleCommonReadErrors(mdtAbstractPort::error_t portError)
71 {
72  Q_ASSERT(pvPort != 0);
73  Q_ASSERT(pvThread != 0);
74 
75  if(!pvThread->runningFlagSet()){
76  mdtError e(MDT_PORT_IO_ERROR, "Non running thread wants to check about errors", mdtError::Warning);
77  MDT_ERROR_SET_SRC(e, "mdtPortThread");
78  e.commit();
79  }
80  if(pvCurrentReadFrame == 0){
81  mdtError e(MDT_PORT_IO_ERROR, "Try to handle a read error, but currently no read frame is set", mdtError::Error);
82  MDT_ERROR_SET_SRC(e, "mdtPortThreadHelper");
83  e.commit();
84  return mdtAbstractPort::UnhandledError;
85  }
86  pvCurrentReadFrame->clear();
87  pvCurrentReadFrame->clearSub();
88  notifyError(portError);
89 
90  switch(portError){
91  case mdtAbstractPort::Disconnected:
92  // We must restore the frame because port will be closed
93  restoreCurrentReadFrameToPool();
94  // Try to reconnect
95  portError = reconnect(true);
96  if(portError != mdtAbstractPort::NoError){
97  pvCurrentReadFrame = 0;
98  return mdtAbstractPort::UnhandledError;
99  }
100  if(!getNewFrameRead()){
101  return mdtAbstractPort::UnhandledError;
102  }
103  portError = mdtAbstractPort::ErrorHandled;
104  break;
105  case mdtAbstractPort::ReadCanceled:
106  portError = mdtAbstractPort::ErrorHandled;
107  break;
108  case mdtAbstractPort::ReadTimeout:
109  portError = mdtAbstractPort::ErrorHandled;
110  break;
111  default:
112  portError = mdtAbstractPort::UnhandledError;
113  break;
114  }
115 
116  Q_ASSERT(pvCurrentReadFrame != 0);
117  return portError;
118 }

180 bool mdtPortThreadHelper::getNewFrameRead()
181 {
182  Q_ASSERT(pvPort != 0);
183  Q_ASSERT(pvThread != 0);
184 
185  if(pvPort->readFramesPool().size() < 1){
186  mdtError e(MDT_PORT_QUEUE_EMPTY_ERROR, "Read frames pool is empty", mdtError::Warning);
187  MDT_ERROR_SET_SRC(e, "mdtPortThreadHelper");
188  e.commit();
189  notifyError(mdtAbstractPort::ReadPoolEmpty);
190  // Wait until a frame is available again, or end of thread.
191  while(pvPort->readFramesPool().size() < 1){
192  pvPort->unlockMutex();
193  msleep(100);
194  pvPort->lockMutex();
195  if(!pvThread->runningFlagSet()){
196  pvCurrentReadFrame = 0;
197  return false;
198  }
199  }
200  // Put info into log that a frame is available again
201  mdtError e2(MDT_NO_ERROR, "Read frames pool has now frames again (no longer empty)", mdtError::Info);
202  MDT_ERROR_SET_SRC(e2, "mdtPortThreadHelper");
203  e2.commit();
204  notifyError(mdtAbstractPort::NoError);
205  }
206  Q_ASSERT(!pvPort->readFramesPool().isEmpty());
207  pvCurrentReadFrame = pvPort->readFramesPool().dequeue();
208  Q_ASSERT(pvCurrentReadFrame != 0);
209  pvCurrentReadFrame->clear();
210  pvCurrentReadFrame->clearSub();
211 
212  return true;
213 }

244 int mdtPortThreadHelper::submitReadenData(const char *data, int size, bool emitNewFrameReaden)
245 {
246  Q_ASSERT(pvPort != 0);
247  Q_ASSERT(pvThread != 0);
248  Q_ASSERT(data != 0);
249 
250  char *bufferCursor = (char*)data;
251  int toStore;
252  int stored;
253  int completeFrames = 0;
254 
255  if(size > 0){
256  emit readProcessBegin();
257  // Store readen data
258  toStore = size;
259  while(toStore > 0){
260  // Check for new frame if needed
261  if(pvCurrentReadFrame == 0){
262  if(!getNewFrameRead()){
263  return mdtAbstractPort::UnhandledError;
264  }
265  }
266  // Store data
267  stored = pvCurrentReadFrame->putData(bufferCursor, toStore);
268  // If frame is full, enqueue to readen frames and get a new one
269  if(pvCurrentReadFrame->bytesToStore() == 0){
270  stored += pvCurrentReadFrame->eofSeqLen();
271  submitCurrentReadFrame(emitNewFrameReaden);
272  if(!getNewFrameRead()){
273  return mdtAbstractPort::UnhandledError;
274  }
275  ++completeFrames;
276  }
277  // When frame becomes full and EOF seq was not reached, stored will be to big
278  if(stored >= pvCurrentReadFrame->capacity()){
279  stored = pvCurrentReadFrame->capacity()-1;
280  toStore = 0;
281  }else{
282  toStore = toStore - stored;
283  }
284  bufferCursor = bufferCursor + stored;
285  Q_ASSERT(bufferCursor < (data + pvCurrentReadFrame->capacity()));
286  }
287  }
288 
289  return completeFrames;
290 }
