63 void mdtUsbPortThreadNew::stop()
64 {

66  // Close device if one was open
67  if(pvDeviceHandle != 0){
68  aboutToClose();
69  int err = libusb_release_interface(pvDeviceHandle, pvbInterfaceNumber);
70  if(err != 0){
71  std::lock_guard<std::mutex> lg(pvLastErrorMutex);
72  pvLastError.setError(tr("Could not release interface."), mdtError::Warning);
73  pvLastError.setSystemError(err, libusb_strerror((libusb_error)err));

79  }
80  // Device is closed, join should not block to long

116 bool mdtUsbPortThreadNew::openDevice()
117 {
118  Q_ASSERT(pvUsbContext != 0);
119  Q_ASSERT(!pvDeviceDescriptor.isEmpty());
120 
121  int err;
122 
123  // Open device
124  pvDeviceHandle = pvDeviceDescriptor.open();
125  if(pvDeviceHandle == 0){
126  std::lock_guard<std::mutex> lg(pvLastErrorMutex);
127  pvLastError = pvDeviceDescriptor.lastError();
128  return false;
129  }
130  Q_ASSERT(pvDeviceHandle != 0);
131  // Tell libusb that we let him detach possibly loaded kernel driver itself
132  err = libusb_set_auto_detach_kernel_driver(pvDeviceHandle, 1);
133  if(err != 0){
134  std::lock_guard<std::mutex> lg(pvLastErrorMutex);
135  pvLastError.setError(QObject::tr("Your platform does not support kernel driver detach."), mdtError::Warning);
136  pvLastError.setSystemError(err, libusb_strerror((libusb_error)err));
137  MDT_ERROR_SET_SRC(pvLastError, "mdtUsbPortThreadNew");
138  pvLastError.commit();
139  }
140  // Claim interface
141  err = libusb_claim_interface(pvDeviceHandle, pvbInterfaceNumber);
142  if(err != 0){
143  std::lock_guard<std::mutex> lg(pvLastErrorMutex);
144  pvLastError.setError(QObject::tr("Could not claim interface."), mdtError::Error);
145  pvLastError.setSystemError(err, libusb_strerror((libusb_error)err));
146  MDT_ERROR_SET_SRC(pvLastError, "mdtUsbPortThreadNew");
147  pvLastError.commit();
148  libusb_close(pvDeviceHandle);
149  pvDeviceHandle = 0;
150  return false;
151  }

153  return true;

160 void mdtUsbPortThreadNew::run()
161 {
162  int err;
163 
164  // Open USB device
165  if(!openDevice()){
166  setCurrentState(State_t::Error);
167  return;
168  }
169  // Tell main thread that we are now running and run..
170  setCurrentState(State_t::Running);
171  while(currentState() == State_t::Running){
172  err = libusb_handle_events(pvUsbContext);
173  if(err != 0){
174  std::lock_guard<std::mutex> lg(pvLastErrorMutex);
175  pvLastError.setError(QObject::tr("USB event handling failed (libusb_handle_events())"), mdtError::Error);
176  pvLastError.setSystemError(err, libusb_strerror((libusb_error)err));
177  MDT_ERROR_SET_SRC(pvLastError, "mdtUsbPortThreadNew");
178  pvLastError.commit();
179  setCurrentState(State_t::Error);
180  return;
181  }
182  // Some event woke us up - Check that it was not a stop request
183  if(currentState() != State_t::Running){
184  break;
185  }
186  }
187  // Tell main thread that we are now stopped
188  setCurrentState(State_t::Stopped);
189 }
