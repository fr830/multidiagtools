130 void mdtPortThreadHelperSocket::onSocketDisconnected()
131 {
132  Q_ASSERT(pvPort != 0);
133  Q_ASSERT(pvThread != 0);
134  Q_ASSERT(pvSocket != 0);
135 
136  pvPort->lockMutex();

138  if(pvConnectMaxTry < 1){
139  mdtError e(MDT_TCP_IO_ERROR, "Connection failed after max try", mdtError::Error);
140  MDT_ERROR_SET_SRC(e, "mdtPortThreadHelperSocket");
141  e.commit();

148  pvConnectMaxTry--;
149  pvSocket->connectToHost(pvHost, pvPortNumber);
150  pvConnectionTimeoutTimer->start();
151  }else{
152  qDebug() << "Socket thread: closing ...";
153  pvSocket->close();
154  qDebug() << "Socket thread: unlock mutex";
155  pvPort->unlockMutex();
156  qDebug() << "Socket thread: notify ConnectionFailed";
157  notifyError(mdtAbstractPort::ConnectionFailed);
158  return;
159  }
160  }
161  qDebug() << "Socket thread: unlock mutex";
162  pvPort->unlockMutex();

470 mdtAbstractPort::error_t mdtPortThreadHelperSocket::readFromSocket()
471 {
472  Q_ASSERT(pvPort != 0);
473  Q_ASSERT(pvThread != 0);
474  Q_ASSERT(pvSocket != 0);
475 
476  int readen;
477  mdtAbstractPort::error_t portError;
478  int submittedFrames;
479 
480  // Check if a new frame is needed
481  if(currentReadFrame() == 0){
482  if(!getNewFrameRead()){
483  return mdtAbstractPort::ReadPoolEmpty;
484  }
485  }
486  // Alloc local buffer if needed
487  if(pvReadBufferSize < currentReadFrame()->capacity()){
488  delete[] pvReadBuffer;
489  pvReadBufferSize = currentReadFrame()->capacity();
490  pvReadBuffer = new char[pvReadBufferSize];
491  if(pvReadBuffer == 0){
492  pvReadBufferSize = 0;
493  mdtError e(MDT_MEMORY_ALLOC_ERROR, "Cannot allocate memory for local buffer", mdtError::Error);
494  MDT_ERROR_SET_SRC(e, "mdtPortThreadHelperSocket");
495  e.commit();
496  return mdtAbstractPort::UnhandledError;
497  }
498  }
499  // Read data from port
500  readen = pvSocket->read(pvReadBuffer, pvReadBufferSize);
501  if(readen < 0){
502  portError = mapSocketError(pvSocket->error(), true);
503  return portError;
504  }
505  // Store to frames
506  submittedFrames = submitReadenData(pvReadBuffer, readen, true);
507  if(submittedFrames < 0){
508  return (mdtAbstractPort::error_t)submittedFrames;
509  }
510  // Stop read timeout timer
511  if(submittedFrames > 0){
512  pvReadTimeoutTimer->stop();
513  }
514 
515  return mdtAbstractPort::NoError;
516 }

518 mdtAbstractPort::error_t mdtPortThreadHelperSocket::writeToSocket()
519 {
520  Q_ASSERT(pvPort != 0);
521  Q_ASSERT(pvThread != 0);
522  Q_ASSERT(pvSocket != 0);
523 
524  int written = 0;
525  mdtFrame *frame = 0;
526  mdtAbstractPort::error_t portError;
527 
528  // Get frame to write
529  Q_ASSERT(currentWriteFrame() == 0);
530  if(pvPort->writeFrames().size() < 1){
531  return mdtAbstractPort::NoError;
532  }
533  setCurrentWriteFrame(pvPort->writeFrames().dequeue());
534  frame = currentWriteFrame();
535  Q_ASSERT(frame != 0);
536  // Start the write timeout timer
537  pvWriteTimeoutTimer->start();
538  // If waitAnAnswer flag is set, we start the read timeout timer
539  if(frame->waitAnAnswer()){
540  pvReadTimeoutTimer->start();
541  }
542  // Write the frame to socket
543  while(!frame->isEmpty()){
544  // Check about flush request
545  if(pvPort->flushOutRequestPending()){
546  pvWriteTimeoutTimer->stop();
547  // Restore frame to pool and return
548  restoreCurrentWriteFrameToPool();
549  return mdtAbstractPort::NoError;
550  }
551  // Send data
552  emit writeProcessBegin();
553  written = pvSocket->write(frame->data(), frame->size());
554  if(written < 0){
555  if(!pvThread->runningFlagSet()){
556  return mdtAbstractPort::ErrorHandled;
557  }
558  portError = mapSocketError(pvSocket->error(), false);
559  return portError;
560  }
561  frame->take(written);
562  }
563  // Here, frame is completly sent
564  Q_ASSERT(frame->isEmpty());
565  restoreCurrentWriteFrameToPool();
566 
567  return mdtAbstractPort::NoError;
568 }

570 mdtAbstractPort::error_t mdtPortThreadHelperSocket::mapSocketError(QAbstractSocket::SocketError error, bool byRead)
571 {
572  Q_ASSERT(pvSocket != 0);
573 
574  switch(error){
575  case QAbstractSocket::ConnectionRefusedError:
576  return mdtAbstractPort::PortAccess;
577  case QAbstractSocket::RemoteHostClosedError:
578  return mdtAbstractPort::Disconnected;
579  case QAbstractSocket::HostNotFoundError:
580  return mdtAbstractPort::PortNotFound;
581  case QAbstractSocket::SocketAccessError:
582  return mdtAbstractPort::PortAccess;
583  case QAbstractSocket::SocketTimeoutError:
584  if(byRead){
585  return mdtAbstractPort::ReadTimeout;
586  }else{
587  return mdtAbstractPort::WriteTimeout;
588  }
589  case QAbstractSocket::NetworkError:
590  return mdtAbstractPort::Disconnected;
591  default:
592  mdtError e(MDT_TCP_IO_ERROR, "Unhandled socket error occured.", mdtError::Error);
593  e.setSystemError(error, pvSocket->errorString());
594  MDT_ERROR_SET_SRC(e, "mdtPortThreadHelperSocket");
595  e.commit();
596  return mdtAbstractPort::UnhandledError;
