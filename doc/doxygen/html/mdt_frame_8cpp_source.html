00041 int mdtFrame::putUntil(const char *data, char token, int maxLen)
00042 {
00043   Q_ASSERT(data != 0);
00044   Q_ASSERT(data != QByteArray::data());
00045 
00046   // Get max size that can be stored
00047   if(maxLen > remainCapacity()){
00048     maxLen = remainCapacity();
00049   }
00050   // Init cursors
00051   char *cursor = (char*)data;
00052   char *end = cursor + maxLen;
00053   int stored = 0;
00054 
00055   // If EOF condition was reached, or frame is full, we do nothing
00056   if(isComplete()||isFull()){
00057     return 0;
00058   }
00059 
00060   // We not want to check ignoreNullValues each byte, so we make 2 loops
00061   if(pvIgnoreNullValues){
00062     while(cursor < end){
00063       if(*cursor == token){
00064         pvEOFcondition = true;
00065         return stored;
00066       }
00067       if(*cursor != 0){
00068         append(*cursor);
00069         ++stored;
00070       }
00071       ++cursor;
00072     }
00073   }else{
00074     while(cursor < end){
00075       if(*cursor == token){
00076         pvEOFcondition = true;
00077         return (cursor - data);
00078       }
00079       append(*cursor);
00080       ++cursor;
00081       ++stored;
00082     }
00083   }
00084 
00085   return stored;
