00080   pvReconnectTimeout = pvPort->config().connectTimeout();

00104 void mdtPortThread::stop()
00105 {
00106   Q_ASSERT(pvPort != 0);
00107 
00108   int err;
00109 
00110   // Unset the running flag
00111   qDebug() << "mdtPortThread::stop() ...";
00112   pvPort->lockMutex();
00113   if(!pvRunning){
00114     pvPort->unlockMutex();
00115     return;
00116   }
00117   pvRunning = false;
00118   pvCurrentError = mdtAbstractPort::UnhandledError;
00119   if(isWriter()){
00120     pvPort->abortFrameToWriteWait();
00121   }
00122 #ifdef Q_OS_UNIX
00123   if(QThread::isRunning()){
00124     Q_ASSERT(pvNativePthreadObject != 0);
00125     err = pthread_kill(pvNativePthreadObject, SIGALRM);
00126     if(err != 0){
00127       mdtError e(MDT_PORT_IO_ERROR, "pthread_kill() failed", mdtError::Error);
00128       switch(err){
00129         case EINVAL:
00130           e.setSystemError(err, "Invalid signal was specified (EINVAL)");
00131         case ESRCH:
00132           e.setSystemError(err, "Specified thread ID not found (ESRCH)");
00133       }
00134       MDT_ERROR_SET_SRC(e, "mdtPortThread");
00135       e.commit();
00136     }
00137   }
00138 #endif
00139   pvPort->unlockMutex();
00140 
00141   // Exit event loop (for subclass that use a event loop)
00142   exit();
00143   // Wait the end of the thread
00144   while(!isFinished()){
00145     qApp->processEvents();
00146     msleep(50);
00147   }
00148   qDebug() << "mdtPortThread::stop() DONE";

00224 mdtAbstractPort::error_t mdtPortThread::reconnect(bool notify)
00225 {
00226   Q_ASSERT(pvPort != 0);
00227 
00228   int count = pvReconnectMaxTry;
00229   mdtAbstractPort::error_t error;
00230 
00231   // Check about buggy caller
00232   if(!pvRunning){
00233     mdtError e(MDT_PORT_IO_ERROR, "Thread called reconnect() during a stop process, aborting", mdtError::Warning);
00234     MDT_ERROR_SET_SRC(e, "mdtPortThread");
00235     e.commit();
00236     return mdtAbstractPort::UnhandledError;
00237   }
00238   if(notify){
00239     notifyError(mdtAbstractPort::Connecting);
00240   }
00241   while(count > 0){
00242     if(!pvRunning){
00243       return mdtAbstractPort::Disconnected;
00244     }
00245     qDebug() << "mdtPortThread::reconnect(): trying ...";
00246     error = pvPort->reconnect(pvReconnectTimeout);
00247     if(error == mdtAbstractPort::NoError){
00248       if(notify){
00249         notifyError(mdtAbstractPort::NoError);
00250       }
00251       return mdtAbstractPort::NoError;
00252     }
00253     if(error != mdtAbstractPort::Disconnected){
00254       if(notify){
00255         notifyError(error);
00256       }
00257       return error;
00258     }
00259     count--;
00260   }
00261   // Max try reached
00262   if(notify){
00263     mdtError e(MDT_PORT_IO_ERROR, "Connection failed after max try", mdtError::Error);
00264     MDT_ERROR_SET_SRC(e, "mdtPortThread");
00265     e.commit();
00266     notifyError(mdtAbstractPort::Disconnected);
00267   }
00268 
00269   return mdtAbstractPort::UnhandledError;
00270 }
