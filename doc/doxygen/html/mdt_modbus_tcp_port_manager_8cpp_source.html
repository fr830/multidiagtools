272 bool mdtModbusTcpPortManager::tryToConnect(const QString &hostName, quint16 port, int timeout)
273 {
274  Q_ASSERT(isClosed());
275 
276  QTcpSocket socket;
277  bool ok = false;
278  QTimer timeoutTimer;
279 
280  timeoutTimer.setSingleShot(true);
281  connect(&timeoutTimer, SIGNAL(timeout()), this, SLOT(abortTryToConnect()));
282  emit(statusMessageChanged(tr("Trying host ") + hostName + " , port " + QString::number(port), "", 500));
283  socket.connectToHost(hostName, port);
284  pvAbortTryToConnect = false;
285  timeoutTimer.start(timeout);
286  while(socket.state() != QAbstractSocket::ConnectedState){
287  if(pvAbortTryToConnect){
288  timeoutTimer.stop();
289  return false;
290  }
291  if(pvAbortScan){
292  timeoutTimer.stop();
293  return false;
294  }
295  QCoreApplication::processEvents(QEventLoop::AllEvents | QEventLoop::WaitForMoreEvents);
296  }
297  timeoutTimer.stop();
298  if(socket.state() != QAbstractSocket::UnconnectedState){
299  // Check if connection was successfull
300  if(socket.state() == QAbstractSocket::ConnectedState){
301  ok = true;
302  }
303  // Disconnect
304  socket.abort();
305  while((socket.state() != QAbstractSocket::ClosingState)&&(socket.state() != QAbstractSocket::UnconnectedState)){
306  QCoreApplication::processEvents(QEventLoop::AllEvents | QEventLoop::WaitForMoreEvents);
307  }
308  }
309

369 int mdtModbusTcpPortManager::getHardwareNodeAddress(int bitsCount, int startFrom)
370 {
371  Q_ASSERT(isReady());
372  Q_ASSERT(bitsCount > 0);
373  Q_ASSERT(startFrom >= 0);
374 
375  int transactionId;
376  mdtPortTransaction *transaction;
377  QByteArray pdu;
378  mdtFrameCodecModbus codec;
379  int i, id;
380 
381  // Setup MODBUS PDU
382  pdu = codec.encodeReadDiscreteInputs(startFrom, bitsCount);
383  if(pdu.isEmpty()){
384  return (int)mdtAbstractPort::UnhandledError;
385  }
386  // Get a new transaction
387  transaction = getNewTransaction();
388  // Send query
389  transaction->setQueryReplyMode(true);
390  transaction->setData(pdu);
391  transactionId = sendData(transaction);
392  if(transactionId < 0){
393  return transactionId;
394  }
395  // Wait on result (use device's defined timeout)
396  if(!waitTransactionDone(transactionId)){
397  return (int)mdtAbstractPort::ReadTimeout;
398  }
399  // Get reply and decode
400  if(codec.decode(readenFrame(transactionId)) != 0x02){
401  return (int)mdtAbstractPort::UnhandledError;
402  }
403  // Check that we have enough states (can be more, because digital inputs are returned as multiple of 8)
404  if(codec.values().size() < bitsCount){
405  mdtError e(MDT_TCP_IO_ERROR, "Have not received enough input states", mdtError::Error);
406  MDT_ERROR_SET_SRC(e, "mdtModbusTcpPortManager");
407  e.commit();
408  return (int)mdtAbstractPort::UnhandledError;
409  }
410  // Ok, have the states, let's decode ID
411  id = 0;
412  for(i=0; i<bitsCount; i++){
413  if((codec.values().at(i).isValid())&&(codec.values().at(i).toBool() == true)){
414  id |= (1<<i);
415  }
416  }
417 
418  return id;
419 }

517 int mdtModbusTcpPortManager::sendData(mdtPortTransaction *transaction)
518 {
519  Q_ASSERT(pvPort != 0);
520  Q_ASSERT(transaction != 0);
521  Q_ASSERT(!transactionsDoneContains(transaction->id()));
522 
523  mdtFrameModbusTcp *frame = 0;
524 
525  // Wait until we can write
526  while(1){
527  // If port manager was stopped, we return
528  if(isClosed()){
529  restoreTransaction(transaction);
530  return mdtAbstractPort::WriteCanceled;
531  }
532  if(isReady()){
533  lockPortMutex();
534  if(pvPort->writeFramesPool().size() > 0){
535  frame = dynamic_cast<mdtFrameModbusTcp*> (pvPort->writeFramesPool().dequeue());
536  Q_ASSERT(frame != 0);
537  // Here we are ready to write - we keep port mutext locked
538  break;
539  }
540  unlockPortMutex();
541  }
542  QCoreApplication::processEvents(QEventLoop::AllEvents | QEventLoop::WaitForMoreEvents);
543  }
544  if(!waitTransactionPossible()){
545  restoreTransaction(transaction);
546  unlockPortMutex();
547  return mdtAbstractPort::WriteCanceled;
548  }
549  // We are ready to write
550  Q_ASSERT(frame != 0);
551  frame->clear();
552  frame->clearSub();
553  // Store data and add frame to write queue
554  incrementCurrentTransactionId(0, 65535);
555  frame->setTransactionId(currentTransactionId());
556  frame->setUnitId(0xFF);
557  frame->setPdu(transaction->data());
558  frame->encode();
559  // We enable waitAnAnswer , used by mdtTcpSocketThread for timeout detection
560  frame->setWaitAnAnswer(true);
561  pvPort->addFrameToWrite(frame);
562  transaction->setId(currentTransactionId());
563  addTransactionPending(transaction);
564  unlockPortMutex();
565 
566  return transaction->id();
567 }
