104 void mdtPortTerm::sendCmd()
105 {
106  QString cmd;
107 
108  if(!pvReady){
109  return;
110  }
111  if(pvCurrentPortManager == 0){
112  setStateError();
113  showStatusMessage(tr("No port manager was set, please choose a port"));
114  return;
115  }
116  cmd = leCmd->text().trimmed();
117  if(cmd.isEmpty()){
118  leCmd->clear();
119  return;
120  }
121  if(pvCurrentPortManager == pvSerialPortManager){
122  if(!sendCommandToSerialPort(cmd)){
123  return;
124  }
125  }else if(pvCurrentPortManager == pvUsbtmcPortManager){
126  if(!sendCommandToUsbtmcPort(cmd)){
127  return;
128  }
129  }
130  leCmd->clear();
131 }

151 void mdtPortTerm::attachToSerialPort()
152 {
153  QList<mdtPortInfo*> ports;
154 
155  detachFromPorts();
156  // Create objects
157  pvSerialPortManager = new mdtSerialPortManager;
158  pvSerialPortCtlWidget = new mdtSerialPortCtlWidget;
159  pvCurrentPortManager = pvSerialPortManager;
160  // Status widget
161  pvStatusWidget->enableTxRxLeds(pvSerialPortManager->writeThread(), pvSerialPortManager->readThread());
162  // Control widget
163  pvStatusWidget->addCustomWidget(pvSerialPortCtlWidget);
164  pvSerialPortCtlWidget->makeConnections(pvSerialPortManager);
165  // Setup port manager and make connections
166  pvSerialPortManager->setKeepTransactionsDone(false);
167  connect(pvSerialPortManager, SIGNAL(stateChanged(int)), this, SLOT(setStateFromPortManager(int)));
168  connect(pvSerialPortManager, SIGNAL(stateChangedForUi(int, const QString&, int, bool)), pvStatusWidget, SLOT(setState(int, const QString&, int, bool)));
169  connect(pvSerialPortManager, SIGNAL(newTransactionDone(mdtPortTransaction*)), this, SLOT(appendReadenData(mdtPortTransaction*)));
170  pvSerialPortManager->notifyCurrentState();
171  // Try to open first port
172  ports = pvSerialPortManager->scan();
173  if(ports.size() < 1){
174  pvStatusWidget->setPermanentText(tr("No free serial port found"));
175  return;
176  }
177  pvSerialPortManager->setPortInfo(*ports.at(0));
178  if(pvSerialPortManager->start()){
179  pvStatusWidget->setPermanentText(tr("Port : ") + pvCurrentPortManager->portInfo().displayText());
180  }
181  qDeleteAll(ports);
182  // Command term char
183  pvCmdTermSequence = "";
184  leCmd->setMaxLength(200);
185 }

303 void mdtPortTerm::setStateFromPortManager(int state)
304 {
305  switch((mdtPortManager::state_t)state){

307  setStatePortClosed();

318  case mdtPortManager::Connecting:

327  setStateReady();
328  break;
329  case mdtPortManager::Busy:

331  break;
332  case mdtPortManager::PortError:

334  break;
335  }

399 bool mdtPortTerm::sendCommandToUsbtmcPort(const QString &command)
400 {
401  Q_ASSERT(pvUsbtmcPortManager != 0);
402 
403  // Send command
404  if(pvUsbtmcPortManager->sendCommand(command.toAscii() + pvCmdTermSequence) < 0){
405  showStatusMessage(tr("Error occured during write process"), 1000);
406  return false;
407  }
408  // If we have a query, send the read request
409  if(command.right(1) == "?"){
410  if(pvUsbtmcPortManager->sendReadRequest(false) < 0){
411  showStatusMessage(tr("Error occured while sending read request"), 1000);
412  return false;
413  }
414  }
415 
416  return true;
