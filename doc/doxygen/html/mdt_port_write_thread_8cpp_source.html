00046 void mdtPortWriteThread::run()
00047 {
00048   qDebug() << "WRTHD: starting ...";
00049   Q_ASSERT(pvPort != 0);
00050 
00051   mdtFrame *frame = 0;
00052   int interframeTime = 0;
00053   int writeMinWaitTime = 0;
00054   bool bytePerByteWrite = false;
00055   mdtAbstractPort::error_t portError;
00056 
00057   pvPort->lockMutex();
00058 #ifdef Q_OS_UNIX
00059   pvNativePthreadObject = pthread_self();
00060   Q_ASSERT(pvNativePthreadObject != 0);
00061 #endif
00062   // Get setup
00063   interframeTime = pvPort->config().writeInterframeTime();
00064   writeMinWaitTime = pvPort->config().writeMinWaitTime();
00065   bytePerByteWrite = pvPort->config().bytePerByteWrite();
00066   // Set the running flag
00067   pvRunning = true;
00068 
00069   // Run...
00070   while(1){
00071     // Read thread state
00072     if(!pvRunning){
00073       break;
00074     }
00075     // Wait for interframe time if needed
00076     if(interframeTime > 0){
00077       pvPort->unlockMutex();
00078       msleep(interframeTime);
00079       pvPort->lockMutex();
00080     }
00081     // Get a frame - will block if nothing is to write
00082     frame = getNewFrameWrite();
00083     // If thread is stopping, it can happen that a Null pointer is returned
00084     if(frame == 0){
00085       break;
00086     }
00087     // Write
00094     portError = writeToPort(frame, bytePerByteWrite, writeMinWaitTime);
00095     if(portError != mdtAbstractPort::NoError){
00096       // Stop
00097       break;
00098     }
00099   }
00100 
00102   
00103   qDebug() << "WRTHD: Cleanup ...";
00104 
00105   pvPort->unlockMutex();
00106   qDebug() << "WRTHD: END";
00107 }
