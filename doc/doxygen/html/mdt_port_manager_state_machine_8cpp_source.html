00050 void mdtPortManagerStateMachine::buildMainStateMachine(mdtPortManager *portManager)
00051 {
00052   Q_ASSERT(portManager != 0);
00053 
00054   // Connect state changed signals
00055   connect(this, SIGNAL(stateChanged(int)), portManager, SIGNAL(stateChanged(int)));
00056   connect(this, SIGNAL(stateChangedForUi(int, const QString&, int, bool)), portManager, SIGNAL(stateChangedForUi(int, const QString&, int, bool)));
00057   // Create states
00058   pvPortClosed = new mdtState(mdtPortManager::PortClosed);
00059   pvPortClosed->setText(tr("Port closed"));
00060   pvPortClosed->setLedColorId(mdtLed::Green);
00061   pvPortClosed->setLedOn(false);
00062   pvPortClosed->setNotifyEnteredToUi(true);
00063   pvStarting = new mdtState(mdtPortManager::Starting);
00064   connect(pvStarting, SIGNAL(entered()), portManager, SLOT(startThreads()));
00065   pvRunning = new mdtState(mdtPortManager::Running);
00066   pvStopping = new mdtState(mdtPortManager::Stopping);
00067   connect(pvStopping, SIGNAL(entered()), portManager, SLOT(stopThreads()));
00068   pvStopped = new mdtState(mdtPortManager::Stopped);
00069   pvPortError = new mdtState(mdtPortManager::PortError);
00070   pvPortError->setText(tr("Fatal error"));
00071   pvPortError->setLedColorId(mdtLed::Red);
00072   pvPortError->setLedOn(true);
00073   pvPortError->setNotifyEnteredToUi(true);
00074   connect(pvPortError, SIGNAL(entered()), portManager, SIGNAL(pmStopThreadsEvent()));
00075   // Add sates to machine
00076   addState(pvPortClosed);
00077   addState(pvStarting);
00078   addState(pvRunning);
00079   addState(pvStopping);
00080   addState(pvStopped);
00081   addState(pvPortError);
00082   // Add pvPortClosed transitions ->
00083   pvPortClosed->addTransition(portManager, SIGNAL(pmStartThreadsEvent()), pvStarting);
00084   // Add pvStarting transitions ->
00085   pvStarting->addTransition(portManager, SIGNAL(pmUnhandledErrorEvent()), pvPortError);
00086   pvStarting->addTransition(portManager, SIGNAL(pmAllThreadsReadyEvent()), pvRunning);
00087   // Add pvRunning transitions ->
00088   pvRunning->addTransition(portManager, SIGNAL(pmStopThreadsEvent()), pvStopping);
00089   pvRunning->addTransition(portManager, SIGNAL(pmUnhandledErrorEvent()), pvPortError);
00090   pvRunning->addTransition(portManager, SIGNAL(pmConnectionFailedEvent()), pvPortError);
00091   // Add pvStopping transitions ->
00092   pvStopping->addTransition(portManager, SIGNAL(pmAllThreadsStoppedEvent()), pvStopped);
00093   // Add pvStopped transitions ->
00094   pvStopped->addTransition(portManager, SIGNAL(pmPortClosedEvent()), pvPortClosed);
00095   // Add pvPortError transitions ->
00096   pvPortError->addTransition(portManager, SIGNAL(pmAllThreadsStoppedEvent()), pvStopped);
00097   pvPortError->addTransition(portManager, SIGNAL(pmStopThreadsEvent()), pvStopping);
00098   // Build Running sub machine
00099   buildRunningSubMachine(portManager, pvRunning);
00100   // Start
00101   setInitialState(pvPortClosed);
00102   start();
00103   waitOnState(mdtPortManager::PortClosed);
00104 }
