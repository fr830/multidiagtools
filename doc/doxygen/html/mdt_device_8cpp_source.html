00089     }
00090     if(pvIos != ios){
00091       pvIos->deleteIos();
00092       delete pvIos;
00093     }
00094   }
00095   // Set new I/Os
00096   pvIos = ios;
00097   if(autoOutputUpdate){
00098     analogOutputs = pvIos->analogOutputs();
