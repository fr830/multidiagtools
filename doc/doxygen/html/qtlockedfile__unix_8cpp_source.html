00056     if (mode == NoLock)
00057         return unlock();

00059     if (mode == m_lock_mode)
00060         return true;
00061 
00062     if (m_lock_mode != NoLock)
00063         unlock();

00066     fl.l_whence = SEEK_SET;

00070     int cmd = block ? F_SETLKW : F_SETLK;

00075             qWarning("QtLockedFile::lock(): fcntl: %s", strerror(errno));
00076         return false;
00077     }
00078

00080     return true;
