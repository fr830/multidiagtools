319 int mdtPortManager::sendData(mdtPortTransaction *transaction)
320 {
321  Q_ASSERT(pvPort != 0);
322  Q_ASSERT(transaction != 0);
323  Q_ASSERT(!pvTransactionsPending.contains(transaction->id()));
324  Q_ASSERT(!transactionsDoneContains(transaction->id()));
325 
326  mdtFrame *frame = 0;
327 
328  // Wait until we can write
329  while(1){
330  // If port manager was stopped, we return
331  if(isClosed()){
332  restoreTransaction(transaction);
333  return mdtAbstractPort::WriteCanceled;
334  }
335  if(isReady()){
336  lockPortMutex();
337  if(pvPort->writeFramesPool().size() > 0){
338  frame = pvPort->writeFramesPool().dequeue();
339  Q_ASSERT(frame != 0);
340  // Here we are ready to write - we keep port mutext locked
341  break;
342  }
343  unlockPortMutex();
344  }
345  QCoreApplication::processEvents(QEventLoop::AllEvents | QEventLoop::WaitForMoreEvents);
346  }
347  if(!waitTransactionPossible()){
348  restoreTransaction(transaction);
349  unlockPortMutex();
350  return mdtAbstractPort::WriteCanceled;
351  }
352  // We are ready to write
353  Q_ASSERT(frame != 0);
354  frame->clear();
355  frame->clearSub();
356  // Store data and add frame to write queue
357  frame->append(transaction->data());
358  pvPort->addFrameToWrite(frame);
359  addTransactionPending(transaction);
360  unlockPortMutex();
361 
362  return transaction->id();
363 }

381 bool mdtPortManager::waitTransactionDone(int id)
382 {
383  Q_ASSERT(pvPort != 0);
384 
385  // Check if transaction is allready done
386  if(transactionsDoneContains(id)){
387  return true;
388  }
389  // Check if transaction exists in pending queue
390  if(!pvTransactionsPending.contains(id)){
391  mdtError e(MDT_PORT_IO_ERROR, "Wait on a frame that was never added to pending queue, id: " + QString::number(id), mdtError::Warning);
392  MDT_ERROR_SET_SRC(e, "mdtPortManager");
393  e.commit();
394  emit pmUnhandledErrorEvent();
395  return false;
396  }
397  /*
398  * If thread handles timeout/disconnection:
399  * In normal way, the thread notifies the error, the state becomes != Ready and this method returns false
400  * But, if many errors are notified, state changes often,
401  * and it happens that we are waiting, in Ready state, on a ID that was flushed, and this for a eternity.
402  * (Keep in mind that we let event loop running, and this way new transactions can be initalized, etc...)
403  * Another problem that is not signaled from thread is a frame the never completes.
404  * To avoid this problem, and keep coherence with port thread,
405  * we add a offset to wait timeout, and return if this time was elapsed.
406  * If thread does not handle timeout:
407  * We simply return after timeout
408  */
409  if(pvThreadHandlesReadTimeout){
410  startTransactionTimer(config().readTimeout() + 200);
411  }else{
412  startTransactionTimer(config().readTimeout());
413  }
414  // Try until success or timeout/error
415  while(!transactionsDoneContains(id)){
416  // Check about timeout or other error
417  // Note: on busy state, we continue waiting (f.ex. ReadPoolEmpty error)
418  if((!isReady())&&(currentState() != Busy)){
419  // Transactions are restored by onThreadsErrorOccured()
420  stopTransactionTimer();
421  return false;
422  }
423  // Check about timeout
424  if(transactionTimeoutOccured()){
425  // Transactions are restored by onTransactionTimeoutOccured()
426  return false;
427  }
428  // Wait
429  QCoreApplication::processEvents(QEventLoop::AllEvents | QEventLoop::WaitForMoreEvents);
430  }
431  stopTransactionTimer();
432 
433  return true;
434 }

816 void mdtPortManager::fromThreadNewFrameReaden()
817 {
818  Q_ASSERT(pvPort != 0);
819 
820  mdtFrame *frame;
821  mdtPortTransaction *transaction;
822  int framesCount = 0;
823 
824  // Get frames in readen queue
825  lockPortMutex();
826  while(pvPort->readenFrames().size() > 0){
827  frame = pvPort->readenFrames().dequeue();

829  if(frame->isComplete()){
830  // Get a transaction
831  transaction = transactionPending(currentTransactionId());

833  incrementCurrentTransactionId();
834  transaction = getNewTransaction();
835  transaction->setId(currentTransactionId());
836  }

838  Q_ASSERT(transaction->id() == currentTransactionId());
839  // Copy data
840  QByteArray data;

853  };

925  flushTransactionsPending();
926  flushTransactionsDone();
