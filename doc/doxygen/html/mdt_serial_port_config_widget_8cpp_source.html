40 void mdtSerialPortConfigWidget::displayConfig(mdtSerialPortConfig &config)

42  int index;
43 
44  // Baud rate
45  index = cbBaudRate->findData(config.baudRate());
46  if(index >= 0){

63  case 5:
64  rbDataBit5->setChecked(true);
65  break;
66  case 6:
67  rbDataBit6->setChecked(true);
68  break;

70  rbDataBit7->setChecked(true);
71  break;

73  rbDataBit8->setChecked(true);
74  break;
75  }
76  // Stop bits count
77  switch(config.stopBitsCount()){
78  case 1:

80  break;

91  rbFlowCtlNone->setChecked(true);
92  }

112 void mdtSerialPortConfigWidget::updateConfig(mdtSerialPortConfig &config)
113 {
114  int index;
115  char xon = 0;
116  char xoff = 0;
117 
118  // Baud rate
119  index = cbBaudRate->currentIndex();
120  if(index >= 0){
121  config.setBaudRate(cbBaudRate->itemData(index).toInt());
122  }
123  // Parity
124  if(rbParityNone->isChecked()){
125  config.setParity(mdtSerialPortConfig::NoParity);
126  }else if(rbParityEven->isChecked()){
127  config.setParity(mdtSerialPortConfig::ParityEven);
128  }else{
129  config.setParity(mdtSerialPortConfig::ParityOdd);
130  }
131  // Data bits count
132  if(rbDataBit5->isChecked()){
133  config.setDataBitsCount(5);
134  }else if(rbDataBit6->isChecked()){
135  config.setDataBitsCount(6);
136  }else if(rbDataBit7->isChecked()){
137  config.setDataBitsCount(7);
138  }else{
139  config.setDataBitsCount(8);
140  }
141  // Stop bits count
142  if(rbStopBit1->isChecked()){
143  config.setStopBitsCount(1);
144  }else{
145  config.setStopBitsCount(2);
146  }
147  // Flow control
148  if(rbFlowCtlRtsCts->isChecked()){
149  config.enableFlowCtlRtsCts();
150  }else{
151  config.diseableFlowCtlRtsCts();
152  }
153  if(rbFlowCtlXonXoff->isChecked()){
154  // Get Xon char
155  index = cbXonChar->currentIndex();
156  if(index >= 0){
157  xon = cbXonChar->itemData(index).toChar().toAscii();
158  }
159  // Get Xoff char
160  index = cbXoffChar->currentIndex();
161  if(index >= 0){
162  xoff = cbXoffChar->itemData(index).toChar().toAscii();
163  }
164  config.enableFlowCtlXonXoff(xon, xoff);
165  }else{
166  config.diseableFlowCtlXonXoff();
167  }
168  if(rbFlowCtlNone->isChecked()){
169  config.diseableFlowCtlRtsCts();
170  config.diseableFlowCtlXonXoff();
171  }
172  // Redisplay the config, so the user can see what was really setup
173  displayConfig(config);
174 }
