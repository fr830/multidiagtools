00064   mdtPortInfo *port;

00066   QList<mdtDeviceInfo*> devices;
00068   int i, j;
00069   mdtAbstractPort::error_t retVal;

00071   // Setup device info
00075   // Scan for ports having a USBTMC device attached
00076   port = 0;
00077   ports = pvUsbtmcPortManager->scan();
00078   for(i=0; i<ports.size(); i++){
00079     Q_ASSERT(ports.at(i) != 0);
00080     // Search in devices list

00082     devices = ports.at(i)->deviceInfoList();
00083     for(j=0; j<devices.size(); j++){
00084       Q_ASSERT(devices.at(j) != 0);
00085       // Check if VID and PID matches
00086       if((devices.at(j)->vendorId() == devInfo.vendorId())&&(devices.at(j)->productId() == devInfo.productId())){
00087         // If requested, check if serial ID matches
00088         if(!devInfo.serialId().isEmpty()){
00089           if(devices.at(j)->serialId() == devInfo.serialId()){
00090             port = ports.at(i);
00091             i = ports.size();
00092             break;
00093           }
00094         }else{
00095           // Only PID and VID must match
00096           port = ports.at(i);
00097           i = ports.size();
00098           break;
00099         }
00100       }
00101     }
00102   }
00103   // Open port if found
00104   if(port != 0){
00105     portManager()->setPortInfo(*port);
00106     if(portManager()->openPort()){
00107       if(portManager()->start()){
00108         retVal = mdtAbstractPort::NoError;
00109       }else{
00110         retVal = mdtAbstractPort::UnhandledError;
00111       }
00112     }else{
00113       retVal = mdtAbstractPort::UnhandledError;
00114     }
00115   }else{
00116     // Device not found
00118     retVal = mdtAbstractPort::PortNotFound;
00119   }
00120   // Free port info list
00121   qDeleteAll(ports);
