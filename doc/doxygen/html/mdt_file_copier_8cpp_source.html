00417 void mdtFileCopier::run()
00418 {
00419   FILE *srcFile;
00420   FILE *destFile;
00421   size_t readen;
00422   size_t written;
00423   size_t charSize = sizeof(char);
00424   mdtFileCopierItem *item;
00425   qint64 copySize = 0;
00426   qint64 copyProgress = 0;
00427   qint64 copySizeDivisor = 1;
00428   QElapsedTimer timer;  // GUI update timer
00429   int previousCopiesCount = 0;
00430   int i;
00431   QString failedCopies = "";
00432   QCryptographicHash hash(QCryptographicHash::Sha1);  // To calculate file "sum"
00433 
00434   // Init states
00435   pvMutex.lock();
00436   pvCopySuccessfull = false;
00437   pvMutex.unlock();
00438 
00439   // Process all available copies
00440   while(1){
00441     pvMutex.lock();
00442     // Check if somthing is to copy
00443     if(pvCopiesInProcess.size() < 1){
00444       pvMutex.unlock();
00445       break;
00446     }
00447     // See if we have to (re-)calculate copy size
00448     if(pvCopiesInProcess.size() > previousCopiesCount){
00449       copySize = 0;
00450       for(i=0; i<pvCopiesInProcess.size(); i++){
00451         item = pvCopiesInProcess.at(i);
00452         Q_ASSERT(item != 0);
00453         copySize += item->fileSize();
00454       }
00455       previousCopiesCount = pvCopiesInProcess.size();
00456       Q_ASSERT(copySize >= Q_INT64_C(0));
00457       // Select divisor regarding total size, and set range
00458       copySizeDivisor = 1;
00459       i=0;
00460       while( copySizeDivisor < (copySize / Q_INT64_C(16384)) ){
00461         copySizeDivisor = (1<<i);
00462         i++;
00463       }
00464       Q_ASSERT(copySizeDivisor > Q_INT64_C(0));
00465       Q_ASSERT((copySize/copySizeDivisor) <= Q_INT64_C(33000) );
00466       Q_ASSERT((copySize/copySizeDivisor) >= Q_INT64_C(0) );
00467       emit(newProgressRange(0, (int)(copySize/copySizeDivisor)));
00468       // Must initialy call setValue() on progress dialog (https://bugreports.qt-project.org/browse/QTBUG-17427)
00469       emit(newProgressValue(0));
00470       timer.start();
00471     }
00472     // Take item and unlock mutex
00473     item = pvCopiesInProcess.takeFirst();
00474     pvMutex.unlock();
00475     Q_ASSERT(item != 0);
00476     // Text to display in progress dialog
00477     emit(newProgressLabelText(item->copyText()));
00478     srcFile = fopen(item->srcFilePath().toAscii().data(), "rb");
00479     if(srcFile == 0){
00480       mdtError e(MDT_FILE_IO_ERROR, "Cannot open file: " + item->srcFilePath(), mdtError::Error);
00481       e.setSystemError(errno, strerror(errno));
00482       MDT_ERROR_SET_SRC(e, "mdtFileCopier");
00483       e.commit();
00484       failedCopies += item->srcFileName() + " -> " + item->destFilePath() + "\n\n";
00485       continue;
00486     }
00487     destFile = fopen(item->destFilePath().toAscii().data(), "wb");
00488     if(destFile == 0){
00489       mdtError e(MDT_FILE_IO_ERROR, "Cannot open file: " + item->destFilePath(), mdtError::Error);
00490       e.setSystemError(errno, strerror(errno));
00491       MDT_ERROR_SET_SRC(e, "mdtFileCopier");
00492       e.commit();
00493       fclose(srcFile);
00494       failedCopies += item->srcFileName() + " -> " + item->destFilePath() + "\n\n";
00495       continue;
00496     }
00497     // Reset hash
00498     hash.reset();
00499     // Copy ..
00500     do{
00501       // See if copy must be aborted
00502       pvMutex.lock();
00503       if(pvCancelCopy){
00504         // Close opened files
00505         fclose(srcFile);
00506         fclose(destFile);
00507         // Remove current destination file (avoid currupted files..)
00508         if(remove(item->destFilePath().toAscii().data()) < 0){
00509           mdtError e(MDT_FILE_IO_ERROR, "Cannot remove file: " + item->destFilePath(), mdtError::Error);
00510           e.setSystemError(errno, strerror(errno));
00511           MDT_ERROR_SET_SRC(e, "mdtFileCopier");
00512           e.commit();
00513         }
00514         qDeleteAll(pvCopiesInProcess);
00515         pvCopiesInProcess.clear();
00516         delete item;
00517         pvMutex.unlock();
00518         return;
00519       }
00520       pvMutex.unlock();
00521       // Process copy ...
00522       readen = fread(pvBuffer, charSize, pvBufferSize, srcFile);
00523       // Hash, if needed
00524       if(item->checkAfterCopy()){
00525         hash.addData(pvBuffer, readen);
00526       }
00527       // Write to destination
00528       written = fwrite(pvBuffer, charSize, readen, destFile);
00529       if(written < readen){
00530         // Write call failed, this is fatal, cancel the copy
00531         fclose(srcFile);
00532         fclose(destFile);
00533         failedCopies += item->srcFileName() + " -> " + item->destFilePath() + "\n\n";
00534         switch(errno){
00535           case ENOSPC:  // Destination device full
00536             emit(copyEnded(failedCopies, tr("Note: destination device is full")));
00537             pvMutex.lock();
00538             qDeleteAll(pvCopiesInProcess);
00539             pvCopiesInProcess.clear();
00540             pvMutex.unlock();
00541             delete item;
00542             return;
00543           default:
00544             mdtError e(MDT_FILE_IO_ERROR, "Write call failed on file: " + item->destFilePath(), mdtError::Error);
00545             e.setSystemError(errno, strerror(errno));
00546             MDT_ERROR_SET_SRC(e, "mdtFileCopier");
00547             e.commit();
00548             emit(copyEnded(failedCopies));
00549             pvMutex.lock();
00550             qDeleteAll(pvCopiesInProcess);
00551             pvCopiesInProcess.clear();
00552             pvMutex.unlock();
00553             delete item;
00554             return;
00555         }
00556       }
00557       copyProgress += (qint64)written;
00558       if(timer.hasExpired(40)){
00559         emit(newProgressValue((int)(copyProgress/copySizeDivisor)));
00560         timer.start();
00561       }
00562     }while(readen > 0);
00563     // Hash, if needed
00564     if(item->checkAfterCopy()){
00565       item->setSrcFileHash(hash.result());
00566     }
00567     // Flush
00568     if(fflush(destFile) != 0){
00569       mdtError e(MDT_FILE_IO_ERROR, "Cannot flush file: " + item->destFilePath(), mdtError::Error);
00570       e.setSystemError(errno, strerror(errno));
00571       MDT_ERROR_SET_SRC(e, "mdtFileCopier");
00572       e.commit();
00573       failedCopies += item->srcFileName() + " -> " + item->destFilePath() + "\n\n";
00574     }
00575     // Sync, if needed
00576     if(item->syncAfterCopy()){
00577       if(!sync(destFile)){
00578         // Sync fail is fatal (device full, not available, ...) - cancel copy
00579         failedCopies += item->srcFileName() + " -> " + item->destFilePath() + "\n\n";
00580         fclose(srcFile);
00581         fclose(destFile);
00582         emit(copyEnded(failedCopies));
00583         pvMutex.lock();
00584         qDeleteAll(pvCopiesInProcess);
00585         pvCopiesInProcess.clear();
00586         pvMutex.unlock();
00587         delete item;
00588         return;
00589       }
00590     }
00591     // Close opened files
00592     if(fclose(srcFile) != 0){
00593       mdtError e(MDT_FILE_IO_ERROR, "Cannot close file: " + item->srcFilePath(), mdtError::Error);
00594       e.setSystemError(errno, strerror(errno));
00595       MDT_ERROR_SET_SRC(e, "mdtFileCopier");
00596       e.commit();
00597     }
00598     srcFile = 0;
