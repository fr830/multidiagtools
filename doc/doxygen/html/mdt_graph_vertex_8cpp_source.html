00079 bool mdtGraphVertex::removeAdjacent(const QVariant &key)
00080 {
00081   int i;
00082 
00083   // Search vertex that contains key in adjacency list
00084   for(i = 0; i < pvAdjacencyList.size(); ++i){
00085     Q_ASSERT(pvAdjacencyList.at(i) != 0);
00086     Q_ASSERT(pvAdjacencyList.at(i)->vertex() != 0);
00087     if(pvAdjacencyList.at(i)->vertex()->key() == key){
00088       Q_ASSERT(pvAdjacencyList.at(i)->edgeData() != 0);
00089       if(pvAutoDeleteEdgeData){
00090         delete pvAdjacencyList.at(i)->edgeData();
00091       }
00092       delete pvAdjacencyList.at(i);
00093       pvAdjacencyList.removeAt(i);
00094       return true;
00095     }
00096   }
00097 
00098   return false;
00099 }
