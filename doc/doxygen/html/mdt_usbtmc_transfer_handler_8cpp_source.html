54  std::lock_guard<std::mutex> lelg(pvLastErrorMutex);

64  * - 1 bulk OUT endpoint
65  * - 1 bulk IN endpoint
66  * - Optionnaly, 1 interrupt IN endpoint
67  */

70  std::lock_guard<std::mutex> lelg(pvLastErrorMutex);

75  pvLastError.commit();
76  return false;
77  }
78  pvBulkOutDescriptor = pvInterfaceDescriptor.bulkOutEndpoints().at(0);
79  // Get bulk IN endpoint descriptor
80  if(pvInterfaceDescriptor.bulkInEndpoints().size() != 1){
81  std::lock_guard<std::mutex> lelg(pvLastErrorMutex);
82  QString msg = deviceIdString();
83  msg += QObject::tr("interface ") + QString::number(bInterfaceNumber) + QObject::tr(" has not exactly 1 bulk IN endpoint (not USBTMC compliant).");
84  pvLastError.setError(msg, mdtError::Error);
85  MDT_ERROR_SET_SRC(pvLastError, "mdtUsbtmcTransferHandler");

87  return false;
88  }
89  pvBulkInDescriptor = pvInterfaceDescriptor.bulkInEndpoints().at(0);
90  // Setup control endpoint
91  if(!pvControlTransferPool.allocTransfers(3, *this, handle)){
92  std::lock_guard<std::mutex> lelg(pvLastErrorMutex);
93  QString msg = deviceIdString();
94  msg += QObject::tr("Allocation of control transfer pool failed.");
95  pvLastError.setError(msg, mdtError::Error);
96  MDT_ERROR_SET_SRC(pvLastError, "mdtUsbtmcTransferHandler");
97  pvLastError.commit();
98  return false;
99  }
100  // Setup bulk-OUT endpoint

102  std::lock_guard<std::mutex> lelg(pvLastErrorMutex);
103  QString msg = deviceIdString();
104  msg += QObject::tr("Allocation of bulk-OUT transfer pool failed.");
105  pvLastError.setError(msg, mdtError::Error);
106  MDT_ERROR_SET_SRC(pvLastError, "mdtUsbtmcTransferHandler");
107  pvLastError.commit();
108  return false;

112  std::lock_guard<std::mutex> lelg(pvLastErrorMutex);

116  MDT_ERROR_SET_SRC(pvLastError, "mdtUsbtmcTransferHandler");
117  pvLastError.commit();
118  return false;

122  pvInterruptInDescriptor = pvInterfaceDescriptor.interruptInEndpoints().at(0);

151 bool mdtUsbtmcTransferHandler::submitCommand(mdtUsbtmcTxMessage & message, bool responseExpected, unsigned int timeout)
152 {
153  // Check if we can accept command now
154  if(!pvStateMachine->is_flag_active<ProcessingFlag>()){
155  std::lock_guard<std::mutex> lelg(pvLastErrorMutex);
156  QString msg = deviceIdString() + QObject::tr("Cannot send a command in a non Processing state.");
157  pvLastError.setError(msg, mdtError::Warning);
158  MDT_ERROR_SET_SRC(pvLastError, "mdtUsbtmcTransferHandler");
159  pvLastError.commit();
160  return false;
161  }
162  // Lock bulk-OUT transfer pool
163  std::lock_guard<std::mutex> bolg(pvBulkOutTransferMutex);
164  qDebug() << "submitCommand() - Queues: control: " << pvControlTransferPool.pendingTransferCount() << " , bulk-OUT: " << pvBulkOutTransferPool.pendingTransferCount() << " , bulk-IN: " << pvBulkInTransferPool.pendingTransferCount();
165  /*
166  * As metionned in USBTMC 1.0 , section 3.2, rule 3
167  * we cannot send a new USBTMC bulk-OUT transfer
168  * if one is pending
169  */
170  if(pvBulkOutTransferPool.pendingTransferCount() > 0){
171  std::lock_guard<std::mutex> lelg(pvLastErrorMutex);
172  QString msg = deviceIdString() + QObject::tr("Cannot send a command while a bulk-OUT transfer is pending.");
173  pvLastError.setError(msg, mdtError::Error);
174  MDT_ERROR_SET_SRC(pvLastError, "mdtUsbtmcTransferHandler");
175  pvLastError.commit();
176  return false;
177  }
178  /*
179  * As metionned in USBTMC 1.0 , section 3.3.2.4, table 12,
180  * we cannot send a new USBTMC bulk-OUT transfer that expects a response
181  * if a Bulk-IN transfer is pending
182  */
183  if(responseExpected){
184  std::lock_guard<std::mutex> bilg(pvBulkInTransferMutex);
185  if(pvBulkInTransferPool.pendingTransferCount() > 0){
186  QString msg = deviceIdString() + QObject::tr("Cannot send a command expecting a response while a bulk-IN transfer is pending.");
187  pvLastError.setError(msg, mdtError::Error);
188  MDT_ERROR_SET_SRC(pvLastError, "mdtUsbtmcTransferHandler");
189  pvLastError.commit();
190  return false;
191  }
192  }
193  /*
194  * Get a transfer from pool, setup bulk-OUT transfer and submit it
195  */
196  mdtUsbtmcBulkTransfer *transfer = pvBulkOutTransferPool.getTransfer(*this, pvBulkOutDescriptor);
197  Q_ASSERT(transfer != 0);
198  transfer->setupDevDepMsgOut(nextBulkOutbTag(), message, responseExpected, timeout);
199  if(!transfer->submit()){
200  ErrorOccuredEvent errorEvent(transfer->lastError());
201  pvStateMachine->process_event(errorEvent);
202  return false;
203  }
204 
205  return true;

460 void mdtUsbtmcTransferHandler::handleControlTransferComplete(mdtUsbtmcControlTransfer *transfer )
461 {
462  Q_ASSERT(transfer != 0);
463 
464  qDebug() << "Control transfer DONE - Control Queue: " << pvControlTransferPool.pendingTransferCount() << " - transfer status: " << transfer->status() << " - length: " << transfer->actualLength();
465 
466  // Check transfer status
467  switch(transfer->status()){
468  case LIBUSB_TRANSFER_COMPLETED: // OK, not a error
469  break;
470  case LIBUSB_TRANSFER_CANCELLED:
471  restoreControlTransferToPool(transfer, true);
472  return;
473  case LIBUSB_TRANSFER_STALL:
474  // We sent a query that is not supported by device.
475  {
476  QString msg = deviceIdString() + QObject::tr("control query with bRequest '") + QString::number(transfer->bRequest()) + "' ";
477  msg += QObject::tr("is not supported by device.");
478  mdtError error(msg, mdtError::Error);
479  MDT_ERROR_SET_SRC(error, "mdtUsbtmcTransferHandler");
480  ErrorOccuredEvent errorEvent(error);
481  pvStateMachine->process_event(errorEvent);
482  }
483  return;
484  case LIBUSB_TRANSFER_NO_DEVICE:
485  pvStateMachine->process_event(DeviceDisconnectedEvent());
486  return;
487  case LIBUSB_TRANSFER_TIMED_OUT:
488  case LIBUSB_TRANSFER_OVERFLOW:
489  case LIBUSB_TRANSFER_ERROR:
490  {
491  QString msg = deviceIdString() + QObject::tr("Control transfer completed with a unhandled status.");
492  mdtError error(msg, mdtError::Error);
493  error.setSystemError(transfer->status(), libusbTransferStatusText(transfer->status()));
494  MDT_ERROR_SET_SRC(error, "mdtUsbtmcTransferHandler");
495  ErrorOccuredEvent errorEvent(error);
496  pvStateMachine->process_event(errorEvent);
497  return;
498  }
499  }
500  // See if transfer contains a response that must be handled
501  if(transfer->actualLength() > 0){
502  switch(transfer->usbtmcRequest()){
503  case mdtUsbtmcControlTransfer::USBTMCbRequest::INITIATE_ABORT_BULK_OUT:
504  handleInitiateAbortBulkOutResponse(transfer);
505  return;
506  case mdtUsbtmcControlTransfer::USBTMCbRequest::CHECK_ABORT_BULK_OUT_STATUS:
507  handleCheckAbortBulkOutStatusResponse(transfer);
508  return;
509  case mdtUsbtmcControlTransfer::USBTMCbRequest::INITIATE_ABORT_BULK_IN:
510  handleInitiateAbortBulkInResponse(transfer);
511  return;
512  case mdtUsbtmcControlTransfer::USBTMCbRequest::CHECK_ABORT_BULK_IN_STATUS:
513  handleCheckAbortBulkInStatusResponse(transfer);
514  return;
515  case mdtUsbtmcControlTransfer::USBTMCbRequest::INITIATE_CLEAR:
516  handleInitiateClearResponse(transfer);
517  return;
518  case mdtUsbtmcControlTransfer::USBTMCbRequest::CHECK_CLEAR_STATUS:
519  handleCheckClearStatusResponse(transfer);
520  return;
521  default:
522  // We received a response that we don't handle
523  QString msg = deviceIdString() + QObject::tr("received a unhandled control response:");
524  msg += " " + transfer->usbtmcRequestText() + " . ";
525  msg += QObject::tr("Will be ignored.");
526  mdtError error(msg, mdtError::Error);
527  MDT_ERROR_SET_SRC(error, "mdtUsbtmcTransferHandler");
528  ErrorOccuredEvent errorEvent(error);
529  pvStateMachine->process_event(errorEvent);
530  return;
531  }
532  }
533  // Nothing more - Restore transfer to pool and signal
534  restoreControlTransferToPool(transfer, true);
535  pvStateMachine->process_event(ControlTransferSuccessfulEvent());
536 }
