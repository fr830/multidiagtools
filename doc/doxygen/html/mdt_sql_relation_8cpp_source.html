00051 bool mdtSqlRelation::addRelation(const QString &parentFieldName, const QString &childFieldName)
00052 {
00053   Q_ASSERT(pvParentModel != 0);
00054   Q_ASSERT(pvChildModel != 0);
00055 
00056   QSqlRecord record;
00057   int parentFieldIndex;
00058   QSqlField parentField;
00059   int childFieldIndex;
00060   QSqlField childField;
00061 
00062   // Get parent  field index, check that it exist
00063   record = pvParentModel->record();
00064   parentFieldIndex = record.indexOf(parentFieldName);
00065   if(parentFieldIndex < 0){
00066     mdtError e(MDT_DATABASE_ERROR, "Field '" + parentFieldName + "' not found in parent table", mdtError::Error);
00067     MDT_ERROR_SET_SRC(e, "mdtSqlRelation");
00068     e.commit();
00069     return false;
00070   }
00071   parentField = record.field(parentFieldIndex);
00072   // Get child field index, check that it exist
00073   record = pvChildModel->record();
00074   childFieldIndex = record.indexOf(childFieldName);
00075   if(childFieldIndex < 0){
00076     mdtError e(MDT_DATABASE_ERROR, "Field '" + childFieldName + "' not found in child table", mdtError::Error);
00077     MDT_ERROR_SET_SRC(e, "mdtSqlRelation");
00078     e.commit();
00079     return false;
00080   }
00081   childField = record.field(childFieldIndex);
00082   // Add relation informations
00083   mdtSqlRelationItem *item = new mdtSqlRelationItem;
00084   item->setParentField(parentField);
00085   item->setParentFieldIndex(parentFieldIndex);
00086   item->setChildField(childField);
00087   item->setChildFieldIndex(childFieldIndex);
00088   pvRelations.append(item);
00089 
00090   return true;
00091 }

00130 void mdtSqlRelation::generateChildModelRelationFilter(int row)
00131 {
00132   Q_ASSERT(pvParentModel != 0);
00133   Q_ASSERT(pvChildModel != 0);
00134   Q_ASSERT(row >= 0);
00135 
00136   int i;
00137   mdtSqlRelationItem *item;
00138   QSqlRecord record;
00139   QVariant data;
00140 
00141   // Get data record of the parent model
00142   record = pvParentModel->record(row);
00143   // Build filter
00144   pvChildModelRelationFilter.clear();
00145   for(i=0; i<pvRelations.size(); ++i){
00146     item = pvRelations.at(i);
00147     Q_ASSERT(item != 0);
00148     // Get parent model's data
00149     data = record.value(item->parentFieldIndex());
00150     if(!data.isValid()){
00151       mdtError e(MDT_DATABASE_ERROR, "Could not get data for field '" + item->parentFieldName() + "' not found in parent table", mdtError::Error);
00152       MDT_ERROR_SET_SRC(e, "mdtSqlRelation");
00153       e.commit();
00154       return;
00155     }
00156     if(i>0){
00157       pvChildModelRelationFilter += " AND";
00158     }
00159     pvChildModelRelationFilter += " \"" + pvChildModel->tableName() + "\".\"" + item->childFieldName() + "\"=";
00160     pvChildModelRelationFilter += item->dataProtection() + data.toString() + item->dataProtection();
00161   }
00162   // Apply filter
00163   generateChildModelFilter();
00164   pvChildModel->setFilter(pvChildModelFilter);
00165 }
