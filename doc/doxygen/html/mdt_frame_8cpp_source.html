41 int mdtFrame::putUntil(const char *data, char token, int maxLen)
42 {
43  Q_ASSERT(data != 0);
44  Q_ASSERT(data != QByteArray::data());
45 
46  // Get max size that can be stored
47  if(maxLen > remainCapacity()){
48  maxLen = remainCapacity();
49  }
50  // Init cursors
51  char *cursor = (char*)data;
52  char *end = cursor + maxLen;
53  int stored = 0;
54 
55  // If EOF condition was reached, or frame is full, we do nothing
56  if(isComplete()||isFull()){
57  return 0;
58  }
59 
60  // We not want to check ignoreNullValues each byte, so we make 2 loops
61  if(pvIgnoreNullValues){
62  while(cursor < end){
63  if(*cursor == token){
64  pvEOFcondition = true;
65  return stored;
66  }
67  if(*cursor != 0){
68  append(*cursor);
69  ++stored;
70  }
71  ++cursor;
72  }
73  }else{
74  while(cursor < end){
75  if(*cursor == token){
76  pvEOFcondition = true;
77  return (cursor - data);
78  }
79  append(*cursor);
80  ++cursor;
81  ++stored;
82  }
83  }
84 
85  return stored;
86 }
