00105       }else{
00106         bufferCursor = frame->data();
00107         Q_ASSERT(bufferCursor < (frame->data() + frame->size()));

00110       }
00111     }
00112   }
00113   // Here, frame is completly sent
00114   Q_ASSERT(frame->isEmpty());
