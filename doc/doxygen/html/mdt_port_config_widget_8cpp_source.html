88 void mdtPortConfigWidget::updateConfig(mdtPortConfig &config)
89 {
90  int index;
91 
92  // Frame type
93  index = cbFrameType->currentIndex();
94  if(index >= 0){
95  switch(cbFrameType->itemData(index).toInt()){
96  case mdtFrame::FT_RAW:
97  config.setFrameType(mdtFrame::FT_RAW);
98  break;
99  case mdtFrame::FT_RAW_TOP:
100  config.setFrameType(mdtFrame::FT_RAW_TOP);

102  case mdtFrame::FT_ASCII:
103  config.setFrameType(mdtFrame::FT_ASCII);
104  break;
105  case mdtFrame::FT_MODBUS_TCP:
106  config.setFrameType(mdtFrame::FT_MODBUS_TCP);
107  break;
108  }
109  }
110  // End of frame sequence - Only for ASCII frame type
111  if(config.frameType() == mdtFrame::FT_ASCII){
112  index = cbEndOfFrameSeq->currentIndex();
113  if(index > -1){
114  config.setEndOfFrameSeq(cbEndOfFrameSeq->itemData(index).toByteArray());
115  }
116  }else{
117  config.setEndOfFrameSeq("");
118  }
119  // Read options
120  config.setReadFrameSize(sbReadFrameSize->value());
121  config.setReadQueueSize(sbReadQueueSize->value());
122  config.setReadTimeout(sbReadTimeout->value());
123  config.setUseReadTimeoutProtocol(cbUseReadTimeoutProtocol->isChecked());
124  // Write options
125  config.setWriteFrameSize(sbWriteFrameSize->value());
126  config.setWriteQueueSize(sbWriteQueueSize->value());
127  config.setWriteTimeout(sbWriteTimeout->value());
128  config.setBytePerByteWrite(cbBytePerByteWrite->isChecked(), sbWriteInterbyteWaitTime->value());
129  config.setWriteInterframeTime(sbWriteInterframeTime->value());
130 
131  // Display real applied setup
