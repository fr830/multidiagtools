00104 void mdtPortTerm::sendCmd()
00105 {
00106   QString cmd;
00107   bool cmdIsQuery = false;
00108 
00109   if(!pvRunning){
00110     return;
00111   }
00112   if(pvCurrentPortManager == 0){
00113     qDebug() << "TERM: err, pvCurrentPortManager == 0";
00114     return;
00115   }
00116   if(leCmd->text().size() < 1){
00117     return;
00118   }
00121   cmd = leCmd->text();
00122   //cmd.remove('\n');
00124   if(cmd.trimmed().right(1) == "?"){
00125     cmdIsQuery = true;
00126   }
00127   cmd.append('\n');
00128   //cmd.append((char)0x04);
00130   // Wait until write is possible and write
00131   if(!pvCurrentPortManager->waitOnWriteReady(1000)){
00132     qDebug() << "TERM: cannot write for the moment";
00133     return;
00134   }
00135   if(pvCurrentPortManager->writeData(cmd.toAscii()) < 0){
00136     qDebug() << "TERM: write error";
00137     return;
00138   }
00139   // If we have a query for USBTMC port, send the read request
00140   if((pvCurrentPortManager == pvUsbtmcPortManager)&&(cmdIsQuery)){
00141     if(!pvCurrentPortManager->waitOnWriteReady(1000)){
00142       qDebug() << "TERM: cannot write for the moment";
00143       return;
00144     }
00145     if(pvUsbtmcPortManager->sendReadRequest(false) < 0){
00146       qDebug() << "TERM: cannot send read request";
00147       return;
00148     }
00149   }
00150   leCmd->clear();
00151 }
