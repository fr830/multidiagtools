00075 {
00076   QMap<QString, QString>::const_iterator it = avaliableTranslations.constBegin();
00077 
00078   // Create a action group
00079   if(pvLanguageActionGroup == 0){
00080     pvLanguageActionGroup = new QActionGroup(this);
00081     connect(pvLanguageActionGroup, SIGNAL(triggered(QAction*)), mdtApp, SLOT(changeLanguage(QAction*)));
00082   }
00083   // Travel available translation and add actions to menu + group
00084   while(it != avaliableTranslations.constEnd()){
00085     QAction *action = new QAction(it.value(), this);
00086     action->setCheckable(true);
00087     action->setData(it.key());
00088     if(it.key() == currentTranslationKey){
00089       action->setChecked(true);
00090     }
00091     menu_Language->addAction(action);
00092     pvLanguageActionGroup->addAction(action);
00093     it++;
00094   }
