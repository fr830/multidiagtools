48  Q_ASSERT(pvValues.at(0).type() == QVariant::String);
49  // Decode <function>

51  if(pvNodes.size() < 1){

58  str = pvNodes.at(0);
59  if(str.left(4) == "VOLT"){
60  pvMeasureType = MT_VOLTAGE_DC;
61  }else if(str.left(4) == "CURR"){
62  pvMeasureType = MT_CURRENT_DC;

67  }else if(str.left(4) == "LRES"){

70  pvMeasureType = MT_CAPACITANCE;
71  }else if(str.left(4) == "DIOD"){
72  pvMeasureType = MT_DIODE;
73  }else if(str.left(4) == "FREQ"){
74  pvMeasureType = MT_FREQUENCY;
75  }else if(str.left(4) == "PWID"){
76  pvMeasureType = MT_PWIDTH;
77  }else if(str.left(4) == "DCYC"){
78  pvMeasureType = MT_DCYCLE;
79  }else{
80  pvMeasureType = MT_UNKNOW;
81  }
82  // Get optional sub types (AC)
83  if(pvNodes.size() > 1){
84  str = pvNodes.at(1);
85  switch(pvMeasureType){
86  case MT_VOLTAGE_DC:
87  if(str.left(2) == "AC"){

98  }
99  }
