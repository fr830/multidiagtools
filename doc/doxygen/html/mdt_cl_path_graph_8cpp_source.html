82 bool mdtClPathGraph::loadLinkList()
83 {
84  QString sql;
85  QSqlError sqlError;
86  int row;
87  QModelIndex index;
88  QVariant data;
89  mdtClPathGraphEdgeData edgeData;
90  vertex_t startVertex, endVertex;
91  edge_t edge;
92  int startConnectionId, endConnectionId;
93 
94  // Clear previous results
95  pvGraph.clear();
96  pvGraphVertices.clear();
97  // Run query to get start and end connection IDs
98  sql = "SELECT UnitConnectionStart_Id_FK, UnitConnectionEnd_Id_FK, LinkDirection_Code_FK FROM LinkList_view";
99  pvLinkListModel->setQuery(sql, pvDatabase);
100  sqlError = pvLinkListModel->lastError();
101  if(sqlError.isValid()){
102  pvLastError.setError(QObject::tr("Cannot load link list from database."), mdtError::Error);
103  pvLastError.setInformativeText(QObject::tr("Plese see details for more informations."));
104  pvLastError.setSystemError(sqlError.number(), sqlError.text());
105  MDT_ERROR_SET_SRC(pvLastError, "mdtClPathGraph");
106  pvLastError.commit();
107  return false;
108  }
109  // Build the graph
110  for(row = 0; row < pvLinkListModel->rowCount(); ++row){
111  // Get start connection ID
112  index = pvLinkListModel->index(row, 0);
113  data = pvLinkListModel->data(index);
114  if(data.isNull()){
115  pvLastError.setError(QObject::tr("Cannot load link list from database."), mdtError::Error);
116  pvLastError.setInformativeText(QObject::tr("Plese see details for more informations."));
117  pvLastError.setSystemError(-1, QObject::tr("UnitConnectionStart_Id_FK returned a NULL value."));
118  MDT_ERROR_SET_SRC(pvLastError, "mdtClPathGraph");
119  pvLastError.commit();
120  return false;
121  }
122  startConnectionId = data.toInt();
123  // Get start vertex or create one if not allready exists
124  if(pvGraphVertices.contains(startConnectionId)){
125  startVertex = pvGraphVertices.value(startConnectionId);
126  }else{
127  startVertex = boost::add_vertex(pvGraph);
128  pvGraphVertices.insert(startConnectionId, startVertex);
129  }
130  // Set start ID to edge data
131  edgeData.startConnectionId = data;
132  // Get end connection ID
133  index = pvLinkListModel->index(row, 1);
134  data = pvLinkListModel->data(index);
135  if(data.isNull()){
136  pvLastError.setError(QObject::tr("Cannot load link list from database."), mdtError::Error);
137  pvLastError.setInformativeText(QObject::tr("Plese see details for more informations."));
138  pvLastError.setSystemError(-1, QObject::tr("UnitConnectionEnd_Id_FK returned a NULL value."));
139  MDT_ERROR_SET_SRC(pvLastError, "mdtClPathGraph");
140  pvLastError.commit();
141  return false;
142  }
143  endConnectionId = data.toInt();
144  // Get end vertex or create one if not allready exists
145  if(pvGraphVertices.contains(endConnectionId)){
146  endVertex = pvGraphVertices.value(endConnectionId);
147  }else{
148  endVertex = boost::add_vertex(pvGraph);
149  pvGraphVertices.insert(endConnectionId, endVertex);
150  }
151  // Set end connection ID to edge data
152  edgeData.endConnectionId = data;
153  // Add edge
154  edgeData.isComplement = false;
155  boost::add_edge(startVertex, endVertex, edgeData, pvGraph);
156  // Add complement edge if link is bidirectinnal
157  index = pvLinkListModel->index(row, 2);
158  data = pvLinkListModel->data(index);
159  if(data.toString() == "BID"){
160  edgeData.isComplement = true;
161  boost::add_edge(endVertex, startVertex, edgeData, pvGraph);
162  }
163  // Fetch more data if possible
164  if((row > 0)&&(pvLinkListModel->canFetchMore())){
165  pvLinkListModel->fetchMore();
166  }
167  }
168 
169  return true;
170 }

172 QList<QVariant> mdtClPathGraph::getLinkedConnectionIdList(const QVariant & fromConnectionId)
173 {
174  QList<QVariant> connectionIdList;
175  mdtClPathGraphEdgeData edge;
176  mdtClPathGraphVisitor visitor(&pvEdgeQueue);
177  vertex_t vertex;
178 
179  // Check if we have requested connection ID in the graph
180  if(!pvGraphVertices.contains(fromConnectionId.toInt())){
181  pvLastError.setError(QObject::tr("Cannot get list of linked connections."), mdtError::Error);
182  pvLastError.setInformativeText(QObject::tr("Plese see details for more informations."));
183  pvLastError.setSystemError(-1, QObject::tr("Start connection ID not found") + " (ID: " + fromConnectionId.toString() + ").");
184  MDT_ERROR_SET_SRC(pvLastError, "mdtClPathGraph");
185  pvLastError.commit();
186  return connectionIdList;
187  }
188  vertex = pvGraphVertices.value(fromConnectionId.toInt());
189  // Clear previous results
190  pvEdgeQueue.clear();
191  // Proceed BFS
192  breadth_first_search(pvGraph, vertex, boost::visitor(visitor));
193  // Get connections
194  while(!pvEdgeQueue.isEmpty()){
195  edge = pvEdgeQueue.dequeue();
196  Q_ASSERT(!edge.startConnectionId.isNull());
197  Q_ASSERT(!edge.endConnectionId.isNull());
198  if(!edge.isComplement){
199  if((edge.startConnectionId != fromConnectionId)&&(!connectionIdList.contains(edge.startConnectionId))){
200  connectionIdList.append(edge.startConnectionId);
201  }
202  if((edge.endConnectionId != fromConnectionId)&&(!connectionIdList.contains(edge.endConnectionId))){
203  connectionIdList.append(edge.endConnectionId);
204  }
205  }
206  }
207 
208  return connectionIdList;
209 }

211 bool mdtClPathGraph::drawPath(const QVariant & fromConnectionId)
212 {
213  mdtClPathGraphEdgeData edge;
214  mdtClPathGraphicsConnection *startConnection, *endConnection;
215  mdtClPathGraphicsLink *link;
216  int startConnectionId, endConnectionId;
217  mdtClPathGraphVisitor visitor(&pvEdgeQueue);
218  vertex_t vertex;
219 
220  // Check if we have requested connection ID in the graph
221  if(!pvGraphVertices.contains(fromConnectionId.toInt())){
222  pvLastError.setError(QObject::tr("Cannot draw requested path."), mdtError::Error);
223  pvLastError.setInformativeText(QObject::tr("Plese see details for more informations."));
224  pvLastError.setSystemError(-1, QObject::tr("Start connection ID not found") + " (ID: " + fromConnectionId.toString() + ").");
225  MDT_ERROR_SET_SRC(pvLastError, "mdtClPathGraph");
226  pvLastError.commit();
227  return false;
228  }
229  vertex = pvGraphVertices.value(fromConnectionId.toInt());
230  // Clear previous results
231  pvEdgeQueue.clear();
232  pvDrawnConnections.clear();
233  pvGraphicsScene->clear();
234  // Proceed BFS
235  breadth_first_search(pvGraph, vertex, boost::visitor(visitor));
236  // draw ...
237  while(!pvEdgeQueue.isEmpty()){
238  edge = pvEdgeQueue.dequeue();
239  Q_ASSERT(!edge.startConnectionId.isNull());
240  Q_ASSERT(!edge.endConnectionId.isNull());
241  if(!edge.isComplement){
242  // Try to get startConnection
243  startConnectionId = edge.startConnectionId.toInt();
244  startConnection = pvDrawnConnections.value(startConnectionId, 0);
245  // Try to get end connection
246  endConnectionId = edge.endConnectionId.toInt();
247  endConnection = pvDrawnConnections.value(endConnectionId, 0);
248  // Create missing connections
249  if((startConnection == 0)&&(endConnection == 0)){
250  startConnection = newConnectionItem(startConnectionId, 0, false);
251  endConnection = newConnectionItem(endConnectionId, startConnection, false);
252  }else if(startConnection == 0){
253  Q_ASSERT(endConnection != 0);
254  startConnection = newConnectionItem(startConnectionId, endConnection, false);
255  }else{
256  Q_ASSERT(startConnection != 0);
257  endConnection = newConnectionItem(endConnectionId, startConnection, false);
258  }
259  Q_ASSERT(startConnection != 0);
260  Q_ASSERT(endConnection != 0);
261  // Draw link
262  link = new mdtClPathGraphicsLink(startConnection, endConnection);
263  pvGraphicsScene->addItem(link);
264  // Set connections and link data
265  setGraphicsItemsData(startConnection, endConnection, link, startConnectionId, endConnectionId);
266  }
267  }
268 
269  return true;
270 }

309 bool mdtClPathGraph::setGraphicsItemsData(mdtClPathGraphicsConnection *startConnection, mdtClPathGraphicsConnection *endConnection, mdtClPathGraphicsLink *link, int startConnectionId, int endConnectionId)
310 {
311  Q_ASSERT(startConnection != 0);
312  Q_ASSERT(endConnection != 0);
313  Q_ASSERT(link != 0);
314 
315  QString sql;
316  QSqlError sqlError;
317  QVariant data;
318  QSqlQuery query(pvDatabase);
319  QSqlRecord record;
320  QString str;
321 
322  sql = "SELECT StartVehicleType, StartVehicleSubType, StartCabinet, StartSchemaPosition, StartUnitConnectorName, StartUnitContactName, StartConnectionType_Code_FK, "\
323  "EndVehicleType, EndVehicleSubType, EndCabinet, EndSchemaPosition, EndUnitConnectorName, EndUnitContactName, EndConnectionType_Code_FK, Identification, "\
324  "LinkType_Code_FK "\
325  "FROM LinkList_view ";
326  sql += " WHERE UnitConnectionStart_Id_FK = " + QString::number(startConnectionId);
327  sql += " AND UnitConnectionEnd_Id_FK = " + QString::number(endConnectionId);
328  if(!query.exec(sql)){
329  sqlError = query.lastError();
330  pvLastError.setError(QObject::tr("Cannot get connections informations from database."), mdtError::Error);
331  pvLastError.setInformativeText(QObject::tr("Plese see details for more informations."));
332  pvLastError.setSystemError(sqlError.number(), sqlError.text());
333  MDT_ERROR_SET_SRC(pvLastError, "mdtClPathGraph");
334  pvLastError.commit();
335  startConnection->setText("??");
336  endConnection->setText("??");
337  return false;
338  }
339  query.next();
340  record = query.record();
341  // Set start data
342  str = record.value("StartVehicleType").toString() + "\n";
343  str += record.value("StartVehicleSubType").toString() + "\n";
344  str += record.value("StartCabinet").toString() + "\n";
345  str += record.value("StartSchemaPosition").toString() + "\n";
346  str += record.value("StartUnitConnectorName").toString() + ";";
347  str += record.value("StartUnitContactName").toString();
348  startConnection->setText(str);
349  // Set start connection type
350  str = record.value("StartConnectionType_Code_FK").toString();
351  if(str == "P"){
352  startConnection->setConnectionType(mdtClPathGraphicsConnection::Pin);
353  }else if(str == "S"){
354  startConnection->setConnectionType(mdtClPathGraphicsConnection::Socket);
355  }else{
356  startConnection->setConnectionType(mdtClPathGraphicsConnection::Terminal);
357  }
358  // Set end data
359  str = record.value("EndVehicleType").toString() + "\n";
360  str += record.value("EndVehicleSubType").toString() + "\n";
361  str += record.value("EndCabinet").toString() + "\n";
362  str += record.value("EndSchemaPosition").toString() + "\n";
363  str += record.value("EndUnitConnectorName").toString() + ";";
364  str += record.value("EndUnitContactName").toString();
365  endConnection->setText(str);
366  // Set end connection type
367  str = record.value("EndConnectionType_Code_FK").toString();
368  if(str == "P"){
369  endConnection->setConnectionType(mdtClPathGraphicsConnection::Pin);
370  }else if(str == "S"){
371  endConnection->setConnectionType(mdtClPathGraphicsConnection::Socket);
372  }else{
373  endConnection->setConnectionType(mdtClPathGraphicsConnection::Terminal);
374  }
375  // Set link data
376  str = record.value("Identification").toString();
377  link->setText(str);
378  str = record.value("LinkType_Code_FK").toString();
379  if(str == "CONNECTION"){
380  link->setLinkType(mdtClPathGraphicsLink::Connection);
381  }else{
382  link->setLinkType(mdtClPathGraphicsLink::CableLink);
383  }
384 
385  return true;
