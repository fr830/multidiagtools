50 void mdtPortManagerStateMachine::buildMainStateMachine(mdtPortManager *portManager)
51 {
52  Q_ASSERT(portManager != 0);
53 
54  // Connect state changed signals
55  connect(this, SIGNAL(stateChanged(int)), portManager, SIGNAL(stateChanged(int)));
56  connect(this, SIGNAL(stateChangedForUi(int, const QString&, int, bool)), portManager, SIGNAL(stateChangedForUi(int, const QString&, int, bool)));
57  // Create states
58  pvPortClosed = new mdtState(mdtPortManager::PortClosed);
59  pvPortClosed->setText(tr("Port closed"));
60  pvPortClosed->setLedColorId(mdtLed::Green);
61  pvPortClosed->setLedOn(false);
62  pvPortClosed->setNotifyEnteredToUi(true);
63  pvStarting = new mdtState(mdtPortManager::Starting);
64  connect(pvStarting, SIGNAL(entered()), portManager, SLOT(startThreads()));
65  pvRunning = new mdtState(mdtPortManager::Running);
66  pvStopping = new mdtState(mdtPortManager::Stopping);
67  connect(pvStopping, SIGNAL(entered()), portManager, SLOT(stopThreads()));
68  pvStopped = new mdtState(mdtPortManager::Stopped);
69  pvPortError = new mdtState(mdtPortManager::PortError);
70  pvPortError->setText(tr("Fatal error"));
71  pvPortError->setLedColorId(mdtLed::Red);
72  pvPortError->setLedOn(true);
73  pvPortError->setNotifyEnteredToUi(true);
74  connect(pvPortError, SIGNAL(entered()), portManager, SIGNAL(pmStopThreadsEvent()));
75  // Add sates to machine
76  addState(pvPortClosed);
77  addState(pvStarting);
78  addState(pvRunning);
79  addState(pvStopping);
80  addState(pvStopped);
81  addState(pvPortError);
82  // Add pvPortClosed transitions ->
83  pvPortClosed->addTransition(portManager, SIGNAL(pmStartThreadsEvent()), pvStarting);
84  // Add pvStarting transitions ->
85  pvStarting->addTransition(portManager, SIGNAL(pmUnhandledErrorEvent()), pvPortError);
86  pvStarting->addTransition(portManager, SIGNAL(pmAllThreadsReadyEvent()), pvRunning);
87  // Add pvRunning transitions ->
88  pvRunning->addTransition(portManager, SIGNAL(pmStopThreadsEvent()), pvStopping);
89  pvRunning->addTransition(portManager, SIGNAL(pmUnhandledErrorEvent()), pvPortError);
90  pvRunning->addTransition(portManager, SIGNAL(pmConnectionFailedEvent()), pvPortError);
91  // Add pvStopping transitions ->
92  pvStopping->addTransition(portManager, SIGNAL(pmAllThreadsStoppedEvent()), pvStopped);
93  // Add pvStopped transitions ->
94  pvStopped->addTransition(portManager, SIGNAL(pmPortClosedEvent()), pvPortClosed);
95  // Add pvPortError transitions ->
96  pvPortError->addTransition(portManager, SIGNAL(pmAllThreadsStoppedEvent()), pvStopped);
97  pvPortError->addTransition(portManager, SIGNAL(pmStopThreadsEvent()), pvStopping);
98  // Build Running sub machine
99  buildRunningSubMachine(portManager, pvRunning);
100  // Start
101  setInitialState(pvPortClosed);
102  start(true);
103  waitOnState(mdtPortManager::PortClosed);
104 }

121  pvConnecting->setNotifyEnteredToUi(true);
122  pvConnected = new mdtState(mdtPortManager::Connected, parentState);
123  pvDisconnected = new mdtState(mdtPortManager::Disconnected, parentState);
124  pvDisconnected->setText(tr("Disconnected"));
125  pvDisconnected->setLedColorId(mdtLed::Green);
126  pvDisconnected->setLedOn(false);
127  pvDisconnected->setNotifyEnteredToUi(true);
128  // Add pvPortReady transitions ->
129  pvPortReady->addTransition(portManager, SIGNAL(pmConnectingEvent()), pvConnecting);
130  // Add pvConnecting transitions ->
131  pvConnecting->addTransition(portManager, SIGNAL(pmDisconnectedEvent()), pvDisconnected);
132  pvConnecting->addTransition(portManager, SIGNAL(pmConnectedEvent()), pvConnected);
133  // Add pvDisconnected transitions ->
134  pvDisconnected->addTransition(portManager, SIGNAL(pmConnectingEvent()), pvConnecting);
135  // Add pvConnected transitions ->
136  pvConnected->addTransition(portManager, SIGNAL(pmDisconnectedEvent()), pvDisconnected);
137  // Add Global sate transitions and set initial state
138  parentState->addTransition(portManager, SIGNAL(pmConnectingEvent()), pvConnecting);
139  parentState->addTransition(portManager, SIGNAL(pmConnectedEvent()), pvConnected);
140  parentState->setInitialState(pvPortReady);
141  // Build connected sub machine
142  buildConnectedSubMachine(portManager, pvConnected);
