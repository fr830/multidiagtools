30 bool mdtPartitionAttributes::setPath(const QString &path)
31 {
32  char driveName[1024];
33  char fs[256];
34  const char *drive;
35  int flags;
36  QByteArray strPath;
37 
38  // Default values
39  pvRootPath = "";
40  pvName = "";
41  pvIsReadOnly = false;
42  pvFileSystem = "";
43 
44  // Get Windows native format of given path
45  strPath = QDir::toNativeSeparators(path).toLocal8Bit();
46  // See if the terminal \ exists
47  if(strPath.size() == 2){
48  strPath.append('\\');
49  }
50  drive = strPath.data();
51  // Get attributes
52  if(GetVolumeInformation(drive, driveName, 1023, 0, 0, (LPDWORD)&flags, fs, 255) == 0){
53  mdtError e(MDT_FILE_IO_ERROR, "GetVolumeInformation() failed", mdtError::Error);
54  e.setSystemErrorWinApi();
55  MDT_ERROR_SET_SRC(e, "mdtPartitionAttributes");
56  e.commit();
57  return false;
58  }
59  // Get flags
60  if(flags & FILE_READ_ONLY_VOLUME){
61  pvIsReadOnly = true;
62  }else{
63  pvIsReadOnly = false;
64  }
65  // Store attributes
66  pvRootPath = QDir::fromNativeSeparators(strPath);
67  pvName = driveName;
68  pvFileSystem = fs;
69 
70  return true;
71 }
