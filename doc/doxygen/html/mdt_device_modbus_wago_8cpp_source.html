84 bool mdtDeviceModbusWago::isWago750()
85 {
86  // If device is not from Wago, server can return a error (Something like: invalid address range)
87  // We put a information into the log, so that we know that we are scanning.
88  mdtError e1(MDT_DEVICE_ERROR, "Device " + name() + ": checking if device is from Wago ...", mdtError::Info);
89  MDT_ERROR_SET_SRC(e1, "mdtDeviceModbusWago");
90  e1.commit();
91 
92  if(!getRegisterValues(0x2011, 1)){
93  return false;
94  }
95  if(registerValues().size() != 1){
96  return false;
97  }
98  mdtError e2(MDT_DEVICE_ERROR, "Device " + name() + ": device is from Wago", mdtError::Info);
99  MDT_ERROR_SET_SRC(e2, "mdtDeviceModbusWago");
100  e2.commit();
101 
102  return (registerValues().at(0) == 750);
103 }

149 bool mdtDeviceModbusWago::detectIos(const QMap<int, mdtDeviceModbusWagoModule*> specialModules)
150 {
151  int analogInputsCnt;
152  int analogOutputsCnt;
153  int digitalInputsCnt;
154  int digitalOutputsCnt;
155  int i;
156  quint16 word;
157  QVariant var;
158  mdtDeviceModbusWagoModule *module;
159  int aiAddress;
160  int aoAddressRead, aoAddressWrite;
161  int diAddress;
162  int doAddressRead, doAddressWrite;
163 
164  // Clear current I/O setup
165  deleteIos();
166  qDeleteAll(pvModules);
167  pvModules.clear();
168  // Detect I/O's count
169  analogInputsCnt = analogInputsCount();
170  if(analogInputsCnt < 0){
171  mdtError e(MDT_DEVICE_ERROR, "Device " + name() + ": error occured during analog inputs count detection", mdtError::Error);
172  MDT_ERROR_SET_SRC(e, "mdtDeviceModbusWago");
173  e.commit();
174  return false;
175  }
176  analogOutputsCnt = analogOutputsCount();
177  if(analogOutputsCnt < 0){
178  mdtError e(MDT_DEVICE_ERROR, "Device " + name() + ": error occured during analog outputs count detection", mdtError::Error);
179  MDT_ERROR_SET_SRC(e, "mdtDeviceModbusWago");
180  e.commit();
181  return false;
182  }
183  digitalInputsCnt = digitalInputsCount();
184  if(digitalInputsCnt < 0){
185  mdtError e(MDT_DEVICE_ERROR, "Device " + name() + ": error occured during digital inputs count detection", mdtError::Error);
186  MDT_ERROR_SET_SRC(e, "mdtDeviceModbusWago");
187  e.commit();
188  return false;
189  }
190  digitalOutputsCnt = digitalOutputsCount();
191  if(digitalOutputsCnt < 0){
192  mdtError e(MDT_DEVICE_ERROR, "Device " + name() + ": error occured during digital outputs count detection", mdtError::Error);
193  MDT_ERROR_SET_SRC(e, "mdtDeviceModbusWago");
194  e.commit();
195  return false;
196  }
197  // Get modules configuration
199  if(!getRegisterValues(0x2030, 62)){
200  qDebug() << "Cannot get I/O config";
201  return false;
202  }
204  // Setup the Fieldbus Coupler.
205  if(registerValues().size() < 1){
206  mdtError e(MDT_DEVICE_ERROR, "Device " + name() + ": no field bus coupler found", mdtError::Error);
207  MDT_ERROR_SET_SRC(e, "mdtDeviceModbusWago");
208  e.commit();
209  return false;
210  }
211  word = registerValues().at(0);
212  if((word < 1)||(word > 999)){
213  mdtError e(MDT_DEVICE_ERROR, "Device " + name() + ": field bus coupler part number is wrong (expected: value from 1 to 999)", mdtError::Error);
214  MDT_ERROR_SET_SRC(e, "mdtDeviceModbusWago");
215  e.commit();
216  return false;
217  }
218  qDebug() << "Found filedbus part number 750 -" << word;
219  // Setup each module and add them to container
220  if(registerValues().size() < 2){
221  mdtError e(MDT_DEVICE_ERROR, "Device " + name() + ": no I/O modules found", mdtError::Error);
222  MDT_ERROR_SET_SRC(e, "mdtDeviceModbusWago");
223  e.commit();
224  return false;
225  }
227  aiAddress = 0;
228  aoAddressRead = 0x0200;
229  aoAddressWrite = 0;
230  diAddress = 0;
231  doAddressRead = 0x0200;
232  doAddressWrite = 0;
233  for(i = 1; i < registerValues().size(); ++i){
234  word = registerValues().at(i);
235  if(word == 0){
236  // We have detected all connected I/Os
237  break;
238  }
240  // Check if we have a special (manually configured) module
241  module = specialModules.value(i);
242  if(module != 0){
243  // Some special modules needs to communicate with device, so we address a first time
244  module->updateAddresses(aiAddress, aoAddressRead, aoAddressWrite, diAddress, doAddressRead, doAddressWrite);
245  if(!module->setupFromRegisterWord(word)){
246  // We must not free given module, simply set pointer Null
247  module = 0;
248  }
249  // Re-address module
250  module->updateAddresses(aiAddress, aoAddressRead, aoAddressWrite, diAddress, doAddressRead, doAddressWrite);
251  }else{
252  // Create a module object, setup and adrress it
254  if(word == 464){
255  module = new mdtDeviceModbusWagoModuleRtd(false, this);
256  }else{
257  module = new mdtDeviceModbusWagoModule(false, this);
258  }
259  if(!module->setupFromRegisterWord(word)){
260  module->clear(true);
261  delete module;
262  module = 0;
263  }
264  module->updateAddresses(aiAddress, aoAddressRead, aoAddressWrite, diAddress, doAddressRead, doAddressWrite);
265  }
266  // If module is null, a problem occured - cleanup and fail
267  if(module == 0){
268  mdtError e(MDT_DEVICE_ERROR, "Device " + name() + ": setup failed on a module (setup word: " + QString::number(word) + ")", mdtError::Error);
269  MDT_ERROR_SET_SRC(e, "mdtDeviceModbusWago");
270  e.commit();
271  qDeleteAll(pvModules);
272  pvModules.clear();
273  deleteIos();
274  return false;
275  }
276  Q_ASSERT(module != 0);
277  Q_ASSERT(module->type() != mdtDeviceModbusWagoModule::Unknown);
279  // Add module's I/Os to container
280  addInputs(module->analogInputs());
281  addOutputs(module->analogOutputs());
282  addInputs(module->digitalInputs());
283  addOutputs(module->digitalOutputs());
284  // Update current first addresses
285  aiAddress = module->nextModuleFirstAiAddress();
286  aoAddressRead = module->nextModuleFirstAoAddressRead();
287  aoAddressWrite = module->nextModuleFirstAoAddressWrite();
288  diAddress = module->nextModuleFirstDiAddress();
289  doAddressRead = module->nextModuleFirstDoAddressRead();
290  doAddressWrite = module->nextModuleFirstDoAddressWrite();
291  // Add module to modules container
292  pvModules.append(module);
293  }
294  // Check coherence between detected setup and I/Os count
295  if(ios()->analogInputsCount() != analogInputsCnt){
296  qDeleteAll(pvModules);
297  pvModules.clear();
298  deleteIos();
299  mdtError e(MDT_DEVICE_ERROR, "Device " + name() + ": analog inputs count not coherent", mdtError::Error);
300  MDT_ERROR_SET_SRC(e, "mdtDeviceModbusWago");
301  e.commit();
302  return false;
303  }
304  if(ios()->analogOutputsCount() != analogOutputsCnt){
305  qDeleteAll(pvModules);
306  pvModules.clear();
307  deleteIos();
308  mdtError e(MDT_DEVICE_ERROR, "Device " + name() + ": analog outputs count not coherent", mdtError::Error);
309  MDT_ERROR_SET_SRC(e, "mdtDeviceModbusWago");
310  e.commit();
311  return false;
312  }
313  if(ios()->digitalInputsCount() != digitalInputsCnt){
314  qDeleteAll(pvModules);
315  pvModules.clear();
316  deleteIos();
317  mdtError e(MDT_DEVICE_ERROR, "Device " + name() + ": digital inputs count not coherent", mdtError::Error);
318  MDT_ERROR_SET_SRC(e, "mdtDeviceModbusWago");
319  e.commit();
320  return false;
321  }
322  if(ios()->digitalOutputsCount() != digitalOutputsCnt){
323  qDeleteAll(pvModules);
324  pvModules.clear();
325  deleteIos();
326  mdtError e(MDT_DEVICE_ERROR, "Device " + name() + ": digital outputs count not coherent", mdtError::Error);
327  MDT_ERROR_SET_SRC(e, "mdtDeviceModbusWago");
328  e.commit();
329  return false;
330  }
331  // Set a default short label for each I/O
332  ios()->setIosDefaultLabelShort();
333 
334  return true;
335 }
