00038 void mdtUsbInterfaceDescriptor::fetchAttributes(libusb_interface_descriptor *descriptor)
00039 {
00040   Q_ASSERT(descriptor != 0);
00041 
00042   quint8 i;
00043   mdtUsbEndpointDescriptor *endpointDescriptor;
00044 
00045   // Get attributes
00046   pvbDescriptorType = descriptor->bDescriptorType;
00047   pvbInterfaceClass = descriptor->bInterfaceClass;
00048   pvbInterfaceSubClass = descriptor->bInterfaceSubClass;
00049   pvbInterfaceProtocol = descriptor->bInterfaceProtocol;
00050   pviInterface = descriptor->iInterface;
00051   // Get endpoints
00052   Q_ASSERT(descriptor->endpoint != 0);
00053   qDeleteAll(pvEndpoints);
00054   pvEndpoints.clear();
00055   for(i=0; i<descriptor->bNumEndpoints; i++){
00056     endpointDescriptor = new mdtUsbEndpointDescriptor;
00057     Q_ASSERT(endpointDescriptor != 0);
00058     endpointDescriptor->fetchAttributes((libusb_endpoint_descriptor*)&descriptor->endpoint[i]);
00059     pvEndpoints.append(endpointDescriptor);
00060   }
00061 }
