77 {
78  Q_ASSERT(lst.size() >= 2);

80 
81  bool ok;

97  if(!ok){

99  MDT_ERROR_SET_SRC(pvLastError, "mdtDeviceAddress");
100  pvLastError.commit();
101  clear();
102  return false;
103  }
104  // Extract idProduct
105  pvUsbDeviceAddress.idProduct = lst.at(2).toUInt(&ok, 16);
106  if(!ok){
107  pvLastError.setError(tr("Could not decode idProduct in address string '") + pvAddressString + tr("'.") , mdtError::Error);
108  MDT_ERROR_SET_SRC(pvLastError, "mdtDeviceAddress");
109  pvLastError.commit();
110  clear();
111  return false;
112  }
113  // Extract serial number
114  pvSerialNumber = lst.at(3);
115  // Extract bInterfaceNumber if given

117  pvUsbDeviceAddress.bInterfaceNumber = lst.at(4).toUInt(&ok);
118  if(!ok){

120  MDT_ERROR_SET_SRC(pvLastError, "mdtDeviceAddress");

172 bool mdtDeviceAddress::decodeModbusTcpAddressString(const QStringList& lst)
173 {
174  Q_ASSERT(lst.size() >= 4);
175  Q_ASSERT(lst.size() <= 5);
176  Q_ASSERT(lst.at(3) == "MODBUS");
177 
178  bool ok;
179 
180  // Extract host name and port
181  pvHostName = lst.at(1);
182  pvTcpipDeviceAddress.port = lst.at(2).toUInt(&ok);
183  if(!ok){
184  pvLastError.setError(tr("Could not decode port number in address string '") + pvAddressString + tr("'.") , mdtError::Error);
185  MDT_ERROR_SET_SRC(pvLastError, "mdtDeviceAddress");
186  pvLastError.commit();
187  clear();
188  return false;
189  }
190  pvDeviceType = DeviceType_t::MODBUS;
191  if(lst.size() < 5){
192  return true;
193  }
194  // Extract MODBUS hardware node address is specified
195  Q_ASSERT(lst.size() == 5);
196  QStringList nid = lst.at(4).split(",");
197  if(nid.size() != 3){
198  pvLastError.setError(tr("Could not decode MODBUS hardware node ID in address string '") + pvAddressString + \
199  tr("'. Expected 3 parameters (ID,bitsCount,firstBit)") , mdtError::Error);
200  MDT_ERROR_SET_SRC(pvLastError, "mdtDeviceAddress");
201  pvLastError.commit();
202  clear();
203  return false;
204  }
205  Q_ASSERT(nid.size() == 3);
206  // Node ID
207  int id = nid.at(0).toInt(&ok);
208  if(!ok){
209  pvLastError.setError(tr("Could not decode MODBUS hardware node ID in address string '") + pvAddressString + \
210  tr("'. ID part is not a number.") , mdtError::Error);
211  MDT_ERROR_SET_SRC(pvLastError, "mdtDeviceAddress");
212  pvLastError.commit();
213  clear();
214  return false;
215  }
216  // Bits count
217  int bc = nid.at(1).toInt(&ok);
218  if(!ok){
219  pvLastError.setError(tr("Could not decode MODBUS hardware node ID in address string '") + pvAddressString + \
220  tr("'. bitsCount part is not a number.") , mdtError::Error);
221  MDT_ERROR_SET_SRC(pvLastError, "mdtDeviceAddress");
222  pvLastError.commit();
223  clear();
224  return false;
225  }
226  // First bit
227  int fb = nid.at(2).toInt(&ok);
228  if(!ok){
229  pvLastError.setError(tr("Could not decode MODBUS hardware node ID in address string '") + pvAddressString + \
230  tr("'. firstBit part is not a number.") , mdtError::Error);
231  MDT_ERROR_SET_SRC(pvLastError, "mdtDeviceAddress");
232  pvLastError.commit();
233  clear();
234  return false;
235  }
236  // Ok, store node ID
237  pvModbusHwNodeId.setId(id, bc, fb);
238 
239  return true;
240 }

280 
281  // Try to open file

307  file.close();
308 
309  return true;
