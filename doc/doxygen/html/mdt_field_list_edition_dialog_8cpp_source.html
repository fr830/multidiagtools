203 bool mdtFieldListEditionDialog::buildResult()
204 {
205  int i;
206  QTableWidgetItem *twItem;
207  QString fieldName;
208  mdtFieldMapField field;
209  int fieldIndex;
210  QString fieldDisplayText;
211 
212  // Check about dupplicate field names a first time
213  if(haveDupplicateFieldName()){
214  return false;
215  }
216  pvResultFields.clear();
217  for(i = 0; i < twFields->rowCount(); ++i){
218  twItem = twFields->item(i, 1);
219  Q_ASSERT(twItem != 0);
220  fieldName = twItem->data(Qt::DisplayRole).toString();
221  // If field name is empty, we simply fail
222  if(fieldName.trimmed().isEmpty()){
223  QMessageBox msgBox(this);
224  msgBox.setText(tr("There are some fields that have no name."));
225  msgBox.setInformativeText(tr("Please correct these fields and try again."));
226  msgBox.setIcon(QMessageBox::Information);
227  msgBox.exec();
228  pvResultFields.clear();
229  return false;
230  }
231  // Check about illegal strings
232  fieldName = correctFieldName(fieldName);
233  if(fieldName.isEmpty()){
234  pvResultFields.clear();
235  return false;
236  }
237  // Store new name
238  twItem->setData(Qt::DisplayRole, fieldName);
239  // Store to result
240  field = mdtFieldMapField();
241  twItem = twFields->item(i, 0);
242  Q_ASSERT(twItem != 0);
243  fieldIndex = twItem->data(Qt::DisplayRole).toInt();
244  field.setIndex(fieldIndex);
245  field.setName(fieldName);
246  twItem = twFields->item(i, 2);
247  Q_ASSERT(twItem != 0);
248  fieldDisplayText = twItem->data(Qt::DisplayRole).toString();
249  field.setDisplayText(fieldDisplayText);
250  pvResultFields.append(field);
251  }
252  // Check about dupplicate field names a after possible string substitution
253  if(haveDupplicateFieldName()){
254  pvResultFields.clear();
255  return false;
256  }
257 
258  return true;
259 }

312 void mdtFieldListEditionDialog::swapRows(int row1, int row2)
313 {
314  QString row1Data, row2Data;
315  QTableWidgetItem *item1, *item2;
316 
317  // If we have less than 2 rows, nothing can be swapped..
318  if(twFields->rowCount() < 2){
319  return;
320  }
321  // Check row1 range
322  if((row1 < 0)||(row1 >= (twFields->rowCount() - 1))){
323  return;
324  }
325  // Check row2 range
326  if((row2 < 1)||(row2 >= twFields->rowCount())){
327  return;
328  }
329  // Get row1 field name
330  item1 = twFields->item(row1, 1);
331  Q_ASSERT(item1 != 0);
332  row1Data = item1->data(Qt::DisplayRole).toString();
333  // Get row2 field name
334  item2 = twFields->item(row2, 1);
335  Q_ASSERT(item2 != 0);
336  row2Data = item2->data(Qt::DisplayRole).toString();
337  // Swap field names
338  item1->setData(Qt::DisplayRole, row2Data);
339  item2->setData(Qt::DisplayRole, row1Data);
340  // Get row1 field display text
341  item1 = twFields->item(row1, 2);
342  Q_ASSERT(item1 != 0);
343  row1Data = item1->data(Qt::DisplayRole).toString();
344  // Get row2 field display text
345  item2 = twFields->item(row2, 2);
346  Q_ASSERT(item2 != 0);
347  row2Data = item2->data(Qt::DisplayRole).toString();
348  // Swap field display texts
349  item1->setData(Qt::DisplayRole, row2Data);
350  item2->setData(Qt::DisplayRole, row1Data);
351 }
