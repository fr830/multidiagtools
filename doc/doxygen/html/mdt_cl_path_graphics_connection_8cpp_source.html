125 QPointF mdtClPathGraphicsConnection::nextPos(bool reverse, qreal linkLength, qreal dAlpha) const
126 {
127  qreal alpha;
128  qreal x, y;
129  qreal newX, newY;
130  qreal pi = M_PI;
131 
132  if(scene() == 0){
133  return QPointF();
134  }
135  // Get scene position of this item
136  x = mapToScene(pvConnectionRect.center()).x();
137  y = mapToScene(pvConnectionRect.center()).y();
138  // Try to find a point directly on the right (or the left if reverse)
139  newY = y;
140  if(reverse){
141  newX = x - linkLength;
142  if(scene()->itemAt(newX, newY, QTransform()) == 0){
143  // No item here, we found a point
144  return QPointF(newX, newY);
145  }
146  }else{
147  newX = x + linkLength;
148  if(scene()->itemAt(newX, newY, QTransform()) == 0){
149  // No item here, we found a point
150  return QPointF(newX, newY);
151  }
152  }
153  // Try to find a free point around us, alternating with alpha 0, alpha > and alpha < 0
154  alpha = dAlpha;
155  while((alpha > -pi)&&(alpha < pi)){
156  // Try with current alpha
157  newX = x + linkLength * cos(alpha);
158  newY = y + linkLength * sin(alpha);
159  if(scene()->itemAt(newX, newY, QTransform()) == 0){
160  // No item here, we found a point
161  return QPointF(newX, newY);
162  }
163  // Try with next alpha
164  alpha += dAlpha;
165  newX = x + linkLength * cos(alpha);
166  newY = y + linkLength * sin(alpha);
167  if(scene()->itemAt(newX, newY, QTransform()) == 0){
168  // No item here, we found a point
169  return QPointF(newX, newY);
170  }
171  // Try with opposite alpha
172  alpha = -alpha;
173  newX = x + linkLength * cos(alpha);
174  newY = y + linkLength * sin(alpha);
175  if(scene()->itemAt(newX, newY, QTransform()) == 0){
176  // No item here, we found a point
177  return QPointF(newX, newY);
178  }
179  // No free point here, continue
180  alpha = -alpha;
181  alpha += dAlpha;
182  }
183 
184  return QPointF();
185 }
