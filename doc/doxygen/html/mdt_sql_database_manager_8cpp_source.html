00076   }else{

00078       pvLastError.setError(tr("Cannot enable foreign keys contraint, because database type is not supported."), mdtError::Error);
00079     }else{

00081     }
00082     MDT_ERROR_SET_SRC(pvLastError, "mdtSqlDatabaseManager");
00083     pvLastError.commit();
00084     return false;
00085   }
00086   // Exec query
00087   if(!query.exec(sql)){
00088     pvLastError.setError(tr("Cannot enable/disable foreign keys support on database '") + pvDatabaseName + "'", mdtError::Error);
00089     MDT_ERROR_SET_SRC(pvLastError, "mdtSqlDatabaseManager");
