83 void mdtClArticleComponentDialog::selectComponent()
84 {
85  mdtSqlSelectionDialog dialog(this);
86  mdtClArticle art(this, pvDatabase);
87  QVariant data;
88 
89  // Setup and show dialog
90  dialog.setMessage(tr("Please select a article to use as component:"));
91  dialog.setQuery(art.sqlForArticleComponentSelection(pvArticleId), pvDatabase, false);
92  dialog.setColumnHidden("Id_PK", true);
93  dialog.setHeaderData("ArticleCode", tr("Article code"));
94  dialog.setHeaderData("DesignationEN", tr("Designation\n(English)"));
95  dialog.setHeaderData("DesignationFR", tr("Designation\n(French)"));
96  dialog.setHeaderData("DesignationDE", tr("Designation\n(German)"));
97  dialog.setHeaderData("DesignationIT", tr("Designation\n(Italian)"));
98  dialog.addColumnToSortOrder("ArticleCode", Qt::AscendingOrder);
99  dialog.sort();
100  dialog.addSelectionResultColumn("Id_PK");
101  dialog.addSelectionResultColumn("ArticleCode");
102  dialog.addSelectionResultColumn("DesignationEN");
103  dialog.addSelectionResultColumn("Unit");
104  dialog.resize(700, 400);
105  dialog.setWindowTitle(tr("Article component selection"));
106  if(dialog.exec() != QDialog::Accepted){
107  return;
108  }
109  // Store result
110  Q_ASSERT(dialog.selectionResult().size() == 4);
111  pvComponentId = dialog.selectionResult().at(0);
112  // Article code
113  data = dialog.selectionResult().at(1);
114  lbArticleCode->setText(data.toString());
115  // Article designation EN
116  data = dialog.selectionResult().at(2);
117  lbArticleDesignationEN->setText(data.toString());
118  // Qty
119  sbQty->setValue(1.0);
120  // Qty unit
121  data = dialog.selectionResult().at(3);
122  lbUnit->setText(data.toString());
123 }

125 void mdtClArticleComponentDialog::displayCurrentComponent()
126 {
127  QModelIndex index;
128  QVariant data;
129  int row;
130 
131  if(pvComponentId.isNull()){
132  lbArticleCode->clear();
133  lbArticleDesignationEN->clear();
134  sbQty->setValue(0.0);
135  lbUnit->clear();
136  return;
137  }
138  // Search component that match given component ID and display its data
139  for(row = 0; row < pvArticleComponentModel->rowCount(); ++row){
140  index = pvArticleComponentModel->index(row, 1);
141  data = pvArticleComponentModel->data(index);
142  if(data == pvComponentId){
143  // Display article code
144  index = pvArticleComponentModel->index(row, 2);
145  data = pvArticleComponentModel->data(index);
146  lbArticleCode->setText(data.toString());
147  // Display article designation EN
148  index = pvArticleComponentModel->index(row, 3);
149  data = pvArticleComponentModel->data(index);
150  lbArticleDesignationEN->setText(data.toString());
151  // Display Qty
152  index = pvArticleComponentModel->index(row, 4);
153  data = pvArticleComponentModel->data(index);
154  sbQty->setValue(data.toDouble());
155  // Display Qty unit
156  index = pvArticleComponentModel->index(row, 5);
157  data = pvArticleComponentModel->data(index);
158  lbUnit->setText(data.toString());
159  break;
160  }
161  }
