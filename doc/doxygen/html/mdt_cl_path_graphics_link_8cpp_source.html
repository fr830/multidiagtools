118 void mdtClPathGraphicsLink::paint(QPainter *painter, const QStyleOptionGraphicsItem *option, QWidget *widget)
119 {
120  QRectF textRect;
121 
122  if((pvStartConnection == 0)||(pvEndConnection == 0)){
123  return;
124  }
125  QLineF line(pvStartPoint, pvEndPoint);
126  if(qFuzzyCompare(line.length(), qreal(0.0))){
127  return;
128  }
129  textRect.setX(pvStartPoint.x() + (line.dx() - pvTextSize.width()) / 2.0);
130  textRect.setY(pvStartPoint.y() + line.dy() / 2.0 - pvTextSize.height());
131  textRect.setSize(pvTextSize);
132  painter->setPen(QPen(pvLinkColor, 1, Qt::SolidLine, Qt::RoundCap, Qt::RoundJoin));
133  painter->drawLine(line);
134  painter->drawText(textRect, Qt::AlignHCenter | Qt::TextDontClip, pvText);
135 }

137 void mdtClPathGraphicsLink::updateBoundingRect()
138 {
139  qreal penWidth = 1.0;
140  qreal extra = (penWidth + 2.0) / 2.0;
141  qreal width;
142  qreal height;
143  qreal xOffset;
144 
145  width = qAbs(pvEndPoint.x() - pvStartPoint.x());
146  if(width < pvTextSize.width()){
147  xOffset = (pvTextSize.width() - width) / 2.0;
148  width = pvTextSize.width();
149  }else{
150  xOffset = 0.0;
151  }
152  height = qAbs(pvEndPoint.y() - pvStartPoint.y());
153  height = qMax(height, height/2.0 + pvTextSize.height());
154  if(pvStartPoint.x() < pvEndPoint.x()){
155  pvBoundingRect.setX(pvStartPoint.x() - xOffset);
156  }else{
157  pvBoundingRect.setX(pvEndPoint.x() - xOffset);
158  }
159  if(pvStartPoint.y() < pvEndPoint.y()){
160  pvBoundingRect.setY(pvEndPoint.y() - height);
161  }else{
162  pvBoundingRect.setY(pvStartPoint.y() - height);
163  }
164  pvBoundingRect.setWidth(width);
165  pvBoundingRect.setHeight(height);
166  pvBoundingRect = pvBoundingRect.normalized();
167  pvBoundingRect = pvBoundingRect.adjusted(-extra, -extra, extra, extra);
168 }
