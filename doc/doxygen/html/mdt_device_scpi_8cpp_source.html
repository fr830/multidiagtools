118 bool mdtDeviceScpi::waitOperationComplete(int timeout, int interval)
119 {
120  Q_ASSERT(interval > 0);
121  Q_ASSERT(pvOperationCompleteTimer != 0);
122 
124  pvOperationComplete = false;
125  pvOperationCompleteTryLeft = timeout / interval;
126  QByteArray response;
127 
129  // Check one time immediatly
130  response = sendQuery("*OPC?\n");
131  if(response.size() > 0){
132  if(response.at(0) == '1'){
133  return true;
134  }
135  }
136  // Check at regular interval
137  pvOperationCompleteTryLeft--;
138  pvOperationCompleteTimer->setInterval(interval);
139  pvOperationCompleteTimer->start();
140  while(!pvOperationComplete){
141  // Check that port manager is not about to close or in error
148  // Check about operation complete timeout
149  if(pvOperationCompleteTryLeft < 1){
150  return false;
151  }
152  // Wait
153  QCoreApplication::processEvents(QEventLoop::AllEvents | QEventLoop::WaitForMoreEvents);
154  }
155 
156  return true;
157 }
