00123 void mdtDeviceStatusWidget::setState(int state)
00124 {
00125   if(state == mdtDevice::Ready){
00126     ldState->setColor(pvReadyColor);
00127     ldState->setOn();
00128     lbMessage->setText(pvReadyText);
00129   }else if(state == mdtDevice::Disconnected){
00130     ldState->setGreen();
00131     ldState->setOff();
00132     lbMessage->setText(pvDisconnectedText);
00133   }else if(state == mdtDevice::Connecting){
00134     ldState->setColor(pvConnectingColor);
00135     ldState->setOn();
00136     lbMessage->setText(pvConnectingText);
00137   }else if(state == mdtDevice::Busy){
00138     ldState->setColor(pvBusyColor);
00139     ldState->setOn();
00140     lbMessage->setText(pvBusyText);
00141   }else{
00142     ldState->setRed();
00143     ldState->setOn();
00144     lbMessage->setText(tr("Unknown state"));
00145   }
00146 }
