00052 bool mdtTcpSocketThread::reconnectToHost()
00053 {
00054   Q_ASSERT(pvPort != 0);
00055 
00056   int counter = pvMaxReconnect + 1;
00057   int timeout = 500;  // Initial timeout
00058 
00059   // If we are connected, return simply
00060   if(pvSocket->state() == QAbstractSocket::ConnectedState){
00061     return true;
00062   }
00063   // If a connection is about to be established, wait and return
00064   if((pvSocket->state() == QAbstractSocket::HostLookupState)||(pvSocket->state() == QAbstractSocket::ConnectingState)){
00065     pvPort->unlockMutex();
00066     if(!pvSocket->waitForConnected(timeout)){
00067       pvPort->lockMutex();
00068       mdtError e(MDT_TCP_IO_ERROR, "Unable to reconnect to host", mdtError::Error);
00069       e.setSystemError(pvSocket->error(), pvSocket->errorString());
00070       MDT_ERROR_SET_SRC(e, "mdtTcpSocketThread");
00071       e.commit();
00072       return false;
00073     }
00074     pvPort->lockMutex();
00075     return true;
00076   }
00077   // Here, we can be disconnecting or disconnected , wait to be shure
00078   if(pvSocket->state() != QAbstractSocket::UnconnectedState){
00079     pvPort->unlockMutex();
00080     if(!pvSocket->waitForDisconnected(timeout)){
00081       pvPort->lockMutex();
00082       mdtError e(MDT_TCP_IO_ERROR, "waitForDisconnected() failed", mdtError::Error);
00083       e.setSystemError(pvSocket->error(), pvSocket->errorString());
00084       MDT_ERROR_SET_SRC(e, "mdtTcpSocketThread");
00085       e.commit();
00086       return false;
00087     }
00088     pvPort->lockMutex();
00089   }
00090   // Ok, try reconnect
00091   while(counter > 0){
00092     pvSocket->connectToHost(pvPeerName, pvPeerPort);
00093     pvPort->unlockMutex();
00094     if(pvSocket->waitForConnected(timeout)){
00095       pvPort->lockMutex();
00096       return true;
00097     }
00098     pvPort->lockMutex();
00099     timeout += 100;
00100     counter--;
00101   }
00102   // Here , the maximum retry is reached
00103   mdtError e(MDT_TCP_IO_ERROR, "Cannot (re-)connect to host " + pvPeerName + ":" + QString::number(pvPeerPort) + " (Max retry reached)" , mdtError::Error);
00104   MDT_ERROR_SET_SRC(e, "mdtTcpSocketThread");
00105   e.commit();
00106 
00107   return false;
00108 }
