125 mdtAbstractPort::error_t mdtPortThreadHelperPort::writeToPort(bool bytePerByteWrite, int interByteTime, int maxWriteTry)
126 {
127  Q_ASSERT(pvPort != 0);
128  Q_ASSERT(pvThread != 0);
129 
130  int written = 0;
131  mdtAbstractPort::error_t portError;
132  mdtFrame *frame = currentWriteFrame();
133 
134  if(frame == 0){
135  return mdtAbstractPort::NoError;
136  }
137  // Write the frame to port
138  while(!frame->isEmpty()){
139  // Wait on write ready event
140  portError = pvPort->waitEventWriteReady();
141  // Check about flush request
142  if(pvPort->flushOutRequestPending()){
143  // Restore frame to pool and return
144  restoreCurrentWriteFrameToPool();
145  return mdtAbstractPort::NoError;
146  }
147  if(portError != mdtAbstractPort::NoError){
148  if(!pvThread->runningFlagSet()){
149  return mdtAbstractPort::ErrorHandled;
150  }
151  return handleCommonWriteErrors(portError);
152  }
153  // Event occured, send the data to port
154  if(bytePerByteWrite){
155  written = writeDataToPort(1);
156  pvPort->unlockMutex();
157  msleep(interByteTime);
158  pvPort->lockMutex();
159  }else{
160  written = writeDataToPort(-1);
161  }
162  if((written == 0)&&(!frame->isEmpty())){
163  if(maxWriteTry <= 0){
164  // Notify error and restore frame into write pool
165  mdtError e(MDT_PORT_IO_ERROR, "Max write try reached after write busy", mdtError::Error);
166  MDT_ERROR_SET_SRC(e, "mdtPortThreadHelperPort");
167  e.commit();
168  if(pvThread->runningFlagSet()){
169  notifyError(mdtAbstractPort::WriteTimeout);
170  }
171  restoreCurrentWriteFrameToPool();
172  // Thread has nothing else to do
173  return mdtAbstractPort::ErrorHandled;
174  }
175  // Cannot write now, sleep some time and try later
176  pvPort->unlockMutex();
177  msleep(100);
178  pvPort->lockMutex();
179  maxWriteTry--;
180  }else if(written < 0){
181  if(!pvThread->runningFlagSet()){
182  return mdtAbstractPort::ErrorHandled;
183  }
184  return handleCommonWriteErrors(portError);
185  }
186  }
187  // Here, frame is completly sent
188  Q_ASSERT(frame->isEmpty());
189  restoreCurrentWriteFrameToPool();
190 
191  return mdtAbstractPort::NoError;
192 }
