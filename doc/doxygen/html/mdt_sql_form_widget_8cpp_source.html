00277 {
00278   Q_ASSERT(model() != 0);
00279 
00280   int row;
00281   QSqlRecord initialRecord;
00282   QSqlError error;
00283 
00284   // Remember current row (will be lost during submit)
00285   row = pvWidgetMapper->currentIndex();
00286   // Remember current record - will help on primary key errors
00287   initialRecord = model()->record(row);
00288 
00289   // Call widget mapper submit() (will commit data from widgets to model)
00290   if(!pvWidgetMapper->submit()){
00291     displayDatabaseError(model()->lastError());
00292     return false;
00293   }
00294   /*
00295    * We use QDataWidgetMapper::ManualSubmit submit policy and QSqlTableModel::OnManualSubmit edit strategy.
00296    * Widget mapper calls submit() on model, but this has no effect with OnManualSubmit edit strategy,
00297    * so we have to call submitAll() on model.
00298    */
00299   if(!model()->submitAll()){
00300     if(!restorePrimaryKeyDataToModel(initialRecord)){
00301       error = model()->lastError();
00302       mdtError e(MDT_DATABASE_ERROR, "Unable to restor primary key after a submitAll error, table: " + model()->tableName(), mdtError::Error);
00303       e.setSystemError(error.number(), error.text());
00304       MDT_ERROR_SET_SRC(e, "mdtSqlFormWidget");
00305       e.commit();
00306       return false;
00307     }
00308     displayDatabaseError(model()->lastError());
00309     return false;
00310   }
00311   // Go back to current row
00312   pvWidgetMapper->setCurrentIndex(row);
00313 
00314   return true;
