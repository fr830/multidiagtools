00082 mdtAbstractPort::error_t mdtTcpSocket::pvOpen()
00083 {
00084   Q_ASSERT(!isOpen());
00085 
00086   QStringList items;
00087   bool ok;
00088 
00089   // Extract host name and port number from port name
00090   items = pvPortName.split(":");
00091   if(items.size() != 2){
00092     mdtError e(MDT_TCP_IO_ERROR, "Cannot extract host name and port name in (format must be hostname:port) " + pvPortName, mdtError::Error);
00093     MDT_ERROR_SET_SRC(e, "mdtTcpSocket");
00094     e.commit();
00095     return SetupError;
00096   }
00097   pvPeerName = items.at(0);
00098   pvPeerPort = items.at(1).toUShort(&ok);
00099   if(!ok){
00100     mdtError e(MDT_TCP_IO_ERROR, "Cannot extract port in (format must be hostname:port) " + pvPortName, mdtError::Error);
00101     MDT_ERROR_SET_SRC(e, "mdtTcpSocket");
00102     e.commit();
00103     return SetupError;
00104   }
00105 
00106   return NoError;
00107 }
