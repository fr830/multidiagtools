00181 void mdtLed::paintEvent(QPaintEvent *)
00182 {
00183   QPainter painter(this);
00184   QPointF center( (qreal)size().width()/2.0  , (qreal)size().height()/2.0 );
00185 
00186   painter.setPen(pvPen);
00187 
00188   // Set brush parameters
00189   if(!pvTextMode){
00190     if(pvGlint){
00191       QRadialGradient gradient( center , center.x() , center);
00192       gradient.setColorAt(0.0 , pvCurrentGlance);
00193       gradient.setColorAt(1.0 , pvCurrentColor);
00194       painter.setBrush(gradient);
00195     }else{
00196       pvFlatBrush.setColor(pvCurrentColor);
00197       painter.setBrush(pvFlatBrush);
00198     }
00199   }
00200 
00201   if(pvAntialiasing){
00202     painter.setRenderHint(QPainter::Antialiasing, true);
00203   }else{
00204     painter.setRenderHint(QPainter::Antialiasing, false);
00205   }
00206 
00207   // Draw
00208   if(pvFixedSize.isValid()){
00209     int leftBorder = ( size().width() - pvFixedSize.width() ) / 2 ;
00210     int topBorder = ( size().height() - pvFixedSize.height() ) / 2;
00211     painter.drawEllipse(leftBorder , topBorder , pvFixedSize.width() , pvFixedSize.height());
00212     if(pvTextMode){
00213       painter.drawText(leftBorder, topBorder, pvFixedSize.width(), pvFixedSize.height(), Qt::AlignHCenter | Qt::AlignVCenter , pvText);
00214     }
00215   }else{
00216     painter.drawEllipse(pvHorzontalBorder , pvVerticalBorder , size().width()-pv2xHorzontalBorder , size().height()-pv2xVerticalBorder);
00217     if(pvTextMode){
00218       painter.drawText(pvHorzontalBorder, pvVerticalBorder, size().width()-pv2xHorzontalBorder, size().height()-pv2xVerticalBorder, Qt::AlignHCenter | Qt::AlignVCenter , pvText);
00219     }
00220   }
