625 bool mdtTtTestNodeEditor::setupTestNodeTable()
626 {
627  Ui::mdtTtTestNodeEditor tne;
628  QSqlTableModel *testNodeModel;
629  QDataWidgetMapper *baseVehicleTypeMapper;
630  QSqlTableModel *baseVehicleTypeModel;
631  mdtSqlRelation *baseVehicleTypeRelation;
632 
633  // Setup main form widget
634  tne.setupUi(mainSqlWidget());
635  connect(tne.pbSetVehicleType, SIGNAL(clicked()), this, SLOT(setBaseVehicleType()));
637  // Setup form
638  if(!setMainTable("TestNode_tbl", "Test node", database())){
639  return false;
640  }
641  /*
642  * Setup base unit widget mapping
643  */
644  testNodeModel = model("TestNode_tbl");
645  Q_ASSERT(testNodeModel != 0);
646  // Setup base article model
647  baseVehicleTypeModel = new QSqlTableModel(this, database());
648  baseVehicleTypeModel->setTable("VehicleType_tbl");
649  if(!baseVehicleTypeModel->select()){
650  return false;
651  }
652  // Setup base article widget mapper
653  baseVehicleTypeMapper = new QDataWidgetMapper(this);
654  baseVehicleTypeMapper->setModel(baseVehicleTypeModel);
655  baseVehicleTypeMapper->addMapping(tne.leType, baseVehicleTypeModel->fieldIndex("Type"));
656  baseVehicleTypeMapper->addMapping(tne.leSubType, baseVehicleTypeModel->fieldIndex("SubType"));
657  baseVehicleTypeMapper->addMapping(tne.leSeriesNumber, baseVehicleTypeModel->fieldIndex("SeriesNumber"));
658  // Setup base article relation
659  baseVehicleTypeRelation = new mdtSqlRelation(this);
660  baseVehicleTypeRelation->setParentModel(testNodeModel);
661  baseVehicleTypeRelation->setChildModel(baseVehicleTypeModel);
662  if(!baseVehicleTypeRelation->addRelation("VehicleType_Id_FK_PK", "Id_PK", false)){
663  return false;
664  }
665  connect(mainSqlWidget(), SIGNAL(currentRowChanged(int)), baseVehicleTypeRelation, SLOT(setParentCurrentIndex(int)));
666  connect(baseVehicleTypeRelation, SIGNAL(childModelFilterApplied()), baseVehicleTypeMapper, SLOT(toFirst()));
667  connect(baseVehicleTypeRelation, SIGNAL(childModelIsEmpty()), baseVehicleTypeMapper, SLOT(revert()));
668  // Force a update
669  mainSqlWidget()->setCurrentIndex(mainSqlWidget()->currentRow());
670 
671  return true;
672 }
