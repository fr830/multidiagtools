63 QList<mdtPortInfo*> mdtUsbPortManager::scan(int bDeviceClass, int bDeviceSubClass, int bDeviceProtocol)
64 {
65  Q_ASSERT(isClosed());
66 
67  QList<mdtPortInfo*> portInfoList;
68  mdtDeviceInfo *deviceInfo;
69  mdtPortInfo *portInfo;
70  libusb_context *ctx;
71  libusb_device **devicesList;
72  mdtUsbDeviceDescriptor deviceDescriptor;
75  QString portName;
76  libusb_device *device;
77  ssize_t devicesCount;
78  ssize_t i;
79  int j, k;
80  int err;
81  QString str;
82 
83  // Init libusb
84  err = libusb_init(&ctx);
85  if(err != 0){
86  mdtError e(MDT_USB_IO_ERROR, "Cannot init libusb", mdtError::Error);
87  MDT_ERROR_SET_SRC(e, "mdtUsbPortManager");
88  e.commit();
89  return portInfoList;
90  }
91  // Get devices list
92  devicesCount = libusb_get_device_list(ctx, &devicesList);
93  if(devicesCount < 0){
94  mdtError e(MDT_USB_IO_ERROR, "Cannot list devices", mdtError::Error);
95  MDT_ERROR_SET_SRC(e, "mdtUsbPortManager");
96  e.commit();
97  return portInfoList;
98  }
99  // Scan ...
100  for(i=0; i<devicesCount; i++){
101  device = devicesList[i];
102  err = deviceDescriptor.fetchAttributes(device, false);
103  if(err != 0){
104  mdtError e(MDT_USB_IO_ERROR, "Device attributes fetch failed", mdtError::Error);
105  MDT_ERROR_SET_SRC(e, "mdtUsbPortManager");
106  e.commit();
107  continue;
108  }
109  // Create a port info object
110  portInfo = new mdtPortInfo;
111  // Search interfaces in various configurations
112  for(j=0; j<deviceDescriptor.configurations().size(); j++){
113  mdtUsbConfigDescriptor configDescriptor = deviceDescriptor.configurations().at(j);
115  for(k=0; k<configDescriptor.interfaces().size(); k++){
116  mdtUsbInterfaceDescriptor ifaceDescriptor = configDescriptor.interfaces().at(k);
118  // See if a specific bDeviceClass was requested
119  if(bDeviceClass >= 0){
120  if(ifaceDescriptor.bInterfaceClass() != (quint8)bDeviceClass){
121  continue;
122  }
123  }
124  // See if a specific bDeviceSubClass was requested
125  if(bDeviceSubClass >= 0){
126  if(ifaceDescriptor.bInterfaceSubClass() != (quint8)bDeviceSubClass){
127  continue;
128  }
129  }
130  // See if a specific bDeviceProtocol was requested
131  if(bDeviceProtocol >= 0){
132  if(ifaceDescriptor.bInterfaceProtocol() != (quint8)bDeviceProtocol){
133  continue;
134  }
135  }
136  // Interresting interface
137  deviceInfo = new mdtDeviceInfo;
138  // Build the display text
139  str = deviceDescriptor.vendorName();
140  str += ", ";
141  str += deviceDescriptor.productName();
142  if(!deviceDescriptor.serialNumber().isEmpty()){
143  str += " (SN: " + deviceDescriptor.serialNumber() + ")";
144  }
145  portInfo->setDisplayText(str);
146  // Set VID, PID, SID and protocol ID
147  deviceInfo->setVendorId(deviceDescriptor.idVendor());
148  deviceInfo->setProductId(deviceDescriptor.idProduct());
149  deviceInfo->setSerialId(deviceDescriptor.serialNumber());
150  deviceInfo->setProtocolId(ifaceDescriptor.bInterfaceProtocol());
151  // Add interface to list
152  portInfo->addDevice(deviceInfo);
153  }
154  }
155  // If port info has no device, delete it, else add it to list
156  if(portInfo->deviceInfoList().size() < 1){
157  delete portInfo;
158  }else{
159  // Build port name
160  deviceInfo = portInfo->deviceInfoList().at(0);
161  Q_ASSERT(deviceInfo != 0);
162  portName = "VID=0x";
163  portName += QString::number((unsigned int)deviceInfo->vendorId(), 16);
164  portName += ":PID=0x";
165  portName += QString::number((unsigned int)deviceInfo->productId(), 16);
166  if(!deviceDescriptor.serialNumber().isEmpty()){
167  portName += ":SID=" + deviceDescriptor.serialNumber();
168  }
169  portInfo->setPortName(portName);
170  // Add port info to list
171  portInfoList.append(portInfo);
172  }
173  }
174  // Release ressources
175  libusb_free_device_list(devicesList, 1);
176  libusb_exit(ctx);
177 
178  return portInfoList;
179 }

236 int mdtUsbPortManager::sendControlRequest(const mdtFrameUsbControl &request, bool setwIndexAsbInterfaceNumber)
237 {
238  Q_ASSERT(pvPort != 0);
239 
240  mdtUsbPort *port;
241  mdtFrameUsbControl *frame;
242 
243  // We need a mdtUsbPort object
244  port = dynamic_cast<mdtUsbPort*>(pvPort);
245  Q_ASSERT(port != 0);
246  // Get a frame in pool
257  // Wait until we can send a new control request
258  while(1){
259  // If port manager was stopped, we return
260  if(isClosed()){
261  return mdtAbstractPort::ControlCanceled;
262  }
263  if(isReady()){
264  lockPortMutex();
265  if(port->controlFramesPool().size() > 0){
266  frame = port->controlFramesPool().dequeue();
267  Q_ASSERT(frame != 0);
268  // Here we are ready to write - we keep port mutext locked
269  break;
270  }
271  unlockPortMutex();
272  }
273  QCoreApplication::processEvents(QEventLoop::AllEvents | QEventLoop::WaitForMoreEvents);
274  }
275  Q_ASSERT(frame != 0);
276  frame->clear();
277  frame->clearSub();
278  // Store data, encode and add frame to queue
279  frame->setDirectionDeviceToHost(request.directionIsDeviceToHost());
280  frame->setbmRequestType(request.bmRequestType());
281  frame->setbRequest(request.bRequest());
282  frame->setwValue(request.wValue());
283  frame->setwIndex(request.wIndex());
284  frame->setwLength(request.wLength());
285  frame->encode();
286  port->addControlRequest(frame, setwIndexAsbInterfaceNumber);
287  unlockPortMutex();
288 
289  return 0;
290 }

360 void mdtUsbPortManager::fromThreadControlResponseReaden()
361 {
362  qDebug() << " *= mdtUsbPortManager::fromThreadControlResponseReaden(): control response !!!!";
363  Q_ASSERT(pvPort != 0);
364 
365  mdtUsbPort *port;
366  mdtFrameUsbControl *frame;
367 
368  // We need a mdtUsbPort object
369  port = dynamic_cast<mdtUsbPort*>(pvPort);
370  Q_ASSERT(port != 0);
371 
372  // Get frames in readen queue
373  lockPortMutex();
374  while(port->controlResponseFrames().size() > 0){
375  frame = port->controlResponseFrames().dequeue();
376  Q_ASSERT(frame != 0);
377  // Copy data
378  qDebug() << " *= mdtUsbPortManager::fromThreadControlResponseReaden(): add data ...";
379  pvReadenControlResponses.enqueue(*frame);
380  // Restore frame back into pool
381  port->controlFramesPool().enqueue(frame);
382  };
383  unlockPortMutex();
384  // Watch queue size
385  qDebug() << " *= mdtUsbPortManager::fromThreadControlResponseReaden(): queue size: " << pvReadenControlResponses.size();
386  if(pvReadenControlResponses.size() > 20){
387  mdtError e(MDT_USB_IO_ERROR, "Readen control resonses queue is largen than 20 items, will be leared (this is a bug)", mdtError::Error);
388  MDT_ERROR_SET_SRC(e, "mdtUsbPortManager");
389  e.commit();
390  pvReadenControlResponses.clear();
391  }
