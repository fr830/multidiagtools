263  template<typename T>
264  std::shared_ptr<T> childController(const QString & tableName)
265  {
266  int i;
267  std::shared_ptr<mdtAbstractSqlTableController> c;
268  for(i = 0; i < pvChildControllerContainers.size(); ++i){
269  c = pvChildControllerContainers.at(i).controller;
270  Q_ASSERT(c);
271  if(c->tableName() == tableName){
272  return std::dynamic_pointer_cast<T>(c);
273  }
274  // Try in child controller's childs
275  auto tc = c->childController<T>(tableName);
276  if(tc){
277  return tc;
278  }
279  }
280  return std::shared_ptr<T>();
