00184   // Calculate number of bytes needed
00185   if((n%8) == 0){
00186     nBytes = n/8;
00187   }else{
00188     nBytes = (n/8)+1;
00189   }
00190   // Function code
00191   pvPdu.append(0x0F);
00192   // Register starting address
00193   pvPdu.append(startAddress >> 8);
00194   pvPdu.append(startAddress & 0x00FF);
00195   // Quantity of coils
00196   pvPdu.append(n >> 8);
00197   pvPdu.append(n & 0x00FF);
00198   // Bytes count
00199   pvPdu.append(nBytes);
00200   // states
00201   for(i=0; i<states.size(); i++){
00202     oneByteStates.append(states.at(i));
00203     // If we have 8 bits, append to PDU
00204     if(oneByteStates.size() == 8){
00205       pvPdu.append(byteFromBooleans(oneByteStates));
00206       oneByteStates.clear();
00207     }
00208   }
00209   // Remaining states
00210   if(oneByteStates.size() > 0){
00211     // See if we have to complete
00212     for(i=oneByteStates.size()-1; i<7; i++){
00213       oneByteStates.append(false);
00214     }
00215     // Add to PDU
00216     pvPdu.append(byteFromBooleans(oneByteStates));

00283 int mdtFrameCodecModbus::decode(const QByteArray &pdu)
00284 {
00285   pvPdu = pdu;
00286   pvValues.clear();
00287   pvLastModbusError = UNKNOW_ERROR;
00288   quint8 functionCode;
00289 
00290   // Case of empty PDU
00291   if(pvPdu.size() < 2){
00292     mdtError e(MDT_FRAME_DECODE_ERROR, "Request a decode of a empty PDU (size < 2 Bytes)", mdtError::Error);
00293     MDT_ERROR_SET_SRC(e, "mdtFrameCodecModbus");
00294     e.commit();
00295     return -1;
00296   }
00297   // Get the function code
00298   functionCode = pvPdu.at(0);
00299   // Case of error code
00300   if(functionCode > 0x80){
00301     decodeModbusError(pvPdu.at(1));
00302     return functionCode;
00303   }
00304   // Decode PDU according functionCode
00305   switch(functionCode){
00306     // Read coils
00307     case 0x01:
00308       if(!decodeReadCoils()){
00309         return -1;
00310       }
00311       return 0x01;
00312       break;
00313     // Read discrete inputs
00314     case 0x02:
00315       if(!decodeReadDiscreteInputs()){
00316         return -1;
00317       }
00318       return 0x02;
00319       break;
00320     // ReadHoldingRegisters
00321     case 0x03:
00322       if(!decodeReadHoldingRegisters()){
00323         return -1;
00324       }
00325       return 0x03;
00326       break;
00327     // ReadInputRegisters
00328     case 0x04:
00329       if(!decodeReadInputRegisters()){
00330         return -1;
00331       }
00332       return 0x04;
00333       break;
00334     // Write single coil
00335     case 0x05:
00336       if(!decodeWriteSingleCoil()){
00337         return -1;
00338       }
00339       return 0x05;
00340       break;
00341     // Write single register
00342     case 0x06:
00343       if(!decodeWriteSingleRegister()){
00344         return -1;
00345       }
00346       return 0x06;
00347       break;
00348     // WriteMultipleCoils
00349     case 0x0F:
00350       if(!decodeWriteMultipleCoils()){
00351         return -1;
00352       }
00353       return 0x0F;
00354       break;
00355     // WriteMultipleRegisters
00356     case 0x10:
00357       if(!decodeWriteMultipleRegisters()){
00358         return -1;
00359       }
00360       return 0x10;
00361       break;
00362     case 0x2B:
00363       if(!decodeReadDeviceIdentification()){
00364         return -1;
00365       }
00366       return 0x2B;
00367       break;
00368     // Unknow code
00369     default:
00370       QByteArray num;
00371       num.setNum(functionCode, 16);
00372       mdtError e(MDT_FRAME_DECODE_ERROR, "Unknow function code: 0x" + num, mdtError::Error);
00373       MDT_ERROR_SET_SRC(e, "mdtFrameCodecModbus");
00374       e.commit();

00376   }
00377

00622 {

00624   int objectLength;
00625   int cursor;
00626 
00627   // We need min. 8 bytes
00628   if(pvPdu.size() < 8){
00629     mdtError e(MDT_FRAME_DECODE_ERROR, "PDU size not valid (size < 8 Bytes)", mdtError::Error);
00630     MDT_ERROR_SET_SRC(e, "mdtFrameCodecModbus");
00631     e.commit();

00633   }
00634   // Check FC
00635   Q_ASSERT((int)pvPdu.at(0) == 0x2B);
00636   // Check MEI Type
00637   if((int)pvPdu.at(1) != 0x0E){
00638     mdtError e(MDT_FRAME_DECODE_ERROR, "MEI Type not valid", mdtError::Error);
00639     MDT_ERROR_SET_SRC(e, "mdtFrameCodecModbus");
00640     e.commit();
00641     return false;
00642   }
00643   // Store More Follows and Next Object Id
00644   if((quint8)pvPdu.at(4) == 0xFF){
00645     pvValues.append(true);
00646   }else{
00647     pvValues.append(false);
00648   }
00649   pvValues.append((int)pvPdu.at(5));
00650   // Get objects
00651   cursor = 7;
00652   while(cursor < pvPdu.size()){
