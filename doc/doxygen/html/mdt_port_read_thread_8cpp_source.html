58 #ifdef Q_OS_UNIX

62  // Get setup

64  pvMinPoolSizeBeforeReadSuspend = pvPort->config().readQueueSize() / 4;
65  if((pvMinPoolSizeBeforeReadSuspend < 1)&&(pvPort->config().readQueueSize() > 0)){
66  pvMinPoolSizeBeforeReadSuspend = 1;
67  }
68  /*
69  * Check read timeout:
70  * - For this thread, read timeout has only sense for read timeout protocol
71  * - For other protocol, port manager must handle read timeout itself if needed (query/reply)
72  * So, if useReadTimeoutProtocol is not set, we set port's readTimeout to infinite.
73  * Port manager will use config's readTimeout, so we not change it here.
74  */
75  if((!useReadTimeoutProtocol)&&(pvPort->config().readTimeout() > -1)){
76  pvPort->setReadTimeout(-1);
77  }
78  // Set the running flag and get a read frame
79  pvRunning = true;
80  // Get frame
81  if(!threadHelper.getNewFrameRead()){
82  return;
83  }
84  // Notify that we are ready
86  emit ready(this);
87

89  while(1){
90  // Read thread state
91  if(!pvRunning){
92  break;
93  }
94  Q_ASSERT(threadHelper.currentReadFrame() != 0);
95  // Wait on Read event
96  portError = pvPort->waitForReadyRead();
97  if(portError != mdtAbstractPort::NoError){
98  // Read thread state
99  if(!pvRunning){
100  break;
101  }
102  if((useReadTimeoutProtocol)&&(portError == mdtAbstractPort::ReadTimeout)){
103  // In timeout protocol, the current frame is considered complete
104  if(!threadHelper.currentReadFrame()->isEmpty()){
105  threadHelper.currentReadFrame()->setComplete();
106  // Submit frame and notify newFrameReaden
107  threadHelper.submitCurrentReadFrame(true);
108  if(!threadHelper.getNewFrameRead()){
109  break;
110  }
111  }
112  }else{
113  portError = threadHelper.handleCommonReadErrors(portError);
114  if(portError != mdtAbstractPort::ErrorHandled){
115  // Unhandled error, stop
116  break;
117  }
118  }
119  }
120  // Check about flush request
121  if(pvPort->flushInRequestPending()){
122  // Put current frame back to pool, get new one and return idle
123  threadHelper.restoreCurrentReadFrameToPool();
124  if(!threadHelper.getNewFrameRead()){
125  break;
126  }
127  continue;
128  }
129  // Read
130  n = threadHelper.readFromPort(true);
131  if(n < 0){
132  // Check about stopping
133  if(!pvRunning){
134  break;
135  }
136  portError = threadHelper.handleCommonReadErrors((mdtAbstractPort::error_t)n);
137  if(portError != mdtAbstractPort::ErrorHandled){
138  // Unhandled error, stop
139  break;
140  }
141  }
142  }
143 
149  pvRunning = false;

151 }
