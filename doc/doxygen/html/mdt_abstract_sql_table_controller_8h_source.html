232  template<typename T>
233  std::shared_ptr<T> childController(const QString & tableName)
234  {
235  int i;
236  std::shared_ptr<mdtAbstractSqlTableController> c;
237  for(i = 0; i < pvChildControllerContainers.size(); ++i){
238  c = pvChildControllerContainers.at(i).controller;
239  Q_ASSERT(c);
240  if(c->tableName() == tableName){
241  return std::dynamic_pointer_cast<T>(c);
242  }
243  // Try in child controller's childs
244  auto tc = c->childController<T>(tableName);
245  if(tc){
246  return tc;
247  }
248  }
249  return std::shared_ptr<T>();
