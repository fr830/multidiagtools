121 void mdtDeviceU3606A::decodeReadenFrame(mdtPortTransaction *transaction)
122 {
123  Q_ASSERT(transaction != 0);
124 
125  bool ok;
126 
128 
129  switch(transaction->type()){
130  case mdtFrameCodecScpi::QT_VALUE:
131  if(transaction->analogIo() == 0){
132  lastErrorW().setError(tr("Device ") + name() + tr(": received value response from device but no I/O was affected to query"), mdtError::Warning);
133  MDT_ERROR_SET_SRC(lastErrorW(), "mdtDeviceU3606A");
134  lastErrorW().commit();
135  }else{
136  ok = pvCodec->decodeValues(transaction->data());
137  if(ok && (pvCodec->values().size() == 1)){
138  transaction->analogIo()->setValue(pvCodec->values().at(0).value<mdtValue>(), false);
139  }else{
140  transaction->analogIo()->setValue(mdtValue(), false);
141  }
142  }
143  break;
144  case mdtFrameCodecScpi::QT_SINGLE_VALUE_FLT:
145  if(transaction->analogIo() == 0){
146  lastErrorW().setError(tr("Device ") + name() + tr(": received value response from device but no I/O was affected to query"), mdtError::Warning);
147  MDT_ERROR_SET_SRC(lastErrorW(), "mdtDeviceU3606A");
148  lastErrorW().commit();
149  }else{
150  transaction->analogIo()->setValue(pvCodec->decodeSingleValueDouble(transaction->data()), false);
151  }
152  break;
153  case mdtFrameCodecScpi::QT_ERR:
154  ok = pvCodec->decodeError(transaction->data());
155  if(!ok){
156  lastErrorW().setError(tr("Device ") + name() + tr(": SYST:ERR? reply could not be decoded"), mdtError::Warning);
157  MDT_ERROR_SET_SRC(lastErrorW(), "mdtDeviceU3606A");
158  lastErrorW().commit();
159  break;
160  }
161  handleDeviceError(pvCodec->values());
162  break;
163  case mdtFrameCodecScpi::QT_CONF:
164  if(!pvCodec->decodeConfigure(transaction->data())){
165  lastErrorW().setError(tr("Device ") + name() + tr(": CONF? reply could not be decoded"), mdtError::Warning);
166  MDT_ERROR_SET_SRC(lastErrorW(), "mdtDeviceU3606A");
167  lastErrorW().commit();
168  break;
169  }
170  updateMeasureSetup();
171  break;
172  default:
173  lastErrorW().setError(tr("Device ") + name() + tr(": received unsupported response from device (type: ") + QString::number(transaction->type()) + tr(")"), mdtError::Warning);
174  MDT_ERROR_SET_SRC(lastErrorW(), "mdtDeviceU3606A");
175  lastErrorW().commit();
176  }
177 }

179 int mdtDeviceU3606A::readAnalogInput(mdtPortTransaction *transaction)
180 {
181  Q_ASSERT(ios() != 0);
182  Q_ASSERT(transaction != 0);
183 
184  int bTag;
185  mdtAnalogIo *aio;
186 
187  aio = transaction->analogIo();
188  Q_ASSERT(aio != 0);
189  switch(aio->addressRead()){
190  case 0: // Measure
191  transaction->setData("READ?");
192  break;
193  case 1: // Sense U (voltage)
194  transaction->setData("SENS:VOLT?");
195  break;
196  case 2: // Sense I (current)
197  transaction->setData("SENS:CURR?");
198  break;
199  default:
200  lastErrorW().setError(tr("Requested to read a unknown analog input. Address: ") + QString::number(aio->addressRead()), mdtError::Error);
201  MDT_ERROR_SET_SRC(lastErrorW(), "mdtDeviceU3606A");
202  lastErrorW().commit();
203  return mdtAbstractPort::UnhandledError;
204  }
205  // Send query
206  bTag = pvUsbtmcPortManager->sendData(transaction);
207  if(bTag < 0){
208  return bTag;
209  }
210  // Remember query type.
211  transaction->setType(mdtFrameCodecScpi::QT_SINGLE_VALUE_FLT);
212  // Send read request
213  bTag = pvUsbtmcPortManager->sendReadRequest(transaction);
214 
215  return bTag;
216 }
