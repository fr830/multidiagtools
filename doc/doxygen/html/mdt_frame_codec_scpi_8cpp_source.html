00046 bool mdtFrameCodecScpi::decodeValues(const QByteArray &data, QString sep)
00047 {
00048   int i;
00049   QVariant value;
00050   double fltValue;
00051 
00052   // Clear previous results
00053   pvValues.clear();
00054   pvNodes.clear();
00055 
00056   // Store raw data in local QString constainer
00057   pvAsciiData = data;
00058 
00059   // Remove begin/end spaces and EOF
00060   if(!clean()){
00061     return false;
00062   }
00063 
00064   // Extract nodes
00065   pvNodes = pvAsciiData.split(sep);
00066   for(i=0; i<pvNodes.size(); i++){
00067     // We ignore empty nodes
00068     if(pvNodes.at(i).isEmpty()){
00069       continue;
00070     }
00071     // Convert node
00072     value = convertData(pvNodes.at(i));
00073     // Check limits if type is double
00074     if(value.type() == QVariant::Double){
00075       fltValue = value.toDouble();
00076       // Check about Nan
00077       if(qAbs(fltValue - pvNan) <= FLT_EPSILON){
00078         value.clear();
00079       }else{
00080         // Check about -OL and OL
00081         if((fltValue <= (pvNinfinity+FLT_EPSILON))||(fltValue >= (pvInfinity-FLT_EPSILON))){
00082           value.clear();
00083         }
00084       }
00085     }
00086     // Add converted value
00087     pvValues.append(value);
00088   }
00089 
00090   return true;

00163 bool mdtFrameCodecScpi::decodeIEEEblock(const QByteArray &data, mdtFrameCodecScpi::waveform_format format)
00164 {
00165   int headerSize;
00166   int dataSize;
00167   bool ok = false;
00168 
00169   // Clear previous results
00170   pvValues.clear();
00171   // If string length is < 3: shure: not an IEEE block
00172   if(data.size() < 3){
00173     mdtError e(MDT_FRAME_DECODE_ERROR, "Frame is not a IEEE block (length < 2)" , mdtError::Warning);
00174     MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpi");
00175     e.commit();
00176     return false;
00177   }
00178   // If first char is not a #, we have not a IEEE block
00179   if(data.at(0) != '#'){
00180     mdtError e(MDT_FRAME_DECODE_ERROR, "Frame is not a IEEE block (beginns not with #)" , mdtError::Warning);
00181     MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpi");
00182     e.commit();
00183     return false;
00184   }
00185   // Get the header length
00186   headerSize = data.mid(1, 1).toInt(&ok);
00187   if(!ok){
00188     mdtError e(MDT_FRAME_DECODE_ERROR, "Cannot get header length in IEEE block" , mdtError::Warning);
00189     MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpi");
00190     e.commit();
00191     return false;
00192   }
00193   // Check that enough bytes are available
00194   if(data.size() < (headerSize+2)){
00195     mdtError e(MDT_FRAME_DECODE_ERROR, "IEEE block contains no data" , mdtError::Warning);
00196     MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpi");
00197     e.commit();
00198     return false;
00199   }
00200   // Get data length
00201   dataSize = data.mid(2, headerSize).toInt(&ok);
00202   if(!ok){
00203     mdtError e(MDT_FRAME_DECODE_ERROR, "Cannot get data length in IEEE block" , mdtError::Warning);
00204     MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpi");
00205     e.commit();
00206     return false;
00207   }
00208   if(dataSize < 1){
00209     mdtError e(MDT_FRAME_DECODE_ERROR, "IEEE block contains no data (info from header)" , mdtError::Warning);
00210     MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpi");
00211     e.commit();
00212     return false;
00213   }
00214   // Calculate real received data size
00215   if(data.at(data.size()-1) == '\n'){
00216     dataSize = data.size()-headerSize-3;
00217   }else{
00218     dataSize = data.size()-headerSize-2;
00219   }
00220   if(dataSize < 1){
00221     mdtError e(MDT_FRAME_DECODE_ERROR, "IEEE block contains no data (info from header)" , mdtError::Warning);
00222     MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpi");
00223     e.commit();
00224     return false;
00225   }
00226   // Decode data part
00227   if(format == BYTE){
00228     return decodeIEEEdataByte(data.mid(headerSize+2, dataSize));
00229   }
00230   if(format == ASCII){
00231     return decodeIEEEdataAscii(data.mid(headerSize+2, dataSize));
00232   }
00233   mdtError e(MDT_FRAME_DECODE_ERROR, "Data format not implemented yet, sorry" , mdtError::Warning);
00234   MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpi");
00235   e.commit();
00236   return false;
00237 }
