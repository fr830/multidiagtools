77 bool mdtSqlFormOld::addChildTable(const QString &tableName, const QString &userFriendlyTableName, QSqlDatabase db)
78 {
79  QSqlTableModel *model;
80  mdtSqlTableWidget *widget;
81  mdtSqlRelation *relation;
82 
83  // Check that main table was set
84  if(pvMainSqlWidget->model() == 0){
85  pvLastError.setError(tr("Cannot add child table '") + tableName + tr("' because no main table was set"), mdtError::Error);
87  MDT_ERROR_SET_SRC(pvLastError, "mdtSqlFormOld");
88  pvLastError.commit();
89  return false;
90  }
91  Q_ASSERT(pvMainSqlWidget->model() != 0);
92  // Setup model
93  model = new QSqlTableModel(this, db);
94  model->setTable(tableName);
95  if(!model->select()){
96  QSqlError sqlError = model->lastError();
97  pvLastError.setError(tr("Unable to select data in table '") + tableName + tr("'"), mdtError::Error);
99  pvLastError.setSystemError(sqlError.number(), sqlError.text());
100  MDT_ERROR_SET_SRC(pvLastError, "mdtSqlFormOld");
101  pvLastError.commit();
102  delete model;
103  return false;
104  }
105  // Setup relation
106  relation = new mdtSqlRelation;
107  relation->setParentModel(pvMainSqlWidget->model());
108  relation->setChildModel(model);
109  pvRelationsByChildTableName.insert(tableName, relation);
110  // Setup child widget
111  widget = new mdtSqlTableWidget;
112  widget->setModel(model);
113  if(!userFriendlyTableName.isEmpty()){
114  widget->setUserFriendlyTableName(userFriendlyTableName);
115  }
116  // Add child widget to main widget
117  pvMainSqlWidget->addChildWidget(widget, relation);
118  addChildWidget(widget);
119 
120  return true;
121 }

124 {

133  pvLastError.commit();

136  // Add relation

138  return false;

142 
143  return true;
