00202 void mdtPortStatusWidget::setState(int state)
00203 {
00204   if(state == mdtPortManager::Ready){
00205     ldState->setColor(pvReadyColor);
00206     ldState->setOn();
00207     pvCurrentStateText = pvReadyText;
00208   }else if(state == mdtPortManager::Disconnected){
00209     ldState->setGreen();
00210     ldState->setOff();
00211     pvCurrentStateText = pvDisconnectedText;
00212   }else if(state == mdtPortManager::Connecting){
00213     ldState->setColor(pvConnectingColor);
00214     ldState->setOn();
00215     pvCurrentStateText = pvConnectingText;
00216   }else if(state == mdtPortManager::Busy){
00217     ldState->setColor(pvBusyColor);
00218     ldState->setOn();
00219     pvCurrentStateText = pvBusyText;
00220   }else{
00221     ldState->setRed();
00222     ldState->setOn();
00223     pvCurrentStateText = tr("Unknown state");
00224   }
00225   // Set state text
00226   if(!pvShowingMessage){
00227     lbMessage->setText(pvCurrentStateText);
00228   }
00229 }
