284 void mdtDoubleEdit::convertAndSetValue(QString str)
285 {
286  double f = 1.0;
287  QChar fc;
288 
289  str = str.trimmed();
290  // Remove unit part
291  str.remove(pvUnit);
292  str = str.trimmed();
293  if(str.isEmpty()){
294  pvValueIsValid = false;
295  displayValue();
296  return;
297  }
298  // If str is not a infinity, get power of 10 factor and remove it from str
299  if(!strIsInfinity(str)){
300  fc = str.at(str.size() - 1);
301  f = factor(fc);
302  if(qAbs<double>(f - 1.0) <= std::numeric_limits<double>::min()){
303  fc = '\0';
304  }
305  if(!fc.isNull()){
306  str.remove(fc);
307  }
308  str = str.trimmed();
309  }
310  // Do conversion
311  pvValue = str.toDouble(&pvValueIsValid);
312  if(pvValueIsValid){
313  pvValue = pvValue * f;
314  }
315 }

317 void mdtDoubleEdit::displayValue()
318 {
319  double f;
320  double x;
321  QChar fc;
322  QString str;
323 
324  if(!pvValueIsValid){
325  pvLineEdit->clear();
326  return;
327  }
328  // Check if we are - or + infinity
329  if(pvValue == -std::numeric_limits<double>::infinity()){
330  pvLineEdit->setText("-" + infinityString());
331  return;
332  }
333  if(pvValue == std::numeric_limits<double>::infinity()){
334  pvLineEdit->setText(infinityString());
335  return;
336  }
337  // Select a power of 10 regarding value - Note: we ignore c and d and h
338  x = qAbs<double>(pvValue);
339  if(x <= std::numeric_limits<double>::min()){
340  f = 1.0;
341  }else if(x < 1.e-15){
342  f = 1.e18;
343  fc = 'a';
344  }else if(isInRange(x, 1e-15, 1e-12)){
345  f = 1e15;
346  fc = 'f';
347  }else if(isInRange(x, 1e-12, 1e-9)){
348  f = 1e12;
349  fc = 'p';
350  }else if(isInRange(x, 1e-9, 1e-6)){
351  f = 1e9;
352  fc = 'n';
353  }else if(isInRange(x, 1e-6, 1e-3)){
354  f = 1e6;
355  fc = 'u';
356  }else if(isInRange(x, 1e-3, 1.0)){
357  f = 1e3;
358  fc = 'm';
359  }else if(isInRange(x, 1.0, 1e3)){
360  f = 1.0;
361  }else if(isInRange(x, 1e3, 1e6)){
362  f = 1e-3;
363  fc = 'k';
364  }else if(isInRange(x, 1e6, 1e9)){
365  f = 1e-6;
366  fc = 'M';
367  }else if(isInRange(x, 1e9, 1e12)){
368  f = 1e-9;
369  fc = 'G';
370  }else if(isInRange(x, 1e12, 1e15)){
371  f = 1e-12;
372  fc = 'T';
373  }else if(isInRange(x, 1e15, 1e18)){
374  f = 1e-15;
375  fc = 'P';
376  }else{
377  f = 1e-18;
378  fc = 'E';
379  }
380  // Build string to display
381  str.setNum(pvValue * f);
382  if((!fc.isNull())||(!pvUnit.isEmpty())){
383  str += " ";
384  }
385  if(!fc.isNull()){
386  str += fc;
387  }
388  str += pvUnit;
389  // Display str
390  pvLineEdit->setText(str);
391 }
