73 bool mdtDeviceModbus::connectToDevice()
74 {
75  // Check that port manager is not running
76  if(!pvTcpPortManager->isClosed()){
77  pvTcpPortManager->stop();
78  }
79  // Check that device address string refers to a TCPIP port
80  if(deviceAddress().portType() != mdtDeviceAddress::PortType_t::TCPIP){
81  pvLastError.setError(tr("Cannot connect to device with address string '") + deviceAddress().addressString() + \
82  tr("': port type '") + deviceAddress().portTypeStr() + tr("' not supported."), mdtError::Error);
83  MDT_ERROR_SET_SRC(pvLastError, "mdtDeviceModbus");
84  pvLastError.commit();
85  return false;
86  }
87  // Check that device address string refers to a MODBUS device
88  if(deviceAddress().deviceType() != mdtDeviceAddress::DeviceType_t::MODBUS){
89  pvLastError.setError(tr("Cannot connect to device with address string '") + deviceAddress().addressString() + \
90  tr("': device type '") + deviceAddress().deviceTypeStr() + tr("' not supported."), mdtError::Error);
91  MDT_ERROR_SET_SRC(pvLastError, "mdtDeviceModbus");
92  pvLastError.commit();
93  return false;
94  }
95  // Connect to device
96  pvTcpPortManager->setPortName(deviceAddress().tcpIpHostName() + ":" + QString::number(deviceAddress().tcpIpPort()) );
97  if(!pvTcpPortManager->start()){
98  pvLastError = pvTcpPortManager->lastError();
99  return false;
100  }
101  // If hardware node ID is valid, check that it matches
102  mdtDeviceAddress da = deviceAddress();
103  if(!da.modbusHwNodeId().isNull()){
104  mdtModbusHwNodeId hwNodeId = da.modbusHwNodeId();
105  mdtModbusHwNodeId deviceHwNodeId = pvTcpPortManager->getHardwareNodeAddress(hwNodeId.bitsCount(), hwNodeId.firstBit());
106  if(deviceHwNodeId.isNull()){
107  pvLastError = pvTcpPortManager->lastError();
108  disconnectFromDevice();
109  return false;
110  }
111  if(deviceHwNodeId.id() != hwNodeId.id()){
112  pvLastError.setError(deviceIdString() + tr(" device's MODBUS hardware node ID does not match requested one. ") + \
113  tr("Requested: ") + QString::number(hwNodeId.id()) + \
114  tr(" , device:") + QString::number(deviceHwNodeId.id()), mdtError::Error);
115  MDT_ERROR_SET_SRC(pvLastError, "mdtDeviceModbus");
116  pvLastError.commit();
117  disconnectFromDevice();
118  return false;
119  }
120  }
121 
122  return true;
123 }

125 bool mdtDeviceModbus::connectToDevice(const mdtModbusHwNodeId & hwNodeId, const QString & alias, int scanTimeout, int port)
126 {
127  mdtDeviceAddressList daList;
128  mdtModbusHwNodeIdList hwNodeIdList;
129 
130  // Check that port manager is not running
131  if(!pvTcpPortManager->isClosed()){
132  pvTcpPortManager->stop();
133  }
134  // Get list of known devices that are reachable and check if node with requested HW ID exists
135  daList = pvTcpPortManager->scanFromKnownHostsFile(scanTimeout);
136  for(const auto & da : daList.internalVector()){
137  if(da.modbusHwNodeId().id() == hwNodeId.id()){
138  setDeviceAddress(da, alias);
139  return connectToDevice();
140  }
141  }
142  // Nothing found in cache, scan the network
143  hwNodeIdList.append(hwNodeId);
144  daList = pvTcpPortManager->scan(QNetworkInterface::allInterfaces(), hwNodeIdList, port, true, scanTimeout);
145  if(daList.isEmpty()){
146  pvLastError.setError(tr("Could not find a MODBUS/TCP device with hardware node ID ") + QString::number(hwNodeId.id()), mdtError::Error);
147  MDT_ERROR_SET_SRC(pvLastError, "mdtDeviceModbus");
148  pvLastError.commit();
149  return false;
150  }
151  Q_ASSERT(daList.size() > 0);
152  /*
153  * Save device addresses to known hosts cache file
154  * We ignore return value, if file cannot be saved,
155  * it's not a problem for all the rest
156  */
157  pvTcpPortManager->saveDeviceAddressList(daList);
158  // Should be ok , connect
159  setDeviceAddress(daList.at(0), alias);
160 
161  return connectToDevice();
162 }

179 void mdtDeviceModbus::decodeReadenFrame(mdtPortTransaction *transaction)
180 {
181  Q_ASSERT(transaction != 0);
182  if(ios() == 0){
183  return;
184  }
185 
186  int fc;
187  QVariant var;
188 
190  // Decode readen frame and update I/O's
191  fc = pvCodec->decode(transaction->data());
192  switch(fc){
193  case 0x01: // Read coils
194  // Check if we have a answer for a single I/O or multiple I/Os
195  if((transaction->ioCount() == 1)&&(transaction->digitalIo() != 0)){
196  // If one DO was requested, 1 byte is returned with just first bit set
197  if(pvCodec->values().size() == 8){
198  var = pvCodec->values().at(0);
199  if((var.isValid())&&(var.type() == QVariant::Bool)){
200  transaction->digitalIo()->setValue(var.toBool(), false);
201  }else{
202  transaction->digitalIo()->setValue(mdtValue(), false);
203  }
204  }else{
205  transaction->digitalIo()->setValue(mdtValue(), false);
206  }
207  transaction->digitalIo()->setEnabled(true);
208  }else{
209  // Transaction contains first I/O address (for read access) and qty of I/Os
210  ios()->updateDigitalOutputValues(pvCodec->values(), transaction->address(), mdtDeviceIos::Read , transaction->ioCount(), true);
211  ios()->setDigitalOutputsEnabled(true);
212  }
213  break;
214  case 0x02: // Read discrete inputs
215  // Check if we have a answer for a single I/O or multiple I/Os
216  if((transaction->ioCount() == 1)&&(transaction->digitalIo() != 0)){
217  // If one DI was requested, 1 byte is returned with just first bit set
218  if(pvCodec->values().size() == 8){
219  var = pvCodec->values().at(0);
220  if((var.isValid())&&(var.type() == QVariant::Bool)){
221  transaction->digitalIo()->setValue(var.toBool(), false);
222  }else{
223  transaction->digitalIo()->setValue(mdtValue(), false);
224  }
225  }else{
226  transaction->digitalIo()->setValue(mdtValue(), false);
227  }
228  }else{
229  // Transaction contains first I/O address and qty of I/Os
230  ios()->updateDigitalInputValues(pvCodec->values(), transaction->address(), transaction->ioCount(), true);
231  }
232  break;
233  case 0x03: // Read holding registers
234  // Check if we have a answer for a single I/O or multiple I/Os
235  if((transaction->ioCount() == 1)&&(transaction->analogIo() != 0)){
236  // Single I/O reply
237  if(pvCodec->values().size() == 1){
238  var = pvCodec->values().at(0);
239  if((var.isValid())&&(var.type() == QVariant::Int)){
240  // Codec only get raw data, so let I/O do needed conversions
241  transaction->analogIo()->setValue(var.toInt(), false);
242  }else{
243  transaction->analogIo()->setValue(mdtValue(), false);
244  }
245  }else{
246  transaction->analogIo()->setValue(mdtValue(), false);
247  }
248  transaction->analogIo()->setEnabled(true);
249  }else{
250  // Multiple I/O reply
251  ios()->updateAnalogOutputValues(pvCodec->values(), transaction->address(), mdtDeviceIos::Read, transaction->ioCount(), true);
252  ios()->setAnalogOutputsEnabled(true);
253  }
254  break;
255  case 0x04: // Read input registers
256  // Check if we have a answer for a single I/O or multiple I/Os
257  if((transaction->ioCount() == 1)&&(transaction->analogIo() != 0)){
258  // Single I/O reply
259  if(pvCodec->values().size() == 1){
260  var = pvCodec->values().at(0);
261  if((var.isValid())&&(var.type() == QVariant::Int)){
262  // Codec only get raw data, so let I/O do needed conversions
263  transaction->analogIo()->setValue(var.toInt(), false);
264  }else{
265  transaction->analogIo()->setValue(mdtValue(), false);
266  }
267  }else{
268  transaction->analogIo()->setValue(mdtValue(), false);
269  }
270  }else{
271  // Multiple I/O reply
272  ios()->updateAnalogInputValues(pvCodec->values(), transaction->address(), transaction->ioCount(), true);
273  }
274  break;
275  case 0x05: // Write single coil reply
276  if(transaction->digitalIo() != 0){
277  // Check validity and update
278  if(pvCodec->values().size() == 2){
279  // Address
280  var = pvCodec->values().at(0);
281  if((var.isValid())&&(var.type() == QVariant::Int)&&(var.toInt() == transaction->digitalIo()->addressWrite())){
282  // Value
283  var = pvCodec->values().at(1);
284  if((var.isValid())&&(var.type() == QVariant::Bool)){
285  transaction->digitalIo()->setValue(var.toBool(), false);
286  }else{
287  transaction->digitalIo()->setValue(mdtValue(), false);
288  }
289  }else{
290  transaction->digitalIo()->setValue(mdtValue(), false);
291  }
292  }else{
293  transaction->digitalIo()->setValue(mdtValue(), false);
294  }
295  transaction->digitalIo()->setEnabled(true);
296  }
297  break;
298  case 0x06: // Write single register reply
299  if(transaction->analogIo() != 0){
300  // Check validity and update
301  if(pvCodec->values().size() == 2){
302  // Address
303  var = pvCodec->values().at(0);
304  if((var.isValid())&&(var.type() == QVariant::Int)&&(var.toInt() == transaction->analogIo()->addressWrite())){
305  // Value
306  var = pvCodec->values().at(1);
307  if((var.isValid())&&(var.type() == QVariant::Int)){
308  // Update (G)UI - Codec only get raw data, so let I/O do needed conversions
309  transaction->analogIo()->setValue(var.toInt(), false);
310  }
311  }else{
312  transaction->analogIo()->setValue(mdtValue(), false);
313  }
314  }else{
315  transaction->analogIo()->setValue(mdtValue(), false);
316  }
317  transaction->analogIo()->setEnabled(true);
318  }
319  break;
320  case 0x0F: // Write multiple coils
321  // Check validitiy and update
322  if(pvCodec->values().size() == 2){
323  // Start address
324  var = pvCodec->values().at(0);
325  if((var.isValid())&&(var.type() == QVariant::Int)&&(var.toInt() == transaction->address())){
326  // Qty of outputs set
327  var = pvCodec->values().at(1);
328  if((!var.isValid())||(var.type() != QVariant::Int)||(var.toInt() != transaction->ioCount())){
329  ios()->setDigitalOutputsValue(mdtValue());
330  }
331  }else{
332  ios()->setDigitalOutputsValue(mdtValue());
333  }
334  }else{
335  ios()->setDigitalOutputsValue(mdtValue());
336  }
337  ios()->setDigitalOutputsEnabled(true);
338  case 0x10: // Write multiple registers reply
339  // Check validitiy and update
340  if(pvCodec->values().size() == 2){
341  // Start address
342  var = pvCodec->values().at(0);
343  if((var.isValid())&&(var.type() == QVariant::Int)&&(var.toInt() == transaction->address())){
344  // Qty of outputs set
345  var = pvCodec->values().at(1);
346  if((!var.isValid())||(var.type() != QVariant::Int)||(var.toInt() != transaction->ioCount())){
347  ios()->setAnalogOutputsValue(mdtValue());
348  }
349  }else{
350  ios()->setAnalogOutputsValue(mdtValue());
351  }
352  }else{
353  ios()->setAnalogOutputsValue(mdtValue());
354  }
355  ios()->setAnalogOutputsEnabled(true);
356  break;
357  default:
359  pvLastError.setError("Received frame with unhandled function code (0x" + QString::number(fc, 16) + ")", mdtError::Warning);
360  MDT_ERROR_SET_SRC(pvLastError, "mdtDeviceModbus");
361  pvLastError.commit();
362  }
