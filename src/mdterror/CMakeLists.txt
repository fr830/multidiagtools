# Project file for Multi Diag Tools
# * Part: mdterror *

# Project
project (multidiagtools)

# List of source files
set (SOURCES_FILES 
      mdtError.cpp
      mdt/error/Logger.cpp
      mdt/error/LoggerBackend.cpp
      mdt/error/LoggerConsoleBackend.cpp
      mdt/error/LoggerFileBackend.cpp
    )

# Output library
add_library (mdterror${MDTLIB_VERSION_SUFFIX} SHARED ${SOURCES_FILES})
target_link_libraries(mdterror${MDTLIB_VERSION_SUFFIX} Qt5::Core)
target_link_libraries(mdterror${MDTLIB_VERSION_SUFFIX} "qtsingleapplication${MDTLIB_VERSION_SUFFIX}")

# Translations - We generate a .pro file, that simplify the usage of lupdate
file(WRITE "${PROJECT_SOURCE_DIR}/../../translations/mdterror.pro" "# Translations project file for lupdate/lrelease\n")
file(APPEND "${PROJECT_SOURCE_DIR}/../../translations/mdterror.pro" "# This file is generated by CMake, all changes will be lost\n")
file(APPEND "${PROJECT_SOURCE_DIR}/../../translations/mdterror.pro" "TRANSLATIONS = ${PROJECT_SOURCE_DIR}/../../translations/mdtutils_fr.ts\n")
file(APPEND "${PROJECT_SOURCE_DIR}/../../translations/mdterror.pro" "TRANSLATIONS += ${PROJECT_SOURCE_DIR}/../../translations/mdtutils_de.ts\n")
foreach(SRCFILE ${SOURCES_FILES})
  file(APPEND "${PROJECT_SOURCE_DIR}/../../translations/mdterror.pro" "SOURCES += ${PROJECT_SOURCE_DIR}/${SRCFILE}\n")
endforeach(SRCFILE)
foreach(FRMFILE ${FORM_FILES})
  file(APPEND "${PROJECT_SOURCE_DIR}/../../translations/mdterror.pro" "FORMS += ${PROJECT_SOURCE_DIR}/${FRMFILE}\n")
endforeach(FRMFILE)

# Headers to install
file(GLOB INSTALL_HEADERS *.h)
install(FILES ${INSTALL_HEADERS} DESTINATION "${HEADERS_DEST_DIR}/mdterror" COMPONENT dev)

# Install targets
install(TARGETS "mdterror${MDTLIB_VERSION_SUFFIX}" DESTINATION "${LIB_DEST_DIR}" COMPONENT lib)

# Strip library and put debug symbols into separate file
if(UNIX)
  add_custom_command(TARGET "mdterror${MDTLIB_VERSION_SUFFIX}"
                    POST_BUILD
                    COMMAND objcopy --only-keep-debug "libmdterror${MDTLIB_VERSION_SUFFIX}.so" "libmdterror${MDTLIB_VERSION_SUFFIX}.dbg"
                    COMMAND strip --strip-debug --strip-unneeded "libmdterror${MDTLIB_VERSION_SUFFIX}.so"
                    COMMAND objcopy --add-gnu-debuglink="libmdterror${MDTLIB_VERSION_SUFFIX}.dbg" "libmdterror${MDTLIB_VERSION_SUFFIX}.so"
                    COMMAND chmod -x "libmdterror${MDTLIB_VERSION_SUFFIX}.dbg")
  install(FILES "libmdterror${MDTLIB_VERSION_SUFFIX}.dbg" DESTINATION "${LIB_DEST_DIR}" COMPONENT dbg)
endif(UNIX)
