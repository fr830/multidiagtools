00037 bool mdtFrameCodecK8055::decode(QByteArray &data)
00038 {
00039   quint8 byte;
00040   int word;
00041 
00042   pvValues.clear();
00043   // Check data
00044   if(data.size() != 8){
00045     mdtError e(MDT_FRAME_DECODE_ERROR, "Wrong size, expected: 8", mdtError::Error);
00046     MDT_ERROR_SET_SRC(e, "mdtFrameCodecK8055");
00047     e.commit();
00048     return false;
00049   }
00051   // Digital in
00052   qDebug() << "mdtFrameCodecK8055::decode() Status: " << (quint8)data.at(1);
00053   byte = data.at(0);
00054   pvValues.append((bool)(byte & 0x10));
00055   pvValues.append((bool)(byte & 0x20));
00056   pvValues.append((bool)(byte & 0x01));
00057   pvValues.append((bool)(byte & 0x40));
00058   pvValues.append((bool)(byte & 0x80));
00059   // Analog in
00060   pvValues.append((int)((quint8)data.at(2)));
00061   pvValues.append((int)((quint8)data.at(3)));
00062   // Counters
00063   word = 0;
00064   word = ((quint8)data.at(4) << 8) | (quint8)data.at(5);
00065   pvValues.append(word);
00066   word = 0;
00067   word = ((quint8)data.at(6) << 8) | (quint8)data.at(7);
00068   pvValues.append(word);
00069 
00070   return true;
00071 }
