00112 void mdtSerialPortConfigWidget::updateConfig(mdtSerialPortConfig &config)
00113 {
00114   int index;
00115   char xon = 0;
00116   char xoff = 0;
00117 
00118   // Baud rate
00119   index = cbBaudRate->currentIndex();
00120   if(index >= 0){
00121     config.setBaudRate(cbBaudRate->itemData(index).toInt());
00122   }
00123   // Parity
00124   if(rbParityNone->isChecked()){
00125     config.setParity(mdtSerialPortConfig::NoParity);
00126   }else if(rbParityEven->isChecked()){
00127     config.setParity(mdtSerialPortConfig::ParityEven);
00128   }else{
00129     config.setParity(mdtSerialPortConfig::ParityOdd);
00130   }
00131   // Data bits count
00132   if(rbDataBit5->isChecked()){
00133     config.setDataBitsCount(5);
00134   }else if(rbDataBit6->isChecked()){
00135     config.setDataBitsCount(6);
00136   }else if(rbDataBit7->isChecked()){
00137     config.setDataBitsCount(7);
00138   }else{
00139     config.setDataBitsCount(8);
00140   }
00141   // Stop bits count
00142   if(rbStopBit1->isChecked()){
00143     config.setStopBitsCount(1);
00144   }else{
00145     config.setStopBitsCount(2);
00146   }
00147   // Flow control
00148   if(rbFlowCtlRtsCts->isChecked()){
00149     config.enableFlowCtlRtsCts();
00150   }else{
00151     config.diseableFlowCtlRtsCts();
00152   }
00153   if(rbFlowCtlXonXoff->isChecked()){
00154     // Get Xon char
00155     index = cbXonChar->currentIndex();
00156     if(index >= 0){
00157       xon = cbXonChar->itemData(index).toChar().toAscii();
00158     }
00159     // Get Xoff char
00160     index = cbXoffChar->currentIndex();
00161     if(index >= 0){
00162       xoff = cbXoffChar->itemData(index).toChar().toAscii();
00163     }
00164     config.enableFlowCtlXonXoff(xon, xoff);
00165   }else{
00166     config.diseableFlowCtlXonXoff();
00167   }
00168   if(rbFlowCtlNone->isChecked()){
00169     config.diseableFlowCtlRtsCts();
00170     config.diseableFlowCtlXonXoff();
00171   }
00172   // Redisplay the config, so the user can see what was really setup
00173   displayConfig(config);
00174 }
