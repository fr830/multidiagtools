00056 QList<mdtPortInfo*> mdtSerialPortManager::scan()
00057 {
00058   Q_ASSERT(isClosed());
00059 
00060   mdtSerialPort *port;
00061   QStringList portNames;
00062   //QStringList availablePorts;
00063   QList<mdtPortInfo*> availablePorts;
00064   mdtPortInfo *portInfo;
00065 
00066 #ifdef Q_OS_UNIX
00067   QFileInfoList filesInfo;
00068   QDir dir;
00069   QStringList extList;
00070 
00071   // We only need system files
00072   dir.setFilter(QDir::Files | QDir::System);
00073   // Default Linux interfaces
00074   extList << "ttyS*";
00075   // USB interfaces
00076   extList << "ttyUSB*";
00077   // Moxa USB interfaces
00078   extList << "ttyMXUSB*";
00079   // Moxa PCI interfaces
00080   extList << "ttyMI*";
00081   // CDC ACM (USB Modems)
00082   extList << "ttyACM*";
00083   // Common device directory
00084   dir.setPath("/dev");
00085   if(!dir.exists()){
00086     mdtError e(MDT_SERIAL_PORT_IO_ERROR, "directory '" + dir.dirName() + "' not exists", mdtError::Error);
00087     MDT_ERROR_SET_SRC(e, "mdtSerialPortManagerPosix");
00088     e.commit();
00089     return availablePorts;
00090   }
00091   dir.setNameFilters(extList);
00092   // Gest list of found files
00093   filesInfo = dir.entryInfoList();
00094   // Construct the list of absolute file paths
00095   for(int i=0; i<filesInfo.size(); ++i){
00096     portNames << filesInfo.at(i).absoluteFilePath();
00097   }
00098   // Sort the list
00099   portNames = mdtAlgorithms::naturalSort(portNames, Qt::CaseInsensitive);
00100 #endif  // #ifdef Q_OS_UNIX
00101 
00102 #ifdef Q_OS_WIN
00103   QString str;
00104 
00105   // I found nothing better than this ..
00106   for(int i=1; i<=20; i++){
00107     str = QString("COM%1").arg(i);
00108     portNames << str;
00109   }
00110 #endif  // #ifdef Q_OS_WIN
00111 
00112   // For each port name, try to open the port and get some attributes (to see if it really exists)
00113   port = new mdtSerialPort;
00114   for(int i=0; i<portNames.size(); ++i){
00115     // Try to open port and get UART type
00116     Q_ASSERT(port != 0);
00117     port->setPortName(portNames.at(i));
00118     if(port->open() == mdtAbstractPort::NoError){
00119       if(port->uartType() != mdtAbstractSerialPort::UT_UNKNOW){
00120         portInfo = new mdtPortInfo;
00121         Q_ASSERT(portInfo != 0);
00122         portInfo->setPortName(portNames.at(i));
00123         portInfo->setDisplayText(portNames.at(i));
00124         //availablePorts.append(portNames.at(i));
00125         availablePorts.append(portInfo);
00126       }
00127       port->close();
00128     }
00129   }
00130   delete port;
00131 
00132   return availablePorts;
00133 }
