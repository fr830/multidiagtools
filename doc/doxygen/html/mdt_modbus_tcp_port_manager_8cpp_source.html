00314 int mdtModbusTcpPortManager::getHardwareNodeAddress(int bitsCount, int startFrom)
00315 {
00316   Q_ASSERT(isRunning());
00317   Q_ASSERT(bitsCount > 0);
00318   Q_ASSERT(startFrom >= 0);
00319 
00320   int transactionId;
00321   mdtPortTransaction *transaction;
00322   QByteArray pdu;
00323   mdtFrameCodecModbus codec;
00324   int i, id;
00325 
00326   // Setup MODBUS PDU
00327   pdu = codec.encodeReadDiscreteInputs(startFrom, bitsCount);
00328   if(pdu.isEmpty()){
00329     return (int)mdtAbstractPort::UnhandledError;
00330   }
00331   // Get a new transaction
00332   transaction = getNewTransaction();
00333   // Send query
00334   transaction->setQueryReplyMode(true);
00335   transactionId = writeData(pdu, transaction);
00336   if(transactionId < 0){
00337     restoreTransaction(transaction);
00338     return transactionId;
00339   }
00340   // Wait on result (use device's defined timeout)
00341   if(!waitOnFrame(transactionId)){
00342     restoreTransaction(transaction);
00343     return (int)mdtAbstractPort::ReadTimeout;
00344   }
00345   // At this state, transaction will be restored by readenFrame()
00346   if(codec.decode(readenFrame(transactionId)) != 0x02){
00347     return (int)mdtAbstractPort::UnhandledError;
00348   }
00349   // Check that we have enough states (can be more, because digital inputs are returned as multiple of 8)
00350   if(codec.values().size() < bitsCount){
00351     mdtError e(MDT_TCP_IO_ERROR, "Have not received enough input states", mdtError::Error);
00352     MDT_ERROR_SET_SRC(e, "mdtModbusTcpPortManager");
00353     e.commit();
00354     return (int)mdtAbstractPort::UnhandledError;
00355   }
00357   // Ok, have the states, let's decode ID
00358   id = 0;
00359   for(i=0; i<bitsCount; i++){
00360     if((codec.values().at(i).isValid())&&(codec.values().at(i).toBool() == true)){
00361       id |= (1<<i);
00362     }
00363   }
00364 
00365   return id;
