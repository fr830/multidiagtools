77 void mdtMvColumnSelectionDialog::displaySelectedColumns()
78 {
79  Q_ASSERT(pvModel != 0);
80  Q_ASSERT(pvView != 0);
81  Q_ASSERT(pvView->horizontalHeader() != 0);
82 
83  int i;
84  unsigned int ui;
85  QList<mdtItemsSelectorDialogItem> items;
86  QHeaderView *hdv = pvView->horizontalHeader();
87  std::vector<int> visualIndexes;
88  int col;
89 
90  // Build list of visual indexes and sort it
91  for(i = 0; i < hdv->count(); ++i){
92  if(!hdv->isSectionHidden(i)){
93  visualIndexes.push_back(hdv->visualIndex(i));
94  }
95  }
96  std::sort(visualIndexes.begin(), visualIndexes.end());
97  // Build items list
98  for(ui = 0; ui < visualIndexes.size(); ++ui){
99  mdtItemsSelectorDialogItem item;
100  col = hdv->logicalIndex(visualIndexes[ui]);
101  item.setColumnIndex(col);
102  item.setText(pvModel->headerData(col, Qt::Horizontal).toString());
103  items.append(item);
104  }
105  setSelectedItems(items);
106 }
