00069 
00070   // Setup device info
00071   device = devInfo;
00072   device.setVendorId(0x0957);
00073   device.setProductId(0x4d18);
00074   // Scan for ports having a USBTMC device attached
00075   port = 0;
00076   ports = pvPortManager->scan();

00078     Q_ASSERT(ports.at(i) != 0);
00079     // Search in devices list
00080     qDebug() << "port " << ports.at(i)->portName();
00081     devices = ports.at(i)->deviceInfoList();
00082     for(j=0; j<devices.size(); j++){
00083       Q_ASSERT(devices.at(j) != 0);
00084       // Check if VID and PID matches
00085       if((devices.at(j)->vendorId() == device.vendorId())&&(devices.at(j)->productId() == device.productId())){
00086         // If requested, check if serial ID matches
00087         if(!device.serialId().isEmpty()){
00088           if(devices.at(j)->serialId() == device.serialId()){
00089             port = ports.at(i);
00090             i = ports.size();
00091             break;
00092           }
00093         }else{
00094           // Only PID and VID must match
00095           port = ports.at(i);
00096           i = ports.size();
00097           break;
00098         }
00099       }
00100     }
00101   }
00102   // Open port if found
00103   if(port != 0){
00104     pvPortManager->setPortInfo(*port);
00105     if(pvPortManager->openPort()){
00106       if(pvPortManager->start()){
00107         retVal = mdtAbstractPort::NoError;
00108       }else{
00109         retVal = mdtAbstractPort::UnhandledError;
00110       }
00111     }else{
00112       retVal = mdtAbstractPort::UnhandledError;
00113     }
00114   }else{
00115     // Device not found
00117     retVal = mdtAbstractPort::PortNotFound;
00118   }
00119   // Free port info list
00120   qDeleteAll(ports);
