86 {

88  QVariant testId;
89  mdtSqlRecord testData;
90 
91  // Setup test data
92  if(!testData.addAllFields("Test_tbl", database())){
93  pvLastError = testData.lastError();
94  return QVariant();
95  }
96  testData.setValue("TestModel_Id_FK", testModelId);
97  testData.setValue("Date", QDate::currentDate());

99  if(!beginTransaction()){

372 mdtTtTestItemNodeSetupData mdtTtTest::getSetupData(const QVariant & testModelItemId, bool & ok)
373 {
374  QString sql;
375  QList<QSqlRecord> dataList;
376  mdtTtTestItemNodeSetupData tiSetupData;
377  mdtTtTestNodeUnitSetupData tnuSetupData;
378  QVector<int> nodeIdList;
379  int i, k;
380  int id;
381 
382  // Get data from db
384  sql = "SELECT * FROM TestItemNodeUnitSetup_view WHERE TestModelItem_Id_FK = " + testModelItemId.toString();
385  dataList = getData(sql, &ok);
386  if(!ok){
387  return tiSetupData;
388  }
389  // Build a list of nodes
390  for(i = 0; i < dataList.size(); ++i){
391  Q_ASSERT(!dataList.at(i).value("TestNode_Id_FK").isNull());
392  id = dataList.at(i).value("TestNode_Id_FK").toInt();
393  if(!nodeIdList.contains(id)){
394  nodeIdList.append(id);
395  }
396  }
398  // For each test node, build test node setup data
399  for(i = 0; i < nodeIdList.size(); ++i){
400  mdtTtTestNodeSetupData tnSetupData;
401  // Add all unit setup data that are part of current node
402  for(k = 0; k < dataList.size(); ++k){
403  id = dataList.at(k).value("TestNode_Id_FK").toInt();
404  if(nodeIdList.at(i) == id){
405  tnuSetupData = dataList.at(k);
406  if(tnSetupData.nodeIdentification().isEmpty()){
407  tnSetupData.setNodeIdentification(tnuSetupData.value("NodeIdentification").toString());
408  }
409  tnSetupData.addUnitSetup(tnuSetupData);
410  }
411  }
412  // Add node setup data to test item setup data
413  tiSetupData.addNodeSetupData(tnSetupData);
414  }
415 
416  return tiSetupData;
417 }

489 bool mdtTtTest::addTestItems(const QVariant & testId, const QVariant & testModelId)
490 {
491  QString sql;
492  QList<QVariant> modelItemsDataList;
493  mdtSqlRecord testItemData;
494  bool ok;
495  int i;
496 
497  // Get current test ID
498  if(testId.isNull()){
499  pvLastError.setError(tr("Could not add test items because current test id (Id_PK) is Null."), mdtError::Error);
500  MDT_ERROR_SET_SRC(pvLastError, "mdtTtTest");
501  pvLastError.commit();
502  return false;
503  }
504  // Get items for given test model
505  sql = "SELECT Id_PK FROM TestModelItem_tbl WHERE TestModel_Id_FK = " + testModelId.toString();
506  modelItemsDataList = getDataList<QVariant>(sql, ok);
507  if(!ok){
508  return false;
509  }
510  // Setup test item data
511  if(!testItemData.addAllFields("TestItem_tbl", database())){
512  pvLastError = testItemData.lastError();
513  return false;
514  }
515  // Add a new test item for each model item
516  for(i = 0; i < modelItemsDataList.size(); ++i){
517  testItemData.clearValues();
518  testItemData.setValue("Test_Id_FK", testId);
519  testItemData.setValue("TestModelItem_Id_FK", modelItemsDataList.at(i));
520  if(!addRecord(testItemData, "TestItem_tbl")){
521  return false;
522  }
523  }
524 
525  return true;
526 }
