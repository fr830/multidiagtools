104 bool mdtClDirectLink::addLink(const QVariant & unitConnectionStartId, const QVariant & unitConnectionEndId)
105 {
106  QList<QSqlRecord> leftDataList; // Data for start part in DirectLink_tbl
107  QList<QSqlRecord> rightDataList; // Data for end part in DirectLink_tbl
108  QList<QSqlRecord> dataList;
109  mdtSqlRecord record;
110  QString sql;
111  bool ok;
112  int i;
113 
114  // Get left data part (that will be stored as start in DirectLink_tbl)
115  dataList = getStartData(unitConnectionStartId, &ok);
116  if(!ok){
117  return false;
118  }
119  addDataList(dataList, leftDataList);
120  dataList = getEndData(unitConnectionStartId, &ok);
121  if(!ok){
122  return false;
123  }
124  addDataList(dataList, leftDataList);
125  // Get right data part (that will be stored as end in DirectLink_tbl)
126  dataList = getStartData(unitConnectionEndId, &ok);
127  if(!ok){
128  return false;
129  }
130  addDataList(dataList, rightDataList);
131  dataList = getEndData(unitConnectionEndId, &ok);
132  if(!ok){
133  return false;
134  }
135  addDataList(dataList, rightDataList);
136  /*
137  * Because this feature is very experimental, we go to error if left and right data part have not the same size
138  */
139  if(leftDataList.size() != rightDataList.size()){
140  pvLastError.setError(tr("Failed to get Left/Right data part properly."), mdtError::Error);
141  MDT_ERROR_SET_SRC(pvLastError, "mdtClDirectLink");
142  pvLastError.commit();
143  return false;
144  }
145  // Add data - Note: transactions are handled outside this method
146  for(i = 0; i < leftDataList.size(); ++i){
147  record = getRecordForTable(leftDataList.at(i), rightDataList.at(i));
148  if(record.isEmpty()){
149  return false;
150  }
151  if(!addRecord(record, "DirectLink_tbl")){
152  return false;
153  }
154  }
155 
156  return true;
157 }

345 bool mdtClDirectLink::connectionsAreTheSame(const QSqlRecord & A, const QSqlRecord & B)
346 {
347  int connectionIdIndexA;
348  int vhctIdIndexA;
349  int connectionIdIndexB;
350  int vhctIdIndexB;
351 
352  // Get unit connection indexe in A
353  connectionIdIndexA = A.indexOf("UnitConnectionStart_Id_FK");
354  if(connectionIdIndexA < 0){
355  connectionIdIndexA = A.indexOf("UnitConnectionEnd_Id_FK");
356  }
357  if(connectionIdIndexA < 0){
358  pvLastError.setError(tr("Cannot find field UnitConnection[Start|End]_Id_FK in A"), mdtError::Error);
359  MDT_ERROR_SET_SRC(pvLastError, "mdtClDirectLink");
360  pvLastError.commit();
361  return false;
362  }
363  // Get vehicle type index in A
364  vhctIdIndexA = A.indexOf("StartVehicleType_Id_PK");
365  if(vhctIdIndexA < 0){
366  vhctIdIndexA = A.indexOf("EndVehicleType_Id_PK");
367  }
368  if(vhctIdIndexA < 0){
369  pvLastError.setError(tr("Cannot find field [Start|End]VehicleType_Id_PK in A"), mdtError::Error);
370  MDT_ERROR_SET_SRC(pvLastError, "mdtClDirectLink");
371  pvLastError.commit();
372  return false;
373  }
374  // Get unit connection indexe in B
375  connectionIdIndexB = B.indexOf("UnitConnectionStart_Id_FK");
376  if(connectionIdIndexB < 0){
377  connectionIdIndexB = B.indexOf("UnitConnectionEnd_Id_FK");
378  }
379  if(connectionIdIndexB < 0){
380  pvLastError.setError(tr("Cannot find field UnitConnection[Start|End]_Id_FK in B"), mdtError::Error);
381  MDT_ERROR_SET_SRC(pvLastError, "mdtClDirectLink");
382  pvLastError.commit();
383  return false;
384  }
385  // Get vehicle type index in B
386  vhctIdIndexB = B.indexOf("StartVehicleType_Id_PK");
387  if(vhctIdIndexB < 0){
388  vhctIdIndexB = B.indexOf("EndVehicleType_Id_PK");
389  }
390  if(vhctIdIndexB < 0){
391  pvLastError.setError(tr("Cannot find field [Start|End]VehicleType_Id_PK in B"), mdtError::Error);
392  MDT_ERROR_SET_SRC(pvLastError, "mdtClDirectLink");
393  pvLastError.commit();
394  return false;
395  }
396  // Check if A and B are equal
397  if((A.value(connectionIdIndexA) == B.value(connectionIdIndexB)) && (A.value(vhctIdIndexA) == B.value(vhctIdIndexB))){
398  return true;
399  }
400 
401  return false;
