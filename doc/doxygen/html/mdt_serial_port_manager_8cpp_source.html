66 #ifdef Q_OS_UNIX

69  QStringList extList;
70 
71  // We only need system files
72  dir.setFilter(QDir::Files | QDir::System);
73  // Default Linux interfaces
74  extList << "ttyS*";
75  // USB interfaces
76  extList << "ttyUSB*";
77  // Moxa USB interfaces
78  extList << "ttyMXUSB*";
79  // Moxa PCI interfaces
80  extList << "ttyMI*";
81  // CDC ACM (USB Modems)
82  extList << "ttyACM*";
83  // Common device directory
84  dir.setPath("/dev");
85  if(!dir.exists()){
86  mdtError e(MDT_SERIAL_PORT_IO_ERROR, "directory '" + dir.dirName() + "' not exists", mdtError::Error);
87  MDT_ERROR_SET_SRC(e, "mdtSerialPortManagerPosix");
88  e.commit();
89  return availablePorts;
90  }
91  dir.setNameFilters(extList);
92  // Gest list of found files
93  filesInfo = dir.entryInfoList();
94  // Construct the list of absolute file paths
95  for(int i=0; i<filesInfo.size(); ++i){
96  portNames << filesInfo.at(i).absoluteFilePath();
97  }
98  // Sort the list
99  portNames = mdtAlgorithms::naturalSort(portNames, Qt::CaseInsensitive);
100 #endif // #ifdef Q_OS_UNIX
101 
102 #ifdef Q_OS_WIN
103  QString str;
104 
105  // I found nothing better than this ..
106  for(int i=1; i<=20; i++){
107  str = QString("COM%1").arg(i);
108  portNames << str;
109  }
110 #endif // #ifdef Q_OS_WIN
111 
112  // For each port name, try to open the port and get some attributes (to see if it really exists)
113  port = new mdtSerialPort;
114  for(int i=0; i<portNames.size(); ++i){
115  // Try to open port and get UART type
116  Q_ASSERT(port != 0);
117  port->setPortName(portNames.at(i));
118  if(port->open() == mdtAbstractPort::NoError){
119  if(port->uartType() != mdtAbstractSerialPort::UT_UNKNOW){
120  portInfo = new mdtPortInfo;
121  Q_ASSERT(portInfo != 0);
122  portInfo->setPortName(portNames.at(i));
123  portInfo->setDisplayText(portNames.at(i));
124  //availablePorts.append(portNames.at(i));
125  availablePorts.append(portInfo);
126  }
127  port->close();
128  }
129  }
130  delete port;
131 
132  return availablePorts;
133 }
