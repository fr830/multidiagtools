00037 bool mdtSqlFormWidgetDataValidator::checkBeforeSubmit()
00038 {
00039   int i;
00040   bool allOk = true;
00041   QWidget *firstNokWidget = 0;
00042 
00043   // We call checkBeforeSubmit() for all items
00044   for(i=0; i<pvFieldHandlers.size(); ++i){
00045     Q_ASSERT(pvFieldHandlers.at(i) != 0);
00046     if(!pvFieldHandlers.at(i)->checkBeforeSubmit()){
00047       allOk = false;
00048       if(firstNokWidget == 0){
00049         firstNokWidget = pvFieldHandlers.at(i)->dataWidget();
00050       }
00051     }
00052   }
00053   // We set the focus on first Nok widget
00054   if(firstNokWidget != 0){
00055     firstNokWidget->setFocus();
00056   }
00057   // Display a warning to the user
00058   if(!allOk){
00059     displayWarning(tr("There are some errors in edited data."), tr("Fields that are not correct should be highlighted.\nMoving cursor over field with error should display the reason\nPlease correct errors, or cancel modifications, and try again.") );
00060   }
00061 
00062   return allOk;
00063 }
