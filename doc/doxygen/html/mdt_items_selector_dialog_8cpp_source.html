147 void mdtItemsSelectorDialog::transferSelectedItemsToSelectedItems()
148 {
149  Q_ASSERT(lwAvailableItems->selectionModel() != 0);
150 
151  QModelIndexList indexes = lwAvailableItems->selectionModel()->selectedIndexes();
152  QModelIndex index;
153  unsigned int ui;
154  std::vector<int> rowList;
155  mdtItemsSelectorDialogItem item;
156 
157  // Build list of rows to remove from available items
158  foreach(index, indexes){
159  rowList.push_back(index.row());
160  }
161  // Transfer items
162  for(ui = 0; ui < rowList.size(); ++ui){
163  Q_ASSERT(rowList[ui] < pvAvailableItems.size());
164  item = pvAvailableItems.at(rowList[ui]);
165  pvSelectedItems.append(item);
166  }
167  // Remove transfered items
168  std::sort(rowList.rbegin(), rowList.rend());
169  for(ui = 0; ui < rowList.size(); ++ui){
170  Q_ASSERT(rowList[ui] < pvAvailableItems.size());
171  pvAvailableItems.removeAt(rowList[ui]);
172  }
173  // Update models
174  updateAvailableItems(-1);
175  updateSelectedItems(-1);
176 }

307  if(pvSelectedItems.at(row).sortOrder() == Qt::AscendingOrder){

315 void mdtItemsSelectorDialog::updateSortOrderOfCurrentItemInSelectedItemsList(int cbSortOrderIndex)

318

321

328  switch(cbSortOrderIndex){

335  }
