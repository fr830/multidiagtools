00042 void mdtSerialPortCtlThread::run()
00043 {
00044   mdtAbstractPort::error_t portError = mdtAbstractPort::NoError;
00045 
00046   Q_ASSERT(pvPort != 0);
00047 
00048   // We need a mdtSerialPort instance here
00049   mdtSerialPort *port = dynamic_cast<mdtSerialPort*>(pvPort);
00050   Q_ASSERT(port != 0);
00051 
00052 #ifdef Q_OS_UNIX
00053   pvNativePthreadObject = pthread_self();
00054   Q_ASSERT(pvNativePthreadObject != 0);
00055 #endif
00056 
00057   // Set the running flag
00058   port->lockMutex();
00059   pvRunning = true;
00060   // Notify that we are ready
00061   notifyError(mdtAbstractPort::NoError);
00062 
00063   // Run...
00064   while(1){
00065     // Read thread state
00066     if(!pvRunning){
00067       break;
00068     }
00069     // Wait on ctl signal event
00070     portError = port->waitEventCtl();
00071     if(portError != mdtAbstractPort::NoError){
00072       if(!pvRunning){
00073         break;
00074       }
00075       // Unhandled error. Signal this and stop
00076       notifyError(portError);
00077       break;
00078     }
00079     // We have a event here, update the flags
00080     if(!port->getCtlStates()){
00081       break;
00082     }
00083   }
00084 
00085   if(portError == mdtAbstractPort::NoError){
00086     notifyError(mdtAbstractPort::Disconnected);
00087   }
00088   pvRunning = false;
00089   port->unlockMutex();
00090 }
