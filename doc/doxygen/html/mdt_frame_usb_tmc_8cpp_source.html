00053 int mdtFrameUsbTmc::putData(const char *data, int maxLen)
00054 {
00055   Q_ASSERT(capacity() > 12);
00056 
00057   int frameSize = 0;
00058   int stored = 0;
00059 
00060   // If frame is allready full, return
00061   if(bytesToStore() < 1){
00062     return 0;
00063   }
00064 
00065   qDebug() << "mdtFrameUsbTmc::putData(): size(): " << size();
00066   qDebug() << "mdtFrameUsbTmc::putData(): maxLen (0): " << maxLen;
00067 
00068   // Check wat's possible to store
00069   if(maxLen > remainCapacity()){
00070     maxLen = remainCapacity();
00071   }
00072   qDebug() << "mdtFrameUsbTmc::putData(): maxLen (1): " << maxLen;
00073   // store
00074   append(data, maxLen);
00075   stored = maxLen;
00076 
00077   // Check if we have enougth data to decode header
00078   if(size() >= 12){
00079     // Check if header was not allready decoded
00080     if(pvTransferSize == 0){
00081       // We have a header here, decode it
00082       pvMsgID = at(0);
00083       pvbTag = at(1);
00084       pvbTagInverse = at(2);
00085       // Check bTag
00086       if(pvbTagInverse != (quint8)(~pvbTag)){
00088         qDebug() << "mdtFrameUsbTmc::putData(): bTag/bTagInverse incoherent, bTag: " << pvbTag << " , expected inverse: " << (quint8)~pvbTag << " , rx inverse: " << pvbTagInverse;
00089       }
00090       pvTransferSize = at(4);
00091       pvTransferSize += (at(5) << 8);
00092       pvTransferSize += (at(6) << 16);
00093       pvTransferSize += (at(7) << 24);
00094       // MsgID specific part
00095       if(pvMsgID == DEV_DEP_MSG_IN){
00096         // EOM
00097         if(at(8) & 0x01){
00098           pvEOM = true;
00099         }else{
00100           pvEOM = false;
00101         }
00103       }else{
00104         qDebug() << "mdtFrameUsbTmc::putData(): MsgID " << pvMsgID << " not supported";
00105       }
00106     }
00107     // Calc total frame size, without alignment bytes
00108     frameSize = pvTransferSize + 12;
00109     // Check if we have a complete frame
00110     if(size() >= frameSize){
00111       pvEOFcondition = true;
00112       qDebug() << "Store: " << right(size()-12).left(pvTransferSize);
00113       pvMessageData += right(size()-12).left(pvTransferSize);
00114     }
00115   }
00116 
00117   return stored;
00118 }

00162 void mdtFrameUsbTmc::encode()
00163 {
00164   // Check minimal needed capacity
00165   if(capacity() < 16){
00166     reserve(16);
00167   }
00168   mdtFrame::clear();
00170   // Get message data size
00171   // Transfer size
00172   if(pvMsgID == DEV_DEP_MSG_OUT){
00173     pvTransferSize = pvMessageData.size();
00174   }else if(pvMsgID == DEV_DEP_MSG_IN){
00175     pvTransferSize = remainCapacity() - 12 - 3; 
00176   }else{
00177     qDebug() << "mdtFrameUsbTmc::encode(): Unknow pvMsgID, transfer size will be set to 0";
00178     pvTransferSize = 0;
00179   }
00180   qDebug() << "Transfer size: " << pvTransferSize;
00181   // Build USBTMC header
00182   append((char)pvMsgID);
00183   append((char)pvbTag);
00184   append((char)pvbTagInverse);
00185   append((char)0x00);
00186   append((char)pvTransferSize & 0x0F);          // TransferSize, LLSB
00187   append((char)((pvTransferSize >> 8) & 0x0F));   // TransferSize, LSB
00188   append((char)((pvTransferSize >> 16) & 0x0F));  // TransferSize, LSB
00189   append((char)((pvTransferSize >> 24) & 0x0F));  // TransferSize, LSB
00190   if(pvEOM){
00191     append((char)0x01);                           // bmTransferAttributes
00192   }else{
00193     append((char)0x00);                           // bmTransferAttributes
00194   }
00195   append((char)0x00);                           // Reserved
00196   append((char)0x00);                           // Reserved
00197   append((char)0x00);                           // Reserved
00198   // Add data
00199   append(pvMessageData);
00200   // Add alignment bytes if requierd
00201   while((size() % 4) != 0){
00202     append((char)0x00);
00204   }
00205 }
