00052 mdtAbstractPort::error_t mdtDeviceU3606A::connectToDevice(const mdtDeviceInfo &devInfo)
00053 {
00054   mdtPortInfo *port;
00055   QList<mdtPortInfo*> ports;
00056   QList<mdtDeviceInfo*> devices;
00057   mdtDeviceInfo device;
00058   int i, j;
00059   mdtAbstractPort::error_t retVal;
00060 
00061   // Setup device info
00062   device = devInfo;
00063   device.setVendorId(0x0957);
00064   device.setProductId(0x4d18);
00065   // Scan for ports having a USBTMC device attached
00066   port = 0;
00067   ports = pvPortManager->scan();
00068   for(i=0; i<ports.size(); i++){
00069     Q_ASSERT(ports.at(i) != 0);
00070     // Search in devices list
00071     qDebug() << "port " << ports.at(i)->portName();
00072     devices = ports.at(i)->deviceInfoList();
00073     for(j=0; j<devices.size(); j++){
00074       Q_ASSERT(devices.at(j) != 0);
00075       // Check if VID and PID matches
00076       if((devices.at(j)->vendorId() == device.vendorId())&&(devices.at(j)->productId() == device.productId())){
00077         // If requested, check if serial ID matches
00078         if(!device.serialId().isEmpty()){
00079           if(devices.at(j)->serialId() == device.serialId()){
00080             port = ports.at(i);
00081             i = ports.size();
00082             break;
00083           }
00084         }else{
00085           // Only PID and VID must match
00086           port = ports.at(i);
00087           i = ports.size();
00088           break;
00089         }
00090       }
00091     }
00092   }
00093   // Open port if found
00094   if(port != 0){
00095     pvPortManager->setPortInfo(*port);
00096     if(pvPortManager->openPort()){
00097       if(pvPortManager->start()){
00098         retVal = mdtAbstractPort::NoError;
00099       }else{
00100         retVal = mdtAbstractPort::UnhandledError;
00101       }
00102     }else{
00103       retVal = mdtAbstractPort::UnhandledError;
00104     }
00105   }else{
00106     // Device not found
00108     retVal = mdtAbstractPort::PortNotFound;
00109   }
00110   // Free port info list
00111   qDeleteAll(ports);
00112   ports.clear();
00113 
00114   return retVal;
00115 }
