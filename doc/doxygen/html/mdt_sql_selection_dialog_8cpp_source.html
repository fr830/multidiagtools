00147   Q_ASSERT(pvModel != 0);

00149 
00150   QList<QVariant> result;

00153   QModelIndexList selectedRows;
00154 
00155   if(pvDialogRejected){
00156     return result;

00158   selectedRows = pvTableView->selectionModel()->selectedRows();
00159   if(selectedRows.size() != 1){
00160     return result;
00161   }
00162   index = selectedRows.at(0);
00163   if(!index.isValid()){
00164     return result;
00165   }
00166   row = index.row();
00167   for(i = 0; i < pvSelectionResultColumns.size(); ++i){
00168     index = pvModel->index(row, pvSelectionResultColumns.at(i));
00169     result.append(pvModel->data(index));
00170   }
00171 
00172   return result;
