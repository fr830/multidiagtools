158 QString mdtError::systemErrorString(QObject *obj) const
159 {
160  QString str;
161  QString num;
162 
163  if(pvSystemText.isEmpty()){
164  return str;
165  }
166  num.setNum(pvSystemNumber);
167  if(obj != 0){
168  str = obj->tr("System returned error number ") + num + obj->tr(":\n");
169  }else{
170  str = "System returned error number " + num + ":\n";
171  }
172  str += pvSystemText;
173 
174  return str;
175 }
