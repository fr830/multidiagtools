116 bool mdtUsbtmcPort::openDevice(const mdtUsbDeviceDescriptor& deviceDescriptor, uint8_t bInterfaceNumber, bool clearDeviceList)
117 {
118  // Check given device descriptor
119  if(deviceDescriptor.isEmpty()){
120  pvLastError.setError(tr("Given device descriptor is empty (did you use scan() to obtain it ?)"), mdtError::Error);
121  MDT_ERROR_SET_SRC(pvLastError, "mdtUsbtmcPort");
122  pvLastError.commit();
123  return false;
124  }
125  // Init thread if required
126  if(pvThread == 0){
127  if(!initThread()){
128  return false;
129  }
130  }
131  Q_ASSERT(pvTransferHandler != 0);
132  // Set device name
133  if(pvDeviceName.isEmpty()){
134  pvDeviceName = deviceDescriptor.idString();
135  }
136  pvTransferHandler->setDeviceName(pvDeviceName);
137  // Start thread
138  if(!pvThread->start(deviceDescriptor, bInterfaceNumber)){
139  pvLastError = pvTransferHandler->lastError();
140  return false;
141  }
142  // Free internal scan result
143  if(clearDeviceList){
144  pvDeviceList->clear();
145  }
146  // Clear Bulk I/O
147  if(!clearBulkIo()){
148  return false;
149  }
150 
151  return true;
152 }

161 bool mdtUsbtmcPort::sendCommand(const QByteArray & command, int timeout)
162 {
163  Q_ASSERT(pvTransferHandler != 0);
164 
165  mdtUsbtmcTxMessage message(command);
166 
167  while(message.hasBytesToRead()){
168  pvTransactionState = TransactionState_t::Pending;
169  if(!pvTransferHandler->submitCommand(message, false, timeout)){
170  setError(pvTransferHandler->lastError());
172  return false;
173  }
174  waitTransactionFinished();
175  if(pvTransactionState != TransactionState_t::Done){
176  return false;
177  }
178  }
179  return true;
180 }

182 QByteArray mdtUsbtmcPort::sendQuery(const QByteArray& query, int timeout)
183 {
184  Q_ASSERT(pvTransferHandler != 0);
185 
186  mdtUsbtmcTxMessage txMessage(query);
187  QByteArray rxBa;
188  mdtUsbtmcRxMessage rxMessage(rxBa);
189 
190  /*
191  * Send query to device.
192  * TODO: if we want to send a long query (approx 4000 bytes),
193  * mdtUsbtmcTransferHandler and this class must be modified
194  * so that we can distingish when Bulk-OUT transfer is done,
195  * then send DEV_DEP_MSG_IN request and finaly get received data.
196  * For most common cases, query are very short, and we assume that
197  * they can be transmitted to device in one shot.
198  */
199  pvTransactionState = TransactionState_t::Pending;
200  if(!pvTransferHandler->submitCommand(txMessage, true, timeout)){
201  setError(pvTransferHandler->lastError());
203  return false;
204  }
205  waitTransactionFinished();
206  if(pvTransactionState != TransactionState_t::Done){
207  return rxBa;
208  }
209  pvTransferHandler->getReceivedData(rxMessage);
210 
211  return rxBa;
212 }
