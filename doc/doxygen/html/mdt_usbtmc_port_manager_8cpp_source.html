138 int mdtUsbtmcPortManager::sendData(mdtPortTransaction *transaction)
139 {
140  Q_ASSERT(pvPort != 0);
141  Q_ASSERT(transaction != 0);
142  Q_ASSERT(!transactionsDoneContains(transaction->id()));
143 
144  mdtFrameUsbTmc *frame;
145 
146  // Wait until we can write
147  while(1){
148  // If port manager was stopped, we return
149  if(isClosed()){
150  restoreTransaction(transaction);
151  return mdtAbstractPort::WriteCanceled;
152  }
153  if(isReady()){
154  lockPortMutex();
155  if(pvPort->writeFramesPool().size() > 0){
156  frame = dynamic_cast<mdtFrameUsbTmc*> (pvPort->writeFramesPool().dequeue());
157  Q_ASSERT(frame != 0);
158  // Here we are ready to write - we keep port mutext locked
159  break;
160  }
161  unlockPortMutex();
162  }
163  QCoreApplication::processEvents(QEventLoop::AllEvents | QEventLoop::WaitForMoreEvents);
164  }
165  if(!waitTransactionPossible()){
166  restoreTransaction(transaction);
167  unlockPortMutex();
168  return mdtAbstractPort::WriteCanceled;
169  }
170  // We are ready to write
171  Q_ASSERT(frame != 0);
172  frame->clear();
173  frame->clearSub();
174  // Store data and add frame to write queue
175  frame->setWaitAnAnswer(false);
176  frame->setMsgID(mdtFrameUsbTmc::DEV_DEP_MSG_OUT);
177  // Increment bTag and enshure it's in correct range (1-255)
178  incrementCurrentTransactionId(1, 255);
179  frame->setbTag(currentTransactionId());
180  frame->setMessageData(transaction->data());
181  frame->setEOM(true);
182  frame->encode();
183  pvPort->addFrameToWrite(frame);
184  transaction->setId(currentTransactionId());
185  unlockPortMutex();
186 
187  return currentTransactionId();
188 }

297 void mdtUsbtmcPortManager::fromThreadNewFrameReaden()
298 {
299  Q_ASSERT(pvPort != 0);
300 
301  mdtFrameUsbTmc *frame;
302  mdtPortTransaction *transaction;
303  int framesCount = 0;
304  // We need access to mdtUsbPort
305  mdtUsbPort *port = dynamic_cast<mdtUsbPort*>(pvPort);
306  Q_ASSERT(port != 0);
307 
308  // Get frames in readen queue
309  lockPortMutex();
310  while(pvPort->readenFrames().size() > 0){
311  frame = dynamic_cast<mdtFrameUsbTmc*> (pvPort->readenFrames().dequeue());
312  Q_ASSERT(frame != 0);
314  port->expectedBulkInbTags().removeOne(frame->bTag());
315  // Check if frame is complete
316  if(!frame->isComplete()){
317  mdtError e(MDT_USB_IO_ERROR, "Received a uncomplete frame", mdtError::Warning);
318  MDT_ERROR_SET_SRC(e, "mdtUsbtmcPortManager");
319  e.commit();
320  // Put frame back into pool
321  pvPort->readFramesPool().enqueue(frame);
322  continue;
323  }
324  // If we have a pending transaction, remove it
325  transaction = transactionPending(frame->bTag());
326  if(transaction == 0){
327  mdtError e(MDT_USB_IO_ERROR, "Received a frame with unexpected bTag number " + QString::number(frame->bTag()) + " (Should be handled by thread, this is a bug)", mdtError::Warning);
328  MDT_ERROR_SET_SRC(e, "mdtUsbtmcPortManager");
329  e.commit();
330  // Put frame back into pool
331  pvPort->readFramesPool().enqueue(frame);
332  continue;
333  }
334  // Check about MsgID
335  if(frame->MsgID() != mdtFrameUsbTmc::DEV_DEP_MSG_IN){
336  mdtError e(MDT_USB_IO_ERROR, \
337  "Received a frame with unhandled MsgID number " + QString::number(frame->MsgID()) + " , will be ignored", \
338  mdtError::Warning);
339  MDT_ERROR_SET_SRC(e, "mdtUsbtmcPortManager");
340  e.commit();
341  // Put frame back into pool and restore transaction
342  pvPort->readFramesPool().enqueue(frame);
343  restoreTransaction(transaction);
344  continue;
345  }
346  // Here we should have a valid frame
348  transaction->setId(frame->bTag());
349  transaction->setData(frame->messageData());
350  // Put frame back into pool and enqueue transaction in done queue
351  pvPort->readFramesPool().enqueue(frame);
352  addTransactionDone(transaction);
353  ++framesCount;
354  }
355  unlockPortMutex();
356  // Commit
357  if(framesCount > 0){
358  commitFrames();
359  }
360 }
