00395   Q_ASSERT(pvPort != 0);
00396   Q_ASSERT(transaction != 0);
00397   Q_ASSERT(!pvTransactionsPending.contains(transaction->id()));
00398   Q_ASSERT(!transactionsDoneContains(transaction->id()));
00399 
00400   mdtFrame *frame;
00401 
00402   // Get a frame in pool
00403   lockPortMutex();

00442 bool mdtPortManager::waitTransactionDone(int id)
00443 {
00444   Q_ASSERT(pvPort != 0);
00445 
00446   int maxIter;
00447 
00448   // Check if transaction exists in pending queue
00449   // Note: we check also done queue, because it's possible that transaction was done
00450   if((!pvTransactionsPending.contains(id))&&(!transactionsDoneContains(id))){
00451     mdtError e(MDT_PORT_IO_ERROR, "Wait on a frame that was never added to pending queue, id: " + QString::number(id), mdtError::Warning);
00452     MDT_ERROR_SET_SRC(e, "mdtPortManager");
00453     e.commit();
00454     emit(unhandledError());
00455     return false;
00456   }
00457   /*
00458    * If thread handles timeout/disconnection:
00459    *  In normal way, the thread notifies the error, the state becomes != Ready and this method returns false
00460    *  But, if many errors are notified, state changes often,
00461    *   and it happens that we are waiting, in Ready state, on a ID that was flushed, and this for a eternity.
00462    *   (Keep in mind that we let event loop running, and this way new transactions can be initalized, etc...)
00463    *  Another problem that is not signaled from thread is a frame the never completes.
00464    *  To avoid this problem, and keep coherence with port thread,
00465    *   we add a offset to wait timeout, and return if this time was elapsed.
00466    * If thread does not handle timeout:
00467    *  We simply return after timeout
00468    */
00469   if(pvThreadHandlesReadTimeout){
00470     maxIter = (config().readTimeout() + 200) / 50;
00471   }else{
00472     maxIter = config().readTimeout() / 50;
00473   }
00474   // Try until success or timeout/error
00475   while(!transactionsDoneContains(id)){
00476     // Check about timeout or other error
00477     //  Note: on busy state, we continue waiting (f.ex. ReadPoolEmpty error)
00478     if((pvCurrentState != Ready)&&(pvCurrentState != Busy)){
00479       // Transactions are restored by onThreadsErrorOccured()
00480       return false;
00481     }
00482     // Check about timeout
00483     if(maxIter <= 0){
00484       flushTransactionsPending();
00485       flushTransactionsDone();
00486       return false;
00487     }
00488     maxIter--;
00489     // Wait
00490     qApp->processEvents();
00491     msleep(50);
00492   }
00493 
00494   return true;
00495 }

00730   Q_ASSERT(!pvTransactionsPending.contains(transaction->id()));
00731 
00732   pvTransactionsPending.insert(transaction->id(), transaction);
00733   // If we have more transactions pending than write queue size, we go to busy state
