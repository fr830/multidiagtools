00089     libusb_close(handle);
00090   }
00091   // Get configuration(s)
00092   qDeleteAll(pvConfigs);

00100         config->fetchAttributes(configDescriptor);
00101         libusb_free_config_descriptor(configDescriptor);
00102         pvConfigs.append(config);
00103         break;
00104       case LIBUSB_ERROR_NOT_FOUND:

00110       }
00111       default:
00112         return err;
00113     }
00114   }else{
00115     for(i=0; i<descriptor.bNumConfigurations; i++){
00116       err = libusb_get_config_descriptor(device, i, &configDescriptor);
00117       switch(err){
00118         case 0:
00119           config = new mdtUsbConfigDescriptor;
00120           Q_ASSERT(config != 0);
00121           config->fetchAttributes(configDescriptor);

00124           break;
00125         case LIBUSB_ERROR_NOT_FOUND:
00126         {
00127           mdtError e(MDT_USB_IO_ERROR, "Unable to find get configuration descriptor at index " + QString::number(i), mdtError::Warning);
00128           MDT_ERROR_SET_SRC(e, "mdtUsbDeviceDescriptor");
00129           e.commit();
00130           break;
00131         }
00132         default:
