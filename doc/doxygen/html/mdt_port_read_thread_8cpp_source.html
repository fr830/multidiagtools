00046 void mdtPortReadThread::run()
00047 {
00048   Q_ASSERT(pvPort != 0);
00049 
00050   mdtAbstractPort::error_t portError = mdtAbstractPort::NoError;
00051   int n;
00052   bool useReadTimeoutProtocol = false;
00053   mdtPortThreadHelperPort threadHelper;
00054   threadHelper.setPort(pvPort);
00055   threadHelper.setThread(this);
00056 
00057   pvPort->lockMutex();
00058 #ifdef Q_OS_UNIX
00059   pvNativePthreadObject = pthread_self();
00060   Q_ASSERT(pvNativePthreadObject != 0);
00061 #endif
00062   // Get setup
00063   useReadTimeoutProtocol = pvPort->config().useReadTimeoutProtocol();
00064   pvMinPoolSizeBeforeReadSuspend = pvPort->config().readQueueSize() / 4;
00065   if((pvMinPoolSizeBeforeReadSuspend < 1)&&(pvPort->config().readQueueSize() > 0)){
00066     pvMinPoolSizeBeforeReadSuspend = 1;
00067   }
00068   /*
00069    * Check read timeout:
00070    *  - For this thread, read timeout has only sense for read timeout protocol
00071    *  - For other protocol, port manager must handle read timeout itself if needed (query/reply)
00072    * So, if useReadTimeoutProtocol is not set, we set port's readTimeout to infinite.
00073    * Port manager will use config's readTimeout, so we not change it here.
00074    */
00075   if((!useReadTimeoutProtocol)&&(pvPort->config().readTimeout() > -1)){
00076     pvPort->setReadTimeout(-1);
00077   }
00078   // Set the running flag and get a read frame
00079   pvRunning = true;
00080   // Get frame
00081   if(!threadHelper.getNewFrameRead()){
00082     return;
00083   }
00084   // Notify that we are ready
00086   emit ready(this);
00087 
00088   // Run...
00089   while(1){
00090     // Read thread state
00091     if(!pvRunning){
00092       break;
00093     }
00094     Q_ASSERT(threadHelper.currentReadFrame() != 0);
00095     // Wait on Read event
00096     portError = pvPort->waitForReadyRead();
00097     if(portError != mdtAbstractPort::NoError){
00098       // Read thread state
00099       if(!pvRunning){
00100         break;
00101       }
00102       if((useReadTimeoutProtocol)&&(portError == mdtAbstractPort::ReadTimeout)){
00103         // In timeout protocol, the current frame is considered complete
00104         if(!threadHelper.currentReadFrame()->isEmpty()){
00105           threadHelper.currentReadFrame()->setComplete();
00106           // Submit frame and notify newFrameReaden
00107           threadHelper.submitCurrentReadFrame(true);
00108           if(!threadHelper.getNewFrameRead()){
00109             break;
00110           }
00111         }
00112       }else{
00113         portError = threadHelper.handleCommonReadErrors(portError);
00114         if(portError != mdtAbstractPort::ErrorHandled){
00115           // Unhandled error, stop
00116           break;
00117         }
00118       }
00119     }
00120     // Check about flush request
00121     if(pvPort->flushInRequestPending()){
00122       // Put current frame back to pool, get new one and return idle
00123       threadHelper.restoreCurrentReadFrameToPool();
00124       if(!threadHelper.getNewFrameRead()){
00125         break;
00126       }
00127       continue;
00128     }
00129     // Read
00130     n = threadHelper.readFromPort(true);
00131     if(n < 0){
00132       // Check about stopping
00133       if(!pvRunning){
00134         break;
00135       }
00136       portError = threadHelper.handleCommonReadErrors((mdtAbstractPort::error_t)n);
00137       if(portError != mdtAbstractPort::ErrorHandled){
00138         // Unhandled error, stop
00139         break;
00140       }
00141     }
00142   }
00143 
00149   pvRunning = false;
00150   pvPort->unlockMutex();
00151 }
