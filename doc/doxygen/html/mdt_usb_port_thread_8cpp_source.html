00038 mdtAbstractPort::error_t mdtUsbPortThread::readUntilShortPacketReceived(int maxReadTransfers)
00039 {
00040   Q_ASSERT(pvPort != 0);
00041 
00042   mdtAbstractPort::error_t portError;
00043   mdtUsbPort *port = dynamic_cast<mdtUsbPort*>(pvPort);
00044   Q_ASSERT(port != 0);
00045   char *buffer = new char[port->readBufferSize()];
00046   int readen;
00047 
00048   while(maxReadTransfers > 0){
00049     readen = port->read(buffer, port->readBufferSize());
00050     if(readen < 0){
00051       delete[] buffer;
00052       return (mdtAbstractPort::error_t)readen;
00053     }
00054     if(readen < port->readBufferSize()){
00055       delete[] buffer;
00056       return mdtAbstractPort::NoError;
00057     }
00058     portError = port->initReadTransfer(port->readBufferSize());
00059     if(portError != mdtAbstractPort::NoError){
00060       delete[] buffer;
00061       return portError;
00062     }
00063     portError = port->handleUsbEvents();
00064     if(portError != mdtAbstractPort::NoError){
00065       delete[] buffer;
00066       return portError;
00067     }
00068     maxReadTransfers--;
00069   }
00070   delete[] buffer;
00071 
00072   return mdtAbstractPort::ReadTimeout;
