00049 bool mdtStateMachine::waitOnState(int state, int timeout)
00050 {
00051   pvTimeoutOccured = false;
00052   if(currentState() == state){
00053     return true;
00054   }
00055   if(timeout > 0){
00056     pvTimeoutTimer->start(timeout);
00057   }
00058   while(currentState() != state){
00059     if(pvTimeoutOccured){
00060       return false;
00061     }
00062     QCoreApplication::processEvents(QEventLoop::AllEvents | QEventLoop::WaitForMoreEvents);
00063   }
00064   pvTimeoutTimer->stop();
00065 
00066   return true;
00067 }

00069 void mdtStateMachine::notifyCurrentState()
00070 {
00071   if(pvCurrentState == 0){
00072     emit stateChanged(-1);
00073   }else{
00074     emit stateChanged(pvCurrentState->id());
00075     if(pvCurrentState->notifyEnteredToUi()){
00076       emit stateChangedForUi(pvCurrentState->id(), pvCurrentState->text(), pvCurrentState->ledColorId(), pvCurrentState->ledIsOn());
00077     }
00078   }
00079 }
