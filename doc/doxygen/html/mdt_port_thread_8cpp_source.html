92 void mdtPortThread::stop()
93 {
94  Q_ASSERT(pvPort != 0);
95 
96  int err;
97 
98  // Unset the running flag
99  qDebug() << "mdtPortThread::stop() ...";
100  pvPort->lockMutex();
101  qDebug() << "mdtPortThread::stop() mutex locked";
102  if(!pvRunning){
103  pvPort->unlockMutex();
104  qDebug() << "mdtPortThread::stop() running flag unset";
105  // Exit event loop (for subclass that use a event loop)
106  exit();
107  return;
108  }
109  pvRunning = false;
110  pvCurrentError = mdtAbstractPort::UnhandledError;
111  if(isWriter()){
112  pvPort->abortFrameToWriteWait();
113  }
114 #ifdef Q_OS_UNIX
115  if(QThread::isRunning()){
116  Q_ASSERT(pvNativePthreadObject != 0);
117  err = pthread_kill(pvNativePthreadObject, SIGALRM);
118  if(err != 0){
119  mdtError e(MDT_PORT_IO_ERROR, "pthread_kill() failed", mdtError::Error);
120  switch(err){
121  case EINVAL:
122  e.setSystemError(err, "Invalid signal was specified (EINVAL)");
123  case ESRCH:
124  e.setSystemError(err, "Specified thread ID not found (ESRCH)");
125  }
126  MDT_ERROR_SET_SRC(e, "mdtPortThread");
127  e.commit();
128  }
129  }
130 #endif
131  pvPort->unlockMutex();
132 
133  // Exit event loop (for subclass that use a event loop)
134  exit();
135 
136  qDebug() << "mdtPortThread::stop() DONE";
137 }

230 mdtAbstractPort::error_t mdtPortThread::reconnect(bool notify)
231 {
232  Q_ASSERT(pvPort != 0);
233 
234  int count = pvReconnectMaxTry;
235  mdtAbstractPort::error_t error;
236 
237  // Check about buggy caller
238  if(!pvRunning){
239  mdtError e(MDT_PORT_IO_ERROR, "Thread called reconnect() during a stop process, aborting", mdtError::Warning);
240  MDT_ERROR_SET_SRC(e, "mdtPortThread");
241  e.commit();
242  return mdtAbstractPort::UnhandledError;
243  }
244  if(notify){
245  notifyError(mdtAbstractPort::Connecting);
246  }
247  while(count > 0){
248  if(!pvRunning){
249  return mdtAbstractPort::Disconnected;
250  }
251  qDebug() << "mdtPortThread::reconnect(): trying ...";
252  error = pvPort->reconnect(pvReconnectTimeout);
253  if(error == mdtAbstractPort::NoError){
254  if(notify){
256  emit connected();
257  }
258  return mdtAbstractPort::NoError;
259  }
260  if(error != mdtAbstractPort::Disconnected){
261  if(notify){
262  notifyError(error);
263  }
264  return error;
265  }
266  count--;
267  }
268  // Max try reached
269  if(notify){
270  mdtError e(MDT_PORT_IO_ERROR, "Connection failed after max try", mdtError::Error);
271  MDT_ERROR_SET_SRC(e, "mdtPortThread");
272  e.commit();
274  notifyError(mdtAbstractPort::ConnectionFailed);
275  }
276 
277  return mdtAbstractPort::UnhandledError;
278 }
