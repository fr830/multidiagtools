00175     if (!socket)

00177 
00178     while (socket->bytesAvailable() < (int)sizeof(quint32))
00179         socket->waitForReadyRead();

00181     QByteArray uMsg;
00182     quint32 remaining;

00184     uMsg.resize(remaining);

00188         got = ds.readRawData(uMsgBuf, remaining);

00192     if (got < 0) {
00193         qWarning("QtLocalPeer: Message reception failed %s", socket->errorString().toLatin1().constData());
