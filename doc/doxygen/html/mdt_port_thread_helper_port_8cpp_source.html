00063 
00064   int readen;

00066   // Check if a new frame is needed

00124 mdtAbstractPort::error_t mdtPortThreadHelperPort::writeToPort(bool bytePerByteWrite, int interByteTime, int maxWriteTry)
00125 {
00126   Q_ASSERT(pvPort != 0);
00127   Q_ASSERT(pvThread != 0);
00128 
00129   int written = 0;
00130   mdtAbstractPort::error_t portError;
00131   mdtFrame *frame = currentWriteFrame();
00132 
00133   if(frame == 0){
00134     return mdtAbstractPort::NoError;
00135   }
00136   // Write the frame to port
00137   while(!frame->isEmpty()){
00138     // Wait on write ready event
00139     portError = pvPort->waitEventWriteReady();
00140     // Check about flush request
00141     if(pvPort->flushOutRequestPending()){
00142       // Restore frame to pool and return
00143       restoreCurrentWriteFrameToPool();
00144       return mdtAbstractPort::NoError;
00145     }
00146     if(portError != mdtAbstractPort::NoError){
00147       if(!pvThread->runningFlagSet()){
00148         return mdtAbstractPort::ErrorHandled;
00149       }
00150       return handleCommonWriteErrors(portError);
00151     }
00152     // Event occured, send the data to port
00153     if(bytePerByteWrite){
00154       written = writeDataToPort(1);
00155       pvPort->unlockMutex();
00156       msleep(interByteTime);
00157       pvPort->lockMutex();
00158     }else{
00159       written = writeDataToPort(-1);
00160     }
00161     if((written == 0)&&(!frame->isEmpty())){
00162       if(maxWriteTry <= 0){
00163         // Notify error and restore frame into write pool
00164         mdtError e(MDT_PORT_IO_ERROR, "Max write try reached after write busy", mdtError::Error);
00165         MDT_ERROR_SET_SRC(e, "mdtPortThreadHelperPort");
00166         e.commit();
00167         if(pvThread->runningFlagSet()){
00168           notifyError(mdtAbstractPort::WriteTimeout);
00169         }
00170         restoreCurrentWriteFrameToPool();
00171         // Thread has nothing else to do
00172         return mdtAbstractPort::ErrorHandled;
00173       }
00174       // Cannot write now, sleep some time and try later
00175       pvPort->unlockMutex();
00176       msleep(100);
00177       pvPort->lockMutex();
00178       maxWriteTry--;
00179     }else if(written < 0){
00180       if(!pvThread->runningFlagSet()){
00181         return mdtAbstractPort::ErrorHandled;
00182       }
00183       return handleCommonWriteErrors(portError);
00184     }
00185   }
00186   // Here, frame is completly sent
00187   Q_ASSERT(frame->isEmpty());
00188   restoreCurrentWriteFrameToPool();
00189 
00190   return mdtAbstractPort::NoError;
00191 }
