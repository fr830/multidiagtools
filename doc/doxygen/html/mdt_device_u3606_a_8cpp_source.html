00065 void mdtDeviceU3606A::decodeReadenFrame(mdtPortTransaction transaction)
00066 {
00067   bool ok;
00068 
00069   qDebug() << "mdtDeviceU3606A::decodeReadenFrame() , ID: " << transaction.id();
00070   
00072   switch(transaction.type()){
00074     case mdtFrameCodecScpi::QT_VALUE:
00075       if(transaction.analogIo() == 0){
00076         mdtError e(MDT_DEVICE_ERROR, "Device " + name() + ": received value response from device but no I/O was affected to query", mdtError::Warning);
00077         MDT_ERROR_SET_SRC(e, "mdtDeviceU3606A");
00078         e.commit();
00079       }else{
00080         ok = pvCodec->decodeValues(transaction.data());
00081         if(ok && (pvCodec->values().size() == 1)){
00083           transaction.analogIo()->setValue(pvCodec->values().at(0).value<mdtValue>(), false);
00084         }else{
00085           transaction.analogIo()->setValue(mdtValue(), false);
00086         }
00087       }
00088       break;
00089     case mdtFrameCodecScpi::QT_SINGLE_VALUE_FLT:
00090       if(transaction.analogIo() == 0){
00091         mdtError e(MDT_DEVICE_ERROR, "Device " + name() + ": received value response from device but no I/O was affected to query", mdtError::Warning);
00092         MDT_ERROR_SET_SRC(e, "mdtDeviceU3606A");
00093         e.commit();
00094       }else{
00096         transaction.analogIo()->setValue(pvCodec->decodeSingleValueDouble(transaction.data()), false);
00097       }
00098       break;
00100     case mdtFrameCodecScpi::QT_ERR:
00101       ok = pvCodec->decodeError(transaction.data());
00102       if(!ok){
00103         mdtError e(MDT_DEVICE_ERROR, "Device " + name() + ": SYST:ERR? reply could not be decoded", mdtError::Error);
00104         MDT_ERROR_SET_SRC(e, "mdtDeviceU3606A");
00105         e.commit();
00106         break;
00107       }
00108       handleDeviceError(pvCodec->values());
00109       break;
00110     default:
00111       mdtError e(MDT_DEVICE_ERROR, "Device " + name() + ": received unsupported response from device (type: " + QString::number(transaction.type()) + ")", mdtError::Warning);
00112       MDT_ERROR_SET_SRC(e, "mdtDeviceU3606A");
00113       e.commit();
00114   }
00115 }
