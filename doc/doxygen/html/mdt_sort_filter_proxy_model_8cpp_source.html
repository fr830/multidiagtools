00043 void mdtSortFilterProxyModel::addColumnToSortOrder(int column, Qt::SortOrder order)
00044 {
00045   // To sort columns in a given order, we simply sort each of them in reverse direction.
00046   // Here, we simply prepend each column. We do column index checking here.

00048     return;
