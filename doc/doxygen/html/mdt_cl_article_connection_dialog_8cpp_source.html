92 void mdtClArticleConnectionDialog::updateConnectionTypeText(const QString& type)
93 {
94  QLocale locale;
95  mdtClArticle art(0, pvDatabase);
96  QString fieldName;
97  QString sql;
98  QList<QSqlRecord> dataList;
99  bool ok;
100 
101  // Select field regarding language
102  switch(locale.language()){
103  case QLocale::French:
104  fieldName = "NameFR";
105  break;
106  case QLocale::German:
107  fieldName = "NameDE";
108  break;
109  case QLocale::Italian:
110  fieldName = "NameIT";
111  break;
112  default:
113  fieldName = "NameEN";
114  }
115  // Get text
116  sql = "SELECT " + fieldName + " FROM ConnectionType_tbl WHERE Code_PK = '" + type + "'";
117  dataList = art.getData(sql , &ok);
118  if(!ok){
119  lbConnectionTypeText->setText(tr("<Error!>"));
120  return;
121  }
122  if(dataList.size() < 1){
123  lbConnectionTypeText->setText("");
124  return;
125  }
126  lbConnectionTypeText->setText(dataList.at(0).value(fieldName).toString());
127 }
