00038 bool mdtSqlRecord::addField(const QString & fieldName, const QString & tableName, const QSqlDatabase & db) 
00039 {
00040   QSqlRecord dbRecord;
00041   QSqlField fld;
00042 
00043   // Get record object to fetch attributes
00044   dbRecord = db.record(tableName);
00045   if(dbRecord.isEmpty()){
00046     QSqlError sqlError = db.lastError();
00047     pvLastError.setError("Cannot get field information. Field: " + fieldName + ", table: " + tableName, mdtError::Error);
00048     if(sqlError.isValid()){
00049       pvLastError.setSystemError(sqlError.number(), sqlError.text());
00050     }
00051     MDT_ERROR_SET_SRC(pvLastError, "mdtSqlRecord");
00052     pvLastError.commit();
00053     return false;
00054   }
00055   // Get field and add it
00056   fld = dbRecord.field(fieldName);
00057   if(!fld.isValid()){
00058     pvLastError.setError("Field '" + fieldName + "' was not found in table '" + tableName + "'", mdtError::Error);
00059     MDT_ERROR_SET_SRC(pvLastError, "mdtSqlRecord");
00060     pvLastError.commit();
00061     return false;
00062   }
00063   fld.setGenerated(false);
00064   append(fld);
00065 
00066   return true;
