00076 {
00077   Q_ASSERT(!pvTcpPortManager->isRunning());

00089       pvTcpPortManager->closePort();

00093     if(pvTcpPortManager->getHardwareNodeAddress(bitsCount, startFrom) == hardwareNodeId){
00094       return mdtAbstractPort::NoError;
00095     }else{
00096       pvTcpPortManager->stop();
00097       pvTcpPortManager->closePort();
