118 bool mdtTtTestCableOffsetTool::isInRange(const mdtValue & value, double min, double max)
119 {
120  double x;
121 
122  // Get double value
123  if(value.isMinusOl()){
124  x = -std::numeric_limits<double>::max();
125  }else if(value.isPlusOl()){
126  x = std::numeric_limits<double>::max();
127  }else{
128  x = value.valueDouble();
129  }
130  // Check range
131  if(!mdtTtTest::isInRange(x, min, max)){
132  QString msg;
133  pvLastError.setError(tr("Range checking failed for given value."), mdtError::Error);
134  msg = tr("Value") + " " + QString::number(x) + " " + tr("is out of range.") + " ";
135  msg += tr("Allowed range is: ") + "[" + QString::number(min) + ";" + QString::number(max) + "]";
136  pvLastError.setInformativeText(msg);
137  MDT_ERROR_SET_SRC(pvLastError, "mdtTtTestCableOffsetTool");
138  pvLastError.commit();
139  return false;
140  }
141 
142  return true;
143 }

151 bool mdtTtTestCableOffsetTool::connectToInstruments()
152 {
153  shared_ptr<mdtDeviceU3606A> multimeter;
154  shared_ptr<mdtDeviceModbusWago> coupler;
155  QVariant couplerNodeId;
156 
157  multimeter = pvTestNodeManager->device<mdtDeviceU3606A>("U3606A");
158  Q_ASSERT(multimeter);
159  coupler = pvTestNodeManager->device<mdtDeviceModbusWago>("W750");
160  Q_ASSERT(coupler);
161 
162  // Get current coupler ID
164  couplerNodeId = "0";
174  // Connect to multimeter
175  if(multimeter->connectToDevice(mdtDeviceInfo()) != mdtAbstractPort::NoError){
176  pvLastError.setError(tr("Cannot connect to ") + multimeter->name(), mdtError::Error);
177  MDT_ERROR_SET_SRC(pvLastError, "mdtTtTestCableOffsetTool");
178  pvLastError.commit();
179  return false;
180  }
181  // Connect to coupler
182  if(coupler->connectToDevice(mdtDeviceInfo()) != mdtAbstractPort::NoError){
183  pvLastError.setError(tr("Cannot connect to ") + coupler->name(), mdtError::Error);
184  MDT_ERROR_SET_SRC(pvLastError, "mdtTtTestCableOffsetTool");
185  pvLastError.commit();
186  return false;
187  }
188  if(!coupler->detectIos()){
189  pvLastError = coupler->lastError();
190  return false;
191  }
192  // Set short labels to coupler units
193  if(!pvTestNodeManager->setDeviceIosLabelShort(couplerNodeId, "W750")){
194  pvLastError = pvTestNodeManager->lastError();
195  disconnectFromInstruments();
196  return false;
197  }
198  // Reset multimeter to factory state
199  if(multimeter->sendCommand("*RST") < 0){
200  pvLastError.setError(tr("*RST failed on device ") + multimeter->name(), mdtError::Error);
201  MDT_ERROR_SET_SRC(pvLastError, "mdtTtTestCableOffsetTool");
202  pvLastError.commit();
203  return false;
204  }
205 
206  return true;
207 }
