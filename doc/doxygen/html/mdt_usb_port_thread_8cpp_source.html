00068     if(portError != mdtAbstractPort::NoError){
00069       delete[] buffer;

00072     errors = port->lastErrors();

00074       if(errors.at(i) != mdtAbstractPort::NoError){
00075         delete[] buffer;
00076         return portError;
00077       }
00078     }
00079     readen = port->read(buffer, port->readBufferSize());
00081     if(readen < 0){
00082       delete[] buffer;
00083       qDebug() << "mdtUsbPortThread::readUntilShortPacketReceived() , error " << readen;
00084       return (mdtAbstractPort::error_t)readen;
00085     }
00086     if(readen < port->readBufferSize()){
00088       delete[] buffer;
00089       return mdtAbstractPort::NoError;
00090     }
00091     maxReadTransfers--;
00092   }
00093   delete[] buffer;
00094   mdtError e(MDT_USB_IO_ERROR, "No short packet received after maxReadTransfers", mdtError::Error);
00095   MDT_ERROR_SET_SRC(e, "mdtUsbPortThread");
00096   e.commit();
00097 
00098   return mdtAbstractPort::ReadTimeout;
