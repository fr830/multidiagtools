100 void mdtClConnectableConnectorDialog::selectEndConnector()
101 {
102  mdtClLink lnk(0, pvDatabase);
103  mdtSqlSelectionDialog selectionDialog(this);
104  mdtSqlTableSelection s;
105  mdtClConnectableCriteria c;
106  QString sql;
107  bool ok;
108 
109  // Get connectability check criteria
110  c = getConnectableCriteria();
111  // Get SQL query
112  sql = lnk.sqlForConnectableUnitConnectorsSelection(pvStartUnitConnectorId, pvEndUnitIdList, c, ok);
113  if(!ok){
114  displayError(lnk.lastError());
115  return;
116  }
117  // Setup and show dialog
118  selectionDialog.setMessage(tr("Select a connector:"));
119  selectionDialog.setQuery(sql, pvDatabase, false);
120  selectionDialog.setColumnHidden("Id_PK", true);
121  selectionDialog.setColumnHidden("Unit_Id_FK", true);
122  selectionDialog.setColumnHidden("Connector_Id_FK", true);
123  selectionDialog.setColumnHidden("ArticleConnector_Id_FK", true);
124  selectionDialog.setColumnHidden("ArticleConnectorName", true);
125  selectionDialog.setColumnHidden("Id_PK", true);
126  selectionDialog.setHeaderData("UnitConnectorName", tr("Connector"));
127  if(selectionDialog.exec() != QDialog::Accepted){
128  return;
129  }
130  // Store selected connector ID
131  s = selectionDialog.selection("Id_PK");
132  if(s.isEmpty()){
133  return;
134  }
135  Q_ASSERT(s.rowCount() == 1);
136  pvEndUnitConnectorId = s.data(0, "Id_PK");
137  displayEndConnector();
138 }

150 void mdtClConnectableConnectorDialog::displayStartConnector()
151 {
152  mdtClLink lnk(0, pvDatabase);
153  QString sql;
154  QList<QSqlRecord> dataList;
155  QSqlRecord data;
156  bool ok;
157 
158  // Clear previously displayed connector
159  clearStartConnectorWidgets();
160  if(pvStartUnitConnectorId.isNull()){
161  return;
162  }
163  // Get connector data
164  sql = "SELECT * FROM UnitConnector_view WHERE Id_PK = " + pvStartUnitConnectorId.toString();
165  dataList = lnk.getDataList<QSqlRecord>(sql, ok);
166  if(!ok){
167  displayError(lnk.lastError());
168  return;
169  }
170  if(dataList.isEmpty()){
171  return;
172  }
173  Q_ASSERT(dataList.size() == 1);
174  data = dataList.at(0);
175  // Update widgets
176  lbStartConnectorName->setText(data.value("UnitConnectorName").toString());
177  lbStartConnectorGender->setText(data.value("Gender").toString());
178  lbStartConnectorForm->setText(data.value("Form").toString());
179  lbStartConnectorInsert->setText(data.value("Insert").toString());
180  lbStartConnectorInsertRotation->setText(data.value("InsertRotation").toString());
181  pbSelectEndConnector->setEnabled(true);
182 }
