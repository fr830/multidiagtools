00126 void mdtModbusIoTool::setup()
00127 {
00128   mdtModbusTcpPortSetupDialog d(this);
00129   d.setPortManager(pvDeviceModbusWago->portManager());
00130   d.exec();
00131   // If portManager is running, we can try to detect I/Os
00132   if(pvDeviceModbusWago->portManager()->isReady()){
00133     if(!pvDeviceModbusWago->isWago750()){
00134       showStatusMessage(tr("Device is not a Wago 750 Fieldbus"), tr("I/O detetcion currently only works with Wago 750 fielbus coupler"));
00136       pvDeviceModbusWago->portManager()->stop();
00137       return;
00138     }
00139     showStatusMessage(tr("I/O detection ..."));
00140     if(!pvDeviceModbusWago->detectIos(pvDeviceIos)){
00141       showStatusMessage(tr("I/O detection failed"));
00143       pvDeviceModbusWago->portManager()->stop();
00144       return;
00145     }
00146     pvDeviceModbusWago->setIos(pvDeviceIos, true);
00147     pvDeviceIosWidget->setDeviceIos(pvDeviceIos);
00148     pvDeviceModbusWago->getDigitalOutputs(0);

00150   }

00153 void mdtModbusIoTool::connectToNode()
00154 {
00155   QList<mdtPortInfo*> portInfoList;
00156   mdtModbusTcpPortManager *m = pvDeviceModbusWago->modbusTcpPortManager();
00157   Q_ASSERT(m != 0);
00158 
00159   if(!m->isClosed()){
00160     showStatusMessage(tr("Please disconnect before reconnect"), 3000);
00161     return;
00162   }
00163   // Update GUI state
00164   pbDisconnect->setEnabled(false);
00165   pbAbortScan->setEnabled(true);
00166   pbConnect->setEnabled(false);
00167   sbHwNodeId->setEnabled(false);
00168   // Scan looking in chache file first
00169   portInfoList = m->scan(m->readScanResult());
00170   // Try to connect ...
00171   if(pvDeviceModbusWago->connectToDevice(portInfoList, sbHwNodeId->value(), 8) != mdtAbstractPort::NoError){
00172     // scan network an try again
00173     pbConnect->setEnabled(false);
00174     qDeleteAll(portInfoList);
00175     portInfoList.clear();
00176     portInfoList = m->scan(QNetworkInterface::allInterfaces(), 502, 100);
00177     if(pvDeviceModbusWago->connectToDevice(portInfoList, sbHwNodeId->value(), 8) != mdtAbstractPort::NoError){
00178       showStatusMessage(tr("Device with HW node ID ") + QString::number(sbHwNodeId->value()) + tr(" not found"));
00182       return;
00183     }
00184     // Ok found, save scan result
00185     if(!m->saveScanResult(portInfoList)){
00186       showStatusMessage(tr("Cannot save scan result in cache file"), 3000);
00187     }
00188     // We no longer need portInfoList
00189     qDeleteAll(portInfoList);
00190     portInfoList.clear();
00191   }
00192   showStatusMessage(tr("I/O detection ..."));
00193   if(!pvDeviceModbusWago->detectIos(pvDeviceIos)){
00194     showStatusMessage(tr("I/O detection failed"));
00196     setStateError();
00200     return;
00201   }
00202   pvDeviceModbusWago->setIos(pvDeviceIos, true);
00203   pvDeviceIosWidget->setDeviceIos(pvDeviceIos);
00204   pvDeviceModbusWago->getDigitalOutputs(0);
00205   showStatusMessage(tr("I/O detection done"), 1000);
00206   pbAbortScan->setEnabled(false);
00208   // GUI state will be updated by portManager when ready
00209 }
