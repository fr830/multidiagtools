307 bool mdtTtBasicTestNodeCalibrationTool::calibrateSenseRelays()
308 {
310  shared_ptr<mdtDeviceModbusWago> coupler;
311  QStringList relays;
312  mdtValue R;
313  double Rlinks;
314  double r;
315  bool ok;
316 
319  coupler = testNodeManager()->device<mdtDeviceModbusWago>("W750");
320  Q_ASSERT(coupler);
321 
322  // Set all relays OFF
323  if(coupler->setDigitalOutputsValue(false, false, false) < 0){
324  pvLastError = coupler->lastError();
325  return false;
326  }
327  // Set K4 and K5 ON
328  coupler->setDigitalOutputValue("K4", true, false, false);
329  coupler->setDigitalOutputValue("K5", true, false, false);
330  // Send relays states to coupler
331  if(!coupler->setDigitalOutputs(true)){
332  pvLastError = coupler->lastError();
333  return false;
334  }
335  coupler->wait(50);
336  // Frit line
337  if(!fritLine()){
338  pvLastError.updateText(tr("Fritting line of realys K4 and K5 failed."));
339  return false;
340  }
341  // Setup pvMultimeter for low resistance measurement
342  if(pvMultimeter->setupLowResistanceMeasure(mdtDeviceU3606A::RangeAuto, mdtDeviceU3606A::ResolutionMin) < 0){
343  pvLastError = pvMultimeter->lastError();
344  return false;
345  }
346  // Check that we have R < 1 Ohm
347  R = pvMultimeter->getMeasureValue();
348  if(!isInRange(R, 0.0, 1.0)){
349  pvLastError.updateText(tr("Calibrating resistance of realys K4 and K5 failed."));
350  return false;
351  }
352  // Get links resistance from SENSE+ to SENSE-
353  relays << "K4" << "K5";
354  Rlinks = getLinkResistanceForRoute("XSENS+", "", "XSENS+", "XSENS-", "", "XSENS-", relays, ok);
355  if(!ok){
356  return false;
357  }
358  // Calibrate K4 and K5: (R-Rlinks)/2
359  r = (R.valueDouble() - Rlinks) / 2.0;
360  qDebug() << "R K4-K5: " << R.valueDouble() << " , Rlinks: " << Rlinks << " , r: " << r;
361  if(!isInRange(r, 0.0, 1.0)){
362  pvLastError.updateText(tr("Calibrating resistance of realys K4 and K5 failed."));
363  return false;
364  }
365  setTestNodeUnitCalibrationOffset("K4", r);
366  setTestNodeUnitCalibrationOffset("K5", r);
367  // Set all relays OFF
368  if(coupler->setDigitalOutputsValue(false, false, false) < 0){
369  pvLastError = coupler->lastError();
370  return false;
371  }
372  // Set K3 and K6 ON
373  coupler->setDigitalOutputValue("K3", true, false, false);
374  coupler->setDigitalOutputValue("K6", true, false, false);
375  // Send relays states to coupler
376  if(!coupler->setDigitalOutputs(true)){
377  pvLastError = coupler->lastError();
378  return false;
379  }
380  coupler->wait(50);
381  // Check that we have R < 1 Ohm
382  R = pvMultimeter->getMeasureValue();
383  if(!isInRange(R, 0.0, 1.0)){
384  pvLastError.updateText(tr("Calibrating resistance of realys K3 and K6 failed."));
385  return false;
386  }
387  // Get links resistance from SENSE+ to SENSE-
388  relays.clear();
389  relays << "K3" << "K6";
390  Rlinks = getLinkResistanceForRoute("XSENS+", "", "XSENS+", "XSENS-", "", "XSENS-", relays, ok);
391  if(!ok){
392  return false;
393  }
394  r = (R.valueDouble() - Rlinks) / 2.0;
395  qDebug() << "R K3-K6: " << R.valueDouble() << " , Rlinks: " << Rlinks << " , r: " << r;
396  if(!isInRange(r, 0.0, 1.0)){
397  pvLastError.updateText(tr("Calibrating resistance of realys K3 and K6 failed."));
398  return false;
399  }
400  // Calibrate K3 and K6: (R-Rlinks)/2
401  setTestNodeUnitCalibrationOffset("K3", r);
402  setTestNodeUnitCalibrationOffset("K6", r);
403 
404  return true;
