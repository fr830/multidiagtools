00066 bool mdtDataTableModel::addRow(const QList<QVariant> &data, bool pkNotInData, int role)

00070   int modelColumnIndex;
00071 
00072   modelRowIndex = rowCount();

00074     return false;
00075   }
00076   modelColumnIndex = 0;
00077   for(dataColumnIndex=0; dataColumnIndex<data.size(); dataColumnIndex++){
00078     // If PK is not in data, we must remap indexes
00079     if((pkNotInData)&&(pvPkIndexes.contains(modelColumnIndex))){
00080       if(modelColumnIndex >= columnCount()){
00081         revertRow(modelRowIndex);

00126 bool mdtDataTableModel::addRows(const QList<QStringList> &rows, const mdtFieldMap &fieldMap, int role)
00127 {
00128   int dataRowIndex;
00129   int modelRowIndex;
00130   int modelColumnIndex;
00131   QVariant data;
00132 
00133   modelRowIndex = rowCount();
00134   if(!insertRows(modelRowIndex, rows.size())){
00135     return false;
00136   }
00137   for(dataRowIndex=0; dataRowIndex<rows.size(); dataRowIndex++){
00138     // Add data of each column
00139     for(modelColumnIndex=0; modelColumnIndex<columnCount(); modelColumnIndex++){
00140       data = fieldMap.dataForFieldIndex(rows.at(dataRowIndex), modelColumnIndex);
00141       // If data is empty, it's possible that we have auto values
00142       if(data.isValid()){
00143         if(!QSqlTableModel::setData(index(modelRowIndex, modelColumnIndex), data, role)){
00144           revertRow(modelRowIndex);
00145           return false;
00146         }
00147       }
00148     }
00149     modelRowIndex++;
00150   }
00151 
00152   return doSubmit();
