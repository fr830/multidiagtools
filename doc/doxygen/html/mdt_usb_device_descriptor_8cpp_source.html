48 int mdtUsbDeviceDescriptor::fetchAttributes(libusb_device *device, bool fetchActiveConfigOnly)
49 {
50  Q_ASSERT(device != 0);
51 
52  struct libusb_device_descriptor descriptor;
53  struct libusb_config_descriptor *configDescriptor;
54  libusb_device_handle *handle;
55  int err;
56  quint8 i;
57  mdtUsbConfigDescriptor *config;
58  unsigned char uSerialNumber[256] = {'\0'};
59  char serialNumber[256] = {'\0'};
60  int serialNumberLength;
61 
62  // Open descriptor
63  err = libusb_get_device_descriptor(device, &descriptor);
64  if(err != 0){
65  return err;
66  }
67  // Store attributes
68  pvbDescriptorType = descriptor.bDescriptorType;
69  pvbcdUSB = descriptor.bcdUSB;
70  pvbDeviceClass = descriptor.bDeviceClass;
71  pvbDeviceSubClass = descriptor.bDeviceSubClass;
72  pvbDeviceProtocol = descriptor.bDeviceProtocol;
73  pvbMaxPacketSize0 = descriptor.bMaxPacketSize0;
74  pvidVendor = descriptor.idVendor;
75  pvidProduct = descriptor.idProduct;
76  pvbcdDevice = descriptor.bcdDevice;
77  // Try to get serial number
78  handle = 0;
79  if(libusb_open(device, &handle) == 0){
80  Q_ASSERT(handle != 0);
81  serialNumberLength = libusb_get_string_descriptor_ascii(handle, descriptor.iSerialNumber, uSerialNumber, 255);
82  if(serialNumberLength > 0){
83  Q_ASSERT(serialNumberLength < 256);
84  for(i=0; i<serialNumberLength; i++){
85  serialNumber[i] = (char)uSerialNumber[i];
86  }
87  pvSerialNumber = serialNumber;
88  }
89  libusb_close(handle);
90  }
91  // Get configuration(s)
92  qDeleteAll(pvConfigs);
93  pvConfigs.clear();
94  if(fetchActiveConfigOnly){
95  err = libusb_get_active_config_descriptor(device, &configDescriptor);
96  switch(err){
97  case 0:
98  config = new mdtUsbConfigDescriptor;
99  Q_ASSERT(config != 0);
100  config->fetchAttributes(configDescriptor);
101  libusb_free_config_descriptor(configDescriptor);
102  pvConfigs.append(config);
103  break;
104  case LIBUSB_ERROR_NOT_FOUND:
105  {
106  mdtError e(MDT_USB_IO_ERROR, "Unable to find get current configuration descriptor", mdtError::Warning);
107  MDT_ERROR_SET_SRC(e, "mdtUsbDeviceDescriptor");
108  e.commit();
109  break;
110  }
111  default:
112  return err;
113  }
114  }else{
115  for(i=0; i<descriptor.bNumConfigurations; i++){
116  err = libusb_get_config_descriptor(device, i, &configDescriptor);
117  switch(err){
118  case 0:
119  config = new mdtUsbConfigDescriptor;
120  Q_ASSERT(config != 0);
121  config->fetchAttributes(configDescriptor);
122  libusb_free_config_descriptor(configDescriptor);
123  pvConfigs.append(config);
124  break;
125  case LIBUSB_ERROR_NOT_FOUND:
126  {
127  mdtError e(MDT_USB_IO_ERROR, "Unable to find get configuration descriptor at index " + QString::number(i), mdtError::Warning);
128  MDT_ERROR_SET_SRC(e, "mdtUsbDeviceDescriptor");
129  e.commit();
130  break;
131  }
132  default:
133  qDeleteAll(pvConfigs);
134  pvConfigs.clear();
135  return err;
136  }
137  }
138  }
139 
140  return 0;
141 }

275 mdtUsbEndpointDescriptor *mdtUsbDeviceDescriptor::firstBulkOutEndpoint(int configIndex, int ifaceIndex, bool dataEndpointOnly)
276 {
277  mdtUsbInterfaceDescriptor *interfaceDescriptor;
278  mdtUsbEndpointDescriptor *endpointDescriptor;
279  int i;
280 
281  interfaceDescriptor = interface(configIndex, ifaceIndex);
282  if(interfaceDescriptor == 0){
283  return 0;
284  }
285  // Search in available enpoints
286  for(i=0; i<interfaceDescriptor->endpoints().size(); i++){
287  endpointDescriptor = interfaceDescriptor->endpoints().at(i);
288  Q_ASSERT(endpointDescriptor != 0);
289  // Check if current endpoint is direction OUT and Bulk transfert
290  if((endpointDescriptor->isDirectionOUT())&&(endpointDescriptor->isTransfertTypeBulk())){
291  // Here we found a bulk out endpoint. If needed, check if it's a data endpoint
292  if(dataEndpointOnly){
293  if(endpointDescriptor->isDataEndpoint()){
294  return endpointDescriptor;
295  }
296  }else{
297  return endpointDescriptor;
298  }
299  }
300  }
301 
302  // Nothing found, return a null pointer
303  return 0;
304 }
