00034 void mdtDigitalIo::setValue(const mdtValue &value, bool emitValueChanged)
00035 {
00036   // Check if state has changed
00037   if((pvValue.isValid() == value.isValid())&&(pvValue.valueBool() == value.valueBool())){
00038     return;
00039   }
00040   // Check validity and store
00041   if(!value.isValid()){
00042     pvValue.clear();
00043   }else{
00044     pvValue.setValue(value.valueBool());
00045   }
00046   // Notify
00047   emit(valueChangedForUi(pvValue));
00048   if(emitValueChanged){
00049     emit(valueChanged(this));

00051   }
