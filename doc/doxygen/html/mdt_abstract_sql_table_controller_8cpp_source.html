504 {

569 bool mdtAbstractSqlTableController::setCurrentRow(const QString& fieldName, const QVariant& matchData)
570 {
571  Q_ASSERT(pvModel);
572  Q_ASSERT(!matchData.isNull());
573 
574  int row;
575  int col;
576  QModelIndex index;
577 
578  if(pvModel->rowCount() < 1){
579  return 0;
580  }
581  // Get column of requested field name
582  col = pvModel->fieldIndex(fieldName);
583  if(col < 0){
584  pvLastError.setError(tr("Requested field") + " '" + fieldName + "' " + tr("was not found in table") + " '" + pvModel->tableName() + "'", mdtError::Error);
585  MDT_ERROR_SET_SRC(pvLastError, "mdtAbstractSqlTableController");
586  pvLastError.commit();
587  return false;
588  }
589  // Search row
590  row = 0;
591  while(true){
592  index = proxyModel()->index(row, col);
593  if(proxyModel()->data(index) == matchData){
594  return setCurrentRow(row);
595  }
596  if(row == (pvModel->rowCount()-1)){
597  if(!pvModel->canFetchMore()){
598  break;
599  }
600  pvModel->fetchMore();
601  }
602  ++row;
603  }
604  // matchData not found
605  QString msg;
606  msg = tr("Could not find value") + " '" + matchData.toString() + "' ";
607  msg += tr("for field") + " '" + fieldName + "' ";
608  msg += tr("in table") + " '" + pvModel->tableName() + "'";
609  pvLastError.setError(msg, mdtError::Error);
610  MDT_ERROR_SET_SRC(pvLastError, "mdtAbstractSqlTableController");
611  pvLastError.commit();
612 
613  return false;
614 }

737 {

739  Q_ASSERT(pvStateMachine.isRunning());

742  if((row == currentRow())&&(!forceSendCurrentRowChanedEvent)){

760  // Check range and send event if OK
761  if(row >= pvModel->rowCount()){
762  return false;
763  }
764  currentRowChangedEvent(row);
