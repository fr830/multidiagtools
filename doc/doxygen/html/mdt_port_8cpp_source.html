72  fd_set input;
73  int n;
74  struct timeval tv = pvReadTimeout; // Create a copy, because select() alter the timeout
75 
76  // Init the select call
77  FD_ZERO(&input);
78  FD_SET(pvFd, &input);
79 
80  pvMutex.unlock();
81  if(tv.tv_sec == -1){
82  n = select(pvFd+1, &input, 0, 0, 0);
83  }else{
84  n = select(pvFd+1, &input, 0, 0, &tv);
85  }
86  pvMutex.lock();
87  if(n == 0){
88  return ReadTimeout;
89  }else{

91  switch(errno){

203 
204  // Try to open port
205  pvFd = pvPortLock->openLocked(pvPortName, O_RDWR | O_NOCTTY | O_NONBLOCK);
206  if(pvFd < 0){
207  err = errno;
208  // Check if port was locked by another

210  mdtError e(MDT_SERIAL_PORT_IO_ERROR, "Port " + pvPortName + " allready locked, cannot open it", mdtError::Error);
211  MDT_ERROR_SET_SRC(e, "mdtPort");
212  e.commit();
213  return PortLocked;
214  }
215  mdtError e(MDT_PORT_IO_ERROR, "Unable to open port: " + pvPortName, mdtError::Error);
216  e.setSystemError(err, strerror(err));
217  MDT_ERROR_SET_SRC(e, "mdtPort");
218  e.commit();
219  pvPortLock->unlock();
220  // Check error and return a possibly correct code
221  switch(err){
222  case EACCES:
223  return PortAccess;
224  case ENOENT:
225  return PortNotFound;
226  default:
227  return UnhandledError;
228  }
229  }
230
