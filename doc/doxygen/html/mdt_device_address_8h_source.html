164  bool setAddressString(const QString & as)
165  {
166  QStringList lst = as.trimmed().split("::");
167 
168  clear();
169  if(lst.size() < 2){
170  pvLastError.setError("'" + as + "' " + tr("contains not a valid address string. Expected at least 2 segments (Port::Something)."), mdtError::Error);
171  MDT_ERROR_SET_SRC(pvLastError, "mdtDeviceAddress");
172  pvLastError.commit();
173  return false;
174  }
175  // We store address string, can be helpful for error messages (if a called decode function fails, it will call clear() ).
176  pvAddressString = as.trimmed();
177  // Call appropriate decode function
178  QString str = lst.at(0);
179  if(str == "TCPIP"){
180  return decodeTcpipAddressString(lst);
181  }
182  if(str == "USB"){
183  return decodeUsbAddressString(lst);
184  }
185  // Error - not supported port type
186  pvLastError.setError("'" + pvAddressString + "' " + tr("contains a invalid or not supported address string."), mdtError::Error);
187  MDT_ERROR_SET_SRC(pvLastError, "mdtDeviceAddress");
188  pvLastError.commit();
189  clear();
190  return false;
191  }
