00032 bool mdtPartitionAttributes::setPath(const QString &path)
00033 {
00034   FILE *fd;
00035   struct mntent *fs;
00036   struct statvfs vfs;
00037   QString mntDir;
00038 
00039   // Default values
00040   pvRootPath = "";
00041   pvName = "";
00042   pvIsReadOnly = false;
00043   pvFileSystem = "";
00044 
00045   // Open mtab FIXME: probably not POSIX, but only Linux ?
00046   fd = setmntent("/etc/mtab", "r");
00047   if(fd == 0){
00048     mdtError e(MDT_FILE_IO_ERROR, "Unable to open /mnt/mtab", mdtError::Error);
00049     e.setSystemError(errno, strerror(errno));
00050     MDT_ERROR_SET_SRC(e, "mdtPartitionAttributes");
00051     e.commit();
00052     return false;
00053   }
00054   // Travel mounted partitions and get attributes
00055   while((fs = getmntent(fd)) != 0){
00056     if(fs->mnt_fsname[0] == '/'){
00057       mntDir = fs->mnt_dir;
00058       if(mntDir == path){
00059         // Store attributes
00060         pvRootPath = fs->mnt_dir;
00061         pvName = fs->mnt_fsname;
00062         pvFileSystem = fs->mnt_type;
00063         // Get flags
00064         if(statvfs(fs->mnt_dir, &vfs) != 0){
00065           mdtError e(MDT_FILE_IO_ERROR, "statvfs() failed on " + pvRootPath, mdtError::Error);
00066           e.setSystemError(errno, strerror(errno));
00067           MDT_ERROR_SET_SRC(e, "mdtPartitionAttributes");
00068           e.commit();
00069           endmntent(fd);
00070           return false;
00071         }
00072         if(vfs.f_flag & ST_RDONLY){
00073           pvIsReadOnly = true;
00074         }else{
00075           pvIsReadOnly = false;
00076         }
00077         // End of attributes fetch - cleanup and return
00078         endmntent(fd);
00079         return true;
00080       }
00081     }
00082   }
00083 
00084   // Close mtab
00085   endmntent(fd);
00086   
00087   return false;
00088 }
