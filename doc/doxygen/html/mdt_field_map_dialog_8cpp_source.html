128 void mdtFieldMapDialog::updateMappingTableView()
129 {
130  QList<mdtFieldMapItem*> items;
131  QMultiMap<int, mdtFieldMapItem*> sortMap;
132  mdtFieldMapItem *fmItem;
133  QTableWidgetItem *twSourceItem, *twDestinationItem;
134  int row, i;
135  QString str;
136  QStringList mappingViewHeader;
137 
138  // Get map items and sort them by source field indexes
139  items = pvFieldMap.items();
140  i = items.size() - 1;
141  while(i >= 0){
142  fmItem = items.at(i);
143  Q_ASSERT(fmItem != 0);
144  sortMap.insert(fmItem->sourceFieldIndex(), fmItem);
145  --i;
146  }
147  items = sortMap.values();
148  // Update view
149  twMapping->setRowCount(items.size());
150  // Set header if needed
151  if(twMapping->horizontalHeaderItem(0) == 0){
152  mappingViewHeader << tr("Source field") << " -> " << tr("Destination field");
153  twMapping->setColumnCount(mappingViewHeader.size());
154  twMapping->setHorizontalHeaderLabels(mappingViewHeader);
155  }
156  for(row = 0; row < items.size(); ++row){
157  fmItem = items.at(row);
158  Q_ASSERT(fmItem != 0);
159  // Update source field
160  twSourceItem = twMapping->item(row, 0);
161  if(twSourceItem == 0){
162  twSourceItem = new QTableWidgetItem;
163  twMapping->setItem(row, 0, twSourceItem);
164  }
165  if(fmItem->sourceFieldDisplayText().isEmpty()){
166  str = fmItem->sourceFieldName();
167  }else{
168  str = fmItem->sourceFieldDisplayText();
169  }
170  if(fmItem->sourceFieldDataStartOffset() > -1){
171  str += " [" + QString::number(fmItem->sourceFieldDataStartOffset()) + "-";
172  if(fmItem->sourceFieldDataEndOffset() > -1){
173  str += QString::number(fmItem->sourceFieldDataEndOffset());
174  }else{
175  str += "*";
176  }
177  str += "]";
178  }
179  twSourceItem->setData(Qt::DisplayRole, str);
180  // Update destination field
181  twDestinationItem = twMapping->item(row, 2);
182  if(twDestinationItem == 0){
183  twDestinationItem = new QTableWidgetItem;
184  twMapping->setItem(row, 2, twDestinationItem);
185  }
186  if(fmItem->destinationFieldDisplayText().isEmpty()){
187  twDestinationItem->setData(Qt::DisplayRole, fmItem->destinationFieldName());
188  }else{
189  twDestinationItem->setData(Qt::DisplayRole, fmItem->destinationFieldDisplayText());
190  }
191  // Store field map item pointer
192  twSourceItem->setData(Qt::UserRole, QVariant::fromValue(fmItem));
193  twDestinationItem->setData(Qt::UserRole, QVariant::fromValue(fmItem));
194  }
195  twMapping->resizeColumnsToContents();
196  twMapping->resizeRowsToContents();
197 }
