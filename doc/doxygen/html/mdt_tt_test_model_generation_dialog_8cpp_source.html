111 void mdtTtTestModelGenerationDialog::populateTestCableComboBox(int index)
112 {
113  mdtTtTestModel tm(0, pvDatabase);
114  QVariant testNodeId;
115  QList<QSqlRecord> dataList;
116  QSqlRecord data;
117  QString sql;
118  QString str;
119  bool ok;
120  int i;
121 
122  cbTestCable->clear();
123  if(index < 0){
124  return;
125  }
126  testNodeId = cbTestNode->itemData(index);
127  Q_ASSERT(!testNodeId.isNull());
128  sql = "SELECT DISTINCT LogicalTestCable_Id_FK, Identification, Key FROM TestNode_LogicalTestCable_view WHERE TestNode_Id_FK = " + testNodeId.toString();
129  sql += " ORDER BY Identification ASC";
130  dataList = tm.getDataList<QSqlRecord>(sql, ok);
131  if(!ok){
132  displayError(tm.lastError());
133  return;
134  }
135  for(i = 0; i < dataList.size(); ++i){
136  data = dataList.at(i);
137  str = data.value("Identification").toString();
138  if(!data.value("Key").isNull()){
139  str += " (" + data.value("Key").toString() + ")";
140  }
141  cbTestCable->addItem(str, data.value("LogicalTestCable_Id_FK"));
142  }
143  // We get measure connection data from here, because this slot is called 1x when a new test node was selected
144  getMeasureConnectionData(testNodeId);
145 }

177 void mdtTtTestModelGenerationDialog::accept()
178 {
179  int index;
180 
181  // Set test model type
183 
184  // Set test node ID
185  index = cbTestNode->currentIndex();
186  if(index < 0){
187  displayErrorAboutMissingParameter(tr("No test node was selected."));
188  return;
189  }
190  pvParameters.testNodeId = cbTestNode->itemData(index);
191  // Set test cable ID
192  index = cbTestCable->currentIndex();
193  if(index < 0){
194  displayErrorAboutMissingParameter(tr("No test cable was selected."));
195  return;
196  }
197  pvParameters.testCableId = cbTestCable->itemData(index);
198  // Set measure connection A
199  index = cbMeasureConnectionA->currentIndex();
200  if(index < 0){
201  displayErrorAboutMissingParameter(tr("Measure connection A was not selected."));
202  return;
203  }
204  pvParameters.measureConnexionIdA = cbMeasureConnectionA->itemData(index);
205  // Set measure connection B
206  index = cbMeasureConnectionB->currentIndex();
207  if(index < 0){
208  displayErrorAboutMissingParameter(tr("Measure connection B was not selected."));
209  return;
210  }
211  pvParameters.measureConnexionIdB = cbMeasureConnectionB->itemData(index);
213  pvParameters.continuityExpectedValue = 0.0;
214  pvParameters.isolationExpectedValue = 1e9;
215  pvParameters.generateForNonLinkedConnections = false;
216 
217  QDialog::accept();
218 }

251 void mdtTtTestModelGenerationDialog::getMeasureConnectionData(const QVariant & testNodeId)
252 {
253  mdtTtTestModel tm(0, pvDatabase);
254  QString sql;
255  bool ok;
256 
257  pvMeasureConnectionDataList.clear();
258  sql = "SELECT UCNX.Id_PK, UCNX.UnitContactName";
259  sql += " FROM UnitConnection_tbl UCNX JOIN TestNodeUnitConnection_tbl TNUCNX ON TNUCNX.UnitConnection_Id_FK_PK = UCNX.Id_PK";
260  sql += " JOIN TestNodeUnit_tbl TNU ON TNU.Unit_Id_FK_PK = TNUCNX.TestNodeUnit_Id_FK";
261  sql += " WHERE TNU.TestNode_Id_FK = " + testNodeId.toString();
262  sql += " AND TNU.Type_Code_FK = 'MEASCONNECTOR'";
263  pvMeasureConnectionDataList = tm.getDataList<QSqlRecord>(sql, ok);
264  if(!ok){
265  pvMeasureConnectionDataList.clear();
266  displayError(tm.lastError());
267  return;
268  }
