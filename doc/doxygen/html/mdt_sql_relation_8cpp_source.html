56 bool mdtSqlRelation::addRelation(const QString &parentFieldName, const QString &childFieldName, bool copyParentToChildOnInsertion, const QString &operatorWithPreviousItem)
57 {
58  Q_ASSERT(pvParentModel != 0);
59  Q_ASSERT(pvChildModel != 0);
60 
61  QSqlRecord record;
62  int parentFieldIndex;
63  QSqlField parentField;
64  int childFieldIndex;
65  QSqlField childField;
66 
67  // Get parent field index, check that it exist
68  record = pvParentModel->record();
69  parentFieldIndex = record.indexOf(parentFieldName);
70  if(parentFieldIndex < 0){
71  mdtError e(MDT_DATABASE_ERROR, "Field '" + parentFieldName + "' not found in parent table", mdtError::Error);
72  MDT_ERROR_SET_SRC(e, "mdtSqlRelation");
73  e.commit();
74  return false;
75  }
76  parentField = record.field(parentFieldIndex);
77  // Get child field index, check that it exist
78  record = pvChildModel->record();
79  childFieldIndex = record.indexOf(childFieldName);
80  if(childFieldIndex < 0){
81  mdtError e(MDT_DATABASE_ERROR, "Field '" + childFieldName + "' not found in child table", mdtError::Error);
82  MDT_ERROR_SET_SRC(e, "mdtSqlRelation");
83  e.commit();
84  return false;
85  }
86  childField = record.field(childFieldIndex);
87  // Add relation informations
88  mdtSqlRelationItem *item = new mdtSqlRelationItem;
89  item->setParentField(parentField);
90  item->setParentFieldIndex(parentFieldIndex);
91  item->setChildField(childField);
92  item->setChildFieldIndex(childFieldIndex);
93  item->setCopyParentToChildOnInsertion(copyParentToChildOnInsertion);
94  item->setRelationOperatorWithPreviousItem(" " + operatorWithPreviousItem.trimmed() + " ");
95  pvRelations.append(item);
96 
97  return true;
98 }

159 void mdtSqlRelation::setChildForeingKeyValues(QSqlRecord &parentRecord, QSqlRecord &childRecord)
160 {
161  Q_ASSERT(pvParentModel != 0);
162  Q_ASSERT(pvChildModel != 0);
163 
164  int i;
165  mdtSqlRelationItem *item;
166  QVariant data;
167 
168  // Copy parent PK -> child FK if needed
169  for(i=0; i<pvRelations.size(); ++i){
170  item = pvRelations.at(i);
171  Q_ASSERT(item != 0);
172  if(item->copyParentToChildOnInsertion()){
173  // Get parent model's data
174  data = parentRecord.value(item->parentFieldIndex());
175  // Pust to child's field
176  childRecord.setValue(item->childFieldIndex(), data);
177  childRecord.setGenerated(item->childFieldIndex(), true);
178  }
179  }
180 }

182 void mdtSqlRelation::generateChildModelRelationFilter(int row)
183 {
184  Q_ASSERT(pvParentModel != 0);
185  Q_ASSERT(pvChildModel != 0);
187 
188  int i;
189  mdtSqlRelationItem *item;
190  QSqlRecord record;
191  QVariant data;
192 
193  // Get data record of the parent model
194  record = pvParentModel->record(row);
195  // Build filter
196  pvChildModelRelationFilter.clear();
197  for(i=0; i<pvRelations.size(); ++i){
198  item = pvRelations.at(i);
199  Q_ASSERT(item != 0);
201  // Get parent model's data
202  if(!record.isNull(item->parentFieldIndex())){
203  data = record.value(item->parentFieldIndex());
204  }
205  if(i>0){
207  pvChildModelRelationFilter += item->relationOperatorWithPreviousItem();
208  }
210  pvChildModelRelationFilter += item->nameProtection() + pvChildModel->tableName() + item->nameProtection() + "." + item->nameProtection() + item->childFieldName() + item->nameProtection() + "=";
211  if(data.isValid()){
212  pvChildModelRelationFilter += item->dataProtection() + data.toString() + item->dataProtection();
213  }else{
214  pvChildModelRelationFilter += item->dataProtection() + item->valueForNoDataFilter().toString() + item->dataProtection();
215  }
216  }
217  // Apply filter
218  generateChildModelFilter();
219  pvChildModel->setFilter(pvChildModelFilter);
220  emit childModelFilterApplied();
221  if(pvChildModel->rowCount() < 1){
222  emit childModelIsEmpty();
223  }
224 }
