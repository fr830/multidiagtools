219  vertex_t vertex;

221 
222  // Check if we have requested connection ID in the graph

244  connectionIdList.append(edgeData.endConnectionId);
245  }
246  }
247  }

297 QList<QVariant> mdtClPathGraph::getShortestPath(const QVariant & fromConnectionId, const QVariant & toConnectionId, bool & ok)
298 {
299  QList<QVariant> connectionIdList;
300  mdtClPathGraphEdgeData edgeData;
301  std::vector<vertex_t> predecessors(boost::num_vertices(pvGraph)); // To store parents
302  std::vector<int> distances(boost::num_vertices(pvGraph)); // To store distances
303  typedef boost::property_map<graph_t, boost::vertex_index_t>::type index_map_t;
304  index_map_t indexMap = boost::get(boost::vertex_index, pvGraph);
305  boost::iterator_property_map<vertex_t*, index_map_t, vertex_t, vertex_t&> predecessorMap(&predecessors[0], indexMap);
306  boost::iterator_property_map<int*, index_map_t, int, int&> distanceMap(&distances[0], indexMap);
307  vertex_t fromVertex, toVertex;
308 
309  // Check if we have requested from connection ID in the graph
310  if(!pvGraphVertices.contains(fromConnectionId.toInt())){
311  ok = false;
312  pvLastError.setError(QObject::tr("Cannot get shortest path."), mdtError::Error);
313  pvLastError.setInformativeText(QObject::tr("Plese see details for more informations."));
314  pvLastError.setSystemError(-1, QObject::tr("Source connection ID not found") + " (ID: " + fromConnectionId.toString() + ").");
315  MDT_ERROR_SET_SRC(pvLastError, "mdtClPathGraph");
316  pvLastError.commit();
317  return connectionIdList;
318  }
319  fromVertex = pvGraphVertices.value(fromConnectionId.toInt());
320  // Check if we have requested to connection ID in the graph
321  if(!pvGraphVertices.contains(toConnectionId.toInt())){
322  ok = false;
323  pvLastError.setError(QObject::tr("Cannot get shortest path."), mdtError::Error);
324  pvLastError.setInformativeText(QObject::tr("Plese see details for more informations."));
325  pvLastError.setSystemError(-1, QObject::tr("Destination connection ID not found") + " (ID: " + toConnectionId.toString() + ").");
326  MDT_ERROR_SET_SRC(pvLastError, "mdtClPathGraph");
327  pvLastError.commit();
328  return connectionIdList;
329  }
330  toVertex = pvGraphVertices.value(toConnectionId.toInt());
331  // Proceed Dijkstra shortest path search
332  boost::dijkstra_shortest_paths(pvGraph, fromVertex, boost::distance_map(distanceMap).predecessor_map(predecessorMap));
333  // Extract the requested shortest path itself
334  vertex_t v = toVertex; // We start at the destination and work our way back to the source
335  for(vertex_t u = predecessorMap[v]; u != v; v = u, u = predecessorMap[v]){
336  std::pair<edge_t, bool> ep = boost::edge(u, v, pvGraph);
337  Q_ASSERT(ep.second);
338  edgeData = pvEdgeDataMap.at(ep.first);
339  Q_ASSERT(!edgeData.startConnectionId.isNull());
340  Q_ASSERT(!edgeData.endConnectionId.isNull());
341  if(edgeData.isComplement){
342  if(!connectionIdList.contains(edgeData.startConnectionId)){
343  connectionIdList.prepend(edgeData.startConnectionId);
344  }
345  if(!connectionIdList.contains(edgeData.endConnectionId)){
346  connectionIdList.prepend(edgeData.endConnectionId);
347  }
348  }else{
349  if(!connectionIdList.contains(edgeData.endConnectionId)){
350  connectionIdList.prepend(edgeData.endConnectionId);
351  }
352  if(!connectionIdList.contains(edgeData.startConnectionId)){
353  connectionIdList.prepend(edgeData.startConnectionId);
354  }
355  }
356  }
357  ok = true;
358 
359  return connectionIdList;
360 }
