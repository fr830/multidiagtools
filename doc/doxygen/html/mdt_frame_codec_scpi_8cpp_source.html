00330     MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpi");
00331     e.commit();
00332     return false;

00334   // Get the header length
00335   headerSize = data.mid(1, 1).toInt(&ok);

00338     MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpi");
00339     e.commit();
00340     return false;
00341   }
00342   // Check that enough bytes are available
00343   if(data.size() < (headerSize+2)){
00344     mdtError e(MDT_FRAME_DECODE_ERROR, "IEEE block contains no data" , mdtError::Warning);
00345     MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpi");
00346     e.commit();
00347     return false;
00348   }
00349   // Get data length
00350   dataSize = data.mid(2, headerSize).toInt(&ok);
00351   if(!ok){
00352     mdtError e(MDT_FRAME_DECODE_ERROR, "Cannot get data length in IEEE block" , mdtError::Warning);
00353     MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpi");
00354     e.commit();
00355     return false;
00356   }
00357   if(dataSize < 1){
00358     mdtError e(MDT_FRAME_DECODE_ERROR, "IEEE block contains no data (info from header)" , mdtError::Warning);
00359     MDT_ERROR_SET_SRC(e, "mdtFrameCodecScpi");

00361     return false;
