217 void mdtAbstractPort::initQueues()
218 {
219  mdtFrame *frame;
220 
221  // Create the read frames pools with requested type
222  for(int i=0; i<config().readQueueSize(); i++){
223  switch(config().frameType()){
224  // Raw (binary) frame
225  case mdtFrame::FT_RAW:
226  frame = new mdtFrame;
227  Q_ASSERT(frame != 0);
228  frame->setDirectlyComplete(true);
229  break;
230  // Raw (binary) frame for use with timeout protocol
231  case mdtFrame::FT_RAW_TOP:
232  frame = new mdtFrame;
233  break;
234  // ASCII frame type
235  case mdtFrame::FT_ASCII:
236  frame = new mdtFrameAscii;
237  dynamic_cast<mdtFrameAscii*>(frame)->setEofSeq(config().endOfFrameSeq());
238  break;
239  // MODBUS/TCP frame type
240  case mdtFrame::FT_MODBUS_TCP:
241  frame = new mdtFrameModbusTcp;
242  break;
243  // USBTMC frame type
247  // Base frame type
248  default:
249  frame = new mdtFrame;
250  }
251  Q_ASSERT(frame != 0);
252  frame->reserve(config().readFrameSize());
253  pvReadFramesPool.enqueue(frame);
254  }
255  // Create the write frames pools
256  for(int i=0; i<config().writeQueueSize(); i++){
257  switch(config().frameType()){
258  // Raw (binary) frame
259  case mdtFrame::FT_RAW:
260  frame = new mdtFrame;
261  break;
262  // Raw (binary) frame for use with timeout protocol
263  case mdtFrame::FT_RAW_TOP:
264  frame = new mdtFrame;
265  break;
266  // ASCII frame type
267  case mdtFrame::FT_ASCII:
268  frame = new mdtFrameAscii;
269  break;
270  // MODBUS/TCP frame type
271  case mdtFrame::FT_MODBUS_TCP:
272  frame = new mdtFrameModbusTcp;
273  break;
274  // USBTMC frame type
278  // Base frame type
279  default:
280  frame = new mdtFrame;
281  }
282  Q_ASSERT(frame != 0);
283  frame->reserve(config().writeFrameSize());
284  pvWriteFramesPool.enqueue(frame);
285  }
286 }

306 mdtFrame *mdtAbstractPort::getFrameToWrite()
307 {
308  mdtFrame *frame;
309 
310  // If queue is empty, wait
311  if(pvWriteFrames.size() < 1){
312  pvWriteFrameAvailable.wait(&pvMutex);
313  // During wait it can happen that a flush was requested,
314  // witch clears the write queue. If this happens,
315  // we must clear the pvFlushOutRequestPending flag,
316  // else thread will not write new frame that must be sent.
317  pvFlushOutRequestPending = false;
318  }
319  // If abortFrameToWriteWait() was called, it can happen that queue is empty
320  if(pvWriteFrames.size() < 1){
321  return 0;
322  }
323  frame = pvWriteFrames.dequeue();
324  Q_ASSERT(frame != 0);
325 
326  return frame;
327 }
