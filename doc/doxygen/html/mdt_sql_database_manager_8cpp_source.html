00323 bool mdtSqlDatabaseManager::createTable(mdtSqlSchemaTable & table, createMode_t createMode) 
00324 {
00325   QString sql;
00326   QSqlError sqlError;
00327 
00328   if(!pvDatabase.isOpen()){
00329     pvLastError.setError(tr("Cannot create table because database is closed."), mdtError::Error);
00330     MDT_ERROR_SET_SRC(pvLastError, "mdtSqlDatabaseManager");
00331     pvLastError.commit();
00332     return false;
00333   }
00334   QSqlQuery query(pvDatabase);
00335   table.setDriverName(pvDatabase.driverName());
00336   // Check if table allready exists
00337   if(pvDatabase.tables().contains(table.tableName())){
00338     switch(createMode){
00339       case KeepExisting:
00340         pvLastError.setError(tr("Table '") + table.tableName()\
00341                              + tr("' allready exists and choose is to keep it.") + "\n"\
00342                              + tr("It is possible that the structure does not match expected one."), mdtError::Warning);
00343         MDT_ERROR_SET_SRC(pvLastError, "mdtSqlDatabaseManager");
00344         pvLastError.commit();
00345         return true;
00346       case FailIfExists:

00348         MDT_ERROR_SET_SRC(pvLastError, "mdtSqlDatabaseManager");
00349         pvLastError.commit();
00350         return false;
00351       case AskUserIfExists:
00352         if(!userChooseToOverwriteTable(table.tableName())){
00353           return false;
00354         }
00355         break;
00356       case OverwriteExisting:
00357         sql = table.sqlForDropTable();
00358         if(!query.exec(sql)){
00359           sqlError = query.lastError();
00360           pvLastError.setError(tr("Cannot drop table '") + table.tableName() + tr("'"), mdtError::Error);
00361           MDT_ERROR_SET_SRC(pvLastError, "mdtSqlDatabaseManager");
00362           pvLastError.setSystemError(sqlError.number(), sqlError.text());
00363           pvLastError.commit();
00364           return false;
00365         }
00366         break;
00367     }
00368   }
00369   // Create table
00370   sql = table.sqlForCreateTable();
00371   if(!query.exec(sql)){
00372     sqlError = query.lastError();
00373     pvLastError.setError(tr("Cannot create table '") + table.tableName() + tr("'"), mdtError::Error);
00374     MDT_ERROR_SET_SRC(pvLastError, "mdtSqlDatabaseManager");
00375     pvLastError.setSystemError(sqlError.number(), sqlError.text());
