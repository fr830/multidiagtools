00299     // Check if connection was successfull
00300     if(socket.state() == QAbstractSocket::ConnectedState){
00301       ok = true;
00302     }
00303     // Disconnect
00304     socket.abort();

00373 int mdtModbusTcpPortManager::getHardwareNodeAddress(int bitsCount, int startFrom)
00374 {
00375   Q_ASSERT(isReady());
00376   Q_ASSERT(bitsCount > 0);
00377   Q_ASSERT(startFrom >= 0);
00378 
00379   int transactionId;
00380   mdtPortTransaction *transaction;
00381   QByteArray pdu;
00382   mdtFrameCodecModbus codec;
00383   int i, id;
00384 
00385   // Setup MODBUS PDU
00386   pdu = codec.encodeReadDiscreteInputs(startFrom, bitsCount);
00387   if(pdu.isEmpty()){
00388     return (int)mdtAbstractPort::UnhandledError;
00389   }
00390   // Get a new transaction
00391   transaction = getNewTransaction();
00392   // Send query
00393   transaction->setQueryReplyMode(true);
00394   transaction->setData(pdu);
00395   transactionId = sendData(transaction);
00396   if(transactionId < 0){
00397     return transactionId;
00398   }
00399   // Wait on result (use device's defined timeout)
00400   if(!waitTransactionDone(transactionId)){
00401     return (int)mdtAbstractPort::ReadTimeout;
00402   }
00403   // Get reply and decode
00404   if(codec.decode(readenFrame(transactionId)) != 0x02){
00405     return (int)mdtAbstractPort::UnhandledError;
00406   }
00407   // Check that we have enough states (can be more, because digital inputs are returned as multiple of 8)
00408   if(codec.values().size() < bitsCount){
00409     mdtError e(MDT_TCP_IO_ERROR, "Have not received enough input states", mdtError::Error);
00410     MDT_ERROR_SET_SRC(e, "mdtModbusTcpPortManager");
00411     e.commit();
00412     return (int)mdtAbstractPort::UnhandledError;
00413   }
00414   // Ok, have the states, let's decode ID
00415   id = 0;
00416   for(i=0; i<bitsCount; i++){
00417     if((codec.values().at(i).isValid())&&(codec.values().at(i).toBool() == true)){
00418       id |= (1<<i);
00419     }
00420   }
00421 
00422   return id;
00423 }

00527 
00528   mdtFrameModbusTcp *frame = 0;
00529 
00530   // Wait until we can write
00531   while(1){
00532     // If port manager was stopped, we return
00533     if(isClosed()){

00543         break;
00544       }
00545       unlockPortMutex();
00546     }
00547     QCoreApplication::processEvents(QEventLoop::AllEvents | QEventLoop::WaitForMoreEvents);
00548   }
00549   if(!waitTransactionPossible()){
00550     restoreTransaction(transaction);
00551     unlockPortMutex();

00554   // We are ready to write

00559   incrementCurrentTransactionId(0, 65535);
00560   frame->setTransactionId(currentTransactionId());

00564   // We enable waitAnAnswer , used by mdtTcpSocketThread for timeout detection
00565   frame->setWaitAnAnswer(true);
00566   pvPort->addFrameToWrite(frame);
