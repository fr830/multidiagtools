00067 QList<mdtPortInfo*> mdtModbusTcpPortManager::scan(const QStringList &hosts, int timeout)
00068 {
00069   Q_ASSERT(!isRunning());
00070 
00071   QList<mdtPortInfo*> portInfoList;
00072   mdtPortInfo *portInfo;
00073   int i;
00074   QStringList host;
00075   QString hostName;
00076   quint16 hostPort;
00077   bool ok;
00078   QTcpSocket socket;
00079   int maxIter;
00080 
00081   for(i=0; i<hosts.size(); i++){
00082     qDebug() << "mdtModbusTcpPortManager::scan(): trying host " << hosts.at(i) << " ...";
00083     // Extract hostname/ip and port
00084     host = hosts.at(i).split(":");
00085     if(host.size() != 2){
00086       mdtError e(MDT_TCP_IO_ERROR, "Cannot extract host and port in " + hosts.at(i), mdtError::Warning);
00087       MDT_ERROR_SET_SRC(e, "mdtModbusTcpPortManager");
00088       e.commit();
00089       continue;
00090     }
00091     hostName = host.at(0);
00092     hostPort = host.at(1).toUShort(&ok);
00093     if(!ok){
00094       mdtError e(MDT_TCP_IO_ERROR, "Cannot extract port in " + hosts.at(i), mdtError::Warning);
00095       MDT_ERROR_SET_SRC(e, "mdtModbusTcpPortManager");
00096       e.commit();
00097       continue;
00098     }
00099     // Try to connect
00100     socket.connectToHost(hostName, hostPort);
00101     maxIter = timeout / 50;
00102     while(socket.state() != QAbstractSocket::ConnectedState){
00103       if(maxIter <= 0){
00104         break;
00105       }
00106       qApp->processEvents();
00107       msleep(50);
00108       maxIter--;
00109     }
00110     if(socket.state() != QAbstractSocket::UnconnectedState){
00111       // Check if connection was successfull
00112       if(socket.state() == QAbstractSocket::ConnectedState){
00113         qDebug() << "Found device at " << host;
00114         portInfo = new mdtPortInfo;
00115         portInfo->setPortName(hosts.at(i));
00116         portInfoList.append(portInfo);
00117       }
00118       // Disconnect
00119       //socket.disconnectFromHost();
00120       socket.abort();
00121       while((socket.state() != QAbstractSocket::ClosingState)&&(socket.state() != QAbstractSocket::UnconnectedState)){
00122         qApp->processEvents();
00123         msleep(50);
00124       }
00125     }
00126   }
00127 
00128   return portInfoList;
00129 }

00132 {
00133   Q_ASSERT(pvPort != 0);
00134 
00135   mdtFrameModbusTcp *frame;

00137   // Get a frame in pool

00140     pvPort->unlockMutex();

00147   Q_ASSERT(frame != 0);

00150   // Store data and add frame to write queue
00152   pvTransactionId++;
00153   frame->setTransactionId(pvTransactionId);
00154   frame->setUnitId(0);    
00155   frame->setPdu(pdu);
00156   frame->encode();
00157   pvPort->addFrameToWrite(frame);
00158   addTransaction(pvTransactionId, enqueueResponse);
