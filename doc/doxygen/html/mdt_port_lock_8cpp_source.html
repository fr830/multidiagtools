00044 int mdtPortLock::openLocked(const QString &portName, int flags)
00045 {
00046   int i;
00047   int err;
00048 
00049   // If this method is called second time without unlocking port before, it's a error
00050   if(pvIsLocked){
00051     mdtError e(MDT_PORT_IO_ERROR , "Port " + portName + " is allready locked by this instance, please unlock it first", mdtError::Info);
00052     MDT_ERROR_SET_SRC(e, "mdtPortLock");
00053     e.commit();
00054     return -1;
00055   }
00056 
00057   pvPort.setFile(portName);
00058   pvIsLockedByAnother = false;
00059 
00060   // Search about lock files
00061   buildLockFilesList();
00062   for(i=0; i<pvLockfiles.size(); i++){
00063     if(pvLockfiles.at(i).exists()){
00064       mdtError e(MDT_PORT_IO_ERROR , "Port " + pvPort.absoluteFilePath() + " is allready locked (Lockfile" + pvLockfiles.at(i).filePath() + " exists)", mdtError::Info);
00065       MDT_ERROR_SET_SRC(e, "mdtPortLock");
00066       e.commit();
00067       pvIsLockedByAnother = true;
00068       return -1;
00069     }
00070   }
00071   // No lock files found, create all
00072   for(i=0; i<pvLockfiles.size(); i++){
00073     if(!createLockFile(pvLockfiles.at(i))){
00074       return -1;
00075     }
00076   }
00077   // Try to open the port
00078   pvFd = ::open(pvPort.absoluteFilePath().toStdString().c_str(), flags);
00079   if(pvFd < 0){
00080     err = errno;
00081     mdtError e(MDT_PORT_IO_ERROR, "Unable to open " + pvPort.absoluteFilePath(), mdtError::Error);
00082     e.setSystemError(err, strerror(err));
00083     MDT_ERROR_SET_SRC(e, "mdtPortLock");
00084     e.commit();
00085     removeLockFiles();
00086     errno = err;
00087     return -1;
00088   }
00089   // Lock the port
00090   pvLock.l_type = F_WRLCK;
00091   pvLock.l_whence = SEEK_SET;
00092   pvLock.l_start = 0;
00093   pvLock.l_len = 0;
00094   if(fcntl(pvFd, F_SETLK, &pvLock) < 0){
00095     err =errno;
00096     mdtError e(MDT_PORT_IO_ERROR, "Unable to lock " + pvPort.absoluteFilePath(), mdtError::Error);
00097     e.setSystemError(err, strerror(err));
00098     MDT_ERROR_SET_SRC(e, "mdtPortLock");
00099     e.commit();
00100     removeLockFiles();
00101     ::close(pvFd);
00102     pvFd = -1;
00103     errno = err;
00104     return -1;
00105   }
00106   // Ok, port is open and locked here
00107   pvIsLocked = true;
00108   pvIsLockedByAnother = false;
00109 
00110   return pvFd;
00111 }

00160     mdtError e(MDT_PORT_IO_ERROR, "No lock directory found", mdtError::Warning);
