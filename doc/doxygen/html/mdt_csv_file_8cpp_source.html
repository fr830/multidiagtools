00251   bool escapeEqualProtection = QString(escapeChar) == dataProtection;
00252   QByteArray escapedDataProtection;

00254   int esDpCount;
00255 
00256   // If no data protection is given, we are never in a protected section
00257   if(dataProtection.isEmpty()){
00258     return false;
00259   }
00260   // Build the escaped data protection version
00261   escapedDataProtection.append(escapeChar.toAscii());
00262   escapedDataProtection += dataProtection.toAscii();
00263   // Count data protection occurences
00264   dpCount = line.count(dataProtection.toAscii());
00265   // Count escaped data protection occurences

00267     esDpCount = 0;

00269     esDpCount = line.count(escapedDataProtection);
00270   }
00271 
00272   return ((dpCount-esDpCount)%2);
