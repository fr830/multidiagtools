119 void mdtSqlWindow::connectEditionElements()
120 {
121  Q_ASSERT(pvForm != 0);
122 
123  std::shared_ptr<mdtSqlDataWidgetController> controller;
124 
125  controller = pvForm->mainTableController<mdtSqlDataWidgetController>();
126  Q_ASSERT(controller);
127  // Connect actions enable/disable
128  connect(controller.get(), SIGNAL(insertEnabledStateChanged(bool)), actInsert, SLOT(setEnabled(bool)));
129  connect(controller.get(), SIGNAL(submitEnabledStateChanged(bool)), actSubmit, SLOT(setEnabled(bool)));
130  connect(controller.get(), SIGNAL(revertEnabledStateChanged(bool)), actRevert, SLOT(setEnabled(bool)));
131  connect(controller.get(), SIGNAL(removeEnabledStateChanged(bool)), actRemove, SLOT(setEnabled(bool)));
132  // Connect actions triggers
133  connect(actInsert, SIGNAL(triggered()), controller.get(), SLOT(insert()));
134  connect(actSubmit, SIGNAL(triggered()), controller.get(), SLOT(submit()));
135  connect(actRevert, SIGNAL(triggered()), controller.get(), SLOT(revert()));
136  connect(actRemove, SIGNAL(triggered()), controller.get(), SLOT(remove()));
