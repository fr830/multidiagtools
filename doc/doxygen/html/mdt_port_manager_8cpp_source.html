00396 {
00397   Q_ASSERT(granularity > 0);
00398 
00399   int maxIter;
00400 
00401   if(timeout == 0){
00402     timeout = adjustedReadTimeout(timeout, false);
00403   }else{
00404     timeout = adjustedReadTimeout(timeout);
00405   }
00406 
00407   // Check if transaction exists in pending queue
00408   // Note: we check also done queue, because it's possible that transaction was done
00409   if((!pvTransactionsPending.contains(id))&&(!pvTransactionsDone.contains(id))){
00410     mdtError e(MDT_PORT_IO_ERROR, "Wait on a frame that was never added to pending queue, id: " + QString::number(id), mdtError::Warning);
00411     MDT_ERROR_SET_SRC(e, "mdtPortManager");
00412     e.commit();
00413     return false;
00414   }
00415   // Try until success or timeout
00416   if(timeout < 0){  // Case of infinite timeout
00417     while(!pvTransactionsDone.contains(id)){
00418       if(readWaitCanceled()){
00419         return false;
00420       }
00421       // Wait
00422       qApp->processEvents();
00423       msleep(granularity);
00424     }
00425   }else{
00426     maxIter = timeout / granularity;
00427     while(!pvTransactionsDone.contains(id)){
00428       if(readWaitCanceled()){
00429         return false;
00430       }
00431       // Check about timeout
00432       if(maxIter <= 0){
00433         return false;
00434       }
00435       // Wait
00436       qApp->processEvents();
00437       msleep(granularity);
00438       maxIter--;
00439     }

00560     if(pvTransactionsEnabled){
00561       QMutableMapIterator<int, mdtPortTransaction*> itPending(pvTransactionsPending);

00579         itDone.next();
00580         pvTransactionsPool.enqueue(itDone.value());
