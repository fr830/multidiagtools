00059 mdtAbstractPort::error_t mdtDeviceScpi::connectToDevice(const mdtDeviceInfo &devInfo)
00060 {
00061   mdtPortInfo *port;
00062   QList<mdtPortInfo*> ports;
00063   QList<mdtDeviceInfo*> devices;
00065   int i, j;
00066   mdtAbstractPort::error_t retVal;
00067 
00068   // Setup device info
00072   // Scan for ports having a USBTMC device attached
00073   port = 0;
00074   ports = pvUsbtmcPortManager->scan();
00075   for(i=0; i<ports.size(); i++){
00076     Q_ASSERT(ports.at(i) != 0);
00077     // Search in devices list
00078     qDebug() << "port " << ports.at(i)->portName();
00079     devices = ports.at(i)->deviceInfoList();
00080     for(j=0; j<devices.size(); j++){
00081       Q_ASSERT(devices.at(j) != 0);
00082       // Check if VID and PID matches
00083       if((devices.at(j)->vendorId() == devInfo.vendorId())&&(devices.at(j)->productId() == devInfo.productId())){
00084         // If requested, check if serial ID matches
00085         if(!devInfo.serialId().isEmpty()){
00086           if(devices.at(j)->serialId() == devInfo.serialId()){
00087             port = ports.at(i);
00088             i = ports.size();
00089             break;
00090           }
00091         }else{
00092           // Only PID and VID must match
00093           port = ports.at(i);
00094           i = ports.size();
00095           break;
00096         }
00097       }
00098     }
00099   }
00100   // Open port if found
00101   if(port != 0){
00102     portManager()->setPortInfo(*port);
00103     if(portManager()->openPort()){
00104       if(portManager()->start()){
00105         retVal = mdtAbstractPort::NoError;
00106       }else{
00107         retVal = mdtAbstractPort::UnhandledError;
00108       }
00109     }else{
00110       retVal = mdtAbstractPort::UnhandledError;
00111     }
00112   }else{
00113     // Device not found
00115     retVal = mdtAbstractPort::PortNotFound;
00116   }
00117   // Free port info list
00118   qDeleteAll(ports);
00119   ports.clear();
00120 
00121   return retVal;
00122 }
